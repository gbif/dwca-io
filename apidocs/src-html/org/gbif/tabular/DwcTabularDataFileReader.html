001package org.gbif.tabular;
002
003import org.gbif.dwc.terms.Term;
004import org.gbif.dwc.terms.TermFactory;
005import org.gbif.dwca.io.ArchiveField;
006import org.gbif.utils.file.tabular.TabularDataFileReader;
007
008import java.io.IOException;
009import java.util.List;
010import java.util.Map;
011
012import com.google.common.base.Strings;
013import com.google.common.collect.Maps;
014
015/**
016 * Specialized TermTabularDataFileReader for DarwinCore archive file.
017 * In addition of reading the underlying csv file this class will handle id field and default values that can
018 * be provided in DarwinCore.
019 */
020public class DwcTabularDataFileReader extends TermTabularDataFileReader {
021
022  //refers to a column in the archive representing the id of the line that is not mapped to a real term
023  public static Term DEFAULT_ID_TERM = TermFactory.instance().findTerm("dwcaId");
024
025  private Map<Term, String> defaultValues = Maps.newHashMap();
026
027  public DwcTabularDataFileReader(TabularDataFileReader<List<String>> tabularDataFileReader, ArchiveField idField,
028                                  List<ArchiveField> fields) {
029    super(tabularDataFileReader, buildColumnMapping(idField, fields));
030
031    for(ArchiveField af : fields) {
032      if(!Strings.isNullOrEmpty(af.getDefaultValue())){
033        defaultValues.put(af.getTerm(), af.getDefaultValue());
034      }
035    }
036  }
037
038  private static Term[] buildColumnMapping(ArchiveField idField, List<ArchiveField> dwcTerms){
039    Map<Integer, Term> columnMapping = Maps.newTreeMap();
040    columnMapping.put(idField.getIndex(), idField.getTerm() != null ? idField.getTerm() : DEFAULT_ID_TERM);
041    for(ArchiveField af : dwcTerms) {
042      if(af.getIndex() != null && af.getIndex() >= 0){
043        columnMapping.put(af.getIndex(), af.getTerm());
044      }
045    }
046    return columnMapping.values().toArray(new Term[0]);
047  }
048
049  @Override
050  public TermTabularDataLine read() throws IOException {
051    TermTabularDataLine line = super.read();
052
053    if(line == null){
054      return null;
055    }
056
057    if(defaultValues.isEmpty()){
058      return line;
059    }
060
061    Map<Term, String> mappedData = Maps.newHashMap(line.getMappedData());
062    int numberOfColumn = line.getNumberOfColumn();
063
064    for(Term term : defaultValues.keySet()) {
065      if(mappedData.containsKey(term)){
066        if(Strings.isNullOrEmpty(mappedData.get(term))){
067          mappedData.put(term, defaultValues.get(term));
068        }
069      }
070      else {
071        mappedData.put(term, defaultValues.get(term));
072        numberOfColumn++;
073      }
074    }
075
076    return new TermTabularDataLine(line.getLineNumber(), mappedData, numberOfColumn);
077  }
078
079}
