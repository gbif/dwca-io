001package org.gbif.tabular;
002
003import org.gbif.dwc.terms.Term;
004import org.gbif.utils.file.tabular.TabularDataFileReader;
005
006import java.io.IOException;
007import java.util.List;
008import java.util.Map;
009
010import com.google.common.collect.Maps;
011
012/**
013 * Decorator of {@link TabularDataFileReader} to map elements of a line to a TermTabularDataLine.
014 * Supports the reading of lines smaller or larger than the declared set of columns.
015 *
016 */
017public class TermTabularDataFileReader implements TabularDataFileReader<TermTabularDataLine> {
018
019  private final TabularDataFileReader<List<String>> tabularDataFileReader;
020  private final Term[] columnMapping;
021  private int lineNumber = 0;
022
023  /**
024   *
025   * @param tabularDataFileReader a TabularDataFileReader implementation to read lines as List of String
026   * @param columnMapping mapping of the columns. Index of the array matched the index of the element in the data line
027   */
028  public TermTabularDataFileReader(TabularDataFileReader<List<String>> tabularDataFileReader, Term[] columnMapping){
029    this.tabularDataFileReader = tabularDataFileReader;
030    this.columnMapping = columnMapping;
031  }
032
033  @Override
034  public List<String> getHeaderLine() throws IOException {
035    return tabularDataFileReader.getHeaderLine();
036  }
037
038  /**
039   *
040   * @return
041   * @throws IOException
042   */
043  public TermTabularDataLine read() throws IOException {
044    List<String> tabularLine = tabularDataFileReader.read();
045
046    // check for end of file
047    if(tabularLine == null){
048      return null;
049    }
050    int numOfColumns = Math.min(columnMapping.length, tabularLine.size());
051    Map<Term, String> line = Maps.newHashMapWithExpectedSize(numOfColumns);
052    for (int i = 0; i < numOfColumns; i++) {
053      line.put(columnMapping[i], tabularLine.get(i));
054    }
055    lineNumber++;
056
057    //if the line contains more data than the number of declared column, set it as unamppedData
058    if(tabularLine.size() > columnMapping.length) {
059      return new TermTabularDataLine(lineNumber, line, tabularLine.size() ,
060              tabularLine.subList(columnMapping.length, tabularLine.size()));
061    }
062    return new TermTabularDataLine(lineNumber, line, tabularLine.size());
063  }
064
065  @Override
066  public void close() {
067    tabularDataFileReader.close();
068  }
069
070}
