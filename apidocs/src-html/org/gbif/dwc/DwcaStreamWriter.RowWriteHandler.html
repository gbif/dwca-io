024import java.util.Map;
025
026import javax.annotation.Nullable;
027
028import org.apache.commons.io.IOUtils;
029import org.slf4j.Logger;

031

033import com.google.common.base.Throwables;
034import com.google.common.collect.Maps;

036/**
037 * An archive writer that writes entire data files at once and does not check integrity of coreids.
038 * In large archives using extensions this yields a much, much better performance than writing star record by star record.
039 */
040public class DwcaStreamWriter implements AutoCloseable {
041  private static final Logger LOG = LoggerFactory.getLogger(DwcaStreamWriter.class);
042
043  private final File dir;
044  private final Term core;
045  private final Term coreIdTerm;
046  private final boolean useHeaders;
047  private final Archive archive = new Archive();
048  private String metadata;
049  private Map<String, String> constituents = Maps.newHashMap();
050
051  /**
052   * @param dir the directory to use as the archive
053   * @param coreRowType the archives core row type
054   * @param coreIdTerm if given used to map the id column of the core
055   * @param useHeaders if true write a single header row for each data file
056   */
057  public DwcaStreamWriter(File dir, Term coreRowType, @Nullable Term coreIdTerm, boolean useHeaders) {
058    this.dir = dir;
059    this.core = coreRowType;
060    this.coreIdTerm = coreIdTerm;
061    this.useHeaders = useHeaders;
062    archive.setLocation(dir);
063  }
064
065  private File dataFile(Term rowType) {
066    return new File(dir, rowType.simpleName()+".tsv");
067  }
068
069  private static ArchiveField idField(int column) {
070    ArchiveField field = new ArchiveField();
071    field.setIndex(column);
072    return field;
073  }
074  /**
075   * @param coreIdColumn zero based index to the rows coreid
076   * @param mapping zero based indexed of the rows
077   */
078  public void write(Term rowType, int coreIdColumn, Map<Term, Integer> mapping, Iterable<String[]> rows) {
079    Preconditions.checkNotNull(rows);
080
081    final int maxMapping = maxMappingColumn(mapping);
082    try (TabWriter writer = addArchiveFile(rowType, coreIdColumn, mapping, maxMapping) ) {
083      // write data
084      for (String[] row : rows) {
085        write(writer, row, maxMapping);
086      }
087    } catch (IOException e) {
088      Throwables.propagate(e);
089    }
090  }
091
092  private static void write(TabWriter writer, String[] row, int maxMapping) throws IOException {
093    if (row != null && row.length < maxMapping) {
094      throw new IllegalArgumentException("Input rows are smaller than the defined mapping of " + maxMapping + " columns.");
095    }
096    writer.write(row);
097  }
098
099  private int maxMappingColumn(Map<Term, Integer> mapping) {
100    return mapping.values().stream().max(Integer::compareTo).get();
101  }
102
103  private TabWriter addArchiveFile(Term rowType, int coreIdColumn, Map<Term, Integer> mapping, int maxMapping) throws IOException {
104    Preconditions.checkNotNull(rowType);
105    Preconditions.checkNotNull(mapping);
106    Preconditions.checkArgument(!mapping.isEmpty());
107    Preconditions.checkArgument(coreIdColumn >= 0);
108
109    final File dataFile = dataFile(rowType);
110    ArchiveFile af = ArchiveFile.buildTabFile();
111    af.setEncoding("UTF-8");
112    af.setRowType(rowType);
113    af.addLocation(dataFile.getName());
114    af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
115    af.setId(idField(coreIdColumn));
116    for (Map.Entry<Term, Integer> entry : mapping.entrySet()) {
117      ArchiveField field = new ArchiveField();
118      field.setTerm(entry.getKey());
119      field.setIndex(entry.getValue());
120      af.addField(field);
121    }
122    if (core.equals(rowType)) {
123      if (coreIdTerm != null) {
124        af.getId().setTerm(coreIdTerm);
125      }
126      archive.setCore(af);
127    } else {
128      archive.addExtension(af);
129    }
130
131    // write headers
132    TabWriter writer = TabWriter.fromFile(dataFile);
133    if (useHeaders){
134      String[] header = new String[maxMapping+1];
135      mapping.entrySet().stream().sorted(Map.Entry.comparingByValue()).forEach((e)->{
136        header[e.getValue()] = e.getKey().simpleName();
137      });
138      if (coreIdTerm != null) {
139        header[0] = coreIdTerm.simpleName();
140      }
141      writer.write(header);
142    }
143
144    return writer;
145  }
146
147  public interface RowWriteHandler extends AutoCloseable {
148    void write(String[] row);
149  }
150
151  private class RowWriteHandlerImpl implements RowWriteHandler {
152    private final TabWriter writer;
153    private final int minColumns;
154
155    RowWriteHandlerImpl(TabWriter writer, int minColumns) {
156      this.writer = writer;
157      this.minColumns = minColumns;
158    }
159
160    @Override
161    public void write(String[] row) {
162      if (row != null && row.length < minColumns) {
163        throw new IllegalArgumentException("Input rows are smaller than the defined mapping of " + minColumns + " columns.");
164      }
165      try {
166        writer.write(row);
167      } catch (IOException e) {
168        throw new IllegalStateException(e);
169      }
170    }
171
172    @Override
173    public void close() throws IOException {
174      writer.close();
175    }
176  }
177
178  /**
179   * Useful for glueing mybatis result handlers directly into the dwca writer.
180   * Make sure to close the write handler properly!
181   * @param rowType
182   * @param coreIdColumn
183   * @param mapping
184   * @return a handler accepting single rows to write into the data file
185   */
186  public RowWriteHandler writeHandler(Term rowType, int coreIdColumn, Map<Term, Integer> mapping) {
187    try {
188      final int maxMapping = maxMappingColumn(mapping);
189      TabWriter writer = addArchiveFile(rowType, coreIdColumn, mapping, maxMapping);
190      return new RowWriteHandlerImpl(writer, maxMapping);
191
192    } catch (IOException e) {
193      throw new IllegalStateException(e);
194    }
195  }
196
197  public void setMetadata(String metadata, String metadataLocation) throws IOException {
198    setMetadata(IOUtils.toInputStream(metadata, StandardCharsets.UTF_8), metadataLocation);
199  }
200
201  public void setMetadata(InputStream metadata, String metadataLocation) throws IOException {
202    DwcaWriter.writeMetadata(metadata, new File(dir, metadataLocation));
203    archive.setMetadataLocation(metadataLocation);
204  }
205
206  public void setMetadataLocation(String metadataLocation) {
207    archive.setMetadataLocation(metadataLocation);
208  }
209
210  /**
211   * Adds a constituent dataset.
212   * The eml file will be called as the datasetID which has to be unique.
213   */
214  public void addConstituent(String datasetID, String metadata) {
215    this.constituents.put(datasetID, metadata);
216  }
217
218  /**
219   * Writes meta.xml and eml.xml to the archive.
220   */
221  @Override
222  public void close() throws IOException {
223    checkCoreRowType();
224    addConstituents();
225    MetaDescriptorWriter.writeMetaFile(archive);
226    LOG.info("Wrote archive to {}", archive.getLocation().getAbsolutePath());
227  }
228
229  /**
230   * check if core row type has been written
231   */
232  private void checkCoreRowType() {
233    if (archive.getCore() == null) {
234      throw new IllegalStateException("The core data file has not yet been written for " + core.qualifiedName());
235    }
236  }
237
238  private void addConstituents() throws IOException {
239    if (!constituents.isEmpty()) {
240      File ddir = new File(dir, Archive.CONSTITUENT_DIR);
241      ddir.mkdirs();
242      for (Map.Entry<String, String> de : constituents.entrySet()) {
243        DwcaWriter.writeMetadata(de.getValue(), new File(ddir, de.getKey()+".xml"));
244      }
245    }
246  }
