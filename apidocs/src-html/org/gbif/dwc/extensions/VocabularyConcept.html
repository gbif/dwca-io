014package org.gbif.dwc.extensions;
015
016import java.net.MalformedURLException;
017import java.net.URL;
018import java.util.Comparator;
019import java.util.HashSet;
020import java.util.Objects;
021import java.util.Set;
022
023/**
024 * A single, identifiable concept in a vocabulary. For example "DE" is an identifier for the concept of Germany, while
025 * "Germany" is the preferred term or representation used in the English language. "Deutschland" represents the
026 * preferred term in German, but both Germany and Deutschland are simply terms representing a single concept in a
027 * vocabulary.
028 *
029 * Taken from https://github.com/gbif/dwca-validator3/
030 */
031public class VocabularyConcept implements Comparable<org.gbif.dwc.extensions.VocabularyConcept> {
032
033  private static final Comparator<org.gbif.dwc.extensions.VocabularyConcept> COMPARATOR =
034          Comparator.comparing(org.gbif.dwc.extensions.VocabularyConcept::getVocabulary,
035                  Comparator.nullsFirst(Comparator.naturalOrder()))
036                  .thenComparing(org.gbif.dwc.extensions.VocabularyConcept::getOrder)
037                  .thenComparing(org.gbif.dwc.extensions.VocabularyConcept::getUri);
038
039  private Vocabulary vocabulary;
040  private String identifier; // usually short, e.g. DE
041  private String description;
042  private String uri; // a URI denoting the concept, mostly used in rdf
043  private URL link; // web link to some more human documentation
044  private int order = -1; // to maintain any custom order not based on a natural concept property
045  private Set<org.gbif.dwc.extensions.VocabularyTerm> alternativeTerms = new HashSet<>();
046  private Set<org.gbif.dwc.extensions.VocabularyTerm> preferredTerms = new HashSet<>();
047
048  public void addAlternativeTerm(org.gbif.dwc.extensions.VocabularyTerm term) {
049    alternativeTerms.add(term);
050  }
051
052  public void addPreferredTerm(org.gbif.dwc.extensions.VocabularyTerm term) {
053    preferredTerms.add(term);
054  }
055
056
057  public Set<org.gbif.dwc.extensions.VocabularyTerm> getAlternativeTerms() {
058    return alternativeTerms;
059  }
060
061  public String getDescription() {
062    return description;
063  }
064
065  public String getIdentifier() {
066    return identifier;
067  }
068
069  public URL getLink() {
070    return link;
071  }
072
073  public Integer getOrder() {
074    return order;
075  }
076
077  public org.gbif.dwc.extensions.VocabularyTerm getPreferredTerm(String lang) {
078    org.gbif.dwc.extensions.VocabularyTerm tEN = null;
079    for (org.gbif.dwc.extensions.VocabularyTerm t : preferredTerms) {
080      if (t.getLang().equalsIgnoreCase(lang)) {
081        return t;
082      } else if (t.getLang().equalsIgnoreCase("en")) {
083        tEN = t;
084      }
085    }
086    return tEN;
087  }
088
089  public Set<org.gbif.dwc.extensions.VocabularyTerm> getPreferredTerms() {
090    return preferredTerms;
091  }
092
093  /**
094   * @return a set of all terms, preferred or alternative, for this concept
095   */
096  public Set<org.gbif.dwc.extensions.VocabularyTerm> getTerms() {
097    Set<org.gbif.dwc.extensions.VocabularyTerm> t = new HashSet<org.gbif.dwc.extensions.VocabularyTerm>(preferredTerms);
098    t.addAll(alternativeTerms);
099    return t;
100  }
101
102  public String getUri() {
103    return uri;
104  }
105
106  public Vocabulary getVocabulary() {
107    return vocabulary;
108  }
109
110
111  public void setAlternativeTerms(Set<org.gbif.dwc.extensions.VocabularyTerm> alternativeTerms) {
112    this.alternativeTerms = alternativeTerms;
113  }
114
115  public void setDescription(String description) {
116    this.description = description;
117  }
118
119  public void setIdentifier(String identifier) {
120    this.identifier = identifier;
121  }
122
123  public void setLink(String link) {
124    try {
125      this.link = new URL(link);
126    } catch (MalformedURLException e) {
127      // silently ignore malformed URLs
128    }
129  }
130
131  public void setLink(URL link) {
132    this.link = link;
133  }
134
135  public void setOrder(int order) {
136    this.order = order;
137  }
138
139  public void setPreferredTerms(Set<VocabularyTerm> preferredTerms) {
140    this.preferredTerms = preferredTerms;
141  }
142
143  public void setUri(String uri) {
144    this.uri = uri;
145  }
146
147  public void setVocabulary(Vocabulary vocabulary) {
148    this.vocabulary = vocabulary;
149  }
150
151
152  @Override
153  public int hashCode() {
154    return Objects.hash(vocabulary, identifier, uri);
155  }
156
157  @Override
158  public int compareTo(org.gbif.dwc.extensions.VocabularyConcept object) {
159    return COMPARATOR.compare(this, object);
160  }
161
162  @Override
163  public boolean equals(Object other) {
164    if (this == other) {
165      return true;
166    }
167    if (!(other instanceof org.gbif.dwc.extensions.VocabularyConcept)) {
168      return false;
169    }
170    org.gbif.dwc.extensions.VocabularyConcept o = (org.gbif.dwc.extensions.VocabularyConcept) other;
171    return Objects.equals(vocabulary, o.vocabulary) &&
172            Objects.equals(identifier, o.identifier) &&
173            Objects.equals(uri, o.uri);
174  }
175
176  @Override
177  public String toString() {
178    return uri;
179  }
180
181}
