001package org.gbif.dwc.extensions;
002
003import org.gbif.digester.CallParamNoNSRule;
004import org.gbif.digester.ThesaurusHandlingRule;
005
006import java.io.IOException;
007import java.io.InputStream;
008import java.net.URL;
009import javax.xml.parsers.ParserConfigurationException;
010import javax.xml.parsers.SAXParserFactory;
011
012import org.apache.commons.digester3.Digester;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015import org.xml.sax.SAXException;
016
017/**
018 *
019 */
020public class ExtensionFactory {
021
022  protected static Logger log = LoggerFactory.getLogger(ExtensionFactory.class);
023  public static final String EXTENSION_NAMESPACE = "http://rs.gbif.org/extension/";
024  private ThesaurusHandlingRule thesaurusRule;
025  private SAXParserFactory saxf;
026
027  public ExtensionFactory(ThesaurusHandlingRule thesaurusRule, SAXParserFactory factory) {
028    super();
029    this.thesaurusRule = thesaurusRule;
030    this.saxf = factory;
031  }
032
033  /**
034   * Builds an extension from the supplied input stream
035   *
036   * @param is For the XML
037   * @return The extension
038   * @throws SAXException
039   * @throws IOException
040   * @throws ParserConfigurationException
041   */
042  public Extension build(InputStream is, URL url, boolean dev) throws IOException, SAXException, ParserConfigurationException {
043
044    // in order to deal with arbitrary namespace prefixes we need to parse namespace aware!
045    Digester digester = new Digester(saxf.newSAXParser());
046    digester.setRuleNamespaceURI(EXTENSION_NAMESPACE);
047
048    Extension e = new Extension(url, dev);
049    digester.push(e);
050
051    digester.addCallMethod("*/extension", "setTitle", 1);
052    digester.addRule("*/extension", new CallParamNoNSRule(0, "title"));
053
054    digester.addCallMethod("*/extension", "setName", 1);
055    digester.addCallParam("*/extension", 0, "name");
056
057    digester.addCallMethod("*/extension", "setNamespace", 1);
058    digester.addCallParam("*/extension", 0, "namespace");
059
060    digester.addCallMethod("*/extension", "setRowType", 1);
061    digester.addCallParam("*/extension", 0, "rowType");
062
063    digester.addCallMethod("*/extension", "setLink", 1);
064    digester.addRule("*/extension", new CallParamNoNSRule(0, "relation"));
065
066    digester.addCallMethod("*/extension", "setDescription", 1);
067    digester.addRule("*/extension", new CallParamNoNSRule(0, "description"));
068
069    digester.addCallMethod("*/extension", "setSubject", 1);
070    digester.addRule("*/extension", new CallParamNoNSRule(0, "subject"));
071
072    // build the properties
073    digester.addObjectCreate("*/property", ExtensionProperty.class);
074
075    digester.addCallMethod("*/property", "setQualname", 1);
076    digester.addCallParam("*/property", 0, "qualName");
077
078    digester.addCallMethod("*/property", "setName", 1);
079    digester.addCallParam("*/property", 0, "name");
080
081    digester.addCallMethod("*/property", "setNamespace", 1);
082    digester.addCallParam("*/property", 0, "namespace");
083
084    digester.addCallMethod("*/property", "setGroup", 1);
085    digester.addCallParam("*/property", 0, "group");
086
087    digester.addCallMethod("*/property", "setType", 1);
088    digester.addCallParam("*/property", 0, "type");
089
090    digester.addCallMethod("*/property", "setRequired", 1);
091    digester.addCallParam("*/property", 0, "required");
092
093    digester.addCallMethod("*/property", "setLink", 1);
094    digester.addRule("*/property", new CallParamNoNSRule(0, "relation"));
095
096    digester.addCallMethod("*/property", "setDescription", 1);
097    digester.addRule("*/property", new CallParamNoNSRule(0, "description"));
098
099    digester.addCallMethod("*/property", "setExamples", 1);
100    digester.addCallParam("*/property", 0, "examples");
101
102    digester.addCallMethod("*/property", "setColumnLength", 1);
103    digester.addCallParam("*/property", 0, "columnLength");
104
105    // This is a special rule that will use the url2ThesaurusMap
106    // to set the Vocabulary based on the attribute "thesaurus"
107    digester.addRule("*/property", thesaurusRule);
108
109    digester.addSetNext("*/property", "addProperty");
110
111    digester.parse(is);
112    return e;
113  }
114}
