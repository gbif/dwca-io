016package org.gbif.dwc.extensions;
017
018import org.gbif.dwc.digester.CallParamNoNSRule;
019import org.gbif.dwc.digester.ThesaurusHandlingRule;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.net.URL;
024
025import javax.xml.parsers.ParserConfigurationException;
026import javax.xml.parsers.SAXParserFactory;
027
028import org.apache.commons.digester3.Digester;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031import org.xml.sax.SAXException;
032
033public class ExtensionFactory {
034
035  protected static Logger log = LoggerFactory.getLogger(ExtensionFactory.class);
036  public static final String EXTENSION_NAMESPACE = "http://rs.gbif.org/extension/";
037  private ThesaurusHandlingRule thesaurusRule;
038  private SAXParserFactory saxf;
039
040  public ExtensionFactory(ThesaurusHandlingRule thesaurusRule, SAXParserFactory factory) {
041    super();
042    this.thesaurusRule = thesaurusRule;
043    this.saxf = factory;
044  }
045
046  /**
047   * Builds an extension from the supplied input stream
048   *
049   * @param is For the XML
050   * @return The extension
051   * @throws SAXException
052   * @throws IOException
053   * @throws ParserConfigurationException
054   */
055  public Extension build(InputStream is, URL url, boolean dev) throws IOException, SAXException, ParserConfigurationException {
056
057    // in order to deal with arbitrary namespace prefixes we need to parse namespace aware!
058    Digester digester = new Digester(saxf.newSAXParser());
059    digester.setRuleNamespaceURI(EXTENSION_NAMESPACE);
060
061    Extension e = new Extension(url, dev);
062    digester.push(e);
063
064    digester.addCallMethod("*/extension", "setTitle", 1);
065    digester.addRule("*/extension", new CallParamNoNSRule(0, "title"));
066
067    digester.addCallMethod("*/extension", "setName", 1);
068    digester.addCallParam("*/extension", 0, "name");
069
070    digester.addCallMethod("*/extension", "setNamespace", 1);
071    digester.addCallParam("*/extension", 0, "namespace");
072
073    digester.addCallMethod("*/extension", "setRowType", 1);
074    digester.addCallParam("*/extension", 0, "rowType");
075
076    digester.addCallMethod("*/extension", "setLink", 1);
077    digester.addRule("*/extension", new CallParamNoNSRule(0, "relation"));
078
079    digester.addCallMethod("*/extension", "setDescription", 1);
080    digester.addRule("*/extension", new CallParamNoNSRule(0, "description"));
081
082    digester.addCallMethod("*/extension", "setSubject", 1);
083    digester.addRule("*/extension", new CallParamNoNSRule(0, "subject"));
084
085    // build the properties
086    digester.addObjectCreate("*/property", ExtensionProperty.class);
087
088    digester.addCallMethod("*/property", "setQualname", 1);
089    digester.addCallParam("*/property", 0, "qualName");
090
091    digester.addCallMethod("*/property", "setName", 1);
092    digester.addCallParam("*/property", 0, "name");
093
094    digester.addCallMethod("*/property", "setNamespace", 1);
095    digester.addCallParam("*/property", 0, "namespace");
096
097    digester.addCallMethod("*/property", "setGroup", 1);
098    digester.addCallParam("*/property", 0, "group");
099
100    digester.addCallMethod("*/property", "setType", 1);
101    digester.addCallParam("*/property", 0, "type");
102
103    digester.addCallMethod("*/property", "setRequired", 1);
104    digester.addCallParam("*/property", 0, "required");
105
106    digester.addCallMethod("*/property", "setLink", 1);
107    digester.addRule("*/property", new CallParamNoNSRule(0, "relation"));
108
109    digester.addCallMethod("*/property", "setDescription", 1);
110    digester.addRule("*/property", new CallParamNoNSRule(0, "description"));
111
112    digester.addCallMethod("*/property", "setExamples", 1);
113    digester.addCallParam("*/property", 0, "examples");
114
115    digester.addCallMethod("*/property", "setColumnLength", 1);
116    digester.addCallParam("*/property", 0, "columnLength");
117
118    // This is a special rule that will use the url2ThesaurusMap
119    // to set the Vocabulary based on the attribute "thesaurus"
120    digester.addRule("*/property", thesaurusRule);
121
122    digester.addSetNext("*/property", "addProperty");
123
124    digester.parse(is);
125    return e;
126  }
127}
