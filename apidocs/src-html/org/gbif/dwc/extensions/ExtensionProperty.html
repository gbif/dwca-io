012import org.gbif.dwc.terms.Term;
013import org.gbif.dwc.terms.TermFactory;
014
015import java.net.URI;
016import java.util.Objects;
017
018/**
019 * Taken from https://github.com/gbif/dwca-validator3/
020 */
021public class ExtensionProperty implements Comparable<org.gbif.dwc.extensions.ExtensionProperty>, Term {
022  private org.gbif.dwc.extensions.Extension extension;
023  private String name;
024  private String namespace;
025  private String qualname;
026  private String type;
027  private String group;
028  private int columnLength = 255; // sensible default
029  private String link;
030  private String examples;
031  private String description;
032  private boolean required;
033  private Vocabulary vocabulary;
034
035  public ExtensionProperty() {
036    super();
037  }
038
039  /**
040   * Construct a new property with a single qualified name. Parses out the name and sets the namespace to end with a
041   * slash or #
042   *
043   * @param qualName
044   */
045  public ExtensionProperty(String qualName) {
046    super();
047    setQualname(qualName);
048  }
049
050  /**
051   * Compare by group and qualified name as default sorting order
052   *
053   * @see Comparable#compareTo(Object)
054   */
055  public int compareTo(org.gbif.dwc.extensions.ExtensionProperty prop) {
056    if (group != null) {
057      int x = this.group.compareTo(prop.group);
058      if (x != 0) {
059        return x;
060      }
061    }
062    return this.qualname.compareTo(prop.qualname);
063  }
064
065  /**
066   * Just compare the unique qualified names to see if extension properties are equal
067   *
068   * @see Object#equals(Object)
069   */
070  @Override
071  public boolean equals(Object other) {
072    if (this == other) {
073      return true;
074    }
075    if (!(other instanceof org.gbif.dwc.extensions.ExtensionProperty)) {
076      return false;
077    }
078    org.gbif.dwc.extensions.ExtensionProperty o = (org.gbif.dwc.extensions.ExtensionProperty) other;
079    return Objects.equals(extension, o.extension) && Objects.equals(qualname, o.qualname);
080  }
081
082  /**
083   * The length of the database column to be generated when the extension property is installed. Also used to trim
084   * incoming data before SQL insert is generated. For LOB columns use -1 or any negative value
085   *
086   * @return
087   */
088  public int getColumnLength() {
089    return columnLength;
090  }
091
092  public String getDescription() {
093    return description;
094  }
095
096  public String getExamples() {
097    return examples;
098  }
099
100  public org.gbif.dwc.extensions.Extension getExtension() {
101    return extension;
102  }
103
104  public String getGroup() {
105    return group;
106  }
107
108  public String getLink() {
109    return link;
110  }
111
112  public String getName() {
113    return name;
114  }
115
116  public String getNamespace() {
117    return namespace;
118  }
119
120  public String getQualname() {
121    return qualname;
122  }
123
124  public String getType() {
125    return type;
126  }
127
128  public Vocabulary getVocabulary() {
129    return vocabulary;
130  }
131
132  /**
133   * @see Object#hashCode()
134   */
135  @Override
136  public int hashCode() {
137    return Objects.hash(extension, qualname);
138  }
139
140  public boolean hasTerms() {
141    return vocabulary != null;
142  }
143
144  public boolean isRequired() {
145    return required;
146  }
147
148  /*
149   * (non-Javadoc)
150   * @see org.gbif.dwc.terms.ConceptTerm#qualifiedName()
151   */
152  @Override
153  public String qualifiedName() {
154    return qualname;
155  }
156
157  @Override
158  public boolean isClass() {
159    return false;
160  }
161
162  @Override
163  public String prefix() {
164    return "";
165  }
166
167  @Override
168  public URI namespace() {
169    return URI.create(namespace);
170  }
171
172  /*
173   * (non-Javadoc)
174   * @see org.gbif.dwc.terms.ConceptTerm#qualifiedNormalisedName()
175   */
176  public String qualifiedNormalisedName() {
177    return TermFactory.normaliseTerm(qualifiedName());
178  }
179
180  public void setColumnLength(int columnLength) {
181    this.columnLength = columnLength;
182  }
183
184  // required for SAX parser
185  public void setColumnLength(String columnLength) {
186    try {
187      this.columnLength = Integer.parseInt(columnLength);
188    } catch (NumberFormatException e) {
189      // swallow stupidity
190    }
191  }
192
193  public void setDescription(String description) {
194    this.description = description;
195  }
196
197  public void setExamples(String examples) {
198    this.examples = examples;
199  }
200
201  public void setExtension(Extension extension) {
202    this.extension = extension;
203  }
204
205  public void setGroup(String group) {
206    this.group = group;
207  }
208
209  public void setLink(String link) {
210    this.link = link;
211  }
212
213  public void setName(String name) {
214    this.name = name;
215  }
216
217  public void setNamespace(String namespace) {
218    this.namespace = namespace;
219  }
220
221  public void setQualname(String qualname) {
222    this.qualname = qualname;
223  }
224
225  public void setRequired(boolean required) {
226    this.required = required;
227  }
228
229  // required by SAX parser
230  public void setRequired(String required) {
231    if ("TRUE".equalsIgnoreCase(required) || "T".equalsIgnoreCase(required) || "1".equalsIgnoreCase(required)) {
232      this.required = true;
233    } else if ("FALSE".equalsIgnoreCase(required) || "F".equalsIgnoreCase(required) || "0".equalsIgnoreCase(required)) {
234      this.required = false;
235    }
236
237    // or we just don't change if not understood
238  }
239
240  public void setType(String type) {
241    this.type = type;
242  }
243
244  public void setVocabulary(Vocabulary vocabulary) {
245    this.vocabulary = vocabulary;
246  }
247
248  /*
249   * (non-Javadoc)
250   * @see org.gbif.dwc.terms.ConceptTerm#simpleName()
251   */
252  public String simpleName() {
253    return name;
254  }
255
256  @Override
257  public String prefixedName() {
258    return simpleName();
259  }
260
261  /*
262   * (non-Javadoc)
263   * @see org.gbif.dwc.terms.ConceptTerm#simpleNormalisedAlternativeNames()
264   */
265  public String[] simpleNormalisedAlternativeNames() {
266    return new String[]{};
267  }
268
269  /*
270   * (non-Javadoc)
271   * @see org.gbif.dwc.terms.ConceptTerm#simpleNormalisedName()
272   */
273  public String simpleNormalisedName() {
274    return TermFactory.normaliseTerm(simpleName());
275  }
276
277  /**
278   * @see Object#toString()
279   */
280  @Override
281  public String toString() {
282    return qualname;
283  }
284
285}
