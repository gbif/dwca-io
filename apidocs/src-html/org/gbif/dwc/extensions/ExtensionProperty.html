014package org.gbif.dwc.extensions;
015
016import org.gbif.dwc.terms.Term;
017import org.gbif.dwc.terms.TermFactory;
018
019import java.net.URI;
020import java.util.Objects;
021
022/**
023 * Taken from https://github.com/gbif/dwca-validator3/
024 */
025public class ExtensionProperty implements Comparable<ExtensionProperty>, Term {
026  private Extension extension;
027  private String name;
028  private String namespace;
029  private String qualname;
030  private String type;
031  private String group;
032  private int columnLength = 255; // sensible default
033  private String link;
034  private String examples;
035  private String description;
036  private boolean required;
037  private Vocabulary vocabulary;
038
039  public ExtensionProperty() {
040    super();
041  }
042
043  /**
044   * Construct a new property with a single qualified name. Parses out the name and sets the namespace to end with a
045   * slash or #
046   *
047   * @param qualName
048   */
049  public ExtensionProperty(String qualName) {
050    super();
051    setQualname(qualName);
052  }
053
054  /**
055   * Compare by group and qualified name as default sorting order
056   *
057   * @see Comparable#compareTo(Object)
058   */
059  @Override
060  public int compareTo(org.gbif.dwc.extensions.ExtensionProperty prop) {
061    if (group != null) {
062      int x = this.group.compareTo(prop.group);
063      if (x != 0) {
064        return x;
065      }
066    }
067    return this.qualname.compareTo(prop.qualname);
068  }
069
070  /**
071   * Just compare the unique qualified names to see if extension properties are equal
072   *
073   * @see Object#equals(Object)
074   */
075  @Override
076  public boolean equals(Object other) {
077    if (this == other) {
078      return true;
079    }
080    if (!(other instanceof org.gbif.dwc.extensions.ExtensionProperty)) {
081      return false;
082    }
083    org.gbif.dwc.extensions.ExtensionProperty o = (org.gbif.dwc.extensions.ExtensionProperty) other;
084    return Objects.equals(extension, o.extension) && Objects.equals(qualname, o.qualname);
085  }
086
087  /**
088   * The length of the database column to be generated when the extension property is installed. Also used to trim
089   * incoming data before SQL insert is generated. For LOB columns use -1 or any negative value
090   *
091   * @return
092   */
093  public int getColumnLength() {
094    return columnLength;
095  }
096
097  public String getDescription() {
098    return description;
099  }
100
101  public String getExamples() {
102    return examples;
103  }
104
105  public org.gbif.dwc.extensions.Extension getExtension() {
106    return extension;
107  }
108
109  public String getGroup() {
110    return group;
111  }
112
113  public String getLink() {
114    return link;
115  }
116
117  public String getName() {
118    return name;
119  }
120
121  public String getNamespace() {
122    return namespace;
123  }
124
125  public String getQualname() {
126    return qualname;
127  }
128
129  public String getType() {
130    return type;
131  }
132
133  public Vocabulary getVocabulary() {
134    return vocabulary;
135  }
136
137  /**
138   * @see Object#hashCode()
139   */
140  @Override
141  public int hashCode() {
142    return Objects.hash(extension, qualname);
143  }
144
145  public boolean hasTerms() {
146    return vocabulary != null;
147  }
148
149  public boolean isRequired() {
150    return required;
151  }
152
153  /*
154   * (non-Javadoc)
155   * @see org.gbif.dwc.terms.ConceptTerm#qualifiedName()
156   */
157  @Override
158  public String qualifiedName() {
159    return qualname;
160  }
161
162  @Override
163  public boolean isClass() {
164    return false;
165  }
166
167  @Override
168  public String prefix() {
169    return "";
170  }
171
172  @Override
173  public URI namespace() {
174    return URI.create(namespace);
175  }
176
177  /*
178   * (non-Javadoc)
179   * @see org.gbif.dwc.terms.ConceptTerm#qualifiedNormalisedName()
180   */
181  public String qualifiedNormalisedName() {
182    return TermFactory.normaliseTerm(qualifiedName());
183  }
184
185  public void setColumnLength(int columnLength) {
186    this.columnLength = columnLength;
187  }
188
189  // required for SAX parser
190  public void setColumnLength(String columnLength) {
191    try {
192      this.columnLength = Integer.parseInt(columnLength);
193    } catch (NumberFormatException e) {
194      // swallow stupidity
195    }
196  }
197
198  public void setDescription(String description) {
199    this.description = description;
200  }
201
202  public void setExamples(String examples) {
203    this.examples = examples;
204  }
205
206  public void setExtension(Extension extension) {
207    this.extension = extension;
208  }
209
210  public void setGroup(String group) {
211    this.group = group;
212  }
213
214  public void setLink(String link) {
215    this.link = link;
216  }
217
218  public void setName(String name) {
219    this.name = name;
220  }
221
222  public void setNamespace(String namespace) {
223    this.namespace = namespace;
224  }
225
226  public void setQualname(String qualname) {
227    this.qualname = qualname;
228  }
229
230  public void setRequired(boolean required) {
231    this.required = required;
232  }
233
234  // required by SAX parser
235  public void setRequired(String required) {
236    if ("TRUE".equalsIgnoreCase(required) || "T".equalsIgnoreCase(required) || "1".equalsIgnoreCase(required)) {
237      this.required = true;
238    } else if ("FALSE".equalsIgnoreCase(required) || "F".equalsIgnoreCase(required) || "0".equalsIgnoreCase(required)) {
239      this.required = false;
240    }
241
242    // or we just don't change if not understood
243  }
244
245  public void setType(String type) {
246    this.type = type;
247  }
248
249  public void setVocabulary(Vocabulary vocabulary) {
250    this.vocabulary = vocabulary;
251  }
252
253  /*
254   * (non-Javadoc)
255   * @see org.gbif.dwc.terms.ConceptTerm#simpleName()
256   */
257  @Override
258  public String simpleName() {
259    return name;
260  }
261
262  @Override
263  public String prefixedName() {
264    return simpleName();
265  }
266
267  /*
268   * (non-Javadoc)
269   * @see org.gbif.dwc.terms.ConceptTerm#simpleNormalisedAlternativeNames()
270   */
271  public String[] simpleNormalisedAlternativeNames() {
272    return new String[]{};
273  }
274
275  /*
276   * (non-Javadoc)
277   * @see org.gbif.dwc.terms.ConceptTerm#simpleNormalisedName()
278   */
279  public String simpleNormalisedName() {
280    return TermFactory.normaliseTerm(simpleName());
281  }
282
283  /**
284   * @see Object#toString()
285   */
286  @Override
287  public String toString() {
288    return qualname;
289  }
290
