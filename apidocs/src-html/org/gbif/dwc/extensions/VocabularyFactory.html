016package org.gbif.dwc.extensions;
017
018import org.gbif.dwc.digester.CallParamNoNSRule;
019
020import java.io.IOException;
021import java.io.InputStream;
022
023import javax.xml.parsers.ParserConfigurationException;
024import javax.xml.parsers.SAXParserFactory;
025
026import org.apache.commons.digester3.Digester;
027import org.xml.sax.SAXException;
028
029/**
030 * Building from XML definitions
031 */
032public class VocabularyFactory {
033
034  public static final String VOCABULARY_NAMESPACE = "http://rs.gbif.org/thesaurus/";
035
036  private final SAXParserFactory saxf;
037
038  public VocabularyFactory(SAXParserFactory saxf) {
039    this.saxf = saxf;
040  }
041
042  /**
043   * Builds a Vocabulary from the supplied input stream
044   *
045   * @param is For the XML
046   * @return The extension
047   * @throws SAXException
048   * @throws IOException
049   * @throws ParserConfigurationException
050   */
051  public Vocabulary build(InputStream is) throws IOException, SAXException, ParserConfigurationException {
052    Digester digester = new Digester(saxf.newSAXParser());
053    digester.setNamespaceAware(true);
054    digester.setXIncludeAware(false);
055    digester.setRuleNamespaceURI(VOCABULARY_NAMESPACE);
056
057    Vocabulary tv = new Vocabulary();
058    digester.push(tv);
059
060    // build the thesaurus
061    digester.addCallMethod("*/thesaurus", "setTitle", 1);
062    digester.addRule("*/thesaurus", new CallParamNoNSRule(0, "title"));
063
064    digester.addCallMethod("*/thesaurus", "setDescription", 1);
065    digester.addRule("*/thesaurus", new CallParamNoNSRule(0, "description"));
066
067    digester.addCallMethod("*/thesaurus", "setLink", 1);
068    digester.addRule("*/thesaurus", new CallParamNoNSRule(0, "relation"));
069
070    digester.addCallMethod("*/thesaurus", "setUri", 1);
071    digester.addRule("*/thesaurus", new CallParamNoNSRule(0, "URI"));
072
073    // build the concept
074    digester.addObjectCreate("*/concept", VocabularyConcept.class);
075
076    digester.addCallMethod("*/concept", "setLink", 1);
077    digester.addRule("*/concept", new CallParamNoNSRule(0, "relation"));
078
079    digester.addCallMethod("*/concept", "setDescription", 1);
080    digester.addRule("*/concept", new CallParamNoNSRule(0, "description"));
081
082    digester.addCallMethod("*/concept", "setUri", 1);
083    digester.addRule("*/concept", new CallParamNoNSRule(0, "URI"));
084
085    digester.addCallMethod("*/concept", "setIdentifier", 1);
086    digester.addRule("*/concept", new CallParamNoNSRule(0, "identifier"));
087
088    // build the terms
089    digester.addObjectCreate("*/preferred/term", VocabularyTerm.class);
090    VocabularyTerm t = new VocabularyTerm();
091
092    digester.addCallMethod("*/preferred/term", "setLang", 1);
093    digester.addRule("*/preferred/term", new CallParamNoNSRule(0, "lang"));
094
095    digester.addCallMethod("*/preferred/term", "setTitle", 1);
096    digester.addRule("*/preferred/term", new CallParamNoNSRule(0, "title"));
097
098    digester.addSetNext("*/preferred/term", "addPreferredTerm");
099
100    // build alternative terms
101    digester.addObjectCreate("*/alternative/term", VocabularyTerm.class);
102    VocabularyTerm talt = new VocabularyTerm();
103
104    digester.addCallMethod("*/alternative/term", "setLang", 1);
105    digester.addRule("*/alternative/term", new CallParamNoNSRule(0, "lang"));
106
107    digester.addCallMethod("*/alternative/term", "setTitle", 1);
108    digester.addRule("*/alternative/term", new CallParamNoNSRule(0, "title"));
109
110    digester.addSetNext("*/alternative/term", "addAlternativeTerm");
111
112    // add concept
113    digester.addSetNext("*/concept", "addConcept");
114
115    digester.parse(is);
116    return tv;
117  }
118
119}
