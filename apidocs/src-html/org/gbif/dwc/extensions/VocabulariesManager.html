001package org.gbif.dwc.extensions;
002
003import java.net.URL;
004import java.util.List;
005import java.util.Map;
006
007/**
008 * Main interface of a {@link org.gbif.dwc.extensions.Vocabulary} manager.
009 */
010public interface VocabulariesManager {
011  /**
012   * Retrieve vocabulary by its unique global URI identifier from installed vocabularies.
013   *
014   * @param uri unique URI identifying the vocabulary as given in the vocabulary definition
015   * @return the installed vocabulary or null if not found
016   */
017  org.gbif.dwc.extensions.Vocabulary get(String uri);
018
019  /**
020   * Returns the parsed vocabulary located at the given URL. If downloaded already it will return the cached copy or
021   * otherwise download it from the URL.
022   *
023   * @param url the resolvable URL that locates the xml vocabulary definition
024   * @return
025   */
026  org.gbif.dwc.extensions.Vocabulary get(URL url);
027
028  /**
029   * Returns a regular map than can be used to populate html select drop downs with
030   * keys=vocabulary concept identifiers and values=preferred term for the given language.
031   * Defaults to english if no term for the requested language exists.
032   *
033   * @param uri the identifier for the vocabulary
034   * @param lang a 2 character iso language code, e.g. DE
035   * @return
036   */
037  Map<String, String> getI18nVocab(String uri, String lang);
038
039  /**
040   * Lists all locally known vocabularies
041   *
042   * @return
043   */
044  List<Vocabulary> list();
045}
