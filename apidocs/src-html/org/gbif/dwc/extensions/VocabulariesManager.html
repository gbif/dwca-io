016package org.gbif.dwc.extensions;
017
018import java.net.URL;
019import java.util.List;
020import java.util.Map;
021
022/**
023 * Main interface of a {@link org.gbif.dwc.extensions.Vocabulary} manager.
024 */
025public interface VocabulariesManager {
026  /**
027   * Retrieve vocabulary by its unique global URI identifier from installed vocabularies.
028   *
029   * @param uri unique URI identifying the vocabulary as given in the vocabulary definition
030   * @return the installed vocabulary or null if not found
031   */
032  org.gbif.dwc.extensions.Vocabulary get(String uri);
033
034  /**
035   * Returns the parsed vocabulary located at the given URL. If downloaded already it will return the cached copy or
036   * otherwise download it from the URL.
037   *
038   * @param url the resolvable URL that locates the xml vocabulary definition
039   * @return
040   */
041  org.gbif.dwc.extensions.Vocabulary get(URL url);
042
043  /**
044   * Returns a regular map than can be used to populate html select drop downs with
045   * keys=vocabulary concept identifiers and values=preferred term for the given language.
046   * Defaults to english if no term for the requested language exists.
047   *
048   * @param uri the identifier for the vocabulary
049   * @param lang a 2 character iso language code, e.g. DE
050   * @return
051   */
052  Map<String, String> getI18nVocab(String uri, String lang);
053
054  /**
055   * Lists all locally known vocabularies
056   *
057   * @return
058   */
059  List<Vocabulary> list();
060}
