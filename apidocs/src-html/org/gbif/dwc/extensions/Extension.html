012import org.gbif.dwc.terms.Term;
013import org.gbif.dwc.terms.TermFactory;
014
015import java.net.MalformedURLException;
016import java.net.URL;
017import java.util.ArrayList;
018import java.util.Comparator;
019import java.util.Date;
020import java.util.List;
021import java.util.Objects;
022
023/**
024 * A Darwin Core extension definition
025 *
026 * Taken from https://github.com/gbif/dwca-validator3/
027 */
028public class Extension implements Comparable<org.gbif.dwc.extensions.Extension>{
029
030  private static final Comparator<org.gbif.dwc.extensions.Extension> COMPARATOR =
031          Comparator.comparing(((org.gbif.dwc.extensions.Extension ext) -> ext.getRowType() == null ? null :
032                  ext.getRowType().qualifiedName()), Comparator.nullsLast(Comparator.naturalOrder()))
033                  .thenComparing(ext -> ext.getUrl() == null ? null : ext.getUrl().toString(),
034                          Comparator.nullsLast(Comparator.naturalOrder()));
035
036  private String title; // human title
037  private String name; // table, file & xml tag naming. no whitespace allowed
038  private URL url;
039  private Term rowType;
040  private String subject;
041  private String description;
042  private String namespace;
043  private URL link; // to documentation
044  private boolean installed;
045  private List<ExtensionProperty> properties = new ArrayList<>();
046  private boolean core = false;
047  private boolean dev = false;
048  private Date modified = new Date();
049
050  public Extension(URL url, boolean dev) {
051    super();
052    this.url = url;
053    this.dev = dev;
054  }
055
056  public void addProperty(ExtensionProperty property) {
057    property.setExtension(this);
058    properties.add(property);
059  }
060
061  @Override
062  public int compareTo(org.gbif.dwc.extensions.Extension object) {
063    return COMPARATOR.compare(this, object);
064  }
065
066  @Override
067  public boolean equals(Object other) {
068    if (this == other) {
069      return true;
070    }
071    if (!(other instanceof org.gbif.dwc.extensions.Extension)) {
072      return false;
073    }
074    org.gbif.dwc.extensions.Extension o = (org.gbif.dwc.extensions.Extension) other;
075    return Objects.equals(rowType, o.rowType) &&
076            Objects.equals(url.toString(), o.url.toString());
077  }
078
079  public String getDescription() {
080    return description;
081  }
082
083  public URL getLink() {
084    return link;
085  }
086
087  public Date getModified() {
088    return modified;
089  }
090
091  public String getName() {
092    return name;
093  }
094
095  public String getNamespace() {
096    return namespace;
097  }
098
099  public List<ExtensionProperty> getProperties() {
100    return properties;
101  }
102
103  public ExtensionProperty getProperty(Term term) {
104    return getProperty(term.qualifiedName());
105  }
106
107  public ExtensionProperty getProperty(String term) {
108    if (term == null) {
109      return null;
110    }
111    for (ExtensionProperty p : properties) {
112      if (term.equalsIgnoreCase(p.getQualname())) {
113        return p;
114      }
115    }
116    return null;
117  }
118
119  public Term getRowType() {
120    return rowType;
121  }
122
123  public String getSubject() {
124    return subject;
125  }
126
127  public String getTitle() {
128    return title;
129  }
130
131  public URL getUrl() {
132    return url;
133  }
134
135  /**
136   * @see Objects#hash
137   */
138  @Override
139  public int hashCode() {
140    return Objects.hash(rowType, url);
141  }
142
143  public boolean hasProperty(Term term) {
144    return getProperty(term) != null;
145  }
146
147  public boolean hasProperty(String term) {
148    return getProperty(term) != null;
149  }
150
151  public boolean isCore() {
152    return core;
153  }
154
155  public boolean isDev() {
156    return dev;
157  }
158
159  public boolean isInstalled() {
160    return installed;
161  }
162
163  public void setCore(boolean core) {
164    this.core = core;
165  }
166
167  public void setDescription(String description) {
168    this.description = description;
169  }
170
171  public void setInstalled(boolean installed) {
172    this.installed = installed;
173  }
174
175  public void setLink(String link) {
176    URL url;
177    try {
178      url = new URL(link);
179      this.link = url;
180    } catch (MalformedURLException e) {
181    }
182  }
183
184  public void setLink(URL link) {
185    this.link = link;
186  }
187
188  public void setModified(Date modified) {
189    this.modified = modified;
190  }
191
192  public void setName(String name) {
193    this.name = name.replaceAll("\\s", "_");
194  }
195
196  public void setNamespace(String namespace) {
197    this.namespace = namespace;
198  }
199
200  public void setProperties(List<ExtensionProperty> properties) {
201    this.properties = properties;
202  }
203
204  public void setRowType(String rowType) {
205    setRowType(TermFactory.instance().findClassTerm(rowType));
206  }
207
208  public void setRowType(Term rowType) {
209    this.rowType = rowType;
210  }
211
212  public void setSubject(String subject) {
213    this.subject = subject;
214  }
215
216  public void setTitle(String title) {
217    this.title = title;
218  }
219
220  @Override
221  public String toString() {
222    return new StringBuilder().append("name:" + this.name).append(", rowType:" + this.rowType).toString();
223  }
224
225}
