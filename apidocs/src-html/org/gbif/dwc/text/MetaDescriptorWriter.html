017package org.gbif.dwc.text;
018
019import org.gbif.registry.metadata.EMLWriter;
020
021import java.io.File;
022import java.io.FileWriter;
023import java.io.IOException;
024import java.io.Writer;
025
026import freemarker.cache.ClassTemplateLoader;
027import freemarker.cache.TemplateLoader;
028import freemarker.template.Configuration;
029import freemarker.template.TemplateException;
030
031/**
032 * Utils class to write a meta.xml descriptor file.
033 */
034public class MetaDescriptorWriter {
035
036  private static final String META_TEMPLATE = "meta.ftl";
037  private static final String TEMPLATE_PATH = "/templates";
038  private static final Configuration FTL = provideFreemarker();
039
040  private MetaDescriptorWriter() {
041  }
042
043  public static void writeMetaFile(File f, Archive archive) throws IOException {
044    Writer writer = new FileWriter(f);
045    try {
046      FTL.getTemplate(META_TEMPLATE).process(archive, writer);
047    } catch (TemplateException e) {
048      throw new IOException("Error while processing the meta.xml template", e);
049    }
050    writer.close();
051  }
052
053  /**
054   * Provides a freemarker template loader. It is configured to access the utf8 templates folder on the classpath, i.e.
055   * /src/resources/templates
056   */
057  private static Configuration provideFreemarker() {
058    // load templates from classpath by prefixing /templates
059    TemplateLoader tl = new ClassTemplateLoader(EMLWriter.class, TEMPLATE_PATH);
060
061    Configuration fm = new Configuration();
062    fm.setDefaultEncoding("utf8");
063    fm.setTemplateLoader(tl);
064
065    return fm;
066  }
067}
