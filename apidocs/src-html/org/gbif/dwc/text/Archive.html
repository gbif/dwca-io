037/**
038 * A darwin core star archive allowing easy reading and iteration over a core record with all its extensions.
039 *
040 * @see <a href="http://darwincore.googlecode.com/svn/trunk/terms/guides/text/index.htm">Darwin Core Text Guide</a>
041 */
042public class Archive implements Iterable<StarRecord> {
043
044  /**
045   * An iterator of fixed DarwinCoreRecords over the core file only. This iterator doesn't need any sorted data files
046   * as it doesn't deal with extensions.
047   */
048  static class ArchiveDwcIterator implements ClosableIterator<DarwinCoreRecord> {
049
050    private CSVReader coreReader;
051    private ArchiveFile core;
052    private int lineCount = 0;
053    private final RecordImpl record;
054    private boolean hasNext = true;
055    private final Set<Term> mappedTerms = new HashSet<Term>();
056
057    ArchiveDwcIterator(Archive archive) {
058      record = new RecordImpl(archive.getCore(), true);
059      core = archive.getCore();
060      // remember used DwC and DC terms
061      for (DwcTerm term : DwcTerm.values()) {
062        if (core.hasTerm(term)) {
063          mappedTerms.add(term);
064        }
065      }
066      for (DcTerm term : DcTerm.values()) {
067        if (core.hasTerm(term)) {
068          mappedTerms.add(term);
069        }
070      }
071      try {
072        coreReader = archive.getCore().getCSVReader();
073        // read first core row
074        record.setRow(coreReader.next());
075        if (!record.hasRow()) {
076          hasNext = false;
077        }
078      } catch (Exception e) {
079        hasNext = false;
080        LOG.warn("Exception caught", e);
081      }
082    }
083
084    public void close() {
085      coreReader.close();
086    }
087
088    public boolean hasNext() {
089      return hasNext;
090    }
091
092    public DarwinCoreRecord next() {
093      DarwinCoreRecord dwc = new DarwinCoreRecord();
094      lineCount++;
095      try {
096        for (Term term : mappedTerms) {
097          dwc.setProperty(term, record.value(term));
098        }
099        dwc.setId(record.id());
100        // read next line to see if it exists at all
101        record.setRow(coreReader.next());
102        if (!record.hasRow()) {
103          hasNext = false;
104        }
105
106      } catch (Exception e) {
107        LOG.warn("Bad row somewhere around core line: {}", lineCount, e);
108      }
109      return dwc;
110    }
111
112    public void remove() {
113      throw new UnsupportedOperationException("Cannot remove a row from archive files");
114    }
115
116  }

118  /**
119   * An iterator over core records of the archive that returns StarRecords, i.e. a single core record with all its
120   * related extension records attached. This is a convenient way to iterate over an entire archive accessing all
121   * information including all extensions.
122   *
123   * Requires underlying data files to be sorted by the coreid column if iteration spans multiple data files, i.e. if
124   * extensions exist.
125   * Extension rows with a non existing coreid are skipped. This requires that we use the same sorting order in the
126   * java code as we use for sorting the data files!
127   */
128  class ArchiveIterator implements ClosableIterator<StarRecord> {
129
130    private final StarRecord rec;
131    private RecordIterator coreIter;
132    private Set<RecordIterator> closables = new HashSet<RecordIterator>();
133    private Map<String, PeekingIterator<Record>> extensionIters = new HashMap<String, PeekingIterator<Record>>();
134    private Map<String, Integer> extensionRecordsSkipped = new HashMap<String, Integer>();
135
136    /**
137     * @param replaceNulls if true replaces common literal null values in all records
138     */
139    ArchiveIterator(Archive archive, boolean replaceNulls) {
140      List<String> rowTypes = new ArrayList<String>();
141
142      try {
143        if (extensions.isEmpty()) {
144          // no need to sort
145          coreIter = RecordIterator.build(archive.getCore(), replaceNulls);
146        } else {
147          // sort data files to align extension records into a single star record
148          if (!archive.sorted) {
149            archive.sortFiles();
150          }
151          coreIter = buildSortedIterator(archive.getCore(), replaceNulls);
152        }
153      } catch (IOException e) {
154        LOG.warn("IOException opening core file", e);
155      }
156
157      for (ArchiveFile af : archive.getExtensions()) {
158        rowTypes.add(af.getRowType());
159        RecordIterator iter =
160          extensions.isEmpty() ? RecordIterator.build(af, replaceNulls) : buildSortedIterator(af, replaceNulls);
161        closables.add(iter);
162        extensionIters.put(af.getRowType(), Iterators.peekingIterator(iter));
163        extensionRecordsSkipped.put(af.getRowType(), 0);
164      }
165
166      rec = new StarRecord(rowTypes);
167    }
168
169    private RecordIterator buildSortedIterator(ArchiveFile af, boolean replaceNulls) {
170      // we need to sort the data files
171      String original = af.getLocation();
172      // temporarily modify archive file to create iterator over sorted file
173      af.getLocations().clear();
174      af.addLocation(ArchiveFile.getLocationSorted(original));
175      RecordIterator iter = RecordIterator.build(af, replaceNulls);
176      // revert to original
177      af.getLocations().clear();
178      af.addLocation(original);
179      return iter;
180    }
181
182    public void close() {
183      coreIter.close();
184      for (ClosableIterator<Record> it : closables) {
185        it.close();
186      }
187      for (Map.Entry<String, Integer> stringIntegerEntry : extensionRecordsSkipped.entrySet()) {
188        Integer skipped = stringIntegerEntry.getValue();
189        if (skipped > 0) {
190          LOG.debug("{} {} extension records without matching core", skipped, stringIntegerEntry.getKey());
191        }
192      }
193    }
194
195    public boolean hasNext() {
196      return coreIter.hasNext();
197    }
198
199    public StarRecord next() {
200      Record core = coreIter.next();
201      rec.newCoreRecord(core);
202      // add extension records if core id exists
203      if (core.id() != null) {
204        String id = core.id();
205        for (Map.Entry<String, PeekingIterator<Record>> ext : extensionIters.entrySet()) {
206          PeekingIterator<Record> it = ext.getValue();
207          String rowType = ext.getKey();
208          while (it.hasNext()) {
209            String extId = it.peek().id();
210            // make sure we have an extid
211            if (Strings.isNullOrEmpty(extId)) {
212              it.next();
213              continue;
214            }
215            if (id.equals(extId)) {
216              // extension row belongs to this core record
217              rec.addRecord(rowType, it.next());
218            } else if (id.compareTo(extId) > 0) {
219              // TODO: we need to use the exact same sorting order, ie comparator, as we use for sorting the data files!!!
220              // this extension id is smaller than the core id and should have been picked up by a core record already
221              // seems to have no matching core record, so lets skip it
222              it.next();
223              extensionRecordsSkipped.put(rowType, extensionRecordsSkipped.get(rowType) + 1);
224            } else {
225              // higher id, we need to wait for this one
226              break;
227            }
228          }
229        }
230      }
231
232      return rec;
233    }
234
235    public void remove() {
236      throw new UnsupportedOperationException("Cannot remove a row from archive files");
237    }
238
239  }

263  public ArchiveFile getExtension(String rowType, boolean allowUnqalifiedMatches) {
264    for (ArchiveFile af : extensions) {
265      if (af.getRowType() != null && af.getRowType().equalsIgnoreCase(rowType)) {
266        return af;
267      }
268    }
269    // no qualified row type matches, try unqualified row type matching?
270    if (allowUnqalifiedMatches) {
271      if (rowType.contains("/")) {
272        rowType = StringUtils.substringAfterLast(rowType, "/").trim();
273      }
274      for (ArchiveFile af : extensions) {
275        String rt = af.getRowType();
276        if (rt != null) {
277          if (rt.contains("/")) {
278            rt = StringUtils.substringAfterLast(rt, "/").trim();
279          }
280          if (rt != null && rt.equalsIgnoreCase(rowType)) {
281            return af;
282          }
283        }
284      }
285    }
286    return null;
287  }

328  /**
329   * Scans the archive for a semi standard support of dataset constituent metadata.
330   * A dataset constituent is a subdataset which is referenced via dwc:datasetID in the data.
331   * The agreement first introduced by catalogue of life for their GSDs is to have a new folder "dataset" that keeps
332   * a metadata file for each constituent named just as the datasetID and suffixed with .xml.
333   *
334   * @return map of constituent datasetID to metadata file inside the archive
335   */
336  public Map<String, File> getConstituentMetadata() {
337    Map<String, File> constituents = Maps.newHashMap();
338    File constDir = new File(location, "dataset");
339    if (constDir.exists()) {
340      File[] files = constDir.listFiles(new FilenameFilter() {
341            public boolean accept(File dir, String filename) { return filename.endsWith(".xml"); }
342          });
343      if (files != null) {
344        for (File cf : files) {
345          String name = cf.getName().split("\\.")[0];
346          constituents.put(name, cf);
347        }
348      }
349    }
350    return constituents;
351  }

393  /**
394   * Sorts all files according to file id, so that we can easily iterate over all files at once.
395   */
396  private void sortFiles() throws IOException {
397    FileUtils futil = new FileUtils();
398    // core
399    try {
400      futil.sort(core.getLocationFile(), ArchiveFile.getLocationFileSorted(core.getLocationFile()), core.getEncoding(),
401        core.getId().getIndex(), core.getFieldsTerminatedBy(), core.getFieldsEnclosedBy(), core.getLinesTerminatedBy(),
402        core.getIgnoreHeaderLines());
403    } catch (IOException e) {
404      LOG.error("Error sorting core file " + core.getLocationFile() + " : " + e.getMessage());
405      throw e;
406    } catch (RuntimeException e) {
407      LOG.error("Error sorting core file " + core.getLocationFile() + " : " + e.getMessage());
408      throw e;
409    }
410    // extensions
411    for (ArchiveFile ext : extensions) {
412      try {
413        futil.sort(ext.getLocationFile(), ArchiveFile.getLocationFileSorted(ext.getLocationFile()), ext.getEncoding(),
414          ext.getId().getIndex(), ext.getFieldsTerminatedBy(), ext.getFieldsEnclosedBy(), ext.getLinesTerminatedBy(),
415          ext.getIgnoreHeaderLines());
416      } catch (IOException e) {
417        LOG.error("Error sorting extension file " + ext.getLocationFile() + " : " + e.getMessage());
418        throw e;
419      } catch (RuntimeException e) {
420        LOG.error("Error sorting extension file " + ext.getLocationFile() + " : " + e.getMessage());
421        throw e;
422      }
423    }
424    sorted = true;
425  }
