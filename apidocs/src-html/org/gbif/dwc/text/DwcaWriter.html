044/**
045 * Simple writer class to create valid dwc archives using tab data files.
046 * The meta.xml descriptor is generated automatically and an optional EML metadata document can be added.
047 * The archive is NOT compressed but the final product is a directory with all the necessary files.
048 * For usage of this class please @see DwcaWriterTest.
049 */
050public class DwcaWriter {
051  private Logger log = LoggerFactory.getLogger(DwcaWriter.class);
052  private final File dir;
053  private final boolean useHeaders;
054  private long recordNum;
055  private String coreId;
056  private Map<Term, String> coreRow;
057  private final Term coreRowType;
058  private final Term coreIdTerm;
059  private final Map<Term, TabWriter> writers = Maps.newHashMap();
060  private final Set<Term> headersOut = Sets.newHashSet();
061  private final Map<Term, String> dataFileNames = Maps.newHashMap();
062  // key=rowType, value=columns
063  private final Map<Term, List<Term>> terms = Maps.newHashMap();
064  // key=rowType, value=default values per column
065  private final Map<Term, Map<Term, String>> defaultValues = Maps.newHashMap();
066  private Dataset eml;
067  
068  /**
069   * Creates a new writer without header rows.
070   * @param coreRowType the core row type.
071   * @param dir         the directory to create the archive in.
072   */
073  public DwcaWriter(Term coreRowType, File dir) throws IOException {
074    this(coreRowType, dir, false);

085  public DwcaWriter(Term coreRowType, File dir, boolean useHeaders) throws IOException {
086    this(coreRowType, null, dir, useHeaders);
087  }

089  /**
090   * If headers are used the first record must include all terms ever used for that file.
091   * If in subsequent rows additional terms are introduced an IllegalArgumentException is thrown.
092   * 
093   * @param coreRowType the core row type
094   * @param coreIdTerm the term of the id column
095   * @param dir the directory to create the archive in
096   * @param useHeaders if true the first row in every data file will include headers
097   */
098  public DwcaWriter(Term coreRowType, Term coreIdTerm, File dir, boolean useHeaders) throws IOException {
099    this.dir = dir;
100    this.coreRowType = coreRowType;
101    this.coreIdTerm = coreIdTerm;
102    this.useHeaders = useHeaders;
103    addRowType(coreRowType);
104  }

106  public static Map<Term, String> recordToMap(Record rec, ArchiveFile af) {
107    Map<Term, String> map = new HashMap<Term, String>();
108    for (Term t : af.getTerms()) {
109      map.put(t, rec.value(t));
110    }
111    return map;
112  }
113
114  public static String dataFileName(Term rowType) {
115    return rowType.simpleName().toLowerCase() + ".txt";
116  }
117
118  private void addRowType(Term rowType) throws IOException {
119    terms.put(rowType, new ArrayList<Term>());
120
121    String dfn = dataFileName(rowType);
122    dataFileNames.put(rowType, dfn);
123    File df = new File(dir, dfn);
124    FileUtils.forceMkdir(df.getParentFile());
125    OutputStream out = new FileOutputStream(df);
126    TabWriter wr = new TabWriter(out);
127    writers.put(rowType, wr);
128  }
129
130  /**
131   * A new core record is started and the last core and all extension records are written.
132   * @param id the new records id
133   * @throws IOException
134   */
135  public void newRecord(String id) throws IOException {
136    // flush last record
137    flushLastCoreRecord();
138    // start new
139    recordNum++;
140    coreId = id;
141    coreRow = new HashMap<Term, String>();
142  }
143
144  private void flushLastCoreRecord() throws IOException {
145    if (coreRow != null) {
146      writeRow(coreRow, coreRowType);
147    }
148  }
149
150  public long getRecordsWritten() {
151    return recordNum;
152  }
153
154  private void writeRow(Map<Term, String> rowMap, Term rowType) throws IOException {
155    TabWriter writer = writers.get(rowType);
156    List<Term> columns = terms.get(rowType);
157    if (useHeaders && !headersOut.contains(rowType)){
158      // write header row
159      writeHeader(writer, rowType, columns);
160    }
161
162    // make sure coreId is not null for extensions
163    if (coreRowType != rowType && coreId == null){
164      log.warn("Adding an {} extension record to a core without an Id! Skip this record", rowType);
165
166    } else {
167      String[] row = new String[columns.size() + 1];
168      row[0] = coreId;
169      for (Map.Entry<Term, String> conceptTermStringEntry : rowMap.entrySet()) {
170        int column = 1 + columns.indexOf(conceptTermStringEntry.getKey());
171        row[column] = conceptTermStringEntry.getValue();
172      }
173      writer.write(row);
174    }
175  }
176
177  private void writeHeader(TabWriter writer, Term rowType, List<Term> columns) throws IOException {
178    int idx = 0;
179    String[] row = new String[columns.size() + 1];
180    Term idTerm;
181    if (DwcTerm.Taxon == coreRowType){
182      idTerm = DwcTerm.taxonID;
183    } else if (DwcTerm.Occurrence == coreRowType){
184      idTerm = DwcTerm.occurrenceID;
185    } else if (DwcTerm.Identification == coreRowType){
186      idTerm = DwcTerm.identificationID;
187    } else if (DwcTerm.Event == coreRowType){
188      idTerm = DwcTerm.eventID;
189    } else {
190      // default to generic dc identifier for id column
191      idTerm = DcTerm.identifier;
192    }
193    row[idx] = idTerm.simpleName();
194
195    for (Term term : columns) {
196      idx ++;
197      row[idx] = term.simpleName();
198    }
199    writer.write(row);
200
201    headersOut.add(rowType);
202  }

205  /**
206   * Add a single value for the current core record.
207   * Calling this method requires that #newRecord() has been called at least once,
208   * otherwise an IllegalStateException is thrown.
209   * @param term
210   * @param value
211   */
212  public void addCoreColumn(Term term, String value) {
213    // ensure we do not overwrite the coreIdTerm if one is defined
214    if (coreIdTerm != null && coreIdTerm.equals(term)) {
215      throw new IllegalStateException("You cannot add a term that was specified as coreId term");
216    }
217    
218    List<Term> coreTerms = terms.get(coreRowType);
219    if (!coreTerms.contains(term)) {
220      if (useHeaders && recordNum>1){
221        throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
222      }
223      coreTerms.add(term);
224    }
225    try {
226      coreRow.put(term, value);
227    } catch (NullPointerException e) {
228      // no core record has been started yet
229      throw new IllegalStateException("No core record has been created yet. Call newRecord() at least once");
230    }
231  }

233  /**
234   * Add a default value to a term of the core.
235   * 
236   * @param term
237   * @param defaultValue
238   */
239  public void addCoreDefaultValue(Term term, String defaultValue){
240    addDefaultValue(coreRowType, term, defaultValue);
241  }
242  
243  /**
244   * Add a default value to a term of the provided rowType.
245   * 
246   * @param rowType
247   * @param term
248   * @param defaultValue
249   */
250  public void addDefaultValue(Term rowType, Term term, String defaultValue){
251    
252    if(!defaultValues.containsKey(rowType)){
253      defaultValues.put(rowType, new HashMap<Term, String>());
254    }
255    Map<Term,String> currentDefaultValues= defaultValues.get(rowType);
256    if(currentDefaultValues.containsKey(term)){
257      throw new IllegalStateException("The default value of term "+ term + " is already defined");
258    }
259    currentDefaultValues.put(term, defaultValue);
260  }

269  /**
270   * Add an extension record associated with the current core record.
271   * 
272   * @param rowType
273   * @param row
274   * @throws IOException
275   */
276  public void addExtensionRecord(Term rowType, Map<Term, String> row) throws IOException {
277    // make sure we know the extension rowtype
278    if (!terms.containsKey(rowType)) {
279      addRowType(rowType);
280    }
281    
282    // make sure we know all terms
283    List<Term> knownTerms = terms.get(rowType);
284    final boolean isFirst = knownTerms.isEmpty();
285    for (Term term : row.keySet()) {
286      if (!knownTerms.contains(term)) {
287        if (useHeaders && !isFirst){
288          throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
289        }
290        knownTerms.add(term);
291      }
292    }
293
294    // write extension record
295    writeRow(row, rowType);
296  }

312  /**
313   * Writes meta.xml and eml.xml to the archive and closes tab writers.
314   */
315  public void close() throws IOException {
316    addEml();
317    addMeta();
318    // flush last record
319    flushLastCoreRecord();
320    // TODO: add missing columns in second iteration of data files
321
322    // close writers
323    for (TabWriter w : writers.values()) {
324      w.close();
325    }
326  }
327
328  private void addEml() throws IOException {
329    if (eml != null) {
330      Writer writer = new FileWriter(new File(dir, "eml.xml"));
331      EMLWriter.write(eml, writer);
332    }
333  }
334
335  private void addMeta() throws IOException {
336    File metaFile = new File(dir, "meta.xml");
337
338    Archive arch = new Archive();
339    if (eml != null) {
340      arch.setMetadataLocation("eml.xml");
341    }
342    arch.setCore(buildArchiveFile(arch, coreRowType, coreIdTerm));
343    for (Term rowType : this.terms.keySet()) {
344      if (!coreRowType.equals(rowType)) {
345        arch.addExtension(buildArchiveFile(arch, rowType, null));
346      }
347    }
348    MetaDescriptorWriter.writeMetaFile(metaFile, arch);
349  }
350
351  private ArchiveFile buildArchiveFile(Archive archive, Term rowType, Term idTerm) {
352    ArchiveFile af = ArchiveFile.buildTabFile();
353    af.setArchive(archive);
354    af.addLocation(dataFileNames.get(rowType));
355
356    af.setEncoding("utf-8");
357    af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
358    af.setRowType(rowType.qualifiedName());
359
360    ArchiveField id = new ArchiveField();
361    id.setIndex(0);
362    af.setId(id);
363    // id an idTerm is provided, always use the index 0
364    if (idTerm != null) {
365      ArchiveField field = new ArchiveField();
366      field.setIndex(0);
367      field.setTerm(idTerm);
368      af.addField(field);
369    }
370    
371    Map<Term,String> termDefaultValueMap = defaultValues.get(rowType);
372    List<Term> rowTypeTerms = terms.get(rowType);
373    int idx = 0;
374    for (Term c : rowTypeTerms) {
375      idx++;
376      ArchiveField field = new ArchiveField();
377      field.setIndex(idx);
378      field.setTerm(c);
379      if(termDefaultValueMap !=null && termDefaultValueMap.containsKey(c)){
380        field.setDefaultValue(termDefaultValueMap.get(c));
381      }
382      af.addField(field);
383    }
384    
385    // check if default values are provided for this rowType
386    if(termDefaultValueMap != null){
387      ArchiveField field = null;
388      for (Term t : termDefaultValueMap.keySet()) {
389        if(!rowTypeTerms.contains(t)){
390          field = new ArchiveField();
391          field.setTerm(t);
392          field.setDefaultValue(termDefaultValueMap.get(t));
393          af.addField(field);
394        }
395      }
396    }
397
398    return af;
399  }
