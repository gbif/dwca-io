048/**
049 * This class encapsulates information about a file contained within a Darwin Core Archive. It
050 * represents the fileType object described in the Darwin Core Archive XSD, i.e. the core or extension elements.
051 *
052 * In the case of a fake single-file archive, it represents that.  In this case, archive is null.
053 *
054 * @see <a href="http://rs.tdwg.org/dwc/text/tdwg_dwc_text.xsd">Darwin Core Archive XSD</a>
055 */

062  public static final Term DEFAULT_ID_TERM = TERM_FACTORY.findPropertyTerm("ARCHIVE_RECORD_ID");

064  public static final String DEFAULT_FIELDS_TERMINATED_BY = ",";

067  /**
068   * Comparator to sort by column index, then term name.
069   */
070  private static final Comparator<ArchiveField> AF_IDX_COMPARATOR = (o1, o2) -> {
071    if (o1.getIndex() == null && o2.getIndex() == null) {
072      return 0;
073    } else if (o1.getIndex() == null) {
074      return -1;
075    } else if (o2.getIndex() == null) {
076      return 1;
077    } else if (o1.getIndex().equals(o2.getIndex())) {
078      return o1.getTerm().qualifiedName().compareTo(o2.getTerm().qualifiedName());
079    } else {
080      return o1.getIndex().compareTo(o2.getIndex());
081    }
082  };

114  protected void validateAsCore(boolean hasExtensions) throws UnsupportedArchiveException {
115    if (hasExtensions) {
116      if (id == null) {
117        LOG.warn("DwC-A core data file »" + title + "« is lacking an id column. No extensions allowed in this case");
118      }
119    }
120    validate();
121  }
122
123  protected void validateAsExtension() throws UnsupportedArchiveException {
124    if (id == null) {
125      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires an id or foreign key to the core id");
126    }
127    validate();
128  }
129
130  protected void validate() throws UnsupportedArchiveException {
131    if (getFirstLocation() == null) {
132      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a location");
133    }
134    for (File f : getLocationFiles()) {
135      if (!f.exists()) {
136        throw new UnsupportedArchiveException("DwC-A data file »" + title + "« does not exist");
137      }
138    }
139    if (encoding == null) {
140      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a character encoding");
141    }
142  }

144  /**
145   * Normalizes and sorts the data file, so it is suitable for in-order iteration as a core or extension file.
146   *
147   * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
148   */
149  /*
150   * Sorting implies a normalization phase to ensure we sort the file properly.
151   * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
152   * we want to sort.
153   *
154   * This method is synchronized to prevent multiple threads trying to normalize/sort a file at the same time,
155   * and uses an advisory lock to avoid conflict between multiple processes.
156   */
157  protected synchronized boolean normalizeAndSort() throws IOException {
158    List<File> filesToSort = getLocationFiles();
159    File sortedFile = getLocationFileSorted(getFirstLocationFile());
160
161    File lockFile = getLocationLockFile(getFirstLocationFile());
162    RandomAccessFile lockFileRA = new RandomAccessFile(lockFile, "rw");
163
164    FileLock lock = lockFileRA.getChannel().tryLock();
165    if (lock == null) {
166      LOG.warn("Another process has locked this DWCA for initialization; waiting until the lock is released.");
167      lock = lockFileRA.getChannel().lock();
168      LOG.warn("Other process has released lock; lock taken, proceeding.");
169    }
170
171    try {
172      // If we already sorted the file and its source didn't change we can avoid doing it again
173      long youngestFileTime = Long.MIN_VALUE;
174      for (File f : filesToSort) {
175        youngestFileTime = Math.max(youngestFileTime, f.lastModified());
176      }
177      if (sortedFile.exists() && sortedFile.lastModified() > youngestFileTime) {
178        LOG.debug("File {} is already sorted ({}B)", sortedFile, sortedFile.length());
179        return false;
180      }
181
182      List<File> normalizedFiles = normalizeIfRequired();
183      if (normalizedFiles != null) {
184        filesToSort = normalizedFiles;
185      }
186
187      FILE_UTILS.sort(filesToSort, sortedFile, getEncoding(),
188          getId().getIndex(), getFieldsTerminatedBy(), getFieldsEnclosedBy(),
189          TabularFileNormalizer.NORMALIZED_END_OF_LINE, getIgnoreHeaderLines());
190
191      if (normalizedFiles != null) {
192        for (File f : normalizedFiles) {
193          Files.deleteIfExists(f.toPath());
194        }
195      }
196
197      return true;
198    } finally {
199      lock.release();
200      lockFileRA.close();
201    }
202  }

204  /**
205   * Apply file normalization if required.
206   **
207   * @return normalizedFile or null if normalization was not applied
208   *
209   * @throws IOException
210   */
211  protected List<File> normalizeIfRequired() throws IOException {
212    // If the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
213    // we can skip normalization
214    boolean normalizationRequired = ! TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(getLinesTerminatedBy())
215        || getFieldsEnclosedBy() != null;
216
217    if (normalizationRequired) {
218      List<File> normalizedFiles = new ArrayList<>();
219      for (File f : getLocationFiles()) {
220        File normalizedFile = getLocationFileNormalized(f);
221        TabularFileNormalizer.normalizeFile(f.toPath(), normalizedFile.toPath(),
222          Charset.forName(getEncoding()), getFieldsTerminatedByChar(),
223          getLinesTerminatedBy(), getFieldsEnclosedBy());
224        normalizedFiles.add(normalizedFile);
225      }
226      return normalizedFiles;
227    }
228    return null;
229  }

311  /**
312   * Generates a list of {@link Term}s mapped to each column in the underlying file.
313   *
314   * A list may be empty when no Term is mapped at a specific position, or may contain more than one Term where
315   * the same column is mapped more than once.
316   *
317   * The length of the list is the maximum index used in the {@link ArchiveField}s + 1.
318   *
319   * @return Data structure representing the header of the file.
320   */
321  public List<List<Term>> getHeader() {
322    List<ArchiveField> archiveFieldsWithIndex = getFieldsSorted()
323            .stream().filter(af -> af.getIndex() != null)
324            .collect(Collectors.toList());
325
326    Optional<Integer> idIndex = id != null ? Optional.of(id.getIndex()) : Optional.empty();
327
328    if (archiveFieldsWithIndex.isEmpty() && !idIndex.isPresent()){
329      return new ArrayList<>();
330    }
331
332    int maxIndex = archiveFieldsWithIndex.stream().mapToInt(ArchiveField::getIndex).max().getAsInt();
333    maxIndex = Math.max(maxIndex, idIndex.orElse(-1));
334
335    List<List<Term>> terms = new ArrayList<>();
336    for (int i = 0; i <= maxIndex; i++) {
337      terms.add(new ArrayList<Term>());
338    }
339    archiveFieldsWithIndex.stream().forEach(af -> terms.get(af.getIndex()).add(af.getTerm()));
340    // Assign default term for id column.
341    idIndex.ifPresent(idx -> terms.get(idx).add(DEFAULT_ID_TERM));
342    return terms;
343  }

345  /**
346   * Get default values mapped to their {@link Term} (if any).
347   *
348   * @return default values or {@code Optional.empty()} if none
349   */
350  public Optional<Map<Term, String>> getDefaultValues() {
351    //check if there is default value(s) defined (look for absence of index)
352    Map<Term, String> defaultValues = fields.values().stream()
353            .filter(af -> StringUtils.isNotBlank(af.getDefaultValue()))
354            .collect(Collectors.toMap(ArchiveField::getTerm, ArchiveField::getDefaultValue));
355    return defaultValues.isEmpty() ? Optional.empty() : Optional.of(defaultValues);
356  }

421  /**
422   * Get the first file location.  Note archive files may use more than one location.
423   * @return first file or null if none
424   */
425  public File getFirstLocationFile() {
426    File dataFile;
427    if (archive != null) {
428      if (getFirstLocation() == null) {
429        // use only archive
430        dataFile = archive.getLocation();
431      } else if (getFirstLocation().startsWith("/")) {
432        // absolute already
433        dataFile = new File(getFirstLocation());
434      } else {
435        // use source file relative to archive dir
436        Path archiveLocation = archive.getLocation().toPath();
437        File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
438        dataFile = new File(directory, getFirstLocation());
439      }
440    } else {
441      dataFile = new File(getFirstLocation());
442    }
443
444    return dataFile;
445  }

461  /**
462   * Get Files for each location.  Note archive files may use more than one location.
463   */
464  public List<File> getLocationFiles() {
465    if (archive != null) {
466      if (getLocations() == null) {
467        // use only archive
468        return Collections.singletonList(archive.getLocation());
469      } else {
470        List<File> locations = new ArrayList<>();
471        for (String s : getLocations()) {
472          if (s.startsWith("/")) {
473            // absolute already
474            locations.add(new File(s));
475          } else {
476            // use source file relative to archive dir
477            Path archiveLocation = archive.getLocation().toPath();
478            File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
479            locations.add(new File(directory, s));
480          }
481        }
482        return locations;
483      }
484    } else {
485      return Collections.singletonList(new File(getFirstLocation()));
486    }
487  }

519  private Reader getReader(boolean sorted) throws IOException {
520    // ArchiveFile location, or Archive in case this is a fake single-file "archive".
521    File file = getFirstLocationFile() != null ? getFirstLocationFile() : getArchive().getLocation();
522    if (sorted) {
523      file = getLocationFileSorted(file);
524    }
525    return Files.newBufferedReader(file.toPath(), Charset.forName(getEncoding()));
526  }
527
528  /**
529   * Get a {@link ClosableIterator} over the records in this file, optionally replacing nulls and entities.
530   *
531   * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
532   * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
533   */
534  public ClosableIterator<Record> iterator(boolean replaceNulls, boolean replaceEntities) {
535    try {
536      // Use the sortedIterator (sorted into a single file) if there are multiple files.
537      if (getLocations().size() > 1) {
538        return sortedIterator(replaceNulls, replaceEntities);
539      }
540
541      TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(false),
542          getFieldsTerminatedByChar(), getLinesTerminatedBy(), getFieldsEnclosedBy(),
543          areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
544      return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
545    } catch (IOException e) {
546      throw new UnsupportedArchiveException(e);
547    }
548  }
549
550  /**
551   * Build an iterator pointing to the sorted tabular file.
552   * The sorted tabular file is also assumed to have been normalized.
553   *
554   * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
555   * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
556   */
557  protected ClosableIterator<Record> sortedIterator(boolean replaceNulls, boolean replaceEntities) throws IOException {
558    TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(true),
559            getFieldsTerminatedByChar(), TabularFileNormalizer.NORMALIZED_END_OF_LINE, getFieldsEnclosedBy(),
560            areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
561    return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
562  }
