066  /**
067   * Comparator to sort by column index, then term name.
068   */
069  private static final Comparator<ArchiveField> AF_IDX_COMPARATOR = (o1, o2) -> {
070    if (o1.getIndex() == null && o2.getIndex() == null) {
071      return 0;
072    } else if (o1.getIndex() == null) {
073      return -1;
074    } else if (o2.getIndex() == null) {
075      return 1;
076    } else if (o1.getIndex().equals(o2.getIndex())) {
077      return o1.getTerm().qualifiedName().compareTo(o2.getTerm().qualifiedName());
078    } else {
079      return o1.getIndex().compareTo(o2.getIndex());
080    }
081  };

113  protected void validateAsCore(boolean hasExtensions) throws UnsupportedArchiveException {
114    if (hasExtensions) {
115      if (id == null) {
116        LOG.warn("DwC-A core data file »" + title + "« is lacking an id column. No extensions allowed in this case");
117      }
118    }
119    validate();
120  }
121
122  protected void validateAsExtension() throws UnsupportedArchiveException {
123    if (id == null) {
124      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires an id or foreign key to the core id");
125    }
126    validate();
127  }
128
129  protected void validate() throws UnsupportedArchiveException {
130    if (getLocation() == null) {
131      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a location");
132    }
133    if (!getLocationFile().exists()) {
134      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« does not exist");
135    }
136    if (encoding == null) {
137      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a character encoding");
138    }
139  }

141  /**
142   * Normalizes and sorts the data file, so it is suitable for in-order iteration as a core or extension file.
143   *
144   * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
145   */
146  /*
147   * Sorting implies a normalization phase to ensure we sort the file properly.
148   * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
149   * we want to sort.
150   *
151   * This method is synchronized to prevent multiple threads trying to normalize/sort a file at the same time,
152   * and uses an advisory lock to avoid conflict between multiple processes.
153   */
154  protected synchronized boolean normalizeAndSort() throws IOException {
155    File fileToSort = getLocationFile();
156    File sortedFile = getLocationFileSorted(getLocationFile());
157
158    File lockFile = getLocationLockFile(getLocationFile());
159    RandomAccessFile lockFileRA = new RandomAccessFile(lockFile, "rw");
160
161    FileLock lock = lockFileRA.getChannel().tryLock();
162    if (lock == null) {
163      LOG.warn("Another process has locked this DWCA for initialization; waiting until the lock is released.");
164      lock = lockFileRA.getChannel().lock();
165      LOG.warn("Other process has released lock; lock taken, proceeding.");
166    }
167
168    try {
169      // If we already sorted the file and its source didn't change we can avoid doing it again
170      if (sortedFile.exists() && Files.getLastModifiedTime(sortedFile.toPath()).toInstant().isAfter(
171          Files.getLastModifiedTime(fileToSort.toPath()).toInstant())) {
172        LOG.debug("File {} is already sorted ({}B)", sortedFile, sortedFile.length());
173        return false;
174      }
175
176      File normalizedFile = normalizeIfRequired();
177      if (normalizedFile != null) {
178        fileToSort = normalizedFile;
179      }
180
181      FILE_UTILS.sort(fileToSort, getLocationFileSorted(getLocationFile()), getEncoding(),
182          getId().getIndex(), getFieldsTerminatedBy(), getFieldsEnclosedBy(),
183          TabularFileNormalizer.NORMALIZED_END_OF_LINE, getIgnoreHeaderLines());
184
185      if (normalizedFile != null) {
186        Files.deleteIfExists(normalizedFile.toPath());
187      }
188
189      return true;
190    } finally {
191      lock.release();
192      lockFileRA.close();
193    }
194  }

196  /**
197   * Apply file normalization if required.
198   **
199   * @return normalizedFile or null if normalization was not applied
200   *
201   * @throws IOException
202   */
203  protected File normalizeIfRequired() throws IOException {
204    // If the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
205    // we can skip normalization
206    boolean normalizationRequired = ! TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(getLinesTerminatedBy())
207        || getFieldsEnclosedBy() != null;
208
209    if (normalizationRequired) {
210      File normalizedFile = getLocationFileNormalized(getLocationFile());
211      TabularFileNormalizer.normalizeFile(getLocationFile().toPath(), normalizedFile.toPath(),
212          Charset.forName(getEncoding()), getFieldsTerminatedByChar(),
213          getLinesTerminatedBy(), getFieldsEnclosedBy());
214      return normalizedFile;
215    }
216    return null;
217  }

300  /**
301   * Generates a list of {@link Term}s mapped to each column in the underlying file.
302   *
303   * A list may be empty when no Term is mapped at a specific position, or may contain more than one Term where
304   * the same column is mapped more than once.
305   *
306   * The length of the list is the maximum index used in the {@link ArchiveField}s + 1.
307   *
308   * @return Data structure representing the header of the file.
309   */
310  public List<List<Term>> getHeader() {
311    List<ArchiveField> archiveFieldsWithIndex = getFieldsSorted()
312            .stream().filter(af -> af.getIndex() != null)
313            .collect(Collectors.toList());
314
315    Optional<Integer> idIndex = id != null ? Optional.of(id.getIndex()) : Optional.empty();
316
317    if (archiveFieldsWithIndex.isEmpty() && !idIndex.isPresent()){
318      return new ArrayList<>();
319    }
320
321    int maxIndex = archiveFieldsWithIndex.stream().mapToInt(ArchiveField::getIndex).max().getAsInt();
322    maxIndex = Math.max(maxIndex, idIndex.orElse(-1));
323
324    List<List<Term>> terms = new ArrayList<>();
325    for (int i = 0; i <= maxIndex; i++) {
326      terms.add(new ArrayList<Term>());
327    }
328    archiveFieldsWithIndex.stream().forEach(af -> terms.get(af.getIndex()).add(af.getTerm()));
329    // Assign default term for id column.
330    idIndex.ifPresent(idx -> terms.get(idx).add(DEFAULT_ID_TERM));
331    return terms;
332  }

334  /**
335   * Get default values mapped to their {@link Term} (if any).
336   *
337   * @return default values or {@code Optional.empty()} if none
338   */
339  public Optional<Map<Term, String>> getDefaultValues() {
340    //check if there is default value(s) defined (look for absence of index)
341    Map<Term, String> defaultValues = fields.values().stream()
342            .filter(af -> StringUtils.isNotBlank(af.getDefaultValue()))
343            .collect(Collectors.toMap(ArchiveField::getTerm, ArchiveField::getDefaultValue));
344    return defaultValues.isEmpty() ? Optional.empty() : Optional.of(defaultValues);
345  }

411  public File getLocationFile() {
412    File dataFile;
413    if (archive != null) {
414      if (getLocation() == null) {
415        // use only archive
416        dataFile = archive.getLocation();
417      } else if (getLocation().startsWith("/")) {
418        // absolute already
419        dataFile = new File(getLocation());
420      } else {
421        // use source file relative to archive dir
422        Path archiveLocation = archive.getLocation().toPath();
423        File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
424        dataFile = new File(directory, getLocation());
425      }
426    } else {
427      dataFile = new File(getLocation());
428    }
429
430    return dataFile;
431  }

467  private Reader getReader(boolean sorted) throws IOException {
468    // ArchiveFile location, or Archive in case this is a fake single-file "archive".
469    File file = getLocationFile() != null ? getLocationFile() : getArchive().getLocation();
470    if (sorted) {
471      file = getLocationFileSorted(file);
472    }
473    return Files.newBufferedReader(file.toPath(), Charset.forName(getEncoding()));
474  }
475
476  /**
477   * Get a {@link ClosableIterator} over the records in this file, optionally replacing nulls and entities.
478   *
479   * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
480   * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
481   */
482  public ClosableIterator<Record> iterator(boolean replaceNulls, boolean replaceEntities) {
483    try {
484      TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(false),
485          getFieldsTerminatedByChar(), getLinesTerminatedBy(), getFieldsEnclosedBy(),
486          areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
487      return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
488    } catch (IOException e) {
489      throw new UnsupportedArchiveException(e);
490    }
491  }
492
493  /**
494   * Build an iterator pointing to the sorted tabular file.
495   * The sorted tabular file is also assumed to have been normalized.
496   *
497   * @param replaceNulls
498   * @param replaceEntities
499   *
500   * @return
501   *
502   * @throws IOException
503   */
504  protected ClosableIterator<Record> sortedIterator(boolean replaceNulls, boolean replaceEntities) throws IOException {
505    TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(true),
506            getFieldsTerminatedByChar(), TabularFileNormalizer.NORMALIZED_END_OF_LINE, getFieldsEnclosedBy(),
507            areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
508    return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
509  }
