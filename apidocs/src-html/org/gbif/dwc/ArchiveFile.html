056  /**
057   * Comparator to sort by column index, then term name.
058   */
059  private static final Comparator<ArchiveField> AF_IDX_COMPARATOR = (o1, o2) -> {
060    if (o1.getIndex() == null && o2.getIndex() == null) {
061      return 0;
062    } else if (o1.getIndex() == null) {
063      return -1;
064    } else if (o2.getIndex() == null) {
065      return 1;
066    } else if (o1.getIndex().equals(o2.getIndex())) {
067      return o1.getTerm().qualifiedName().compareTo(o2.getTerm().qualifiedName());
068    } else {
069      return o1.getIndex().compareTo(o2.getIndex());
070    }
071  };

103  protected void validateAsCore(boolean hasExtensions) throws UnsupportedArchiveException {
104    if (hasExtensions) {
105      if (id == null) {
106        LOG.warn("DwC-A core data file »" + title + "« is lacking an id column. No extensions allowed in this case");
107      }
108    }
109    validate();
110  }

112  protected void validateAsExtension() throws UnsupportedArchiveException {
113    if (id == null) {
114      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires an id or foreign key to the core id");
115    }
116    validate();
117  }

119  protected void validate() throws UnsupportedArchiveException {
120    if (this == null) {
121      throw new UnsupportedArchiveException("DwC-A data file is NULL");
122    }
123    if (getLocation() == null) {
124      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a location");
125    }
126    if (!getLocationFile().exists()) {
127      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« does not exist");
128    }
129    if (encoding == null) {
130      throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a character encoding");
131    }
132  }

134  /**
135   * Normalizes and sorts the data file, so it is suitable for in-order iteration as a core or extension file.
136   *
137   * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
138   */
139  /*
140   * Sorting implies a normalization phase to ensure we sort the file properly.
141   * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
142   * we want to sort.
143   *
144   * This method is synchronized to prevent multiple threads trying to normalize/sort a file at the same time,
145   * and uses an advisory lock to avoid conflict between multiple processes.
146   */
147  protected synchronized boolean normalizeAndSort() throws IOException {
148    File fileToSort = getLocationFile();
149    File sortedFile = getLocationFileSorted(getLocationFile());
150
151    File lockFile = getLocationLockFile(getLocationFile());
152    RandomAccessFile lockFileRA = new RandomAccessFile(lockFile, "rw");
153
154    FileLock lock = lockFileRA.getChannel().tryLock();
155    if (lock == null) {
156      LOG.warn("Another process has locked this DWCA for initialization; waiting until the lock is released.");
157      lock = lockFileRA.getChannel().lock();
158      LOG.warn("Other process has released lock; lock taken, proceeding.");
159    }
160
161    try {
162      // If we already sorted the file and its source didn't change we can avoid doing it again
163      if (sortedFile.exists() && Files.getLastModifiedTime(sortedFile.toPath()).toInstant().isAfter(
164          Files.getLastModifiedTime(fileToSort.toPath()).toInstant())) {
165        LOG.debug("File {} is already sorted ({}B)", sortedFile, sortedFile.length());
166        return false;
167      }
168
169      File normalizedFile = normalizeIfRequired();
170      if (normalizedFile != null) {
171        fileToSort = normalizedFile;
172      }
173
174      FILE_UTILS.sort(fileToSort, getLocationFileSorted(getLocationFile()), getEncoding(),
175          getId().getIndex(), getFieldsTerminatedBy(), getFieldsEnclosedBy(),
176          TabularFileNormalizer.NORMALIZED_END_OF_LINE, getIgnoreHeaderLines());
177
178      if (normalizedFile != null) {
179        Files.deleteIfExists(normalizedFile.toPath());
180      }
181
182      return true;
183    } finally {
184      lock.release();
185      lockFileRA.close();
186    }
187  }

189  /**
190   * Apply file normalization if required.
191   **
192   * @return normalizedFile or null if normalization was not applied
193   *
194   * @throws IOException
195   */
196  @VisibleForTesting
197  protected File normalizeIfRequired() throws IOException {
198    // If the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
199    // we can skip normalization
200    boolean normalizationRequired = ! TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(getLinesTerminatedBy())
201        || getFieldsEnclosedBy() != null;
202
203    if (normalizationRequired) {
204      File normalizedFile = getLocationFileNormalized(getLocationFile());
205      TabularFileNormalizer.normalizeFile(getLocationFile().toPath(), normalizedFile.toPath(),
206          Charset.forName(getEncoding()), getFieldsTerminatedByChar(),
207          getLinesTerminatedBy(), getFieldsEnclosedBy());
208      return normalizedFile;
209    }
210    return null;
211  }

294  /**
295   * Generates a list of {@link Term}s mapped to each column in the underlying file.
296   *
297   * A list may be empty when no Term is mapped at a specific position, or may contain more than one Term where
298   * the same column is mapped more than once.
299   *
300   * The length of the list is the maximum index used in the {@link ArchiveField}s + 1.
301   *
302   * @return Data structure representing the header of the file.
303   */
304  public List<List<Term>> getHeader() {
305    List<ArchiveField> archiveFieldsWithIndex = getFieldsSorted()
306            .stream().filter(af -> af.getIndex() != null)
307            .collect(Collectors.toList());
308
309    Optional<Integer> idIndex = id != null ? Optional.of(id.getIndex()) : Optional.empty();
310
311    if (archiveFieldsWithIndex.isEmpty() && !idIndex.isPresent()){
312      return new ArrayList<>();
313    }
314
315    int maxIndex = archiveFieldsWithIndex.stream().mapToInt(ArchiveField::getIndex).max().getAsInt();
316    maxIndex = Math.max(maxIndex, idIndex.orElse(-1));
317
318    List<List<Term>> terms = new ArrayList<>();
319    for (int i = 0; i <= maxIndex; i++) {
320      terms.add(new ArrayList<Term>());
321    }
322    archiveFieldsWithIndex.stream().forEach(af -> terms.get(af.getIndex()).add(af.getTerm()));
323    // Assign default term for id column.
324    idIndex.ifPresent(idx -> terms.get(idx).add(DEFAULT_ID_TERM));
325    return terms;
326  }

328  /**
329   * Get default values mapped to their {@link Term} (if any).
330   *
331   * @return default values or {@code Optional.empty()} if none
332   */
333  public Optional<Map<Term, String>> getDefaultValues() {
334    //check if there is default value(s) defined (look for absence of index)
335    Map<Term, String> defaultValues = fields.values().stream()
336            .filter(af -> StringUtils.isNotBlank(af.getDefaultValue()))
337            .collect(Collectors.toMap(ArchiveField::getTerm, ArchiveField::getDefaultValue));
338    return defaultValues.isEmpty() ? Optional.empty() : Optional.of(defaultValues);
339  }

405  public File getLocationFile() {
406    File dataFile;
407    if (archive != null) {
408      if (getLocation() == null) {
409        // use only archive
410        dataFile = archive.getLocation();
411      } else if (getLocation().startsWith("/")) {
412        // absolute already
413        dataFile = new File(getLocation());
414      } else {
415        // use source file relative to archive dir
416        Path archiveLocation = archive.getLocation().toPath();
417        File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
418        dataFile = new File(directory, getLocation());
419      }
420    } else {
421      dataFile = new File(getLocation());
422    }
423
424    return dataFile;
425  }

460  private Reader getReader(boolean sorted) throws IOException {
461    // ArchiveFile location, or Archive in case this is a fake single-file "archive".
462    File file = getLocationFile() != null ? getLocationFile() : getArchive().getLocation();
463    if (sorted) {
464      file = getLocationFileSorted(file);
465    }
466    return Files.newBufferedReader(file.toPath(), Charset.forName(getEncoding()));
467  }
468
469  /**
470   * Get a {@link ClosableIterator} over the records in this file, optionally replacing nulls and entities.
471   *
472   * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
473   * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
474   */
475  public ClosableIterator<Record> iterator(boolean replaceNulls, boolean replaceEntities) {
476    try {
477      TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(false),
478          getFieldsTerminatedByChar(), getLinesTerminatedBy(), getFieldsEnclosedBy(),
479          areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
480      return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
481    } catch (IOException e) {
482      throw new UnsupportedArchiveException(e);
483    }
484  }
485
486  /**
487   * Build an iterator pointing to the sorted tabular file.
488   * The sorted tabular file is also assumed to have been normalized.
489   *
490   * @param replaceNulls
491   * @param replaceEntities
492   *
493   * @return
494   *
495   * @throws IOException
496   */
497  protected ClosableIterator<Record> sortedIterator(boolean replaceNulls, boolean replaceEntities) throws IOException {
498    TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(true),
499            getFieldsTerminatedByChar(), TabularFileNormalizer.NORMALIZED_END_OF_LINE, getFieldsEnclosedBy(),
500            areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
501    return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
502  }
