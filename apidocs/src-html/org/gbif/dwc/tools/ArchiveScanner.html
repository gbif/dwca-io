001package org.gbif.dwc.tools;
002
003import org.gbif.dwc.record.Record;
004import org.gbif.dwc.record.StarRecord;
005import org.gbif.dwc.terms.DwcTerm;
006import org.gbif.dwc.text.Archive;
007import org.gbif.dwc.text.ArchiveFactory;
008import org.gbif.dwc.text.UnsupportedArchiveException;
009
010import java.io.File;
011import java.io.IOException;
012import java.util.ArrayList;
013import java.util.List;
014
015public class ArchiveScanner {
016
017  public static void main(String[] args) throws IOException, UnsupportedArchiveException {
018    if (args.length < 1) {
019      System.out.println(
020        "Please specify a Darwin Core Archive folder or single Darwin Core CSV file as the first argument.\nAn optional second integer argument lets you specify the number of records to view, a third the offset to skip.");
021      System.exit(0);
022    }
023    File archiveFile = new File(args[0]);
024    if (!archiveFile.exists()) {
025      System.out.println("Cannot find archive file: " + archiveFile);
026      System.exit(0);
027    }
028    int maxNumbers = 0;
029    if (args.length > 1) {
030      try {
031        maxNumbers = Integer.parseInt(args[1]);
032      } catch (NumberFormatException ignored) {
033        System.out.println("Second argument should an integer defining maxNumbers");
034      }
035    }
036    int offset = 0;
037    if (args.length > 2) {
038      try {
039        offset = Integer.parseInt(args[2]);
040      } catch (NumberFormatException ignored) {
041        System.out.println("Third offset argument is no integer");
042      }
043    }
044
045    System.out.println("Opening archive: " + archiveFile.getAbsolutePath());
046    Archive arch = ArchiveFactory.openArchive(archiveFile);
047
048    if (arch.getCore() == null) {
049      System.out.println("Cannot locate the core data file");
050      System.exit(0);
051    }
052    System.out.println("Core file(s) found: " + arch.getCore().getLocations());
053    System.out.println("Core row type: " + arch.getCore().getRowType());
054    System.out.println("Core identifier column: " + arch.getCore().getId().getIndex());
055
056    List<DwcTerm> terms = new ArrayList<DwcTerm>();
057    terms.add(DwcTerm.scientificName);
058    terms.add(DwcTerm.taxonRank);
059    terms.add(DwcTerm.parentNameUsageID);
060    terms.add(DwcTerm.acceptedNameUsageID);
061    terms.add(DwcTerm.taxonomicStatus);
062    terms.add(DwcTerm.nomenclaturalStatus);
063    terms.add(DwcTerm.kingdom);
064    terms.add(DwcTerm.family);
065    for (DwcTerm t : terms) {
066      if (!arch.getCore().hasTerm(t)) {
067        System.out.println("Cannot locate term " + t);
068      }
069    }
070
071    System.out.println("Number of extensions " + arch.getExtensions().size());
072
073    // count records
074    int i = 0;
075    for (StarRecord rec : arch) {
076      i++;
077    }
078    System.out.println("Archive contains " + i + " core records.");
079
080    // show some records
081    i = 0;
082    if (maxNumbers > 0) {
083      for (StarRecord rec : arch) {
084        i++;
085        if (i > offset + maxNumbers) {
086          break;
087        }
088        if (offset < i) {
089          System.out.println("record " + i + ":  " + rec);
090          for (Record erec : rec) {
091            System.out.println("  : " + erec);
092          }
093        }
094      }
095    }
096  }
097
098}
