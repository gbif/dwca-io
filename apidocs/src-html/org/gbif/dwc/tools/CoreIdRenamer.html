042/**
043 * Tool that takes a dwca file and renames its core ids according to some id generator.
044 * All extension records and ID terms pointing to the core ID are renamed accordingly to preserve the archive
045 * integrity.
046 */
047public class CoreIdRenamer {
048
049  private final IdGenerator idGen;
050
051  public CoreIdRenamer() {
052    idGen = new IntSequenceGenerator(1);
053  }
054
055  public CoreIdRenamer(IdGenerator idGen) {
056    this.idGen = idGen;
057  }
058
059  public CoreIdRenamer(int firstId) {
060    idGen = new IntSequenceGenerator(firstId);
061  }
062
063  public void rename(File dwca) throws IOException {
064    rename(ArchiveFactory.openArchive(dwca));
065  }
066
067  public void rename(Archive arch) throws IOException {
068    ArchiveField id = arch.getCore().getId();
069    TermFactory fact = TermFactory.instance();
070    Term rowType = arch.getCore().getRowType();
071
072    // detect dependent terms with ids to be renamed accordingly
073    List<Term> relatedTerms = new ArrayList<Term>();
074    if (DwcTerm.Taxon == rowType) {
075      // potentially 3 foreign keys with values pointing to the coreID
076      if (arch.getCore().hasTerm(DwcTerm.parentNameUsageID)) {
077        relatedTerms.add(DwcTerm.parentNameUsageID);
078      }
079      if (arch.getCore().hasTerm(DwcTerm.acceptedNameUsageID)) {
080        relatedTerms.add(DwcTerm.acceptedNameUsageID);
081      }
082      if (arch.getCore().hasTerm(DwcTerm.originalNameUsageID)) {
083        relatedTerms.add(DwcTerm.originalNameUsageID);
084      }
085    } else if (DwcTerm.Occurrence == rowType) {
086      // none as far as we know
087    } else {
088      throw new IllegalStateException("Unknown core rowtype - cannot rename the archive ids");
089    }
090
091    // map old to new ids and store them in memory
092    Map<String, String> old2new = new HashMap<String, String>();
093    ClosableIterator<DarwinCoreRecord> iter = arch.iteratorDwc();
094    System.out.println("Mapping record ids ...");
095    while (iter.hasNext()) {
096      DarwinCoreRecord rec = iter.next();
097      old2new.put(rec.id(), idGen.nextId());
098    }
099    System.out.println("Mapped " + old2new.size() + " record ids. Rewrite files ...");
100
101    // rename all files
102    rewriteFile(arch.getCore(), old2new, relatedTerms);
103    for (ArchiveFile af : arch.getExtensions()) {
104      rewriteFile(arch.getCore(), old2new, relatedTerms);
105    }
106  }
107
108  private void rewriteFile(ArchiveFile original, Map<String, String> old2new, List<Term> relatedTerms)
109    throws IOException {
110    File newFile = new File(original.getLocationFile().getParent(), original.getLocationFile().getName() + "-NEW");
111    Writer writer =
112      new BufferedWriter(new OutputStreamWriter(new FileOutputStream(newFile, false), original.getEncoding()));
113
114    List<Integer> renameCols = new ArrayList<Integer>();
115    renameCols.add(original.getId().getIndex());
116    for (Term t : relatedTerms) {
117      if (original.hasTerm(t)) {
118        renameCols.add(original.getField(t).getIndex());
119      }
120    }
121    // iterate thru rows
122    CSVReader reader = original.getCSVReader();
123    final String quote = original.getFieldsEnclosedBy() == null ? "" : original.getFieldsEnclosedBy().toString();
124    final String quoteDelimQuote = quote + original.getFieldsTerminatedBy() + quote;
125
126    // headers
127    for (int hr = reader.headerRows; hr > 0; hr--) {
128      // write row with given delimter, quotation and line ending
129      String rowAsString = quote + StringUtils.join(reader.header, quoteDelimQuote) + quote;
130      writer.write(rowAsString);
131      writer.write('\n');
132    }
133    // records
134    for (String[] row : reader) {
135      for (Integer col : renameCols) {
136        row[col] = old2new.get(row[col]);
137      }
138      // write row with given delimter, quotation and line ending
139      String rowAsString = quote + StringUtils.join(row, quoteDelimQuote) + quote;
140      writer.write(rowAsString);
141      writer.write('\n');
142    }
143    writer.flush();
144    writer.close();
145  }
146
147  public static void main(String[] args) throws IOException {
148    CoreIdRenamer renamer = new CoreIdRenamer();
149    Archive arch = ArchiveFactory.openArchive(new File("/Users/markus/Desktop/ecat_checklist"));
150    arch.getCore().setRowType(DwcTerm.Taxon);
151    renamer.rename(arch);
152  }
