001package org.gbif.dwc.record;
002
003import java.util.regex.Pattern;
004
005import org.apache.commons.lang3.StringEscapeUtils;
006
007public class CleanUtils {
008  private static final Pattern NULL_REPL = Pattern.compile("^\\s*(null|\\\\N)?\\s*$", Pattern.CASE_INSENSITIVE);
009
010  private CleanUtils() {
011  }
012
013  /**
014   * Does basic entity replacements if requested to string values.
015   * @param value the original string
016   * @param nulls if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
017   * @param entities if true replaces html4, xml and numerical entities with their unicode character
018   */
019  public static String clean(String value, boolean nulls, boolean entities) {
020    if (value == null || (nulls && NULL_REPL.matcher(value).find()) ) {
021      return null;
022    }
023    return entities ? StringEscapeUtils.unescapeHtml4(value) : value;
024  }
025
026}
