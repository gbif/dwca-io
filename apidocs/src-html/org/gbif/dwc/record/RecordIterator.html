017package org.gbif.dwc.record;
018
019import org.gbif.dwc.terms.Term;
020import org.gbif.dwc.text.ArchiveField;
021import org.gbif.dwc.text.ArchiveFile;
022import org.gbif.file.CSVReader;
023import org.gbif.utils.file.ClosableIterator;
024
025import java.io.IOException;
026import java.util.Iterator;
027import java.util.Map;
028
029import com.google.common.collect.Iterators;
030import com.google.common.collect.PeekingIterator;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034public class RecordIterator implements ClosableIterator<Record> {
035
036  private static final Logger LOG = LoggerFactory.getLogger(RecordIterator.class);
037  private ClosableIterator<String[]> closable;
038  private final PeekingIterator<String[]> iter;
039  private int lineCount = 0;
040  // needed to create RecordImpl
041  private final ArchiveField id;
042  private final Map<Term, ArchiveField> fields;
043  private final Term rowType;
044  private final boolean replaceEntities;
045  private final boolean replaceNulls;
046
047  /**
048   * @param replaceNulls if true record values will have literal nulls replaced with NULL.
049   * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
050   */
051  public RecordIterator(ClosableIterator<String[]> recordSource, ArchiveField id, Map<Term, ArchiveField> fields,
052    Term rowType, boolean replaceNulls, boolean replaceEntities) {
053    this.id = id;
054    this.fields = fields;
055    this.rowType = rowType;
056    this.replaceNulls = replaceNulls;
057    this.replaceEntities = replaceEntities;
058    closable = recordSource;
059    if (closable == null) {
060      Iterator<String[]> empty = Iterators.emptyIterator();
061      iter = Iterators.peekingIterator(empty);
062    } else {
063      iter = Iterators.peekingIterator(closable);
064    }
065  }
066
067  /**
068   * @param replaceNulls if true record values will have literal nulls replaced with NULL.
069   * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
070   */
071  public static RecordIterator build(ArchiveFile source, boolean replaceNulls, boolean replaceEntities) {
072    try {
073      CSVReader csvr = CSVReader.build(source);
074      return new RecordIterator(csvr, source.getId(), source.getFields(), source.getRowType(), replaceNulls, replaceEntities);
075    } catch (IOException e) {
076      LOG.error("Can't open archive file " + source + " for building a record iterator", e);
077    }
078    return null;
079  }
080
081  public void close() {
082    closable.close();
083  }
084
085  public boolean hasNext() {
086    return iter.hasNext() && iter.peek() != null;
087  }
088
089  public Record next() {
090    lineCount++;
091    RecordImpl record = null;
092    try {
093      // update record with cached row
094      record = new RecordImpl(id, fields, rowType, replaceNulls, replaceEntities);
095      String[] row = iter.next();
096      while (row.length == 0) {
097        // ignore rows without a single column
098        row = iter.next();
099      }
100      record.setRow(row);
101    } catch (Exception e) {
102      LOG.warn("Bad row somewhere around line: {}", lineCount, e);
103
104    }
105    return record;
106  }
107
108  public void remove() {
109    throw new UnsupportedOperationException("Cannot remove a row from an archive file");
110  }
111
112}
