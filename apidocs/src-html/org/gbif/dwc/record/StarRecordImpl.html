014package org.gbif.dwc.record;
015
016import org.gbif.dwc.terms.Term;
017
018import java.util.ArrayList;
019import java.util.Collection;
020import java.util.HashMap;
021import java.util.Iterator;
022import java.util.List;
023import java.util.Map;
024import java.util.Set;
025
026public class StarRecordImpl implements StarRecord {
027
028  private Record core;
029  private final Map<Term, List<Record>> extensions;
030
031  public StarRecordImpl(Collection<Term> extensions) {
032    this.extensions = new HashMap<Term, List<Record>>();
033    for (Term rowType : extensions) {
034      this.extensions.put(rowType, new ArrayList<Record>());
035    }
036  }
037
038  public void addRecord(Term rowType, Record record) {
039    if (!extensions.containsKey(rowType)) {
040      throw new IllegalArgumentException("RowType not supported");
041    }
042    extensions.get(rowType).add(record);
043  }
044
045  /**
046   * @return the core record
047   */
048  @Override
049  public Record core() {
050    return core;
051  }
052
053  @Override
054  public boolean hasExtension(Term rowType) {
055    return extensions.containsKey(rowType) && !extensions.get(rowType).isEmpty();
056  }
057
058  /**
059   * Retrieves all extension records of a specific extension.
060   * If the requested extension is not mapped null will be returned.
061   *
062   * @param rowType the Term representing the rowType
063   *
064   * @return possibly empty list of extension record or null if extension is not mapped at all
065   */
066  @Override
067  public List<Record> extension(Term rowType) {
068    return extensions.get(rowType);
069  }
070
071  /**
072   * Retrieves all extension records related to the core record across all extensions as a map.
073   */
074  @Override
075  public Map<Term, List<Record>> extensions() {
076    return extensions;
077  }
078
079  /**
080   * Get a new iterator over all extension records, no matter to which extension they belong.
081   *
082   * @return the ExtensionRecord iterator, reusing the same instance for each call
083   */
084  @Override
085  public Iterator<Record> iterator() {
086    List<Record> records = new ArrayList<>();
087    for (List<Record> recs : extensions.values()) {
088      records.addAll(recs);
089    }
090    return records.iterator();
091  }
092
093  public void newCoreRecord(Record core) {
094    this.core = core;
095    for (List<Record> lists : extensions.values()) {
096      lists.clear();
097    }
098  }
099
100  /**
101   * @return set of extension rowTypes associated with this star record
102   */
103  @Override
104  public Set<Term> rowTypes() {
105    return extensions.keySet();
106  }
107
108  /**
109   * @return the number of associated extension records across all rowTypes
110   */
111  @Override
112  public int size() {
113    int x = 0;
114    for (List<Record> recs : extensions.values()) {
115      x += recs.size();
116    }
117    return x;
118  }
119
120  @Override
121  public String toString() {
122    return "StarRecord with core "+core.toString()+" and extensions "+extensions;
123  }
124}
