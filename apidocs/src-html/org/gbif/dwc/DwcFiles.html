036   */
037  private DwcFiles() {

040  /**
041   * Get a {@link ClosableIterator} on the provided {@link ArchiveFile}.
042   *
043   * @param source
044   * @param replaceNulls    if true record values will have literal nulls replaced with NULL.
045   * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
046   */
047  public static ClosableIterator<Record> iterator(ArchiveFile source, boolean replaceNulls, boolean replaceEntities) throws IOException {
048    Objects.requireNonNull(source, "source ArchiveFile shall be provided");
049    TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(
050            Files.newBufferedReader(source.getLocationFile() != null ? source.getLocationFile().toPath() : source.getArchive().getLocation().toPath()),
051            getFieldsTerminatedBy(source.getFieldsTerminatedBy()), source.getLinesTerminatedBy(), source.getFieldsEnclosedBy(),
052            isHeaderLineIncluded(source.getIgnoreHeaderLines()), getLineToSkipBeforeHeader(source.getIgnoreHeaderLines()));
053    return new DwcRecordIterator(tabularFileReader, source.getId(), source.getFields(), source.getRowType(), replaceNulls, replaceEntities);
054  }
055
056  /**
057   * Get the fieldsTerminatedBy as char or throw exception.
058   *
059   * @param fieldsTerminatedBy
060   *
061   * @return
062   */
063  private static char getFieldsTerminatedBy(String fieldsTerminatedBy) {
064    Objects.requireNonNull(fieldsTerminatedBy, "fieldsTerminatedBy shall be provided");
065    if (fieldsTerminatedBy.length() != 1) {
066      throw new IllegalArgumentException();
067    }
068    return fieldsTerminatedBy.charAt(0);
069  }
070
071  /**
072   * Determines if a header line is included based on an optional integer.
073   *
074   * @param ignoreHeaderLines
075   *
076   * @return
077   */
078  private static boolean isHeaderLineIncluded(Integer ignoreHeaderLines) {
079    return ignoreHeaderLines != null && ignoreHeaderLines > 0;
080  }
081
082  /**
083   * Determines the number of line to skip before the header line or return null.
084   *
085   * @param ignoreHeaderLines
086   *
087   * @return
088   */
089  private static Integer getLineToSkipBeforeHeader(Integer ignoreHeaderLines) {
090    if (ignoreHeaderLines != null && ignoreHeaderLines > 1) {
091      return ignoreHeaderLines - 1;
092    }
093    return null;
094  }
095
096  /**
097   * Prepare an {@link Archive} into a {@link NormalizedDwcArchive} which allows to get {@link StarRecord} {@link
098   * ClosableIterator}.
099   * This method will initiate the normalization process. This process can take some times depending on the size of
100   * files and number of extension.
101   *
102   * @param archive
103   * @param replaceNulls
104   * @param replaceEntities
105   *
106   * @return new {@link NormalizedDwcArchive} instance
107   *
108   * @throws IOException
109   */
110  public static NormalizedDwcArchive prepareArchive(final Archive archive, boolean replaceNulls, boolean replaceEntities) throws IOException {
111
112    //if no extensions are provided we don't need to sort the file
113    if (archive.getExtensions().isEmpty()) {
114      return new NormalizedDwcArchive(() -> iterator(archive.getCore(), replaceNulls, replaceEntities));
115    }
116
117    //otherwise, we need to sort core + extensions
118    normalizeAndSortArchiveFiles(archive);
119
120    return new NormalizedDwcArchive(() -> buildSortedIterator(archive.getCore(), replaceNulls, replaceEntities),
121            () -> buildSortedIteratorExt(archive, replaceNulls, replaceEntities));
122  }
123
124  /**
125   * Sorts all files according in the {@link Archive} so that we can easily iterate over all files at once.
126   */
127  private static void normalizeAndSortArchiveFiles(Archive archive) throws IOException {
128    normalizeAndSort(archive.getCore());
129    for (ArchiveFile archiveFile : archive.getExtensions()) {
130      normalizeAndSort(archiveFile);
131    }
132  }
133
134  /**
135   * Sort a single {@link ArchiveFile}. Sorting implies a normalization phase to ensure we sort the file properly.
136   * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
137   * we want to sort.
138   *
139   * @param archiveFile
140   * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
141   *
142   * @throws IOException
143   */
144  @VisibleForTesting
145  protected static boolean normalizeAndSort(ArchiveFile archiveFile) throws IOException {
146
147    File fileToSort = archiveFile.getLocationFile();
148    File sortedFile = ArchiveFile.getLocationFileSorted(archiveFile.getLocationFile());
149    String linesTerminatedBy = archiveFile.getLinesTerminatedBy();
150
151    //if we already sorted the file and its source didn't change we can avoid doing it again
152    if (sortedFile.exists() && Files.getLastModifiedTime(sortedFile.toPath()).toInstant().isAfter(
153            Files.getLastModifiedTime(fileToSort.toPath()).toInstant())) {
154      return false;
155    }
156
157    File normalizedFile = normalizeIfRequired(archiveFile);
158    if(normalizedFile != null){
159      fileToSort = normalizedFile;
160    }
161
162    F_UTILS.sort(fileToSort, ArchiveFile.getLocationFileSorted(archiveFile.getLocationFile()), archiveFile.getEncoding(),
163            archiveFile.getId().getIndex(), archiveFile.getFieldsTerminatedBy(), archiveFile.getFieldsEnclosedBy(),
164            TabularFileNormalizer.NORMALIZED_END_OF_LINE, archiveFile.getIgnoreHeaderLines());
165
166    if(normalizedFile != null ){
167      Files.deleteIfExists(normalizedFile.toPath());
168    }
169
170    return true;
171  }
172
173  /**
174   * Apply file normalization if required.
175   *
176   * @param archiveFile
177   *
178   * @return normalizedFile or null if normalization was not applied
179   *
180   * @throws IOException
181   */
182  @VisibleForTesting
183  protected static File normalizeIfRequired(ArchiveFile archiveFile) throws IOException {
184    //if the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
185    //we can skip normalization
186    boolean normalizationRequired =
187            !TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(archiveFile.getLinesTerminatedBy()) ||
188                    archiveFile.getFieldsEnclosedBy() != null;
189
190    if (normalizationRequired) {
191      File normalizedFile = getLocationFileNormalized(archiveFile.getLocationFile());
192      TabularFileNormalizer.normalizeFile(archiveFile.getLocationFile().toPath(), normalizedFile.toPath(),
193              Charset.forName(archiveFile.getEncoding()), getFieldsTerminatedBy(archiveFile.getFieldsTerminatedBy()),
194              archiveFile.getLinesTerminatedBy(), archiveFile.getFieldsEnclosedBy());
195      return normalizedFile;
196    }
197    return null;
198  }
199
200  private static File getLocationFileNormalized(File location) {
201    return new File(location.getParentFile(), location.getName() + "-normalized");
202  }
203
204  /**
205   * Build an iterator (pointing to the sorted tabular file) for each extensions of the {@link Archive}.
206   *
207   * @param archive
208   * @param replaceNulls
209   * @param replaceEntities
210   *
211   * @return
212   *
213   * @throws IOException
214   */
215  private static Map<Term, ClosableIterator<Record>> buildSortedIteratorExt(Archive archive,
216                                                                            boolean replaceNulls, boolean replaceEntities) throws IOException {
217    Map<Term, ClosableIterator<Record>> extensionIts = new HashMap<>();
218    for (ArchiveFile ext : archive.getExtensions()) {
219      extensionIts.put(ext.getRowType(), buildSortedIterator(ext, replaceNulls, replaceEntities));
220    }
221    return extensionIts;
222  }
223
224  /**
225   * Build an iterator on top of the provided {@link ArchiveFile} pointing to the sorted tabular file.
226   * The sorted tabular file is also assumed to have been normalized.
227   *
228   * @param af
229   * @param replaceNulls
230   * @param replaceEntities
231   *
232   * @return
233   *
234   * @throws IOException
235   */
236  private static ClosableIterator<Record> buildSortedIterator(ArchiveFile af,
237                                                              boolean replaceNulls, boolean replaceEntities) throws IOException {
238    TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(
239            Files.newBufferedReader(getLocationFileSorted(af.getLocationFile()).toPath()),
240            getFieldsTerminatedBy(af.getFieldsTerminatedBy()),
241            TabularFileNormalizer.NORMALIZED_END_OF_LINE, af.getFieldsEnclosedBy(),
242            isHeaderLineIncluded(af.getIgnoreHeaderLines()), getLineToSkipBeforeHeader(af.getIgnoreHeaderLines()));
243    return new DwcRecordIterator(tabularFileReader, af.getId(), af.getFields(), af.getRowType(), replaceNulls, replaceEntities);
244  }
