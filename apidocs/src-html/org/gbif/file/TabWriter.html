001package org.gbif.file;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005import java.io.OutputStream;
006import java.io.OutputStreamWriter;
007import java.io.UnsupportedEncodingException;
008import java.io.Writer;
009import java.util.regex.Pattern;
010
011import org.apache.commons.lang3.StringUtils;
012
013public class TabWriter {
014
015  private static final Pattern escapeChars = Pattern.compile("[\t\n\r]");
016  private int records = 0;
017  private Writer writer;
018
019  public TabWriter(OutputStream stream) {
020    try {
021      writer = new BufferedWriter(new OutputStreamWriter(stream, "UTF8"));
022    } catch (UnsupportedEncodingException e) {
023      throw new IllegalStateException(e);
024    }
025  }
026
027  public void write(String[] row) throws IOException {
028    if (row == null || row.length == 0) {
029      return;
030    }
031    String rowString = tabRow(row);
032    if (rowString != null) {
033      writer.write(rowString);
034    }
035  }
036
037  private String tabRow(String[] columns) {
038    // escape \t \n \r chars !!!
039    boolean empty = true;
040    for (int i = 0; i < columns.length; i++) {
041      if (columns[i] != null) {
042        empty = false;
043        columns[i] = StringUtils.trimToNull(escapeChars.matcher(columns[i]).replaceAll(" "));
044      }
045    }
046    if (empty) {
047      // dont create a row at all!
048      return null;
049    }
050    return StringUtils.join(columns, '\t') + "\n";
051  }
052
053  public void close() throws IOException {
054    writer.close();
055  }
056}
