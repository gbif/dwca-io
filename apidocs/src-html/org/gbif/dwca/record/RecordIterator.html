017package org.gbif.dwca.record;
018
019import org.gbif.dwc.terms.Term;
020import org.gbif.dwca.io.ArchiveField;
021import org.gbif.dwca.io.ArchiveFile;
022import org.gbif.io.CSVReader;
023import org.gbif.utils.file.ClosableIterator;
024
025import java.io.IOException;
026import java.util.Iterator;
027import java.util.Map;
028
029import com.google.common.collect.Iterators;
030import com.google.common.collect.PeekingIterator;
031
032import org.slf4j.Logger;
033import org.slf4j.LoggerFactory;
034
035public class RecordIterator implements ClosableIterator<Record> {
036
037  private static final Logger LOG = LoggerFactory.getLogger(RecordIterator.class);
038  private ClosableIterator<String[]> closable;
039  private final PeekingIterator<String[]> iter;
040  private int lineCount = 0;
041  // needed to create RecordImpl
042  private final ArchiveField id;
043  private final Map<Term, ArchiveField> fields;
044  private final Term rowType;
045  private final boolean replaceEntities;
046  private final boolean replaceNulls;
047
048  /**
049   * @param replaceNulls if true record values will have literal nulls replaced with NULL.
050   * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
051   */
052  public RecordIterator(ClosableIterator<String[]> recordSource, ArchiveField id, Map<Term, ArchiveField> fields,
053    Term rowType, boolean replaceNulls, boolean replaceEntities) {
054    this.id = id;
055    this.fields = fields;
056    this.rowType = rowType;
057    this.replaceNulls = replaceNulls;
058    this.replaceEntities = replaceEntities;
059    closable = recordSource;
060    if (closable == null) {
061      Iterator<String[]> empty = Iterators.emptyIterator();
062      iter = Iterators.peekingIterator(empty);
063    } else {
064      iter = Iterators.peekingIterator(closable);
065    }
066  }
067
068  /**
069   * @param replaceNulls if true record values will have literal nulls replaced with NULL.
070   * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
071   */
072  public static RecordIterator build(ArchiveFile source, boolean replaceNulls, boolean replaceEntities) {
073    try {
074      CSVReader csvr = CSVReader.build(source);
075      return new RecordIterator(csvr, source.getId(), source.getFields(), source.getRowType(), replaceNulls, replaceEntities);
076    } catch (IOException e) {
077      LOG.error("Can't open archive file " + source + " for building a record iterator", e);
078    }
079    return null;
080  }
081
082  public void close() {
083    closable.close();
084  }
085
086  public boolean hasNext() {
087    return iter.hasNext() && iter.peek() != null;
088  }
089
090  public Record next() {
091    lineCount++;
092    RecordImpl record = null;
093    try {
094      // update record with cached row
095      record = new RecordImpl(id, fields, rowType, replaceNulls, replaceEntities);
096      String[] row = iter.next();
097      while (row.length == 0) {
098        // ignore rows without a single column
099        row = iter.next();
100      }
101      record.setRow(row);
102    } catch (Exception e) {
103      LOG.warn("Bad row somewhere around line: {}", lineCount, e);
104
105    }
106    return record;
107  }
108
109  public void remove() {
110    throw new UnsupportedOperationException("Cannot remove a row from an archive file");
111  }
112
113}
