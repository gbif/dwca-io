080  public String buildHigherClassification(Character lowestRank) {
081    StringBuilder buf = new StringBuilder();
082    if (kingdom != null) {
083      buf.append(StringUtils.capitalize(kingdom.toLowerCase()));
084      buf.append(CLASSIFICATION_DELIMITER);
085    }
086    if (lowestRank == null || lowestRank != 'k') {
087      if (phylum != null) {
088        buf.append(StringUtils.capitalize(phylum.toLowerCase()));
089        buf.append(CLASSIFICATION_DELIMITER);
090      }
091      if (lowestRank == null || lowestRank != 'p') {
092        if (classs != null) {
093          buf.append(StringUtils.capitalize(classs.toLowerCase()));
094          buf.append(CLASSIFICATION_DELIMITER);
095        }
096        if (lowestRank == null || lowestRank != 'c') {
097          if (order != null) {
098            buf.append(StringUtils.capitalize(order.toLowerCase()));
099            buf.append(CLASSIFICATION_DELIMITER);
100          }
101          if (lowestRank == null || lowestRank != 'o') {
102            if (family != null) {
103              buf.append(StringUtils.capitalize(family.toLowerCase()));
104              buf.append(CLASSIFICATION_DELIMITER);
105            }
106            if (lowestRank == null || lowestRank != 'f') {
107              if (genus != null) {
108                buf.append(StringUtils.capitalize(genus.toLowerCase()));
109                buf.append(CLASSIFICATION_DELIMITER);
110              }
111              if (lowestRank == null || lowestRank != 'g') {
112                if (subgenus != null) {
113                  buf.append(StringUtils.capitalize(subgenus.toLowerCase()));
114                  buf.append(CLASSIFICATION_DELIMITER);
115                }
116              }
117            }
118          }
119        }
120      }
121    }
122
123    if (buf.length() > CLASSIFICATION_DELIMITER.length()) {
124      buf.delete(buf.length() - CLASSIFICATION_DELIMITER.length(), buf.length());
125    }
126    return buf.toString();
127  }

129  public String getFullScientificName() {
130    if (scientificNameAuthorship != null && scientificName != null) {
131      String normedSciName = NORM_AUTHORS.matcher(scientificName.toLowerCase()).replaceAll(" ");
132      String normedAuthors = NORM_AUTHORS.matcher(scientificNameAuthorship.toLowerCase()).replaceAll(" ");
133      if (!normedSciName.contains(normedAuthors)) {
134        return scientificName + " " + scientificNameAuthorship;
135      }
136    }
137    if (scientificName == null) {
138      String sciname = null;
139      if (this.genus != null) {
140        if (specificEpithet != null) {
141          sciname = genus + " " + specificEpithet;
142          if (infraspecificEpithet != null) {
143            sciname += " " + infraspecificEpithet;
144          }
145          // potentially add authorship in this case
146          if (sciname != null && scientificNameAuthorship != null) {
147            sciname += " " + scientificNameAuthorship;
148          }
149        } else {
150          sciname = subgenus != null ? subgenus : genus;
151        }
152      }
153      return sciname;
154    }
155    return scientificName;
156  }

158  public String getLowestScientificName() {
159    String sciname = getFullScientificName();
160    if (sciname == null) {
161      if (family != null) {
162        sciname = family;
163      } else if (order != null) {
164        sciname = order;
165      } else if (classs != null) {
166        sciname = classs;
167      } else if (phylum != null) {
168        sciname = phylum;
169      } else if (kingdom != null) {
170        sciname = kingdom;
171      }
172    }
173    return sciname;
174  }

176  /**
177   * Gets a dwc property by its concept term.
178   * This method is only able to access official Darwin Core or Dublin Core terms, not any custom extensions.
179   *
180   * @param prop the concept term to lookup
181   *
182   * @return the terms value, null or IllegalArgumentException for unsupported terms
183   */
184  public String getProperty(Term prop) {
185    String getter = String.format("get%s", getPropertyName(prop));
186    try {
187      Method m = getClass().getMethod(getter);
188      Object val = m.invoke(this);
189      return val == null ? null : val.toString();
190
191    } catch (NoSuchMethodException e) {
192      throw new IllegalArgumentException("non existing dwc property: " + prop);
193
194    } catch (Exception e) {
195      // should never happen
196      throw new RuntimeException(e);
197    }
198  }

200  private String getPropertyName(Term prop) {
201    String propName = StringUtils.capitalize(prop.simpleName());
202    if (propName.equalsIgnoreCase("Class")) {
203      propName = "Classs";
204    }
205    return propName;
206  }

208  public String getTaxonRankInterpreted() {
209    if (scientificName != null || taxonRank != null) {
210      return taxonRank;
211    }
212    if (genus != null && specificEpithet != null && infraspecificEpithet != null) {
213      if (taxonRank != null) {
214        return taxonRank;
215      }
216      if (verbatimTaxonRank != null) {
217        return verbatimTaxonRank;
218      }
219      return "infraspecies";
220    }
221    if (genus != null && specificEpithet != null) {
222      return "species";
223    }
224    if (subgenus != null) {
225      return "subgenus";
226    }
227    if (genus != null) {
228      return "genus";
229    }
230    if (family != null) {
231      return "family";
232    }
233    if (order != null) {
234      return "order";
235    }
236    if (classs != null) {
237      return "class";
238    }
239    if (phylum != null) {
240      return "phylum";
241    }
242    if (kingdom != null) {
243      return "kingdom";
244    }
245    return null;
246  }

248  /**
249   * The method tries to best assemble the most complete scientific name possible incl the name authorship.
250   * It first tries to use scientificName and scientificNameAuthorship if existing.
251   * Otherwise it uses the atomized name parts. Warning: this uses the genus property which might be wrong in case
252   * of synonym names, see https://code.google.com/p/darwincore/issues/detail?id=151
253   * Note also that the assembled name never includes a rank marker.
254   *
255   * @return the best guess of the full scientific name with authorship
256   */
257  protected String norm(String x) {
258    x = StringUtils.trimToNull(x);
259    if (x != null && (x.equalsIgnoreCase("\\N") || x.equalsIgnoreCase("NULL"))) {
260      x = null;
261    }
262    return x;
263  }

265  public boolean setProperty(Term prop, String value) {
266    if (prop == null) {
267      return false;
268    }
269    String setterName = null;
270    try {
271      setterName = String.format("set%s", getPropertyName(prop));
272      Method m = getClass().getMethod(setterName, String.class);
273      m.invoke(this, value);
274    } catch (Exception e) {
275      LOG.warn("Trying to access bad dwc property setter: " + setterName + " for property " + prop);
276      return false;
277    }
278    return true;
279  }
