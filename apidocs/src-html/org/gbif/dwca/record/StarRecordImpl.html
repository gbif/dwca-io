001package org.gbif.dwca.record;
002
003import org.gbif.dwc.terms.Term;
004
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.HashMap;
008import java.util.Iterator;
009import java.util.List;
010import java.util.Map;
011import java.util.Set;
012
013public class StarRecordImpl implements StarRecord {
014
015  private Record core;
016  private final Map<Term, List<Record>> extensions;
017
018  public StarRecordImpl(Collection<Term> extensions) {
019    this.extensions = new HashMap<Term, List<Record>>();
020    for (Term rowType : extensions) {
021      this.extensions.put(rowType, new ArrayList<Record>());
022    }
023  }
024
025  public void addRecord(Term rowType, Record record) {
026    if (!extensions.containsKey(rowType)) {
027      throw new IllegalArgumentException("RowType not supported");
028    }
029    extensions.get(rowType).add(record);
030  }
031
032  /**
033   * @return the core record
034   */
035  public Record core() {
036    return core;
037  }
038
039  public boolean hasExtension(Term rowType) {
040    return extensions.containsKey(rowType);
041  }
042
043  /**
044   * Retrieves all extension records of a specific extension.
045   * If the requested extension is not mapped null will be returned.
046   *
047   * @param rowType the Term representing the rowType
048   *
049   * @return possibly empty list of extension record or null if extension is not mapped at all
050   */
051  public List<Record> extension(Term rowType) {
052    return extensions.get(rowType);
053  }
054
055  /**
056   * Retrieves all extension records related to the core record across all extensions as a map.
057   */
058  public Map<Term, List<Record>> extensions() {
059    return extensions;
060  }
061
062  /**
063   * Get a new iterator over all extension records, no matter to which extension they belong.
064   *
065   * @return the ExtensionRecord iterator, reusing the same instance for each call
066   */
067  public Iterator<Record> iterator() {
068    List<Record> records = new ArrayList<Record>();
069    for (List<Record> recs : extensions.values()) {
070      records.addAll(recs);
071    }
072    return records.iterator();
073  }
074
075  public void newCoreRecord(Record core) {
076    this.core = core;
077    for (List<Record> lists : extensions.values()) {
078      lists.clear();
079    }
080  }
081
082  /**
083   * @return set of extension rowTypes associated with this star record
084   */
085  public Set<Term> rowTypes() {
086    return extensions.keySet();
087  }
088
089  /**
090   * @return the number of associated extension records across all rowTypes
091   */
092  public int size() {
093    int x = 0;
094    for (List<Record> recs : extensions.values()) {
095      x += recs.size();
096    }
097    return x;
098  }
099}
