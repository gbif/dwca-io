022/**
023 * An archive writer that writes entire data files at once and does not check integrity of coreids.
024 * In large archives using extensions this yields a much, much better performance than writing star record by star record.
025 */
026public class DwcaStreamWriter implements AutoCloseable {
027    private static final Logger LOG = LoggerFactory.getLogger(DwcaStreamWriter.class);
028
029    private final File dir;
030    private final Term core;
031    private final Term coreIdTerm;
032    private final boolean useHeaders;
033    private final Archive archive = new Archive();
034    private Dataset metadata;
035    private Map<String, Dataset> constituents = Maps.newHashMap();
036
037    /**
038     * @param dir the directory to use as the archive
039     * @param coreRowType the archives core row type
040     * @param coreIdTerm if given used to map the id column of the core
041     * @param useHeaders if true write a single header row for each data file
042     */
043    public DwcaStreamWriter(File dir, Term coreRowType, @Nullable Term coreIdTerm, boolean useHeaders) {
044        this.dir = dir;
045        this.core = coreRowType;
046        this.coreIdTerm = coreIdTerm;
047        this.useHeaders = useHeaders;
048        archive.setLocation(dir);
049    }
050
051    private File dataFile(Term rowType) {
052        return new File(dir, rowType.simpleName()+".tsv");
053    }
054
055    private static ArchiveField idField(int column) {
056        ArchiveField field = new ArchiveField();
057        field.setIndex(column);
058        return field;
059    }
060    /**
061     * @param coreIdColumn zero based index to the rows coreid
062     * @param mapping zero based indexed of the rows
063     */
064    public void write(Term rowType, int coreIdColumn, Map<Term, Integer> mapping, Iterable<String[]> rows) {
065        Preconditions.checkNotNull(rows);
066
067        final int maxMapping = maxMappingColumn(mapping);
068        try (TabWriter writer = addArchiveFile(rowType, coreIdColumn, mapping, maxMapping) ) {
069            // write data
070            for (String[] row : rows) {
071                write(writer, row, maxMapping);
072            }
073        } catch (IOException e) {
074            Throwables.propagate(e);
075        }
076    }
077
078    private static void write(TabWriter writer, String[] row, int maxMapping) throws IOException {
079        if (row != null && row.length < maxMapping) {
080            throw new IllegalArgumentException("Input rows are smaller than the defined mapping of " + maxMapping + " columns.");
081        }
082        writer.write(row);
083    }
084
085    private int maxMappingColumn(Map<Term, Integer> mapping) {
086        return mapping.values().stream().max(Integer::compareTo).get();
087    }
088
089    private TabWriter addArchiveFile(Term rowType, int coreIdColumn, Map<Term, Integer> mapping, int maxMapping) throws IOException {
090        Preconditions.checkNotNull(rowType);
091        Preconditions.checkNotNull(mapping);
092        Preconditions.checkArgument(!mapping.isEmpty());
093        Preconditions.checkArgument(coreIdColumn >= 0);
094
095        final File dataFile = dataFile(rowType);
096        ArchiveFile af = ArchiveFile.buildTabFile();
097        af.setEncoding("UTF8");
098        af.setRowType(rowType);
099        af.addLocation(dataFile.getName());
100        af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
101        af.setId(idField(coreIdColumn));
102        for (Map.Entry<Term, Integer> entry : mapping.entrySet()) {
103            ArchiveField field = new ArchiveField();
104            field.setTerm(entry.getKey());
105            field.setIndex(entry.getValue());
106            af.addField(field);
107        }
108        if (core.equals(rowType)) {
109            if (coreIdTerm != null) {
110                af.getId().setTerm(coreIdTerm);
111            }
112            archive.setCore(af);
113        } else {
114            archive.addExtension(af);
115        }
116
117        // write headers
118        TabWriter writer = TabWriter.fromFile(dataFile);
119        if (useHeaders){
120            String[] header = new String[maxMapping+1];
121            mapping.entrySet().stream().sorted(Map.Entry.comparingByValue()).forEach((e)->{
122                header[e.getValue()] = e.getKey().simpleName();
123            });
124            if (coreIdTerm != null) {
125                header[0] = coreIdTerm.simpleName();
126            }
127            writer.write(header);
128        }
129
130        return writer;
131    }
132
133    public interface RowWriteHandler extends AutoCloseable {
134        void write(String[] row);
135    }
136
137    private class RowWriteHandlerImpl implements RowWriteHandler {
138        private final TabWriter writer;
139        private final int minColumns;
140
141        RowWriteHandlerImpl(TabWriter writer, int minColumns) {
142            this.writer = writer;
143            this.minColumns = minColumns;
144        }
145
146        public void write(String[] row) {
147            if (row != null && row.length < minColumns) {
148                throw new IllegalArgumentException("Input rows are smaller than the defined mapping of " + minColumns + " columns.");
149            }
150            try {
151                writer.write(row);
152            } catch (IOException e) {
153                throw new IllegalStateException(e);
154            }
155        }
156
157        @Override
158        public void close() throws IOException {
159            writer.close();
160        }
161    }
162
163    /**
164     * Useful for glueing mybatis result handlers directly into the dwca writer.
165     * Make sure to close the write handler properly!
166     * @param rowType
167     * @param coreIdColumn
168     * @param mapping
169     * @return a handler accepting single rows to write into the data file
170     */
171    public RowWriteHandler writeHandler(Term rowType, int coreIdColumn, Map<Term, Integer> mapping) {
172        try {
173            final int maxMapping = maxMappingColumn(mapping);
174            TabWriter writer = addArchiveFile(rowType, coreIdColumn, mapping, maxMapping);
175            return new RowWriteHandlerImpl(writer, maxMapping);
176
177        } catch (IOException e) {
178            throw new IllegalStateException(e);
179        }
180    }
181
182    public void setMetadata(Dataset d) {
183        metadata = d;
184    }
185
186    /**
187     * Adds a constituent dataset using the dataset key as the datasetID
188     */
189    public void addConstituent(Dataset eml) {
190        addConstituent(eml.getKey().toString(), eml);
191    }
192
193    /**
194     * Adds a constituent dataset.
195     * The eml file will be called as the datasetID which has to be unique.
196     */
197    public void addConstituent(String datasetID, Dataset eml) {
198        this.constituents.put(datasetID, eml);
199    }
200
201    /**
202     * Writes meta.xml and eml.xml to the archive.
203     */
204    @Override
205    public void close() throws IOException {
206        checkCoreRowType();
207        addEml();
208        addConstituents();
209        MetaDescriptorWriter.writeMetaFile(archive);
210        LOG.info("Wrote archive to {}", archive.getLocation().getAbsolutePath());
211    }
212
213    /**
214     * check if core row type has been written
215     */
216    private void checkCoreRowType() {
217        if (archive.getCore() == null) {
218            throw new IllegalStateException("The core data file has not yet been written for " + core.qualifiedName());
219        }
220    }
221
222    private void addEml() throws IOException {
223        if (metadata != null) {
224            DwcaWriter.writeEml(metadata, new File(dir, "eml.xml"));
225            archive.setMetadataLocation("eml.xml");
226        }
227    }
228
229    private void addConstituents() throws IOException {
230        if (!constituents.isEmpty()) {
231            File ddir = new File(dir, Archive.CONSTITUENT_DIR);
232            ddir.mkdirs();
233            for (Map.Entry<String, Dataset> de : constituents.entrySet()) {
234                DwcaWriter.writeEml(de.getValue(), new File(ddir, de.getKey()+".xml"));
235            }
236        }
237    }
