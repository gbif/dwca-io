048  public class ArchiveFieldIndexComparator implements Comparator<ArchiveField> {
049
050    public int compare(ArchiveField o1, ArchiveField o2) {
051      if (o1.getIndex() == null && o2.getIndex() == null) {
052        return 0;
053      } else if (o1.getIndex() == null) {
054        return -1;
055      } else if (o2.getIndex() == null) {
056        return 1;
057      } else if (o1.getIndex().equals(o2.getIndex())) {
058        return o1.getTerm().qualifiedName().compareToIgnoreCase(o2.getTerm().qualifiedName());
059      } else {
060        return o1.getIndex().compareTo(o2.getIndex());
061      }
062    }
063
064  }

164  /**
165   * Generates an ordered array representing all the {@link Term} matching the position in the underlying file.
166   * The array can contain {@code null} if no {@link Term} is mapped at a specific position.
167   * The size of the array is defined by the maximum index used within {@link ArchiveField} + 1 (since indices are 0
168   * based).
169   *
170   * @return Array of {@link Term} representing the header or an empty Array if no headers are present.
171   */
172  public Term[] getHeader() {
173    List<ArchiveField> archiveFieldsWithIndex = getFieldsSorted()
174            .stream().filter(af -> af.getIndex() != null)
175            .collect(Collectors.toList());
176
177    Optional<Integer> idIndex = id != null ? Optional.of(id.getIndex()) : Optional.empty();
178
179    if(archiveFieldsWithIndex.isEmpty() && !idIndex.isPresent()){
180      return new Term[0];
181    }
182
183    int maxIndex = archiveFieldsWithIndex.stream()
184            .mapToInt(ArchiveField::getIndex).max().getAsInt();
185    maxIndex = Math.max(maxIndex, idIndex.orElse(-1));
186
187    Term[] terms = new Term[maxIndex + 1];
188    // handle id column, assign default Term, it will be rewritten below if assigned to a term
189    idIndex.ifPresent(idx -> terms[idx] = DEFAULT_ID_TERM);
190    archiveFieldsWithIndex.stream().forEach(af -> terms[af.getIndex()] = af.getTerm());
191    return terms;
192  }
193
194  /**
195   * Get default values mapped to their {@link Term} (if any).
196   *
197   * @return default values or {@code Optional.empty()} if none
198   */
199  public Optional<Map<Term, String>> getDefaultValues() {
200    //check if there is default value(s) defined (look for absence of index)
201    Map<Term, String> defaultValues = fields.values().stream()
202            .filter(af -> af.getIndex() == null && af.getDefaultValue() != null)
203            .collect(Collectors.toMap(k -> k.getTerm(), v -> v.getDefaultValue()));
204    return defaultValues.isEmpty() ? Optional.empty() : Optional.of(defaultValues);
205  }

235  public File getLocationFile() {
236    File dataFile;
237    if (archive != null) {
238      if (getLocation() == null) {
239        // use only archive
240        dataFile = archive.getLocation();
241      } else if (getLocation().startsWith("/")) {
242        // absolute already
243        dataFile = new File(getLocation());
244      } else {
245        // use source file relative to archive dir
246        dataFile = new File(archive.getLocation(), getLocation());
247      }
248    } else {
249      dataFile = new File(getLocation());
250    }
251
252    return dataFile;
253  }
