022import java.io.File;
023import java.io.IOException;
024import java.util.ArrayList;
025import java.util.Collections;

027import java.util.HashMap;
028import java.util.Iterator;

036/**
037 * This class can be used to encapsulate information about a file contained within a Darwin Core Archive. It generally
038 * represents the fileType object described in the Darwin Core Archive XSD.
039 *
040 * @see <a href="http://rs.tdwg.org/dwc/text/tdwg_dwc_text.xsd">Darwin Core Archive XSD</a>
041 */
042public class ArchiveFile implements Iterable<Record> {
043  private static final TermFactory TERM_FACTORY = TermFactory.instance();
044
045  public class ArchiveFieldIndexComparator implements Comparator<ArchiveField> {
046
047    public int compare(ArchiveField o1, ArchiveField o2) {
048      if (o1.getIndex() == null && o2.getIndex() == null) {
049        return 0;
050      } else if (o1.getIndex() == null) {
051        return -1;
052      } else if (o2.getIndex() == null) {
053        return 1;
054      } else if (o1.getIndex().equals(o2.getIndex())) {
055        return o1.getTerm().qualifiedName().compareToIgnoreCase(o2.getTerm().qualifiedName());
056      } else {
057        return o1.getIndex().compareTo(o2.getIndex());
058      }
059    }
060
061  }
062
063  private ArchiveField id;
064  private Archive archive;
065  private final LinkedList<String> locations = new LinkedList<String>();
066  private String title;
067  private String fieldsTerminatedBy;
068  private Character fieldsEnclosedBy;
069  // this is actually ignored by the CSVReader and any of \n, \r or \n\r is used
070  private String linesTerminatedBy = "\n";
071  private String encoding = "utf8";
072  private Term rowType;
073  private Integer ignoreHeaderLines = 0;
074
075  private String dateFormat = "YYYY-MM-DD";
076
077  private Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
078
079  public static ArchiveFile buildCsvFile() {
080    ArchiveFile af = new ArchiveFile();
081    af.setFieldsEnclosedBy('"');
082    af.setFieldsTerminatedBy(",");
083    return af;
084  }
085
086  public static ArchiveFile buildTabFile() {
087    ArchiveFile af = new ArchiveFile();
088    af.setFieldsEnclosedBy(null);
089    af.setFieldsTerminatedBy("\t");
090    return af;
091  }
092
093  public static File getLocationFileSorted(File location) {
094    return new File(location.getParentFile(), location.getName() + "-sorted");
095  }
096
097  public static String getLocationSorted(String location) {
098    return location + "-sorted";
099  }
100
101  public void addField(ArchiveField field) {
102    fields.put(field.getTerm(), field);
103  }
104
105  public void addLocation(String location) {
106    if (title == null) {
107      if (location != null && location.lastIndexOf('/') > 1) {
108        title = location.substring(location.lastIndexOf('/') + 1, location.length());
109      } else {
110        title = location;
111      }
112    }
113    locations.add(location);
114  }
115
116  public Archive getArchive() {
117    return archive;
118  }
119
120  public CSVReader getCSVReader() throws IOException {
121    return CSVReaderFactory.build(this);
122  }
123
124  public String getDateFormat() {
125    return dateFormat;
126  }
127
128  public String getEncoding() {
129    return encoding;
130  }
131
132  public ArchiveField getField(Term term) {
133    if (term == null) {
134      return null;
135    }
136    return fields.get(term);
137  }
138
139  public ArchiveField getField(String term) {
140    return getField(TERM_FACTORY.findTerm(term));
141  }
142
143  public Map<Term, ArchiveField> getFields() {
144    return fields;
145  }
146
147  public Character getFieldsEnclosedBy() {
148    return fieldsEnclosedBy;
149  }
150
151  public List<ArchiveField> getFieldsSorted() {
152    List<ArchiveField> list = new ArrayList<ArchiveField>(fields.values());
153    Collections.sort(list, new ArchiveFieldIndexComparator());
154    return list;
155  }
156
157  public String getFieldsTerminatedBy() {
158    return fieldsTerminatedBy;
159  }
160
161  public ArchiveField getId() {
162    return id;
163  }
164
165  public Integer getIgnoreHeaderLines() {
166    return ignoreHeaderLines;
167  }
168
169  public String getLinesTerminatedBy() {
170    return linesTerminatedBy;
171  }
172
173  /**
174   * Get the first file location.
175   * TODO: check if we got more than 1 file and implement some unix concat into a single file first before sorting that
176   */
177  public String getLocation() {
178    return locations.getFirst();
179  }
180
181  public File getLocationFile() {
182    File dataFile;
183    if (archive != null) {
184      if (getLocation() == null) {
185        // use only archive
186        dataFile = archive.getLocation();
187      } else if (getLocation().startsWith("/")) {
188        // absolute already
189        dataFile = new File(getLocation());
190      } else {
191        // use source file relative to archive dir
192        dataFile = new File(archive.getLocation(), getLocation());
193      }
194    } else {
195      dataFile = new File(getLocation());
196    }
197
198    return dataFile;
199  }
200
201  public List<String> getLocations() {
202    return locations;
203  }
204
205  public Term getRowType() {
206    return rowType;
207  }
208
209  public Set<Term> getTerms() {
210    return fields.keySet();
211  }
212
213  public String getTitle() {
214    return title;
215  }
216
217  public boolean hasTerm(Term term) {
218    return getField(term) != null;
219  }
220
221  public boolean hasTerm(String term) {
222    return getField(term) != null;
223  }
224
225  public Iterator<Record> iterator() {
226    return RecordIterator.build(this, true, true);
227  }
228
229  public void setArchive(Archive archive) {
230    this.archive = archive;
231  }
232
233  public void setDateFormat(String dateFormat) {
234    this.dateFormat = dateFormat;
235  }
236
237  public void setEncoding(String encoding) {
238    this.encoding = StringUtils.trimToNull(encoding);
239  }
240
241  public void setFieldsEnclosedBy(Character fieldsEnclosedBy) {
242    this.fieldsEnclosedBy = fieldsEnclosedBy;
243  }
244
245  public void setFieldsTerminatedBy(String fieldsTerminatedBy) {
246    this.fieldsTerminatedBy = StringUtils.defaultIfEmpty(fieldsTerminatedBy, null);
247  }
248
249  public void setId(ArchiveField id) {
250    this.id = id;
251  }
252
253  public void setIgnoreHeaderLines(Integer ignoreHeaderLines) {
254    if (ignoreHeaderLines == null || ignoreHeaderLines < 0) {
255      ignoreHeaderLines = 0;
256    }
257    this.ignoreHeaderLines = ignoreHeaderLines;
258  }
259
260  public void setLinesTerminatedBy(String linesTerminatedBy) {
261    this.linesTerminatedBy = StringUtils.defaultIfEmpty(linesTerminatedBy, null);
262  }
263
264  public void setRowType(Term rowType) {
265    this.rowType = rowType;
266  }
267}
