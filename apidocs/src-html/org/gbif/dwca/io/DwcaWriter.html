049/**
050 * Simple writer class to create valid dwc archives using tab data files.
051 * The meta.xml descriptor is generated automatically and an optional EML metadata document can be added.
052 * The archive is NOT compressed but the final product is a directory with all the necessary files.
053 * For usage of this class please @see DwcaWriterTest.
054 */
055public class DwcaWriter {
056  private Logger log = LoggerFactory.getLogger(DwcaWriter.class);
057  private final File dir;
058  private final boolean useHeaders;
059  private long recordNum;
060  private String coreId;
061  private Map<Term, String> coreRow;
062  private final Term coreRowType;
063  private final Term coreIdTerm;
064  private final Map<Term, TabWriter> writers = Maps.newHashMap();
065  private final Set<Term> headersOut = Sets.newHashSet();
066  private final Map<Term, String> dataFileNames = Maps.newHashMap();
067  // key=rowType, value=columns
068  private final Map<Term, List<Term>> terms = Maps.newHashMap();
069  // key=rowType, value=default values per column
070  private final Map<Term, Map<Term, String>> defaultValues = Maps.newHashMap();
071  private Dataset eml;
072  private Map<String, Dataset> constituents = Maps.newHashMap();

099   * @param coreRowType the core row type

112  public static Map<Term, String> recordToMap(Record rec, ArchiveFile af) {
113    Map<Term, String> map = new HashMap<Term, String>();
114    for (Term t : af.getTerms()) {

120  public static String dataFileName(Term rowType) {
121    return rowType.simpleName().toLowerCase() + ".txt";

124  private void addRowType(Term rowType) throws IOException {
125    terms.put(rowType, new ArrayList<Term>());
126
127    String dfn = dataFileName(rowType);
128    dataFileNames.put(rowType, dfn);
129    File df = new File(dir, dfn);
130    FileUtils.forceMkdir(df.getParentFile());
131    OutputStream out = new FileOutputStream(df);
132    TabWriter wr = new TabWriter(out);
133    writers.put(rowType, wr);
134  }

136  /**
137   * A new core record is started and the last core and all extension records are written.
138   * @param id the new records id
139   * @throws IOException
140   */
141  public void newRecord(String id) throws IOException {
142    // flush last record
143    flushLastCoreRecord();
144    // start new
145    recordNum++;
146    coreId = id;
147    coreRow = new HashMap<Term, String>();
148  }
149
150  private void flushLastCoreRecord() throws IOException {
151    if (coreRow != null) {
152      writeRow(coreRow, coreRowType);
153    }
154  }
155
156  public long getRecordsWritten() {
157    return recordNum;
158  }

160  private void writeRow(Map<Term, String> rowMap, Term rowType) throws IOException {
161    TabWriter writer = writers.get(rowType);
162    List<Term> columns = terms.get(rowType);
163    if (useHeaders && !headersOut.contains(rowType)){
164      // write header row
165      writeHeader(writer, rowType, columns);
166    }
167
168    // make sure coreId is not null for extensions
169    if (coreRowType != rowType && coreId == null){
170      log.warn("Adding an {} extension record to a core without an Id! Skip this record", rowType);
171
172    } else {
173      String[] row = new String[columns.size() + 1];
174      row[0] = coreId;
175      for (Map.Entry<Term, String> conceptTermStringEntry : rowMap.entrySet()) {
176        int column = 1 + columns.indexOf(conceptTermStringEntry.getKey());
177        row[column] = conceptTermStringEntry.getValue();
178      }
179      writer.write(row);
180    }
181  }
182
183  private void writeHeader(TabWriter writer, Term rowType, List<Term> columns) throws IOException {
184    int idx = 0;
185    String[] row = new String[columns.size() + 1];
186    Term idTerm;
187    if (DwcTerm.Taxon == coreRowType){
188      idTerm = DwcTerm.taxonID;
189    } else if (DwcTerm.Occurrence == coreRowType){
190      idTerm = DwcTerm.occurrenceID;
191    } else if (DwcTerm.Identification == coreRowType){
192      idTerm = DwcTerm.identificationID;
193    } else if (DwcTerm.Event == coreRowType){
194      idTerm = DwcTerm.eventID;
195    } else {
196      // default to generic dc identifier for id column
197      idTerm = DcTerm.identifier;
198    }
199    row[idx] = idTerm.simpleName();
200
201    for (Term term : columns) {
202      idx ++;
203      row[idx] = term.simpleName();
204    }
205    writer.write(row);
206
207    headersOut.add(rowType);
208  }

211  /**
212   * Add a single value for the current core record.
213   * Calling this method requires that #newRecord() has been called at least once,
214   * otherwise an IllegalStateException is thrown.
215   * @param term
216   * @param value
217   */
218  public void addCoreColumn(Term term, String value) {
219    // ensure we do not overwrite the coreIdTerm if one is defined
220    if (coreIdTerm != null && coreIdTerm.equals(term)) {
221      throw new IllegalStateException("You cannot add a term that was specified as coreId term");
222    }
223    
224    List<Term> coreTerms = terms.get(coreRowType);
225    if (!coreTerms.contains(term)) {
226      if (useHeaders && recordNum>1){
227        throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
228      }
229      coreTerms.add(term);
230    }
231    try {
232      coreRow.put(term, value);
233    } catch (NullPointerException e) {
234      // no core record has been started yet
235      throw new IllegalStateException("No core record has been created yet. Call newRecord() at least once");
236    }
237  }

288  /**
289   * Add a default value to a term of the provided rowType.
290   * 
291   * @param rowType
292   * @param term
293   * @param defaultValue
294   */
295  public void addDefaultValue(Term rowType, Term term, String defaultValue){
296    
297    if(!defaultValues.containsKey(rowType)){
298      defaultValues.put(rowType, new HashMap<Term, String>());
299    }
300    Map<Term,String> currentDefaultValues= defaultValues.get(rowType);
301    if(currentDefaultValues.containsKey(term)){
302      throw new IllegalStateException("The default value of term "+ term + " is already defined");
303    }
304    currentDefaultValues.put(term, defaultValue);
305  }

314  /**
315   * Add an extension record associated with the current core record.
316   * 
317   * @param rowType
318   * @param row
319   * @throws IOException
320   */
321  public void addExtensionRecord(Term rowType, Map<Term, String> row) throws IOException {
322    // make sure we know the extension rowtype
323    if (!terms.containsKey(rowType)) {
324      addRowType(rowType);
325    }
326    
327    // make sure we know all terms
328    List<Term> knownTerms = terms.get(rowType);
329    final boolean isFirst = knownTerms.isEmpty();
330    for (Term term : row.keySet()) {
331      if (!knownTerms.contains(term)) {
332        if (useHeaders && !isFirst){
333          throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
334        }
335        knownTerms.add(term);
336      }
337    }
338
339    // write extension record
340    writeRow(row, rowType);
341  }

379  /**
380   * Writes meta.xml and eml.xml to the archive and closes tab writers.
381   *
382   */
383  public void close() throws IOException {
384    addEml();
385    addConstituents();
386    addMeta();
387    // flush last record
388    flushLastCoreRecord();
389    // TODO: add missing columns in second iteration of data files
390
391    // close writers
392    for (TabWriter w : writers.values()) {
393      w.close();
394    }
395  }

397  private static void writeEml(Dataset d, File f) throws IOException {
398    if (d != null) {
399      Writer writer = null;
400      try {
401        writer = new FileWriter(f);
402        EMLWriter.write(d, writer);
403      } finally {
404        Closeables.close(writer, true);
405      }
406    }
407  }
408
409  private void addEml() throws IOException {
410    writeEml(eml, new File(dir, "eml.xml"));
411  }
412
413  private void addConstituents() throws IOException {
414    if (!constituents.isEmpty()) {
415      File ddir = new File(dir, Archive.CONSTITUENT_DIR);
416      ddir.mkdirs();
417      for (Map.Entry<String, Dataset> de : constituents.entrySet()) {
418        writeEml(de.getValue(), new File(ddir, de.getKey()+".xml"));
419      }
420    }
421  }
422
423  private void addMeta() throws IOException {
424    File metaFile = new File(dir, Archive.META_FN);
425
426    Archive arch = new Archive();
427    if (eml != null) {
428      arch.setMetadataLocation("eml.xml");
429    }
430    arch.setCore(buildArchiveFile(arch, coreRowType, coreIdTerm));
431    for (Term rowType : this.terms.keySet()) {
432      if (!coreRowType.equals(rowType)) {
433        arch.addExtension(buildArchiveFile(arch, rowType, null));
434      }
435    }
436    MetaDescriptorWriter.writeMetaFile(metaFile, arch);
437  }

439  /**
440   * Build an ArchiveFile for core or extension(s).
441   *
442   * @param archive
443   * @param rowType
444   * @param idTerm the term of the id column, may be null
445   * @return
446   */
447  private ArchiveFile buildArchiveFile(Archive archive, Term rowType, Term idTerm) {
448    ArchiveFile af = ArchiveFile.buildTabFile();
449    af.setArchive(archive);
450    af.addLocation(dataFileNames.get(rowType));
451
452    af.setEncoding("utf-8");
453    af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
454    af.setRowType(rowType);
455
456    ArchiveField id = new ArchiveField();
457    id.setIndex(0);
458    af.setId(id);
459    // always use the index 0 for idTerm
460    if (idTerm != null) {
461      af.addField(buildArchiveField(0, idTerm));
462    }
463    
464    Map<Term,String> termDefaultValueMap = defaultValues.get(rowType);
465    List<Term> rowTypeTerms = terms.get(rowType);
466    int idx = 0;
467    String defaultValue;
468    for (Term c : rowTypeTerms) {
469      idx++;
470      defaultValue = (termDefaultValueMap !=null ? termDefaultValueMap.get(c) : null);
471      af.addField(buildArchiveField(idx, c, defaultValue));
472    }
473    
474    // check if default values are provided for this rowType
475    if(termDefaultValueMap != null){
476      for (Term t : termDefaultValueMap.keySet()) {
477        if(!rowTypeTerms.contains(t)){
478          af.addField(buildArchiveFieldDefaultValue(t, termDefaultValueMap.get(t)));
479        }
480      }
481    }
482
483    return af;
484  }

486  /**
487   * Build an ArchiveField with a defaultValue and no index.
488   *
489   * @param term
490   * @param defaultValue
491   * @return
492   */
493  private ArchiveField buildArchiveFieldDefaultValue(Term term, String defaultValue){
494    Preconditions.checkNotNull(term, "Can't use a null term");
495    Preconditions.checkNotNull(defaultValue, "Can't use a null defaultValue");
496
497    return new ArchiveField(term, defaultValue);
498  }

500  /**
501   * Build an ArchiveField with no defaultValue.
502   *
503   * @param idx
504   * @param term
505   * @return
506   */
507  private ArchiveField buildArchiveField(Integer idx, Term term){
508    return buildArchiveField(idx, term, null);
509  }

511  /**
512   *
513   * Build an ArchiveField from optional parameters.
514   *
515   * @param idx index or null
516   * @param term term or null
517   * @param defaultValue default value or null
518   * @return
519   */
520  private ArchiveField buildArchiveField(Integer idx, Term term, String defaultValue){
521    Preconditions.checkNotNull(idx, "Can't use a null index");
522    Preconditions.checkNotNull(term, "Can't use a null term");
523
524    ArchiveField field = new ArchiveField(idx, term);
525
526    if (StringUtils.isNotBlank(defaultValue)){
527      field.setDefaultValue(defaultValue);
528    }
529    return field;
