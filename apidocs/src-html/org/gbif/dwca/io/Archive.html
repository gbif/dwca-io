037/**
038 * A darwin core star archive allowing easy reading and iteration over a core record with all its extensions.
039 *
040 * @see <a href="http://tdwg.github.io/dwc/terms/guides/text/">Darwin Core Text Guide</a>
041 */
042public class Archive implements Iterable<StarRecord> {
043  public static final String CONSTITUENT_DIR = "dataset";
044  public static final String META_FN = "meta.xml";
045
046  /**
047   * An iterator of fixed DarwinCoreRecords over the core file only. This iterator doesn't need any sorted data files
048   * as it doesn't deal with extensions.
049   */
050  static class ArchiveDwcIterator implements ClosableIterator<DarwinCoreRecord> {
051
052    private CSVReader coreReader;
053    private ArchiveFile core;
054    private int lineCount = 0;
055    private final RecordImpl record;
056    private boolean hasNext = true;
057    private final Set<Term> mappedTerms = new HashSet<Term>();
058
059    ArchiveDwcIterator(Archive archive) {
060      record = new RecordImpl(archive.getCore(), true, true);
061      core = archive.getCore();
062      // remember used DwC and DC terms
063      for (DwcTerm term : DwcTerm.values()) {
064        if (core.hasTerm(term)) {
065          mappedTerms.add(term);
066        }
067      }
068      for (DcTerm term : DcTerm.values()) {
069        if (core.hasTerm(term)) {
070          mappedTerms.add(term);
071        }
072      }
073      try {
074        coreReader = archive.getCore().getCSVReader();
075        // read first core row
076        record.setRow(coreReader.next());
077        if (!record.hasRow()) {
078          hasNext = false;
079        }
080      } catch (Exception e) {
081        hasNext = false;
082        LOG.warn("Exception caught", e);
083      }
084    }
085
086    public void close() {
087      coreReader.close();
088    }
089
090    public boolean hasNext() {
091      return hasNext;
092    }
093
094    public DarwinCoreRecord next() {
095      DarwinCoreRecord dwc = new DarwinCoreRecord();
096      lineCount++;
097      try {
098        for (Term term : mappedTerms) {
099          dwc.setProperty(term, record.value(term));
100        }
101        dwc.setId(record.id());
102        // read next line to see if it exists at all
103        record.setRow(coreReader.next());
104        if (!record.hasRow()) {
105          hasNext = false;
106        }
107
108      } catch (Exception e) {
109        LOG.warn("Bad row somewhere around core line: {}", lineCount, e);
110      }
111      return dwc;
112    }
113
114    public void remove() {
115      throw new UnsupportedOperationException("Cannot remove a row from archive files");
116    }
117
118  }

120  /**
121   * An iterator over core records of the archive that returns StarRecords, i.e. a single core record with all its
122   * related extension records attached. This is a convenient way to iterate over an entire archive accessing all
123   * information including all extensions.
124   *
125   * Requires underlying data files to be sorted by the coreid column if iteration spans multiple data files, i.e. if
126   * extensions exist.
127   * Extension rows with a non existing coreid are skipped. This requires that we use the same sorting order in the
128   * java code as we use for sorting the data files!
129   */
130  class ArchiveIterator implements ClosableIterator<StarRecord> {
131
132    private final StarRecordImpl rec;
133    private RecordIterator coreIter;
134    private Set<RecordIterator> closables = new HashSet<RecordIterator>();
135    private Map<Term, PeekingIterator<Record>> extensionIters = new HashMap<Term, PeekingIterator<Record>>();
136    private Map<Term, Integer> extensionRecordsSkipped = new HashMap<Term, Integer>();
137
138    /**
139     * @param replaceNulls if true replaces common literal null values in all records
140     */
141    ArchiveIterator(Archive archive, boolean replaceNulls, boolean replaceEntities) {
142      List<Term> rowTypes = new ArrayList<Term>();
143
144      try {
145        if (extensions.isEmpty()) {
146          // no need to sort
147          coreIter = RecordIterator.build(archive.getCore(), replaceNulls, replaceEntities);
148        } else {
149          // sort data files to align extension records into a single star record
150          if (!archive.sorted) {
151            archive.sortFiles();
152          }
153          coreIter = buildSortedIterator(archive.getCore(), replaceNulls, replaceEntities);
154        }
155      } catch (IOException e) {
156        LOG.warn("IOException opening core file", e);
157      }
158
159      for (ArchiveFile af : archive.getExtensions()) {
160        rowTypes.add(af.getRowType());
161        RecordIterator iter =
162          extensions.isEmpty() ? RecordIterator.build(af, replaceNulls, replaceEntities) : buildSortedIterator(af,
163            replaceNulls, replaceNulls);
164        closables.add(iter);
165        extensionIters.put(af.getRowType(), Iterators.peekingIterator(iter));
166        extensionRecordsSkipped.put(af.getRowType(), 0);
167      }
168
169      rec = new StarRecordImpl(rowTypes);
170    }
171
172    private RecordIterator buildSortedIterator(ArchiveFile af, boolean replaceNulls, boolean replaceEntities) {
173      // we need to sort the data files
174      String original = af.getLocation();
175      // temporarily modify archive file to create iterator over sorted file
176      af.getLocations().clear();
177      af.addLocation(ArchiveFile.getLocationSorted(original));
178      RecordIterator iter = RecordIterator.build(af, replaceNulls, replaceEntities);
179      // revert to original
180      af.getLocations().clear();
181      af.addLocation(original);
182      return iter;
183    }
184
185    public void close() {
186      coreIter.close();
187      for (ClosableIterator<Record> it : closables) {
188        it.close();
189      }
190      for (Map.Entry<Term, Integer> stringIntegerEntry : extensionRecordsSkipped.entrySet()) {
191        Integer skipped = stringIntegerEntry.getValue();
192        if (skipped > 0) {
193          LOG.debug("{} {} extension records without matching core", skipped, stringIntegerEntry.getKey());
194        }
195      }
196    }
197
198    public boolean hasNext() {
199      return coreIter.hasNext();
200    }
201
202    public StarRecord next() {
203      Record core = coreIter.next();
204      rec.newCoreRecord(core);
205      // add extension records if core id exists
206      if (core.id() != null) {
207        String id = core.id();
208        for (Map.Entry<Term, PeekingIterator<Record>> ext : extensionIters.entrySet()) {
209          PeekingIterator<Record> it = ext.getValue();
210          Term rowType = ext.getKey();
211          while (it.hasNext()) {
212            String extId = it.peek().id();
213            // make sure we have an extid
214            if (Strings.isNullOrEmpty(extId)) {
215              it.next();
216              continue;
217            }
218            if (id.equals(extId)) {
219              // extension row belongs to this core record
220              rec.addRecord(rowType, it.next());
221            } else if (id.compareTo(extId) > 0) {
222              // TODO: we need to use the exact same sorting order, ie comparator, as we use for sorting the data files!!!
223              // this extension id is smaller than the core id and should have been picked up by a core record already
224              // seems to have no matching core record, so lets skip it
225              it.next();
226              extensionRecordsSkipped.put(rowType, extensionRecordsSkipped.get(rowType) + 1);
227            } else {
228              // higher id, we need to wait for this one
229              break;
230            }
231          }
232        }
233      }
234
235      return rec;
236    }
237
238    public void remove() {
239      throw new UnsupportedOperationException("Cannot remove a row from archive files");
240    }
241
242  }

279  public Dataset getMetadata() throws MetadataException {
280    if (metadata == null) {
281      File mf = getMetadataLocationFile();
282      try {
283        InputStream stream;
284        if (mf.exists()) {
285          stream = FileUtils.getInputStream(mf);
286        } else {
287          // try as url
288          URL url = new URL(metadataLocation);
289          stream = url.openStream();
290        }
291        metadata = DatasetParser.build(stream);
292      } catch (IOException e) {
293        throw new MetadataException(e);
294      } catch (RuntimeException e) {
295        throw new MetadataException(e);
296      }
297    }
298    return metadata;
299  }

302    return metadataLocation;
303  }

312  /**
313   * Scans the archive for a semi standard support of dataset constituent metadata.
314   * A dataset constituent is a subdataset which is referenced via dwc:datasetID in the data.
315   * The agreement first introduced by catalogue of life for their GSDs is to have a new folder "dataset" that keeps
316   * a metadata file for each constituent named just as the datasetID and suffixed with .xml.
317   *
318   * @return map of constituent datasetID to metadata file inside the archive
319   */
320  public Map<String, File> getConstituentMetadata() {
321    Map<String, File> constituents = Maps.newHashMap();
322    File constDir = new File(location, CONSTITUENT_DIR);
323    if (constDir.exists()) {
324      File[] files = constDir.listFiles(new FilenameFilter() {
325            public boolean accept(File dir, String filename) { return filename.endsWith(".xml"); }
326          });
327      if (files != null) {
328        for (File cf : files) {
329          String name = cf.getName().split("\\.")[0];
330          constituents.put(name, cf);
331        }
332      }
333    }
334    return constituents;
335  }

378  /**
379   * Sorts all files according to file id, so that we can easily iterate over all files at once.
380   */
381  private void sortFiles() throws IOException {
382    FileUtils futil = new FileUtils();
383    // core
384    try {
385      futil.sort(core.getLocationFile(), ArchiveFile.getLocationFileSorted(core.getLocationFile()), core.getEncoding(),
386        core.getId().getIndex(), core.getFieldsTerminatedBy(), core.getFieldsEnclosedBy(), core.getLinesTerminatedBy(),
387        core.getIgnoreHeaderLines());
388    } catch (IOException e) {
389      LOG.error("Error sorting core file " + core.getLocationFile() + " : " + e.getMessage());
390      throw e;
391    } catch (RuntimeException e) {
392      LOG.error("Error sorting core file " + core.getLocationFile() + " : " + e.getMessage());
393      throw e;
394    }
395    // extensions
396    for (ArchiveFile ext : extensions) {
397      try {
398        futil.sort(ext.getLocationFile(), ArchiveFile.getLocationFileSorted(ext.getLocationFile()), ext.getEncoding(),
399          ext.getId().getIndex(), ext.getFieldsTerminatedBy(), ext.getFieldsEnclosedBy(), ext.getLinesTerminatedBy(),
400          ext.getIgnoreHeaderLines());
401      } catch (IOException e) {
402        LOG.error("Error sorting extension file " + ext.getLocationFile() + " : " + e.getMessage());
403        throw e;
404      } catch (RuntimeException e) {
405        LOG.error("Error sorting extension file " + ext.getLocationFile() + " : " + e.getMessage());
406        throw e;
407      }
408    }
409    sorted = true;
410  }
