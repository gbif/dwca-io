013package org.gbif.dwca.io;
014
015import org.gbif.dwc.DwcFiles;
016import org.gbif.dwc.meta.DwcMetaFiles;
017import org.gbif.util.DownloadUtil;
018
019import java.io.File;
020import java.io.IOException;
021import java.net.URL;
022
023import com.google.common.io.Files;
024import org.apache.commons.lang3.StringUtils;
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028/**
029 * Factory used to build {@link Archive} object from a DarwinCore Archive file.
030 * 
031 * @author mdoering
032 *
033 */
034public class ArchiveFactory {
035
036  private static final Logger LOG = LoggerFactory.getLogger(ArchiveFactory.class);
037
038  /**
039   * Opens an archive from a URL, downloading and decompressing it.
040   *
041   * @param archiveUrl the location of a compressed archive or single data file
042   * @param workingDir writable directory to download to and decompress archive
043   */
044  public static Archive openArchive(URL archiveUrl, File workingDir) throws IOException, UnsupportedArchiveException {
045    File downloadTo = new File(workingDir, "dwca-download");
046    File dwca = new File(workingDir, "dwca");
047    DownloadUtil.download(archiveUrl, downloadTo);
048    return openArchive(downloadTo, dwca);
049  }
050
051  /**
052   * Opens an archive from a local file and decompresses or copies it into the given archive directory.
053   * Make sure the archive directory does not contain files already, any existing files will be removed!
054   *
055   * If the source archive is an uncompressed, single data file and a valid archive, it will be copied as is
056   * to the archiveDir.
057   *
058   * @param archiveFile the location of a compressed archive or single data file
059   * @param archiveDir  empty, writable directory used to keep decompress archive in
060   */
061  public static Archive openArchive(File archiveFile, File archiveDir) throws IOException, UnsupportedArchiveException {
062
063    try {
064      Archive archive = DwcFiles.fromCompressed(archiveFile.toPath(), archiveDir.toPath());
065      // keep validation for backward compatibility
066      return validateArchive(archive);
067    }
068    catch (UnsupportedArchiveException uaEx){
069      LOG.debug("Could not uncompress archive [{}], try to read as single text file", archiveFile, uaEx);
070      // If its a text file only we will get this exception - but also for corrupt compressions
071      // try to open as text file only and if successful copy file to archive dir
072      Archive arch = openArchiveDataFile(archiveFile);
073      Files.copy(archiveFile, new File(archiveDir, archiveFile.getName()));
074      return arch;
075    }
076  }
077
078  /**
079   * Opens a dwca archive which is just a single decompressed data file with headers, e.g. a csv or tab delimited file
080   */
081  public static Archive openArchiveDataFile(File dataFile) throws IOException, UnsupportedArchiveException {
082    Archive archive = DwcFiles.fromLocation(dataFile.toPath());
083
084    // for backward compatibility reason we keep this behavior to check if we have a metadata file next to
085    // a single file.
086    if(StringUtils.isBlank(archive.getMetadataLocation())) {
087      // check if we also have a metadata file next to this data file
088      DwcMetaFiles.discoverMetadataFile(dataFile.getParentFile().toPath())
089              .ifPresent(archive::setMetadataLocation);
090    }
091
092    // final validation
093    return validateArchive(archive);
094  }
095
096  /**
097   * @param dwcaFolder the location of an expanded dwc archive directory or just a single dwc text file
098   */
099  public static Archive openArchive(File dwcaFolder) throws IOException, UnsupportedArchiveException {
100
101    Archive archive = DwcFiles.fromLocation(dwcaFolder.toPath());
102
103    // final validation
104    return validateArchive(archive);
105  }
106
107  private static Archive validateArchive(Archive archive) throws UnsupportedArchiveException {
108    validateCoreFile(archive.getCore(), !archive.getExtensions().isEmpty());
109    for (ArchiveFile af : archive.getExtensions()) {
110      validateExtensionFile(af);
111    }
112    // report basic stats
113    LOG.debug("Archive contains " + archive.getExtensions().size() + " described extension files");
114    LOG.debug("Archive contains " + archive.getCore().getFields().size() + " core properties");
115    return archive;
116  }
117
118  private static void validateCoreFile(ArchiveFile f, boolean hasExtensions) throws UnsupportedArchiveException {
119    if (hasExtensions) {
120      if (f.getId() == null) {
121        LOG.warn(
122          "DwC-A core data file " + f.getTitle() + " is lacking an id column. No extensions allowed in this case");
123      }
124    }
125    validateFile(f);
126  }
127
128  private static void validateExtensionFile(ArchiveFile f) throws UnsupportedArchiveException {
129    if (f.getId() == null) {
130      throw new UnsupportedArchiveException(
131        "DwC-A data file " + f.getTitle() + " requires an id or foreign key to the core id");
132    }
133    validateFile(f);
134  }
135
136  private static void validateFile(ArchiveFile f) throws UnsupportedArchiveException {
137    if (f == null) {
138      throw new UnsupportedArchiveException("DwC-A data file is NULL");
139    }
140    if (f.getLocationFile() == null) {
141      throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a location");
142    }
143    if (f.getEncoding() == null) {
144      throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a character encoding");
145    }
146  }
