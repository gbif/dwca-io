070  /**
071   * Predefined mapping between {@link Term} and its rowType.
072   * Ordering is important since the first found will be used.
073   */
074  private static final Map<Term, Term> TERM_TO_ROW_TYPE;
075  static {
076    Map<Term, Term> idToRowType = new LinkedHashMap<>();
077    idToRowType.put(DwcTerm.occurrenceID, DwcTerm.Occurrence);
078    idToRowType.put(DwcTerm.taxonID, DwcTerm.Taxon);
079    idToRowType.put(DwcTerm.eventID, DwcTerm.Event);
080    TERM_TO_ROW_TYPE = Collections.unmodifiableMap(idToRowType);
081  }

084   * Terms that can represent an identifier within a file

086  private static final List<Term> ID_TERMS = Collections.unmodifiableList(
087          Arrays.asList(DwcTerm.occurrenceID, DwcTerm.taxonID, DwcTerm.eventID, DcTerm.identifier));

097  /**
098   * Opens an archive from a URL, downloading and decompressing it.
099   *
100   * @param archiveUrl the location of a compressed archive or single data file
101   * @param workingDir writable directory to download to and decompress archive
102   */
103  public static Archive openArchive(URL archiveUrl, File workingDir) throws IOException, UnsupportedArchiveException {
104    File downloadTo = new File(workingDir, "dwca-download");
105    File dwca = new File(workingDir, "dwca");
106    DownloadUtil.download(archiveUrl, downloadTo);
107    return openArchive(downloadTo, dwca);
108  }

110  /**
111   * Opens an archive from a local file and decompresses or copies it into the given archive directory.
112   * Make sure the archive directory does not contain files already, any existing files will be removed!
113   *
114   * If the source archive is an uncompressed, single data file and a valid archive, it will be copied as is
115   * to the archiveDir.
116   *
117   * @param archiveFile the location of a compressed archive or single data file
118   * @param archiveDir  empty, writable directory used to keep decompress archive in
119   */
120  public static Archive openArchive(File archiveFile, File archiveDir) throws IOException, UnsupportedArchiveException {
121    if (archiveDir.exists()) {
122      // clean up any existing folder
123      LOG.debug("Deleting existing archive folder [{}]", archiveDir.getAbsolutePath());
124      org.gbif.utils.file.FileUtils.deleteDirectoryRecursively(archiveDir);
125    }
126    FileUtils.forceMkdir(archiveDir);
127    // try to decompress archive
128    try {
129      CompressionUtil.decompressFile(archiveDir, archiveFile, true);
130      // we keep subfolder, but often the entire archive is within one subfolder. Remove that root folder if present
131      File[] rootFiles = archiveDir.listFiles((FileFilter) HiddenFileFilter.VISIBLE);
132      if (rootFiles.length == 1) {
133        File root = rootFiles[0];
134        if (root.isDirectory()) {
135          // single root dir, flatten structure
136          LOG.debug("Removing single root folder {} found in decompressed archive", root.getAbsoluteFile());
137          for (File f : FileUtils.listFiles(root, TrueFileFilter.TRUE, null)) {
138            File f2 = new File(archiveDir, f.getName());
139            f.renameTo(f2);
140          }
141        }
142      }
143      // continue to read archive from the tmp dir
144      return openArchive(archiveDir);
145
146    } catch (CompressionUtil.UnsupportedCompressionType e) {
147      LOG.debug("Could not uncompress archive [{}], try to read as single text file", archiveFile, e);
148      // If its a text file only we will get this exception - but also for corrupt compressions
149      // try to open as text file only and if successful copy file to archive dir
150      Archive arch = openArchiveDataFile(archiveFile);
151      Files.copy(archiveFile, new File(archiveDir, archiveFile.getName()));
152      return arch;
153    }
154  }

156  /**
157   * Opens a dwca archive which is just a single decompressed data file with headers, e.g. a csv or tab delimited file
158   */
159  public static Archive openArchiveDataFile(File dataFile) throws IOException, UnsupportedArchiveException {
160    Archive archive = new Archive();
161    archive.setLocation(dataFile);
162
163    ArchiveFile coreFile = readFileHeaders(dataFile);
164    archive.setCore(coreFile);
165
166    // check if we also have a metadata file next to this data file
167    discoverMetadataFile(archive, dataFile.getParentFile());
168
169    // final validation
170    return validateArchive(archive);
171  }

173  /**
174   * @param dwcaFolder the location of an expanded dwc archive directory or just a single dwc text file
175   */
176  public static Archive openArchive(File dwcaFolder) throws IOException, UnsupportedArchiveException {
177    if (!dwcaFolder.exists()) {
178      throw new FileNotFoundException("Archive folder not existing: " + dwcaFolder.getAbsolutePath());
179    }
180    // delegate to open data file method if its a single file, not a folder
181    if (dwcaFolder.isFile()) {
182      return openArchiveDataFile(dwcaFolder);
183    }
184
185    Archive archive = new Archive();
186    archive.setLocation(dwcaFolder);
187
188    // Accommodate archives coming from legacy IPTs which put a "\" before each filename
189    // http://dev.gbif.org/issues/browse/POR-2396
190    // https://code.google.com/p/gbif-providertoolkit/issues/detail?id=1015
191    Iterator<File> iter = FileUtils.iterateFiles(dwcaFolder, new String[] {"xml", "txt"}, false);
192    while (iter.hasNext()) {
193      File f = iter.next();
194      if (f.getName().startsWith("\\")) {
195        String orig = f.getName();
196        String replacement = f.getName().replaceFirst("\\\\", "");
197        LOG.info("Renaming file from {} to {}", orig, replacement);
198        f.renameTo(new File(dwcaFolder, replacement));
199      }
200    }
201
202    // read metadata
203    File mf = new File(dwcaFolder, Archive.META_FN);
204    if (mf.exists()) {
205      // read metafile
206      readMetaDescriptor(archive, new FileInputStream(mf));
207
208    } else {
209      // meta.xml lacking.
210      // Try to detect data files ourselves as best as we can.
211      // look for a single, visible text data file
212      List<File> dataFiles = new ArrayList<File>();
213      for (String suffix : DATA_FILE_SUFFICES) {
214        FileFilter ff = FileFilterUtils.and(
215            FileFilterUtils.suffixFileFilter(suffix, IOCase.INSENSITIVE), HiddenFileFilter.VISIBLE
216        );
217        dataFiles.addAll(Arrays.asList(dwcaFolder.listFiles(ff)));
218      }
219
220      if (dataFiles.size() == 1) {
221        File dataFile = new File(dwcaFolder, dataFiles.get(0).getName());
222        ArchiveFile coreFile = readFileHeaders(dataFile);
223        coreFile.getLocations().clear();
224        coreFile.addLocation(dataFile.getName());
225        archive.setCore(coreFile);
226
227      } else {
228        throw new UnsupportedArchiveException(
229          "The archive given is a folder with more or less than 1 data files having a csv, txt or tab suffix");
230      }
231    }
232
233    // check if we also have a metadata file next to this data file
234    discoverMetadataFile(archive, mf.getParentFile());
235
236    // final validation
237    return validateArchive(archive);
238  }

241  private static void discoverMetadataFile(Archive archive, File folder) {
242    if (archive.getMetadataLocation() == null) {
243      // search for popular metadata filenames
244      for (String metadataFN : Lists.newArrayList("eml.xml", "metadata.xml")) {
245        File emlFile = new File(folder, metadataFN);
246        if (emlFile.exists()) {
247          archive.setMetadataLocation(metadataFN);
248          break;
249        }
250      }
251    }
252  }

254  private static ArchiveFile readFileHeaders(File dataFile) throws UnsupportedArchiveException, IOException {
255    ArchiveFile dwcFile = new ArchiveFile();
256    dwcFile.addLocation(null);
257    dwcFile.setIgnoreHeaderLines(1);
258
259    String[] headers;
260    try (CSVReader reader = CSVReaderFactory.build(dataFile)) {
261      // copy found delimiters & encoding
262      dwcFile.setEncoding(reader.encoding);
263      dwcFile.setFieldsTerminatedBy(reader.delimiter);
264      dwcFile.setFieldsEnclosedBy(reader.quoteChar);
265      headers = reader.getHeader();
266    }
267
268    // detect dwc terms as good as we can based on header row
269    int index = 0;
270    for (String head : headers) {
271      // there are never any quotes in term names - remove them just in case the csvreader didnt recognize them
272      if (head != null && head.length() > 1) {
273        try {
274          Term dt = TERM_FACTORY.findTerm(head);
275          ArchiveField field = new ArchiveField(index, dt, null, DataType.string);
276          dwcFile.addField(field);
277        } catch (IllegalArgumentException e) {
278          LOG.warn("Illegal term name >>{}<< found in header, ignore column {}", head, index);
279        }
280      }
281      index++;
282    }
283
284    List<Term> headerAsTerm = dwcFile.getFields().keySet()
285            .stream()
286            .collect(Collectors.toList());
287
288    determineRecordIdentifier(headerAsTerm).ifPresent(
289            t -> dwcFile.setId(dwcFile.getField(t))
290    );
291
292    determineRowType(headerAsTerm).ifPresent(
293            t -> dwcFile.setRowType(t)
294    );
295    return dwcFile;
296  }

298  @VisibleForTesting
299  protected static void readMetaDescriptor(Archive archive, InputStream metaDescriptor) throws UnsupportedArchiveException {
300
301    try {
302      SAXParser p = SAX_FACTORY.newSAXParser();
303      MetaXMLSaxHandler mh = new MetaXMLSaxHandler(archive);
304      LOG.debug("Reading archive metadata file");
305      p.parse(new BOMInputStream(metaDescriptor), mh);
306    } catch (Exception e1) {
307      LOG.warn("Exception caught", e1);
308      throw new UnsupportedArchiveException(e1);
309    }
310  }

312  private static Archive validateArchive(Archive archive) throws UnsupportedArchiveException {
313    validateCoreFile(archive.getCore(), !archive.getExtensions().isEmpty());
314    for (ArchiveFile af : archive.getExtensions()) {
315      validateExtensionFile(af);
316    }
317    // report basic stats
318    LOG.debug("Archive contains " + archive.getExtensions().size() + " described extension files");
319    LOG.debug("Archive contains " + archive.getCore().getFields().size() + " core properties");
320    return archive;
321  }
322
323  private static void validateCoreFile(ArchiveFile f, boolean hasExtensions) throws UnsupportedArchiveException {
324    if (hasExtensions) {
325      if (f.getId() == null) {
326        LOG.warn(
327          "DwC-A core data file " + f.getTitle() + " is lacking an id column. No extensions allowed in this case");
328      }
329    }
330    validateFile(f);
331  }
332
333  private static void validateExtensionFile(ArchiveFile f) throws UnsupportedArchiveException {
334    if (f.getId() == null) {
335      throw new UnsupportedArchiveException(
336        "DwC-A data file " + f.getTitle() + " requires an id or foreign key to the core id");
337    }
338    validateFile(f);
339  }
340
341  private static void validateFile(ArchiveFile f) throws UnsupportedArchiveException {
342    if (f == null) {
343      throw new UnsupportedArchiveException("DwC-A data file is NULL");
344    }
345    if (f.getLocationFile() == null) {
346      throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a location");
347    }
348    if (f.getEncoding() == null) {
349      throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a character encoding");
350    }
351  }

353  /**
354   * Tries to determine the rowType based on a list of {@link Term}.
355   *
356   * @param terms the list can contain null values
357   *
358   * @return {@link Term} as {@code Optional} or {@code Optional.empty()} if can not be determined
359   */
360  static Optional<Term> determineRowType(List<Term> terms) {
361    return TERM_TO_ROW_TYPE.entrySet().stream()
362            .filter(ke -> terms.contains(ke.getKey()))
363            .map(Map.Entry::getValue).findFirst();
364  }
365
366  /**
367   * Tries to determine the record identifier based on a list of {@link Term}.
368   *
369   * @param terms the list can contain null values
370   *
371   * @return {@link Term} as {@code Optional} or {@code Optional.empty()} if can not be determined
372   */
373  static Optional<Term> determineRecordIdentifier(List<Term> terms) {
374    //try to find the first matching term respecting the order defined by ID_TERMS
375    return ID_TERMS.stream()
376            .filter(t -> terms.contains(t))
377            .findFirst();
378  }
