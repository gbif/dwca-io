001package org.gbif.dwca.io;
002
003import java.util.regex.Pattern;
004
005import org.apache.commons.lang3.StringUtils;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008import org.xml.sax.Attributes;
009import org.xml.sax.SAXException;
010import org.xml.sax.helpers.DefaultHandler;
011
012/**
013 * A simple metadata sax base handler that collects all character data inside elements into a string buffer, resetting
014 * the buffer with every element start and storing the string version of the buffer in this.content when the end of the
015 * element is reached.
016 * Make sure to call the super methods when implementing this handler!
017 *
018 * @author markus
019 *
020 */
021public abstract class SimpleSaxHandler extends DefaultHandler {
022  protected final Logger log = LoggerFactory.getLogger(getClass());
023  protected String content;
024  private StringBuffer chars;
025  private Pattern normWhitespace = Pattern.compile("\\s+");
026  protected String parents = "";
027
028  @Override
029  public void characters(char[] ch, int start, int length) {
030    chars.append(ch, start, length);
031  }
032
033  @Override
034  public void endElement(String uri, String localName, String qName) throws SAXException {
035    content = StringUtils.trimToNull(chars.toString());
036    // norm whitespace
037    if (content != null) {
038      content = normWhitespace.matcher(content).replaceAll(" ");
039    }
040    parents = parents.substring(0, parents.length() - localName.length() - 1);
041  }
042
043  @Override
044  public void startDocument() {
045    parents = "";
046  }
047
048  @Override
049  public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
050    chars = new StringBuffer();
051    parents += "/" + localName.toLowerCase();
052  }
053
054  @Override
055  public String toString() {
056    return getClass().getSimpleName();
057  }
058
059}
