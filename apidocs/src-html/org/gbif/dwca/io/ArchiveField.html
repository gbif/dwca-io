001package org.gbif.dwca.io;
002
003import org.gbif.dwc.terms.Term;
004
005public class ArchiveField {
006
007  public enum DataType {
008    string("xs:string"), bool("xs:boolean"), integer("xs:integer"), decimal("xs:decimal"), date("xs:dateTime"),
009    uri("xs:URI");
010
011    private final String xsdType;
012
013    DataType(String xsd) {
014      xsdType = xsd;
015    }
016
017    public static DataType findByExtensionEnumTypeName(String type) {
018      if (type.equalsIgnoreCase("boolean")) {
019        return bool;
020      }
021      for (DataType dt : DataType.values()) {
022        if (dt.name().equalsIgnoreCase(type)) {
023          return dt;
024        }
025      }
026      return null;
027    }
028
029    public static DataType findByXmlSchemaType(String type) {
030      for (DataType dt : DataType.values()) {
031        if (dt.xsdType.equalsIgnoreCase(type)) {
032          return dt;
033        }
034      }
035      return null;
036    }
037  }
038
039  private Integer index;
040  private Term term;
041  private String defaultValue;
042  private String delimitedBy;
043
044  private DataType type = DataType.string;
045
046  public ArchiveField() {
047
048  }
049
050  public ArchiveField(Integer index, Term term, String defaultValue, DataType type) {
051    this.index = index;
052    this.term = term;
053    this.defaultValue = defaultValue;
054    this.type = type;
055  }
056
057  public ArchiveField(Integer index, Term term, String defaultValue, DataType type, String delimitedBy) {
058    this.index = index;
059    this.term = term;
060    this.defaultValue = defaultValue;
061    this.type = type;
062    this.delimitedBy = delimitedBy;
063  }
064
065  public String getDefaultValue() {
066    return defaultValue;
067  }
068
069  public Integer getIndex() {
070    return index;
071  }
072
073  public Term getTerm() {
074    return term;
075  }
076
077  public DataType getType() {
078    return type;
079  }
080
081  public String getDelimitedBy() {
082    return delimitedBy;
083  }
084
085  public void setDefaultValue(String defaultValue) {
086    this.defaultValue = defaultValue;
087  }
088
089  public void setIndex(Integer index) {
090    this.index = index;
091  }
092
093  public void setTerm(Term term) {
094    this.term = term;
095  }
096
097  public void setType(DataType type) {
098    this.type = type;
099  }
100
101  public void setDelimitedBy(String delimitedBy) {
102    this.delimitedBy = delimitedBy;
103  }
104
105}
