013package org.gbif.dwca.io;
014
015import org.apache.commons.lang3.StringUtils;
016import org.gbif.dwc.terms.Term;
017import org.gbif.dwc.terms.TermFactory;
018import org.gbif.dwca.record.Record;
019import org.gbif.dwca.record.RecordIterator;
020import org.gbif.utils.file.csv.CSVReader;
021
022import java.io.File;
023import java.io.IOException;
024import java.util.*;
025
026/**
027 * This class can be used to encapsulate information about a file contained within a Darwin Core Archive. It generally
028 * represents the fileType object described in the Darwin Core Archive XSD.
029 *
030 * @see <a href="http://rs.tdwg.org/dwc/text/tdwg_dwc_text.xsd">Darwin Core Archive XSD</a>
031 */
032public class ArchiveFile implements Iterable<Record> {
033  private static final TermFactory TERM_FACTORY = TermFactory.instance();
034
035  public class ArchiveFieldIndexComparator implements Comparator<ArchiveField> {
036
037    public int compare(ArchiveField o1, ArchiveField o2) {
038      if (o1.getIndex() == null && o2.getIndex() == null) {
039        return 0;
040      } else if (o1.getIndex() == null) {
041        return -1;
042      } else if (o2.getIndex() == null) {
043        return 1;
044      } else if (o1.getIndex().equals(o2.getIndex())) {
045        return o1.getTerm().qualifiedName().compareToIgnoreCase(o2.getTerm().qualifiedName());
046      } else {
047        return o1.getIndex().compareTo(o2.getIndex());
048      }
049    }
050
051  }
052
053  private ArchiveField id;
054  private Archive archive;
055  private final LinkedList<String> locations = new LinkedList<String>();
056  private String title;
057  private String fieldsTerminatedBy;
058  private Character fieldsEnclosedBy;
059  // this is actually ignored by the CSVReader and any of \n, \r or \n\r is used
060  private String linesTerminatedBy = "\n";
061  private String encoding = "utf8";
062  private Term rowType;
063  private Integer ignoreHeaderLines = 0;
064
065  private String dateFormat = "YYYY-MM-DD";
066
067  private Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
068
069  public static ArchiveFile buildCsvFile() {
070    ArchiveFile af = new ArchiveFile();
071    af.setFieldsEnclosedBy('"');
072    af.setFieldsTerminatedBy(",");
073    return af;
074  }
075
076  public static ArchiveFile buildTabFile() {
077    ArchiveFile af = new ArchiveFile();
078    af.setFieldsEnclosedBy(null);
079    af.setFieldsTerminatedBy("\t");
080    return af;
081  }
082
083  public static File getLocationFileSorted(File location) {
084    return new File(location.getParentFile(), location.getName() + "-sorted");
085  }
086
087  public static String getLocationSorted(String location) {
088    return location + "-sorted";
089  }
090
091  public void addField(ArchiveField field) {
092    fields.put(field.getTerm(), field);
093  }
094
095  public void addLocation(String location) {
096    if (title == null) {
097      if (location != null && location.lastIndexOf('/') > 1) {
098        title = location.substring(location.lastIndexOf('/') + 1, location.length());
099      } else {
100        title = location;
101      }
102    }
103    locations.add(location);
104  }
105
106  public Archive getArchive() {
107    return archive;
108  }
109
110  public CSVReader getCSVReader() throws IOException {
111    return org.gbif.io.CSVReaderFactory.build(this);
112  }
113
114  public String getDateFormat() {
115    return dateFormat;
116  }
117
118  public String getEncoding() {
119    return encoding;
120  }
121
122  public ArchiveField getField(Term term) {
123    if (term == null) {
124      return null;
125    }
126    return fields.get(term);
127  }
128
129  public ArchiveField getField(String term) {
130    return getField(TERM_FACTORY.findTerm(term));
131  }
132
133  public Map<Term, ArchiveField> getFields() {
134    return fields;
135  }
136
137  public Character getFieldsEnclosedBy() {
138    return fieldsEnclosedBy;
139  }
140
141  public List<ArchiveField> getFieldsSorted() {
142    List<ArchiveField> list = new ArrayList<ArchiveField>(fields.values());
143    Collections.sort(list, new ArchiveFieldIndexComparator());
144    return list;
145  }
146
147  public String getFieldsTerminatedBy() {
148    return fieldsTerminatedBy;
149  }
150
151  public ArchiveField getId() {
152    return id;
153  }
154
155  public Integer getIgnoreHeaderLines() {
156    return ignoreHeaderLines;
157  }
158
159  public String getLinesTerminatedBy() {
160    return linesTerminatedBy;
161  }
162
163  /**
164   * Get the first file location.
165   * TODO: check if we got more than 1 file and implement some unix concat into a single file first before sorting that
166   */
167  public String getLocation() {
168    return locations.getFirst();
169  }
170
171  public File getLocationFile() {
172    File dataFile;
173    if (archive != null) {
174      if (getLocation() == null) {
175        // use only archive
176        dataFile = archive.getLocation();
177      } else if (getLocation().startsWith("/")) {
178        // absolute already
179        dataFile = new File(getLocation());
180      } else {
181        // use source file relative to archive dir
182        dataFile = new File(archive.getLocation(), getLocation());
183      }
184    } else {
185      dataFile = new File(getLocation());
186    }
187
188    return dataFile;
189  }
190
191  public List<String> getLocations() {
192    return locations;
193  }
194
195  public Term getRowType() {
196    return rowType;
197  }
198
199  public Set<Term> getTerms() {
200    return fields.keySet();
201  }
202
203  public String getTitle() {
204    return title;
205  }
206
207  public boolean hasTerm(Term term) {
208    return getField(term) != null;
209  }
210
211  public boolean hasTerm(String term) {
212    return getField(term) != null;
213  }
214
215  public Iterator<Record> iterator() {
216    return RecordIterator.build(this, true, true);
217  }
218
219  public void setArchive(Archive archive) {
220    this.archive = archive;
221  }
222
223  public void setDateFormat(String dateFormat) {
224    this.dateFormat = dateFormat;
225  }
226
227  public void setEncoding(String encoding) {
228    this.encoding = StringUtils.trimToNull(encoding);
229  }
230
231  public void setFieldsEnclosedBy(Character fieldsEnclosedBy) {
232    this.fieldsEnclosedBy = fieldsEnclosedBy;
233  }
234
235  public void setFieldsTerminatedBy(String fieldsTerminatedBy) {
236    this.fieldsTerminatedBy = StringUtils.defaultIfEmpty(fieldsTerminatedBy, null);
237  }
238
239  public void setId(ArchiveField id) {
240    this.id = id;
241  }
242
243  public void setIgnoreHeaderLines(Integer ignoreHeaderLines) {
244    if (ignoreHeaderLines == null || ignoreHeaderLines < 0) {
245      ignoreHeaderLines = 0;
246    }
247    this.ignoreHeaderLines = ignoreHeaderLines;
248  }
249
250  public void setLinesTerminatedBy(String linesTerminatedBy) {
251    this.linesTerminatedBy = StringUtils.defaultIfEmpty(linesTerminatedBy, null);
252  }
253
254  public void setRowType(Term rowType) {
255    this.rowType = rowType;
256  }
257}
