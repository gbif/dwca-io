001package org.gbif.dwca.tools;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.StringReader;
006import javax.xml.transform.Source;
007import javax.xml.transform.stream.StreamSource;
008import javax.xml.validation.Schema;
009import javax.xml.validation.SchemaFactory;
010import javax.xml.validation.Validator;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014import org.xml.sax.SAXException;
015
016/**
017 * Utility class that exposes a singleton instance of an xml validator for the dwc archive meta.xml descriptor.
018 * @Deprecated use validator for that
019 */
020@Deprecated
021public class MetaValidator {
022  private static final Logger LOG = LoggerFactory.getLogger(MetaValidator.class);
023  private static final String SCHEMA_LANG = "http://www.w3.org/2001/XMLSchema";
024  private static final String XSD_SCHEMA = "https://raw.githubusercontent.com/tdwg/dwc/master/standard/documents/text/tdwg_dwc_text.xsd";
025  private static Validator VALIDATOR;
026
027  public static void validate(String xml) throws IllegalArgumentException, IOException, SAXException {
028    validate(new StreamSource(new StringReader(xml)));
029  }
030
031  public static void validate(InputStream xml) throws IllegalArgumentException, IOException, SAXException {
032    validate(new StreamSource(xml));
033  }
034
035  /**
036   * @throws IllegalArgumentException
037   *      If the <code>Source</code>
038   *      is an XML artifact that the implementation cannot
039   *      validate (for example, a processing instruction).
040   *
041   * @throws SAXException
042   *      If the {@link org.xml.sax.ErrorHandler} throws a {@link SAXException} or
043   *      if a fatal error is found and the {@link org.xml.sax.ErrorHandler} returns
044   *      normally.
045   *
046   * @throws IOException
047   *      If the validator is processing a
048   *      {@link javax.xml.transform.sax.SAXSource} and the
049   *      underlying {@link org.xml.sax.XMLReader} throws an
050   *      {@link IOException}.
051   *
052   *
053   * @throws NullPointerException If <code>source</code> is
054   *   <code>null</code>.
055   */
056  public static void validate(Source source) throws IllegalArgumentException, IOException, SAXException {
057    getValidator().validate(source);
058    LOG.debug("XML passed validation");
059  }
060
061  public static Validator getValidator() throws IOException, SAXException {
062    if (VALIDATOR == null) {
063      // define the type of schema - we use W3C:
064      // resolve validation driver:
065      SchemaFactory factory = SchemaFactory.newInstance(SCHEMA_LANG);
066      // create schema by reading it from gbif online resources:
067      Schema schema = factory.newSchema(new StreamSource(XSD_SCHEMA));
068      VALIDATOR = schema.newValidator();
069    }
070    return VALIDATOR;
071  }
072}
