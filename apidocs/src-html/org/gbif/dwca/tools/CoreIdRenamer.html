036/**
037 * Tool that takes a dwca file and renames its core ids according to some id generator.
038 * All extension records and ID terms pointing to the core ID are renamed accordingly to preserve the archive
039 * integrity.
040 */
041public class CoreIdRenamer {
042
043  private final IdGenerator idGen;
044
045  public CoreIdRenamer() {
046    idGen = new IntSequenceGenerator(1);
047  }
048
049  public CoreIdRenamer(IdGenerator idGen) {
050    this.idGen = idGen;
051  }
052
053  public CoreIdRenamer(int firstId) {
054    idGen = new IntSequenceGenerator(firstId);
055  }
056
057  public void rename(File dwca) throws IOException {
058    rename(ArchiveFactory.openArchive(dwca));
059  }
060
061  public void rename(Archive arch) throws IOException {
062    ArchiveField id = arch.getCore().getId();
063    TermFactory fact = TermFactory.instance();
064    Term rowType = arch.getCore().getRowType();
065
066    // detect dependent terms with ids to be renamed accordingly
067    List<Term> relatedTerms = new ArrayList<Term>();
068    if (DwcTerm.Taxon == rowType) {
069      // potentially 3 foreign keys with values pointing to the coreID
070      if (arch.getCore().hasTerm(DwcTerm.parentNameUsageID)) {
071        relatedTerms.add(DwcTerm.parentNameUsageID);
072      }
073      if (arch.getCore().hasTerm(DwcTerm.acceptedNameUsageID)) {
074        relatedTerms.add(DwcTerm.acceptedNameUsageID);
075      }
076      if (arch.getCore().hasTerm(DwcTerm.originalNameUsageID)) {
077        relatedTerms.add(DwcTerm.originalNameUsageID);
078      }
079    } else if (DwcTerm.Occurrence == rowType) {
080      // none as far as we know
081    } else {
082      throw new IllegalStateException("Unknown core rowtype - cannot rename the archive ids");
083    }
084
085    // map old to new ids and store them in memory
086    Map<String, String> old2new = new HashMap<String, String>();
087    ClosableIterator<DarwinCoreRecord> iter = arch.iteratorDwc();
088    System.out.println("Mapping record ids ...");
089    while (iter.hasNext()) {
090      DarwinCoreRecord rec = iter.next();
091      old2new.put(rec.id(), idGen.nextId());
092    }
093    System.out.println("Mapped " + old2new.size() + " record ids. Rewrite files ...");
094
095    // rename all files
096    rewriteFile(arch.getCore(), old2new, relatedTerms);
097    for (ArchiveFile af : arch.getExtensions()) {
098      rewriteFile(arch.getCore(), old2new, relatedTerms);
099    }
100  }
101
102  private void rewriteFile(ArchiveFile original, Map<String, String> old2new, List<Term> relatedTerms)
103    throws IOException {
104    File newFile = new File(original.getLocationFile().getParent(), original.getLocationFile().getName() + "-NEW");
105    Writer writer =
106      new BufferedWriter(new OutputStreamWriter(new FileOutputStream(newFile, false), original.getEncoding()));
107
108    List<Integer> renameCols = new ArrayList<Integer>();
109    renameCols.add(original.getId().getIndex());
110    for (Term t : relatedTerms) {
111      if (original.hasTerm(t)) {
112        renameCols.add(original.getField(t).getIndex());
113      }
114    }
115    // iterate thru rows
116    CSVReader reader = original.getCSVReader();
117    final String quote = original.getFieldsEnclosedBy() == null ? "" : original.getFieldsEnclosedBy().toString();
118    final String quoteDelimQuote = quote + original.getFieldsTerminatedBy() + quote;
119
120    // headers
121    for (int hr = reader.headerRows; hr > 0; hr--) {
122      // write row with given delimter, quotation and line ending
123      String rowAsString = quote + StringUtils.join(reader.header, quoteDelimQuote) + quote;
124      writer.write(rowAsString);
125      writer.write('\n');
126    }
127    // records
128    for (String[] row : reader) {
129      for (Integer col : renameCols) {
130        row[col] = old2new.get(row[col]);
131      }
132      // write row with given delimter, quotation and line ending
133      String rowAsString = quote + StringUtils.join(row, quoteDelimQuote) + quote;
134      writer.write(rowAsString);
135      writer.write('\n');
136    }
137    writer.flush();
138    writer.close();
139  }
140
141  public static void main(String[] args) throws IOException {
142    CoreIdRenamer renamer = new CoreIdRenamer();
143    Archive arch = ArchiveFactory.openArchive(new File("/Users/markus/Desktop/ecat_checklist"));
144    arch.getCore().setRowType(DwcTerm.Taxon);
145    renamer.rename(arch);
146  }
