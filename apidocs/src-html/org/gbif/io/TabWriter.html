001package org.gbif.io;
002
003import java.io.*;
004import java.util.regex.Pattern;
005
006import org.apache.commons.lang3.StringUtils;
007
008public class TabWriter implements AutoCloseable {
009
010  private static final Pattern escapeChars = Pattern.compile("[\t\n\r]");
011  private int records = 0;
012  private Writer writer;
013
014  public static TabWriter fromFile(File file) throws FileNotFoundException {
015    return new TabWriter(new FileOutputStream(file));
016  }
017
018  public TabWriter(OutputStream stream) {
019    try {
020      writer = new BufferedWriter(new OutputStreamWriter(stream, "UTF8"));
021    } catch (UnsupportedEncodingException e) {
022      throw new IllegalStateException(e);
023    }
024  }
025
026  public TabWriter(Writer writer) {
027    this.writer = writer;
028  }
029
030  public void write(String[] row) throws IOException {
031    if (row == null || row.length == 0) {
032      return;
033    }
034    String rowString = tabRow(row);
035    if (rowString != null) {
036      writer.write(rowString);
037    }
038  }
039
040  private String tabRow(String[] columns) {
041    // escape \t \n \r chars !!!
042    boolean empty = true;
043    for (int i = 0; i < columns.length; i++) {
044      if (columns[i] != null) {
045        empty = false;
046        columns[i] = StringUtils.trimToNull(escapeChars.matcher(columns[i]).replaceAll(" "));
047      }
048    }
049    if (empty) {
050      // dont create a row at all!
051      return null;
052    }
053    return StringUtils.join(columns, '\t') + "\n";
054  }
055
056  @Override
057  public void close() throws IOException {
058    writer.close();
059  }
060}
