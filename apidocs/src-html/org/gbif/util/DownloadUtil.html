001package org.gbif.util;
002
003import java.io.File;
004import java.io.FileOutputStream;
005import java.io.IOException;
006import java.net.URL;
007import java.nio.channels.Channels;
008import java.nio.channels.ReadableByteChannel;
009
010/**
011 * A simple download utils that uses the native http capabilities of the URL class and NIO.
012 * It does not handle redirects and other advanced http options - please use the gbif-httputil HttpUtils class for
013 * those cases which makes use of the heavier but more robust HttpClient.
014 */
015public class DownloadUtil {
016
017  /**
018   * A private Utils class.
019   */
020  private DownloadUtil() {
021  }
022
023  /**
024   * Copies the content of a URL to a local file.
025   *
026   * @param url source url
027   * @param out file to write content to
028   */
029  public static void download(URL url, File out) throws IOException {
030    ReadableByteChannel rbc = Channels.newChannel(url.openStream());
031    FileOutputStream fos = new FileOutputStream(out);
032    fos.getChannel().transferFrom(rbc, 0, 1 << 24);
033    fos.close();
034    rbc.close();
035  }
036
037  /**
038   * Copies the content of a URL to a local file.
039   *
040   * @param url source url
041   * @param out file to write content to
042   */
043  public static void download(String url, File out) throws IOException {
044    URL u = new URL(url);
045    download(u, out);
046  }
047
048}
