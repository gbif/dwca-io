017package org.gbif.utils.collection;
018
019import java.lang.reflect.Array;
020import java.util.AbstractSet;
021import java.util.Collection;
022import java.util.ConcurrentModificationException;
023import java.util.Iterator;
024import java.util.NoSuchElementException;
025
026/**
027 * A HashSet implementation taken from Ontopia
028 * which is both faster and more compact than java.util.HashSet
029 * <p/>
030 * INTERNAL: Implements the Set interface more compactly than
031 * java.util.HashSet by using a closed hashtable.
032 *
033 * @see <a href="http://ontopia.wordpress.com/2009/09/23/a-faster-and-more-compact-set/">Ontopia blog</a>
034 * @see <a href="http://code.google.com/p/ontopia/source/browse/trunk/ontopia/src/java/net/ontopia/utils/">Ontopia
035 *      source</a>
036 */
037public class CompactHashSet<T> extends AbstractSet<T> {
038
039  private class CompactHashIterator implements Iterator<T> {
040
041    private int index;
042    private int lastReturned = -1;
043
044    /**
045     * The modCount value that the iterator believes that the backing
046     * CompactHashSet should have. If this expectation is violated,
047     * the iterator has detected concurrent modification.
048     */
049    private int expectedModCount;
050
051    CompactHashIterator() {
052      for (index = 0; index < objects.length && (objects[index] == null || objects[index] == deletedObject); index++) {
053      }
054      expectedModCount = modCount;
055    }
056
057    public boolean hasNext() {
058      return index < objects.length;
059    }
060
061    public T next() {
062      if (modCount != expectedModCount) {
063        throw new ConcurrentModificationException();
064      }
065      int length = objects.length;
066      if (index >= length) {
067        lastReturned = -2;
068        throw new NoSuchElementException();
069      }
070
071      lastReturned = index;
072      for (index += 1; index < length && (objects[index] == null || objects[index] == deletedObject); index++) {
073      }
074      if (objects[lastReturned] == nullObject) {
075        return null;
076      } else {
077        return (T) objects[lastReturned];
078      }
079    }
080
081    public void remove() {
082      if (modCount != expectedModCount) {
083        throw new ConcurrentModificationException();
084      }
085      if (lastReturned == -1 || lastReturned == -2) {
086        throw new IllegalStateException();
087      }
088      // delete object
089      if (objects[lastReturned] != null && objects[lastReturned] != deletedObject) {
090        objects[lastReturned] = deletedObject;
091        elements--;
092        modCount++;
093        expectedModCount = modCount; // this is expected; we made the change
094      }
095    }
096  }
097
098  protected static final int INITIAL_SIZE = 3;
099
100  protected static final double LOAD_FACTOR = 0.75;
101  /**
102   * This object is used to represent null, should clients add that to the set.
103   */
104  protected static final Object nullObject = new Object();
105  /**
106   * When an object is deleted this object is put into the hashtable
107   * in its place, so that other objects with the same key
108   * (collisions) further down the hashtable are not lost after we
109   * delete an object in the collision chain.
110   */
111  protected static final Object deletedObject = new Object();
112  protected int elements;
113  protected int freecells;
114  protected Object[] objects;
115
116  protected int modCount;
117
118  /**
119   * Constructs a new, empty set.
120   */
121  public CompactHashSet() {
122    objects = new Object[INITIAL_SIZE];
123    elements = 0;
124    freecells = objects.length;
125    modCount = 0;
126  }
127
128  /**
129   * Constructs a new set containing the elements in the specified
130   * collection.
131   *
132   * @param c the collection whose elements are to be placed into this set.
133   */
134  public CompactHashSet(Collection c) {
135    this(c.size());
136    addAll(c);
137  }
138
139  // ===== SET IMPLEMENTATION =============================================
140
141  /**
142   * Constructs a new, empty set.
143   */
144  public CompactHashSet(int size) {
145    // NOTE: If array size is 0, we get a
146    // "java.lang.ArithmeticException: / by zero" in add(Object).
147    objects = new Object[size == 0 ? 1 : size];
148    elements = 0;
149    freecells = objects.length;
150    modCount = 0;
151  }
152
153  /**
154   * Adds the specified element to this set if it is not already
155   * present.
156   *
157   * @param x element to be added to this set.
158   *
159   * @return <tt>true</tt> if the set did not already contain the specified
160   *         element.
161   */
162  @Override
163  public boolean add(T x) {
164    Object o = x;
165    if (o == null) {
166      o = nullObject;
167    }
168
169    int hash = o.hashCode();
170    int index = (hash & 0x7FFFFFFF) % objects.length;
171    int offset = 1;
172    int deletedix = -1;
173
174    // search for the object (continue while !null and !this object)
175    while (objects[index] != null && !(objects[index].hashCode() == hash && objects[index].equals(o))) {
176
177      // if there's a deleted object here we can put this object here,
178      // provided it's not in here somewhere else already
179      if (objects[index] == deletedObject) {
180        deletedix = index;
181      }
182
183      index = ((index + offset) & 0x7FFFFFFF) % objects.length;
184      offset = offset * 2 + 1;
185
186      if (offset == -1) {
187        offset = 2;
188      }
189    }
190
191    if (objects[index] == null) { // wasn't present already
192      if (deletedix != -1) {
193        index = deletedix;
194      } else {
195        freecells--;
196      }
197
198      modCount++;
199      elements++;
200      objects[index] = o;
201
202      // rehash with same capacity
203      if (1 - (freecells / (double) objects.length) > LOAD_FACTOR) {
204        rehash(objects.length);
205        // rehash with increased capacity
206        if (1 - (freecells / (double) objects.length) > LOAD_FACTOR) {
207          rehash(objects.length * 2 + 1);
208        }
209      }
210      return true;
211    } else {
212      return false;
213    }
214  }
215
216  /**
217   * Removes all of the elements from this set.
218   */
219  @Override
220  public void clear() {
221    elements = 0;
222    for (int ix = 0; ix < objects.length; ix++) {
223      objects[ix] = null;
224    }
225    freecells = objects.length;
226    modCount++;
227  }
228
229  /**
230   * Returns <tt>true</tt> if this set contains the specified element.
231   *
232   * @param o element whose presence in this set is to be tested.
233   *
234   * @return <tt>true</tt> if this set contains the specified element.
235   */
236  @Override
237  public boolean contains(Object o) {
238    if (o == null) {
239      o = nullObject;
240    }
241
242    int hash = o.hashCode();
243    int index = (hash & 0x7FFFFFFF) % objects.length;
244    int offset = 1;
245
246    // search for the object (continue while !null and !this object)
247    while (objects[index] != null && !(objects[index].hashCode() == hash && objects[index].equals(o))) {
248      index = ((index + offset) & 0x7FFFFFFF) % objects.length;
249      offset = offset * 2 + 1;
250
251      if (offset == -1) {
252        offset = 2;
253      }
254    }
255
256    return objects[index] != null;
257  }
258
259  /**
260   * INTERNAL: Used for debugging only.
261   */
262  public void dump() {
263    System.out.println("Size: " + objects.length);
264    System.out.println("Elements: " + elements);
265    System.out.println("Free cells: " + freecells);
266    System.out.println();
267    for (int ix = 0; ix < objects.length; ix++) {
268      System.out.println("[" + ix + "]: " + objects[ix]);
269    }
270  }
271
272  /**
273   * Returns <tt>true</tt> if this set contains no elements.
274   */
275  @Override
276  public boolean isEmpty() {
277    return elements == 0;
278  }
279
280  /**
281   * Returns an iterator over the elements in this set. The elements
282   * are returned in no particular order.
283   *
284   * @return an Iterator over the elements in this set.
285   *
286   * @see ConcurrentModificationException
287   */
288  @Override
289  public Iterator iterator() {
290    return new CompactHashIterator();
291  }
292
293  /**
294   * INTERNAL: Rehashes the hashset to a bigger size.
295   */
296  protected void rehash(int newCapacity) {
297    int oldCapacity = objects.length;
298    Object[] newObjects = new Object[newCapacity];
299
300    for (int ix = 0; ix < oldCapacity; ix++) {
301      Object o = objects[ix];
302      if (o == null || o == deletedObject) {
303        continue;
304      }
305
306      int hash = o.hashCode();
307      int index = (hash & 0x7FFFFFFF) % newCapacity;
308      int offset = 1;
309
310      // search for the object
311      while (newObjects[index] != null) { // no need to test for duplicates
312        index = ((index + offset) & 0x7FFFFFFF) % newCapacity;
313        offset = offset * 2 + 1;
314
315        if (offset == -1) {
316          offset = 2;
317        }
318      }
319
320      newObjects[index] = o;
321    }
322
323    objects = newObjects;
324    freecells = objects.length - elements;
325  }
326
327  /**
328   * Removes the specified element from the set.
329   */
330  @Override
331  public boolean remove(Object o) {
332    if (o == null) {
333      o = nullObject;
334    }
335
336    int hash = o.hashCode();
337    int index = (hash & 0x7FFFFFFF) % objects.length;
338    int offset = 1;
339
340    // search for the object (continue while !null and !this object)
341    while (objects[index] != null && !(objects[index].hashCode() == hash && objects[index].equals(o))) {
342      index = ((index + offset) & 0x7FFFFFFF) % objects.length;
343      offset = offset * 2 + 1;
344
345      if (offset == -1) {
346        offset = 2;
347      }
348    }
349
350    // we found the right position, now do the removal
351    if (objects[index] != null) {
352      // we found the object
353      objects[index] = deletedObject;
354      modCount++;
355      elements--;
356      return true;
357    } else {
358      // we did not find the object
359      return false;
360    }
361  }
362
363  // ===== INTERNAL METHODS ===============================================
364
365  /**
366   * Returns the number of elements in this set (its cardinality).
367   */
368  @Override
369  public int size() {
370    return elements;
371  }
372
373  @Override
374  public Object[] toArray() {
375    Object[] result = new Object[elements];
376    Object[] objects = this.objects;
377    int pos = 0;
378    for (Object object : objects) {
379      if (object != null && object != deletedObject) {
380        result[pos++] = object == nullObject ? null : object;
381      }
382    }
383    return result;
384  }
385
386  // ===== ITERATOR IMPLEMENTATON =========================================
387
388  @Override
389  public Object[] toArray(Object[] a) {
390    int size = elements;
391    if (a.length < size) {
392      a = (Object[]) Array.newInstance(a.getClass().getComponentType(), size);
393    }
394    Object[] objects = this.objects;
395    int pos = 0;
396    for (Object object : objects) {
397      if (object != null && object != deletedObject) {
398        a[pos++] = object == nullObject ? null : object;
399      }
400    }
401    return a;
402  }
