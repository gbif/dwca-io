020package org.gbif.utils.file;
021
022import java.io.File;
023import java.io.FileInputStream;
024import java.io.FileOutputStream;
025import java.io.IOException;
026import java.nio.ByteBuffer;
027import java.nio.channels.FileChannel;
028import java.util.ArrayList;
029import java.util.List;
030
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034/**
035 * A utility to split files into chucks.
036 * This is done on file size, but then extended to read to the end of the current line.
037 * Therefore a chunksize of 32meg will result in a split files of slightly more (assuming lines are
038 * not very long).
039 * This is done using NIO libraries for high performance.
040 */
041public class FileSplitter {
042
043  private static final Logger LOG = LoggerFactory.getLogger(FileSplitter.class);
044  public static final String SEPARATOR = "_";
045  public static final int READ_AHEAD_BYTES = 256;
046
047  // for the file, gives the Byte markers for reading lines, such that the lines read will approximately
048  // equate be the chunk size (slightly more as it reads to the end of the row)
049  public static List<Long> scanToChunk(File from, long chunkSizeBytes) throws IOException {
050    List<Long> chunkBytes = new ArrayList<Long>();
051    FileInputStream fis = new FileInputStream(from);
052    FileChannel fcin = fis.getChannel();
053
054    long byteCount = chunkSizeBytes;
055
056    // now we need to read and transfer to the end of the line...
057    ByteBuffer bb = ByteBuffer.allocate(READ_AHEAD_BYTES);
058
059    while (byteCount < fcin.size()) {
060      fcin.read(bb, byteCount);
061      int i = 0;
062      for (i = 0; i < bb.limit(); i++) {
063        if ((char) bb.get(i) == '\n') {
064          i++;
065          break;
066        }
067      }
068      // bb.rewind();
069      chunkBytes.add(i + byteCount);
070      byteCount += chunkSizeBytes;
071    }
072    fcin.close();
073    fis.close();
074    return chunkBytes;
075  }
076
077  /**
078   * Splits a file "pumaConcolor.txt" into the target directory using the suffix ("part") like so:
079   * - pumaConcolor_part_0.txt
080   * - pumaConcolor_part_1.txt
081   * - pumaConcolor_part_2.txt
082   * Returns the files parts
083   */
084  public static List<File> split(File from, File targetDirectory, String suffix, long chunkSizeBytes)
085    throws IOException {
086    List<File> files = new ArrayList<File>();
087    FileInputStream fis = new FileInputStream(from);
088    FileChannel fcin = fis.getChannel();
089
090    String filePartNamePrefix = "";
091    String filePartNameSuffix = "";
092    if (from.getName().contains(".")) {
093      filePartNamePrefix = from.getName().substring(0, from.getName().indexOf("."));
094      filePartNameSuffix = from.getName().substring(from.getName().indexOf("."));
095    } else {
096      filePartNamePrefix = from.getName();
097    }
098
099    long byteCount = 0;
100    int filePartCount = 0;
101
102    while (byteCount < fcin.size()) {
103      long time = System.currentTimeMillis();
104      // create the output file
105      String fileName = filePartNamePrefix + SEPARATOR + suffix + SEPARATOR + filePartCount + filePartNameSuffix;
106      File to = new File(targetDirectory, fileName);
107      files.add(to);
108
109      // copy to the new file
110      FileOutputStream fos = new FileOutputStream(to);
111      FileChannel fcout = fos.getChannel();
112      fcin.transferTo(byteCount, chunkSizeBytes, fcout);
113      byteCount += chunkSizeBytes;
114
115      // now we need to read and transfer to the end of the line...
116      ByteBuffer bb = ByteBuffer.allocate(READ_AHEAD_BYTES);
117      fcin.read(bb, byteCount);
118      int i = 0;
119      for (i = 0; i < bb.limit(); i++) {
120        if ((char) bb.get(i) == '\n') {
121          i++;
122          break;
123        }
124      }
125      bb.rewind();
126      bb.limit(i);
127      fcout.write(bb);
128      byteCount += i;
129
130      fcout.close();
131      fos.close();
132      filePartCount++;
133      LOG.debug("Filepart[" + fileName + "] created in " + (1 + System.currentTimeMillis() - time) / 1000 + " secs");
134    }
135
136    // TODO - have tested but need to test thoroughly...
137    // what if the file was smaller than 32 meg?
138    // what if the chunk size was exactly the file size?
139    // did the joins line up properly?
140    // what if the read ahead line did not get to the end of the line?
141
142    fcin.close();
143    fis.close();
144    return files;
145  }
146
147  private FileSplitter() {
148    throw new UnsupportedOperationException("Can't initialize class");
149  }
