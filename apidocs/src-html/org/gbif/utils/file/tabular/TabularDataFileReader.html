017package org.gbif.utils.file.tabular;
018
019import java.io.Closeable;
020import java.io.IOException;
021import java.text.ParseException;
022import java.util.List;
023
024/**
025 * Interface defining a reader for tabular data file (e.g. CSV)
026 */
027public interface TabularDataFileReader<T> extends Closeable {
028
029  /**
030   * Get the header line of the tabular data file (if possible).
031   *
032   * @return headers or null
033   */
034  List<String> getHeaderLine() throws IOException;
035
036  /**
037   * Read a non-empty line of the tabular data file.
038   * An empty line represents a line with no printable characters. A line with only the defined separators is
039   * expected to be returned.
040   *
041   * @return the next line of the tabular data file or null if the end of the file is reached.
042   *
043   * @throws IOException
044   * @throws ParseException
045   */
046  T read() throws IOException, ParseException;
047
048
049  /**
050   * The line number of where the last record returned by {@link #read()} starts.
051   * If no records have been returned yet this method is expected to return 0.
052   * Once {@link #read()} returned null, this methods will return the number of the last line in the file.
053   * Note that if the very last line is an empty line with only an endline character it will not be counted.
054   * Line number includes header line and empty lines (if applicable).
055   *
056   * @return line number of where the last record starts
057   */
058  long getLastRecordLineNumber();
059
060  /**
061   * Represents the number of record returned by the {@link #read()} method.
062   * If no records have been returned yet this method is expected to return 0.
063   * A record can span over multiple line.
064   *
065   * @return
066   */
067  long getLastRecordNumber();
068
069}
