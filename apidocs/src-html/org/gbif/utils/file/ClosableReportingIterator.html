017package org.gbif.utils.file;
018
019/**
020 * Interface that extends the ClosableIterator providing a means to check if there is an error, log what it is, store
021 * the actual Exception, and skip over it to continue reading.
022 *
023 * @param <T> the type of elements returned by the iterator
024 */
025public interface ClosableReportingIterator<T> extends ClosableIterator<T> {
026
027  /**
028   * Return true if an error was encountered while iterating over row.
029   */
030  boolean hasRowError();
031
032  /**
033   * Return an informative message about the error encountered while iterating over row.
034   * Different from the Exception's message, used to store the row number, row string, etc.
035   */
036  String getErrorMessage();
037
038  /**
039   * Return the Exception encountered while iterating over row.
040   */
041  Exception getException();
042}
