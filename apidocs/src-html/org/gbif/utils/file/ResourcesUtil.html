017package org.gbif.utils.file;
018
019import java.io.File;
020import java.io.FileOutputStream;
021import java.io.IOException;
022import java.io.OutputStream;
023import java.net.URISyntaxException;
024import java.net.URL;
025import java.net.URLDecoder;
026import java.util.Enumeration;
027import java.util.HashSet;
028import java.util.Set;
029import java.util.jar.JarEntry;
030import java.util.jar.JarFile;
031
032import com.google.common.io.Closer;
033import com.google.common.io.Files;
034import com.google.common.io.Resources;
035import org.apache.commons.lang3.StringUtils;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039/**
040 * Utils class dealing with classpath resources in addition to guavas {@link Resources}.
041 */
042public class ResourcesUtil {
043
044  private static final Logger LOG = LoggerFactory.getLogger(ResourcesUtil.class);
045
046  /**
047   * Static utils class.
048   */
049  private ResourcesUtil() {
050
051  }
052
053  /**
054   * Copies classpath resources to real files.
055   *
056   * @param folder                 to copy resource files into
057   * @param ignoreMissingResources if true ignores missing resources, throws IOException otherwise
058   * @param classpathPrefix        common prefix added to all classpath resources which is not used for the result file
059   *                               path.
060   * @param classpathResources     list of classpath resources to be copied into folder
061   */
062  public static void copy(File folder, String classpathPrefix, boolean ignoreMissingResources,
063    String... classpathResources) throws IOException {
064    for (String classpathResource : classpathResources) {
065      String res = classpathPrefix + classpathResource;
066      URL url = null;
067      try {
068        url = Resources.getResource(res);
069        if (url == null) {
070          throw new IllegalArgumentException("Classpath resource " + res + " not existing");
071        }
072      } catch (IllegalArgumentException e) {
073        if (ignoreMissingResources) {
074          LOG.debug("Resource {} not found", res);
075          continue;
076        }
077        throw new IOException(e);
078      }
079
080      Closer closer = Closer.create();
081      try {
082        File f = new File(folder, classpathResource);
083        Files.createParentDirs(f);
084        OutputStream out = closer.register(new FileOutputStream(f));
085        Resources.copy(url, out);
086      } catch (Throwable e) {
087        // must catch Throwable for closer to work, see https://code.google.com/p/guava-libraries/wiki/ClosingResourcesExplained
088        throw closer.rethrow(e);
089      } finally {
090        closer.close();
091      }
092    }
093  }
094
095    /**
096     * List directory contents for a resource folder. Not recursive.
097     * Works for regular files and also JARs.
098     *
099     * Based on code from Greg Briggs, slightly modified.
100     *
101     * @param clazz Any java class that lives in the same place as the resources you want.
102     * @param path Should end with "/", but not start with one.
103     * @return Just the name of each member item, not the full paths. Empty array in case folder cannot be found
104     * @throws IOException
105     */
106    public static String[] list(Class clazz, String path) throws IOException {
107        if (!path.endsWith("/")) {
108            path = path + "/";
109        }
110        URL dirURL = clazz.getClassLoader().getResource(path);
111        if (dirURL != null && dirURL.getProtocol().equals("file")) {
112        /* A file path: easy enough */
113            try {
114                return new File(dirURL.toURI()).list();
115            } catch (URISyntaxException e) {
116                throw new IOException("Bad URI. Cannot list files for path " + path + " in class " + clazz, e);
117            }
118        }
119
120        if (dirURL == null) {
121        /*
122         * In case of a jar file, we can't actually find a directory.
123         * Have to assume the same jar as clazz.
124         */
125            String me = clazz.getName().replace(".", "/")+".class";
126            dirURL = clazz.getClassLoader().getResource(me);
127        }
128
129        if (dirURL.getProtocol().equals("jar")) {
130        /* A JAR path */
131            String jarPath = dirURL.getPath().substring(5, dirURL.getPath().indexOf("!")); //strip out only the JAR file
132            JarFile jar = new JarFile(URLDecoder.decode(jarPath, FileUtils.UTF8));
133            Enumeration<JarEntry> entries = jar.entries(); //gives ALL entries in jar
134            Set<String> result = new HashSet<String>(); //avoid duplicates in case it is a subdirectory
135            while(entries.hasMoreElements()) {
136                String name = entries.nextElement().getName();
137                if (name.startsWith(path)) { //filter according to the path
138                    String entry = name.substring(path.length());
139                    if (!StringUtils.isBlank(entry)) {
140                        int checkSubdir = entry.indexOf("/");
141                        if (checkSubdir >= 0) {
142                            // if it is a subdirectory, we just return the directory name
143                            entry = entry.substring(0, checkSubdir);
144                        }
145                        result.add(entry);
146                    }
147                }
148            }
149            return result.toArray(new String[result.size()]);
150        }
151
152        return new String[]{};
153    }
