008import org.apache.poi.ss.usermodel.DateUtil;
009import org.apache.poi.ss.usermodel.FormulaEvaluator;
010import org.apache.poi.ss.usermodel.Row;
011import org.apache.poi.ss.usermodel.Sheet;
012import org.apache.poi.ss.usermodel.Workbook;
013import org.apache.poi.ss.usermodel.WorkbookFactory;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017import java.io.IOException;
018import java.nio.file.Path;
019import java.util.ArrayList;
020import java.util.List;
021import java.util.ListIterator;
022import java.util.Optional;
023import java.util.stream.Collectors;
024import java.util.stream.IntStream;

026/**
027 * This converter reads a Workbook (.xlsx, .xls) file and produces a CSV file.
028 *
029 * Inspired from Apache POI example:
030 * http://svn.apache.org/repos/asf/poi/trunk/src/examples/src/org/apache/poi/ss/examples/ToCSV.java
031 */
032public class ExcelConverter {
033
034  private static final Logger LOG = LoggerFactory.getLogger(ExcelConverter.class);
035  private static final int FLUSH_INTERVAL = 1000;
036  private static final int FIRST_SHEET_INDEX = 0;
037
038  private final DataFormatter formatter = new IsoDateDataFormatter();
039
040  private final Workbook workbook;
041  /**
042   * Destination for data
043   */
044  private final SpreadsheetConsumer writer;
045
046  ExcelConverter(Workbook workbook, SpreadsheetConsumer writer) {
047    this.workbook = workbook;
048    this.writer = writer;
049  }
050
051  /**
052   * Convert a workbook (.xls format) to some other format like CSV.
053   *
054   * Example: ExcelConverter.convert(xlsPath, new CsvSpreadsheetConsumer(new FileWriter(outputCsvFile)));
055   *
056   * @param workbookFile Path to the workbook file
057   * @param writer A writer which will be given many List<String>s
058   * @throws IOException
059   * @throws InvalidFormatException Thrown if invalid XML is found whilst parsing an input SpreadsheetML file.
060   */
061  public static long convert(Path workbookFile, SpreadsheetConsumer writer) throws IOException, InvalidFormatException {
062    Workbook workbook = WorkbookFactory.create(workbookFile.toFile());
063
064    ExcelConverter converter = new ExcelConverter(workbook, writer);
065
066    long count = converter.process();
067
068    writer.flush();
069    return count;
070  }
071
072  /**
073   * Initiates the processing of the XLS workbook file to CSV.
074   *
075   * @throws IOException If reading the data from the package fails.
076   */
077  private int process() throws IOException {
078    int numberOfLineWritten = 0;
079
080    if (workbook.getNumberOfSheets() == 0) {
081      LOG.warn("No sheet found in the workbook");
082      return numberOfLineWritten;
083    }
084
085    int sheetIndex = FIRST_SHEET_INDEX;
086    if (workbook.getNumberOfSheets() > 1) {
087      List<String> sheetNames = IntStream.range(0, workbook.getNumberOfSheets())
088              .mapToObj(idx -> workbook.getSheetName(idx))
089              .collect(Collectors.toList());
090      //run the sheetSelector function to see if we can get a sheet name from it otherwise, keep the first sheet
091      Optional<String> sheetName = Optional.empty(); //sheetSelector.apply(sheetNames);
092      if (sheetName.isPresent()) {
093        sheetIndex = sheetName.map(name -> workbook.getSheetIndex(name)).get();
094      } else{
095        LOG.warn("Detected more than 1 sheet and can not get a selection from sheetSelector(), only reading the first one.");
096      }
097    }
098
099    Sheet sheet = workbook.getSheetAt(sheetIndex); //get the first Sheet, and only get this
100    if (sheet.getPhysicalNumberOfRows() > 0) {
101      //pass the evaluator to other methods since it's unknown how expensive is to get one!
102      numberOfLineWritten = writeSheetContent(sheet, workbook.getCreationHelper().createFormulaEvaluator());
103    }
104
105    return numberOfLineWritten;
106  }
107
108  /**
109   * Writes the content of a Sheet into the csvWriter.
110   */
111  private int writeSheetContent(Sheet sheet, FormulaEvaluator evaluator) throws IOException {
112    // extract headers (first line)
113    List<String> headers = getHeaders(sheet, evaluator);
114    writer.write(headers);
115
116    int rowSize = headers.size() - 1; //this is done avoid the calculation on each call
117    for (int j = 1; j <= sheet.getLastRowNum(); j++) {
118      writer.write(rowToCSV(sheet.getRow(j), evaluator, rowSize));
119      if (j % FLUSH_INTERVAL == 0) {
120        writer.flush();
121      }
122    }
123    // +1 to add the header line
124    return sheet.getLastRowNum() + 1;
125  }
126
127  /**
128   * Extract the first row from as the header.
129   */
130  private List<String> getHeaders(Sheet sheet, FormulaEvaluator evaluator) {
131    Row headerRow = sheet.getRow(0);
132    //we want to loop until maxColumnIdx (if provided) even if it's greater than getLastCellNum()
133    //we shall have the same number of entries on every line in the CSV
134    List<String> headers = rowToCSV(headerRow, evaluator, headerRow.getLastCellNum());
135
136    //reverse iteration (from the right side of the Workbook) to remove empty columns
137    ListIterator<String> iterator = headers.listIterator(headers.size());
138    while (iterator.hasPrevious()) {
139      if (StringUtils.isBlank(iterator.previous())) {
140        iterator.remove();
141      }
142    }
143    return headers;
144  }
145
146  /**
147   * Called to convert a row of cells into a line of data that can later be
148   * output to the CSV file.
149   *
150   * @param row can be HSSFRow or XSSFRow classes or null
151   * @param evaluator workbook formula evaluator
152   * @param maxCellNum maximum number of cell to evaluate
153   * @return
154   */
155  private List<String> rowToCSV(Row row, FormulaEvaluator evaluator, int maxCellNum) {
156    List<String> csvLine = new ArrayList<>();
157    if (row != null) {
158      for (int i = 0; i <= maxCellNum; i++) {
159        Cell cell = row.getCell(i);
160        //add an empty string when we have no data
161        if (cell == null) {
162          csvLine.add("");
163        } else {
164          //getCellTypeEnum deprecation explanation: see https://bz.apache.org/bugzilla/show_bug.cgi?id=60228
165          if (CellType.FORMULA == cell.getCellTypeEnum()) {
166            csvLine.add(formatter.formatCellValue(cell, evaluator));
167          } else if (cell.getCellTypeEnum() == CellType.NUMERIC && DateUtil.isCellDateFormatted(cell)) {
168            csvLine.add(formatter.formatRawCellContents(cell.getNumericCellValue(), cell.getCellStyle().getIndex(), cell.getCellStyle().getDataFormatString()));
169          } else {
170            csvLine.add(formatter.formatCellValue(cell));
171          }
172        }
173      }
174    }
175    return csvLine;
176  }
177}
