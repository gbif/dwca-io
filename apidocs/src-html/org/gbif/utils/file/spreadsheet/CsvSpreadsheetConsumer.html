001package org.gbif.utils.file.spreadsheet;
002
003import com.fasterxml.jackson.databind.SequenceWriter;
004import com.fasterxml.jackson.dataformat.csv.CsvMapper;
005import com.fasterxml.jackson.dataformat.csv.CsvParser;
006import com.fasterxml.jackson.dataformat.csv.CsvSchema;
007
008import java.io.IOException;
009import java.io.Writer;
010import java.util.List;
011
012/**
013 * Convert rows from a spreadsheet into CSV.
014 */
015public class CsvSpreadsheetConsumer implements SpreadsheetConsumer {
016  final SequenceWriter sequenceWriter;
017  long count = 0;
018
019  public CsvSpreadsheetConsumer(Writer writer) throws IOException {
020    CsvMapper mapper = new CsvMapper();
021    mapper.enable(CsvParser.Feature.WRAP_AS_ARRAY);
022
023    CsvSchema schema = CsvSchema.emptySchema().withColumnSeparator(',').withLineSeparator("\n").withQuoteChar('"');
024
025    sequenceWriter = mapper.writerFor(List.class).with(schema).writeValues(writer);
026  }
027
028  @Override
029  public void write(List<String> row) throws IOException {
030    sequenceWriter.write(row);
031    count++;
032  }
033
034  @Override
035  public void flush() throws IOException {
036    sequenceWriter.flush();
037  }
038
039  @Override
040  public long getCount() {
041    return count;
042  }
043}
