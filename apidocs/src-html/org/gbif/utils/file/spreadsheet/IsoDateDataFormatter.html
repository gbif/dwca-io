001package org.gbif.utils.file.spreadsheet;
002
003import org.apache.poi.ss.usermodel.DataFormatter;
004import org.apache.poi.ss.usermodel.DateUtil;
005
006import java.text.DateFormat;
007import java.text.Format;
008import java.text.SimpleDateFormat;
009import java.util.Date;
010import java.util.Locale;
011
012/**
013 * Override Excel formats to use ISO dates, but leave everything else as it was.
014 */
015public class IsoDateDataFormatter extends DataFormatter {
016
017  private DateFormat dateDateformat = new SimpleDateFormat("yyyy-MM-DD");
018  private DateFormat dateTimeDateformat = new SimpleDateFormat("yyyy-MM-DD'T'HH:mm:ss'Z'");
019
020  public IsoDateDataFormatter() {
021    super(Locale.UK);
022  }
023
024  private Format getDateFormat(String pFormatStr) {
025    // Within [] is an elapsed duration (hours?)
026
027    // Just decide whether a time is present
028    if (pFormatStr.contains("h")) {
029      return dateTimeDateformat;
030    } else {
031      return dateDateformat;
032    }
033  }
034
035  /**
036   * Formats a date to ISO 8601, otherwise hands over to the superclass.
037   */
038  public String formatRawCellContents(double value, int formatIndex, String formatString) {
039    // Is it a date?
040    if (DateUtil.isADateFormat(formatIndex, formatString) && DateUtil.isValidExcelDate(value)) {
041      Format dateFormat = getDateFormat(formatString);
042      Date d = DateUtil.getJavaDate(value, false);
043      return dateFormat.format(d);
044    }
045
046    return super.formatRawCellContents(value, formatIndex, formatString, false);
047  }
048}
