017package org.gbif.utils.concurrent;
018
019import java.util.concurrent.ExecutorService;
020import java.util.concurrent.TimeUnit;
021
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 *
027 */
028public class ExecutorUtils {
029  private static final Logger LOG = LoggerFactory.getLogger(ExecutorUtils.class);
030
031  /**
032   * Shuts down an executor and waits up to 1 hour for the already submitted jobs to finish.
033   * @param exec
034   */
035  public static void stop(ExecutorService exec) {
036    stop(exec, 1, TimeUnit.HOURS);
037  }
038
039  /**
040   * Shuts down an executor and waits for the job to finish until the given timeout is reached
041   * before forcing an immediate shutdown.
042   * @param exec
043   * @param timeout
044   * @param unit
045   */
046  public static void stop(ExecutorService exec, int timeout, TimeUnit unit) {
047    LOG.debug("Shutting down executor service {}", exec);
048    exec.shutdown(); // Disable new tasks from being submitted
049    try {
050      if (!exec.awaitTermination(timeout, unit)) {
051        LOG.warn("Forcing shut down of executor service, pending tasks will be lost! {}", exec);
052        exec.shutdownNow();
053      }
054    } catch (InterruptedException ie) {
055      LOG.warn("Forcing shut down of executor service, pending tasks will be lost! {}", exec);
056      exec.shutdownNow();
057      // Preserve interrupt status
058      Thread.currentThread().interrupt();
059    }
060  }
061}
