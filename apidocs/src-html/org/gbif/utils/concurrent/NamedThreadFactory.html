001package org.gbif.utils.concurrent;
002
003import java.util.concurrent.Executors;
004import java.util.concurrent.ThreadFactory;
005import java.util.concurrent.atomic.AtomicInteger;
006
007/**
008 * Modified Executors DefaultThreadFactory to allow custom named thread pools.
009 * Otherwise, this factory yields the same semantics as the thread factory returned by
010 * {@link Executors#defaultThreadFactory()}.
011 */
012public class NamedThreadFactory implements ThreadFactory {
013  private final ThreadGroup group;
014  private final AtomicInteger threadNumber = new AtomicInteger(1);
015  private final String namePrefix;
016
017  /**
018   * @param poolName the name prefix of the thread pool which will be appended -number for the individual thread
019   */
020  public NamedThreadFactory(String poolName) {
021    SecurityManager s = System.getSecurityManager();
022    group = (s != null) ? s.getThreadGroup() :
023        Thread.currentThread().getThreadGroup();
024    namePrefix = poolName + "-";
025  }
026
027  public Thread newThread(Runnable r) {
028    Thread t = new Thread(group, r,
029        namePrefix + threadNumber.getAndIncrement(),
030        0);
031    if (t.isDaemon())
032      t.setDaemon(false);
033    if (t.getPriority() != Thread.NORM_PRIORITY)
034      t.setPriority(Thread.NORM_PRIORITY);
035    return t;
036  }
037}
