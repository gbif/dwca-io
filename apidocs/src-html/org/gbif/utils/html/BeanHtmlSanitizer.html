017package org.gbif.utils.html;
018
019import java.lang.reflect.InvocationTargetException;
020import java.util.Map;
021import java.util.Set;
022
023import com.google.common.base.Preconditions;
024import org.apache.commons.beanutils.PropertyUtils;
025import org.owasp.html.PolicyFactory;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029/**
030 * Simple wrapper around OWASP Html Sanitizer to apply it to all String properties of a bean.
031 */
032public class BeanHtmlSanitizer {
033
034  private static final Logger LOG = LoggerFactory.getLogger(BeanHtmlSanitizer.class);
035
036  /**
037   * Apply the {@link PolicyFactory} to all String properties of the provided bean.
038   * This function will only apply sanitization to properties that are String and will only
039   * be applied to the variable of the provided bean. In other words, it won't be applied to inner-beans.
040   *
041   * @param obj
042   * @param policy
043   * @param exclusion set of property that should not be sanitized
044   * @param <T>
045   * @return same instance of the provided bean with all String properties sanitized by the {@link PolicyFactory}.
046   */
047  public static <T> T sanitize(T obj, PolicyFactory policy, Set<String> exclusion) {
048    Preconditions.checkNotNull(obj, "The provided object can not be null");
049    Preconditions.checkNotNull(policy, "The provided policy can not be null");
050
051    try {
052      Map<String, Object> properties = PropertyUtils.describe(obj);
053      for (String property : properties.keySet()) {
054        if ((exclusion == null || !exclusion.contains(property)) &&
055                String.class.equals(PropertyUtils.getPropertyType(obj, property))) {
056          PropertyUtils.setProperty(obj, property, policy.sanitize((String) properties.get(property)));
057        }
058      }
059      //TODO change for Java 7 multicatch when possible
060    } catch (IllegalAccessException e) {
061      LOG.error("Issue while applying HTML sanitization", e);
062    } catch (InvocationTargetException e) {
063      LOG.error("Issue while applying HTML sanitization", e);
064    } catch (NoSuchMethodException e) {
065      LOG.error("Issue while applying HTML sanitization", e);
066    }
067    return obj;
068  }
069
070  /**
071   * Apply the {@link PolicyFactory} to all String properties of the provided bean.
072   * This function will only apply sanitization to properties that are String and will only
073   * be applied to the variable of the provided bean. In other words, it won't be applied to inner-beans.
074   *
075   * @return same instance of the provided bean with all String properties sanitized by the {@link PolicyFactory}.
076   */
077  public static <T> T sanitize(T obj, PolicyFactory policy) {
078    return sanitize(obj, policy, null);
079  }
080
081}
