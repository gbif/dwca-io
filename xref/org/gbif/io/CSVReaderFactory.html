17  package org.gbif.io;
18  
19  import org.gbif.dwca.io.ArchiveFile;
20  import org.gbif.dwca.io.UnkownDelimitersException;
21  import org.gbif.utils.file.csv.CSVReader;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.io.File;
26  import java.io.IOException;
27  
28  /**
29   * Available for backward compatibility only.
30   *
31   * Use org.gbif.utils.file.csv.CSVReaderFactory instead
32   */
33  public class CSVReaderFactory {
34  
35    private static final Logger LOG = LoggerFactory.getLogger(CSVReaderFactory.class);
36  
37    public static CSVReader build(ArchiveFile source) throws IOException {
38      return new CSVReader(source.getLocationFile(), source.getEncoding(), source.getFieldsTerminatedBy(),
39              source.getFieldsEnclosedBy(), source.getIgnoreHeaderLines());
40    }
41  
42    /**
43     * Available for backward compatibility
44     * @param source
45     * @param encoding
46     * @return
47     * @throws UnkownDelimitersException
48     */
49    private static ArchiveFile buildArchiveFile(File source, String encoding) throws UnkownDelimitersException {
50      ArchiveFile dwcFile = new ArchiveFile();
51      // add file
52      dwcFile.addLocation(source.getAbsolutePath());
53  
54      // detect character encoding
55      dwcFile.setEncoding(encoding);
56  
57      // set header row to 1
58      dwcFile.setIgnoreHeaderLines(1);
59  
60      org.gbif.utils.file.csv.CSVReaderFactory.CSVMetadata csvMetadata = org.gbif.utils.file.csv.CSVReaderFactory.extractCsvMetadata(source, encoding);
61  
62      dwcFile.setFieldsTerminatedBy(csvMetadata.getDelimiter());
63      dwcFile.setFieldsEnclosedBy(csvMetadata.getQuotedBy());
64  
65  
66      String msg = "Detected field delimiter >>>" + dwcFile.getFieldsTerminatedBy() + "<<<";
67      if (dwcFile.getFieldsEnclosedBy() != null) {
68        msg += " and quoted by >>>" + dwcFile.getFieldsEnclosedBy() + "<<<";
69      }
70      LOG.debug(msg);
71  
72      return dwcFile;
73    }
74  
75  }
