36    private static final Logger LOG = LoggerFactory.getLogger(CSVReaderFactory.class);
37    private static final int ROWS_TO_INSPECT = 10;
38  
39    public static CSVReader build(ArchiveFile source) throws IOException {
40      return new CSVReader(source.getLocationFile(), source.getEncoding(), source.getFieldsTerminatedBy(),
41        source.getFieldsEnclosedBy(), source.getIgnoreHeaderLines());
42    }
43  
44    public static CSVReader build(File source) throws IOException {
45      String encoding = detectEncoding(source);
46      ArchiveFile af = buildArchiveFile(source, encoding);
47      return build(af);
48    }
49  
50    public static CSVReader build(File source, String delimiter, boolean header) throws IOException {
51      return new CSVReader(source, detectEncoding(source), delimiter, null, header ? 1 : 0);
52    }
53  
54    public static CSVReader build(File source, String encoding, String delimiter, Character quotes, Integer headerRows)
55      throws IOException {
56      return new CSVReader(source, encoding, delimiter, quotes, headerRows);
57    }
58  
59    public static CSVReader build(File source, String encoding, String delimiter, Integer headerRows) throws IOException {
60      return new CSVReader(source, encoding, delimiter, '"', headerRows);
61    }
62  
63    public static CSVReader build(InputStream stream, String encoding, String delimiter, Character quotes,
64      Integer headerRows) throws IOException {
65      return new CSVReader(stream, encoding, delimiter, quotes, headerRows);
66    }

68    private static ArchiveFile buildArchiveFile(File source, String encoding) throws UnkownDelimitersException {
69      ArchiveFile dwcFile = new ArchiveFile();
70      // add file
71      dwcFile.addLocation(source.getAbsolutePath());
72  
73      // detect character encoding
74      dwcFile.setEncoding(encoding);
75  
76      // set header row to 1
77      dwcFile.setIgnoreHeaderLines(1);
78  
79      // try csv, tab and then other popular delimiters
80      // keep number of resulting columns for comparisons
81      int maxColumns = 0;
82  
83      // CSV
84      String[] potentialDelimiters = {",", "\t", ";", "|"};
85      for (String delim : potentialDelimiters) {
86        // test with various quotes including a dynamic one if the first char in each field is consistently the same
87        List<Character> potentialQuotes = new ArrayList<Character>();
88  
89        CSVReader reader;
90        try {
91          reader = CSVReader.build(source, encoding, delim, null, 1);
92          Character firstChar = likelyQuoteChar(reader);
93          reader.close();
94          if (firstChar != null) {
95            potentialQuotes.add(firstChar);
96          }
97        } catch (IOException ignored) {
98        }
99        // prefer quotes for CSVs
100       if (delim.equals(",")) {
101         potentialQuotes.add('"');
102         potentialQuotes.add('\'');
103         potentialQuotes.add(null);
104       } else {
105         potentialQuotes.add(null);
106         potentialQuotes.add('"');
107         potentialQuotes.add('\'');
108       }
109 
110       for (Character quote : potentialQuotes) {
111         try {
112           reader = CSVReader.build(source, encoding, delim, quote, 0);
113           int x = consistentRowSize(reader);
114           //          System.out.println("Delim >>>"+delim+"<<<  Quote >>>"+quote+"<<<   rowSize="+x);
115           if (x > maxColumns) {
116             dwcFile.setFieldsTerminatedBy(delim);
117             dwcFile.setFieldsEnclosedBy(quote);
118             maxColumns = x;
119           }
120           reader.close();
121         } catch (IOException ignored) {
122           // swallow, maybe different delimiters work
123           // if all fail we will throw an exception at the end
124         }
125       }
126     }
127 
128     if (maxColumns < 1) {
129       throw new UnkownDelimitersException("Unable to detect field delimiter");
130     }
131 
132     String msg = "Detected field delimiter >>>" + dwcFile.getFieldsTerminatedBy() + "<<<";
133     if (dwcFile.getFieldsEnclosedBy() != null) {
134       msg += " and quoted by >>>" + dwcFile.getFieldsEnclosedBy() + "<<<";
135     }
136     LOG.debug(msg);
137 
138     return dwcFile;
139   }

141   /**
142    * @return the number of consistent columns, -1 if non consistent or column numbers-2 in case the column numbers only
143    *         differ by 1 at max.
144    */
145   private static int consistentRowSize(CSVReader reader) {
146     int rowNum = 0;
147     int columns = 0;
148     boolean plusMinusOne = false;
149     while (reader.hasNext() && rowNum < ROWS_TO_INSPECT) {
150       String[] row = reader.next();
151       if (rowNum == 0) {
152         columns = row.length;
153       }
154       if (Math.abs(columns - row.length) > 1) {
155         return -1;
156       }
157       if (columns != row.length) {
158         plusMinusOne = true;
159       }
160       rowNum++;
161     }
162     if (plusMinusOne) {
163       return columns - 2;
164     }
165     return columns;
166   }
167 
168   private static String detectEncoding(File source) throws UnkownCharsetException {
169     Charset encoding;
170     try {
171       encoding = CharsetDetection.detectEncoding(source, 16384);
172       if (encoding == null) {
173         throw new UnkownCharsetException("Unable to detect the files character encoding");
174       }
175     } catch (IOException e) {
176       throw new UnkownCharsetException(e);
177     }
178     return encoding.displayName();
179   }

181   /**
182    * Checks if all non empty/null fields start with the same character.
183    *
184    * @return the first character if consistent, otherwise null
185    */
186   private static Character likelyQuoteChar(CSVReader reader) {
187     Character quote = null;
188     int line = 0;
189     while (reader.hasNext() && line < 10) {
190       line++;
191       String[] row = reader.next();
192       if (row != null) {
193         for (String col : row) {
194           if (col != null && col.length() > 0) {
195             // same char at start & end?
196             if (col.length() > 1 && col.charAt(0) == col.charAt(col.length() - 1)) {
197               // only consider non alphanumerics
198               char potQuote = col.charAt(0);
199               if (Character.isLetterOrDigit(potQuote)) {
200                 break;
201               }
202               if (quote == null) {
203                 quote = potQuote;
204               } else {
205                 if (!quote.equals(potQuote)) {
206                   quote = null;
207                   break;
208                 }
209               }
210             }
211           }
212         }
213       }
214     }
215     return quote;
216   }
