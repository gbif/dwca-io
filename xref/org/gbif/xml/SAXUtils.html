3   import javax.xml.parsers.SAXParserFactory;
4   
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   /**
9    * Utility classes related for SAX based parser to parse XML documents
10   */
11  public class SAXUtils {
12  
13    private static final Logger LOG = LoggerFactory.getLogger(SAXUtils.class);
14  
15    public static SAXParserFactory getNsAwareSaxParserFactory() {
16      SAXParserFactory saxf = null;
17      try {
18        saxf = SAXParserFactory.newInstance();
19        saxf.setValidating(false);
20        saxf.setNamespaceAware(true);
21      } catch (Exception e) {
22        LOG.error("Can't create namespace aware SAX Parser Factory: " + e.getMessage(), e);
23      }
24      return saxf;
25    }
26  }
