1   package org.gbif.digester;
2   
3   import org.apache.commons.digester3.CallParamRule;
4   import org.xml.sax.Attributes;
5   
6   /**
7    * Will not honor namespaces in attributes in any way
8    *
9    * This class allows for getting access to attributes in elements where the
10   * namespaces differ, but will ignore the namespace of the attribute
11   * Additionally, it only allows for accessing a single attribute (since NS are
12   * ignored) It would not be difficult to honor namespaces, but is not currently
13   * needed.
14   *
15   */
16  public class CallParamNoNSRule extends CallParamRule {
17    /**
18     * @param paramIndex The index to use (e.g. in the param list for a call
19     *          method rule)
20     * @param attributeName The local name of the attribute only
21     */
22    public CallParamNoNSRule(int paramIndex, String attributeName) {
23      super(paramIndex, attributeName);
24    }
25  
26    @Override
27    public void begin(String namespace, String name, Attributes attributes) throws Exception {
28      Object param = null;
29      for (int i = 0; i < attributes.getLength(); i++) {
30  
31        // if it has no prefix, or has SOME prefix and ends in the attribute name
32        // (___:attributeName)
33        if (attributes.getQName(i).equals(attributeName)
34                || attributes.getQName(i).endsWith(":" + attributeName)) {
35          param = attributes.getValue(i);
36          break;
37        }
38      }
39      // add to the params stack
40      if (param != null) {
41        Object parameters[] = getDigester().peekParams();
42        parameters[paramIndex] = param;
43      }
44    }
45  }
