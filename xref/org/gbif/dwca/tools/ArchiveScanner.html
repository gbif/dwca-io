1   package org.gbif.dwca.tools;
2   
3   import org.gbif.dwc.terms.DwcTerm;
4   import org.gbif.dwca.io.Archive;
5   import org.gbif.dwca.io.ArchiveFactory;
6   import org.gbif.dwca.io.UnsupportedArchiveException;
7   import org.gbif.dwca.record.Record;
8   import org.gbif.dwca.record.StarRecord;
9   
10  import java.io.File;
11  import java.io.IOException;
12  import java.util.ArrayList;
13  import java.util.List;
14  
15  public class ArchiveScanner {
16  
17    public static void main(String[] args) throws IOException, UnsupportedArchiveException {
18      if (args.length < 1) {
19        System.out.println(
20          "Please specify a Darwin Core Archive folder or single Darwin Core CSV file as the first argument.\nAn optional second integer argument lets you specify the number of records to view, a third the offset to skip.");
21        System.exit(0);
22      }
23      File archiveFile = new File(args[0]);
24      if (!archiveFile.exists()) {
25        System.out.println("Cannot find archive file: " + archiveFile);
26        System.exit(0);
27      }
28      int maxNumbers = 0;
29      if (args.length > 1) {
30        try {
31          maxNumbers = Integer.parseInt(args[1]);
32        } catch (NumberFormatException ignored) {
33          System.out.println("Second argument should an integer defining maxNumbers");
34        }
35      }
36      int offset = 0;
37      if (args.length > 2) {
38        try {
39          offset = Integer.parseInt(args[2]);
40        } catch (NumberFormatException ignored) {
41          System.out.println("Third offset argument is no integer");
42        }
43      }
44  
45      System.out.println("Opening archive: " + archiveFile.getAbsolutePath());
46      Archive arch = ArchiveFactory.openArchive(archiveFile);
47  
48      if (arch.getCore() == null) {
49        System.out.println("Cannot locate the core data file");
50        System.exit(0);
51      }
52      System.out.println("Core file(s) found: " + arch.getCore().getLocations());
53      System.out.println("Core row type: " + arch.getCore().getRowType());
54      System.out.println("Core identifier column: " + arch.getCore().getId().getIndex());
55  
56      List<DwcTerm> terms = new ArrayList<DwcTerm>();
57      terms.add(DwcTerm.scientificName);
58      terms.add(DwcTerm.taxonRank);
59      terms.add(DwcTerm.parentNameUsageID);
60      terms.add(DwcTerm.acceptedNameUsageID);
61      terms.add(DwcTerm.taxonomicStatus);
62      terms.add(DwcTerm.nomenclaturalStatus);
63      terms.add(DwcTerm.kingdom);
64      terms.add(DwcTerm.family);
65      for (DwcTerm t : terms) {
66        if (!arch.getCore().hasTerm(t)) {
67          System.out.println("Cannot locate term " + t);
68        }
69      }
70  
71      System.out.println("Number of extensions " + arch.getExtensions().size());
72  
73      // count records
74      int i = 0;
75      for (StarRecord rec : arch) {
76        i++;
77      }
78      System.out.println("Archive contains " + i + " core records.");
79  
80      // show some records
81      i = 0;
82      if (maxNumbers > 0) {
83        for (StarRecord rec : arch) {
84          i++;
85          if (i > offset + maxNumbers) {
86            break;
87          }
88          if (offset < i) {
89            System.out.println("record " + i + ":  " + rec);
90            for (Record erec : rec) {
91              System.out.println("  : " + erec);
92            }
93          }
94        }
95      }
96    }
97  
98  }
