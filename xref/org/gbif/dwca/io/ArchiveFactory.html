59          private static Character getFirstChar(String x) throws UnsupportedArchiveException {
60              if (x == null || x.length() == 0) {
61                  return null;
62              }
63              if (x.length() == 1) {
64                  return x.charAt(0);
65              }
66              if (x.equalsIgnoreCase("\\t")) {
67                  return '\t';
68              }
69              if (x.equalsIgnoreCase("\\n")) {
70                  return '\n';
71              }
72              if (x.equalsIgnoreCase("\\r")) {
73                  return '\r';
74              }
75              if (x.length() > 1) {
76                  throw new UnsupportedArchiveException(
77                      "Only darwin core archives with a single quotation character are supported, but found >>>" + x + "<<<");
78              }
79              return ' ';
80          }

82          private static void makeLocationPathsAbsolute(ArchiveFile af, File root) {
83              // I know this is verbose and stupid, but its easy coded now without the hassle of deep copying lists, etc...
84              List<String> newLocs = new ArrayList<String>();
85              for (String loc : af.getLocations()) {
86                  newLocs.add(new File(root, af.getLocation()).getAbsolutePath());
87              }
88              af.getLocations().clear();
89              for (String loc : newLocs) {
90                  af.getLocations().add(loc);
91              }
92          }

94          private static String unescapeBackslash(String x) {
95              if (x == null || x.length() == 0) {
96                  return null;
97              }
98              return x.replaceAll("\\\\t", String.valueOf('\t')).replaceAll("\\\\n", String.valueOf('\n'))
99                  .replaceAll("\\\\r", String.valueOf('\r')).replaceAll("\\\\f", String.valueOf('\f'));
100         }

102         private ArchiveFile buildArchiveFile(Attributes attr) throws UnsupportedArchiveException {
103             ArchiveFile dwcFile = new ArchiveFile();
104 
105             // extract the File attributes
106             if (getAttr(attr, "encoding") != null) {
107                 dwcFile.setEncoding(getAttr(attr, "encoding"));
108             }
109             if (getAttr(attr, "fieldsTerminatedBy") != null) {
110                 dwcFile.setFieldsTerminatedBy(unescapeBackslash(getAttr(attr, "fieldsTerminatedBy")));
111             }
112             if (getAttr(attr, "fieldsEnclosedBy") != null) {
113                 dwcFile.setFieldsEnclosedBy(getFirstChar(getAttr(attr, "fieldsEnclosedBy")));
114             }
115             if (getAttr(attr, "linesTerminatedBy") != null) {
116                 dwcFile.setLinesTerminatedBy(unescapeBackslash(getAttr(attr, "linesTerminatedBy")));
117             }
118             if (getAttr(attr, "rowType") != null) {
119                 dwcFile.setRowType(TERM_FACTORY.findTerm(getAttr(attr, "rowType")));
120             }
121             String ignoreHeaderLines = getAttr(attr, "ignoreHeaderLines");
122             try {
123                 dwcFile.setIgnoreHeaderLines(Integer.parseInt(ignoreHeaderLines));
124             } catch (NumberFormatException ignored) { // swallow null or bad value
125             }
126             return dwcFile;
127         }

129         /**
130          * Build an ArchiveField object based on xml attributes.
131          */
132         private ArchiveField buildField(Attributes attributes) {
133             // build field
134             Term term = TERM_FACTORY.findTerm(getAttr(attributes, "term"));
135             String defaultValue = getAttr(attributes, "default");
136             DataType type = DataType.findByXmlSchemaType(getAttr(attributes, "type"));
137             if (type == null) {
138                 type = DataType.string;
139             }
140             String indexAsString = getAttr(attributes, "index");
141             Integer index = null;
142             if (indexAsString != null) {
143                 // let bad errors be thrown up
144                 try {
145                     index = Integer.parseInt(indexAsString);
146                 } catch (NumberFormatException e) {
147                     throw new UnsupportedArchiveException(e);
148                 }
149             }
150             String delimiter = getAttr(attributes, "delimitedBy");
151             return new ArchiveField(index, term, defaultValue, type, delimiter);
152         }

181         private String getAttr(Attributes attributes, String key) {
182             String val = null;
183             if (attributes != null) {
184                 // try without NS
185                 val = attributes.getValue("", key);
186                 if (val == null) {
187                     // try with dwca NS if nothing found
188                     val = attributes.getValue(NS_DWCA, key);
189                 }
190             }
191             return Strings.isNullOrEmpty(val) ? null : val;
192         }

194         @Override
195         public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
196             super.startElement(uri, localName, qName, attributes);
197             if (localName.equalsIgnoreCase("archive") || localName.equalsIgnoreCase("stararchive")) {
198                 // metadata location
199                 archive.setMetadataLocation(getAttr(attributes, "metadata"));
200             } else if (localName.equalsIgnoreCase("core") || localName.equalsIgnoreCase("extension")) {
201                 // archive/extension
202                 af = new ArchiveFile();
203                 if (localName.equalsIgnoreCase("core") || localName.equalsIgnoreCase("extension")) {
204                     // archive/core or archive/extension
205                     af = buildArchiveFile(attributes);
206                 }
207             } else if (localName.equalsIgnoreCase("coreid") || localName.equalsIgnoreCase("id")) {
208                 ArchiveField field = buildField(attributes);
209                 if (af != null) {
210                     af.setId(field);
211                 } else {
212                     log.warn(localName + " field found outside of an archive file");
213                 }
214             } else if (localName.equalsIgnoreCase("field")) {
215                 ArchiveField field = buildField(attributes);
216                 if (af != null) {
217                     af.addField(field);
218                 } else {
219                     log.warn("field found outside of an archive file");
220                 }
221             }
222         }

236      * @param archiveUrl the location of a compressed archive or single data file
237      * @param workingDir writable directory to download to and decompress archive

239     public static Archive openArchive(URL archiveUrl, File workingDir) throws IOException, UnsupportedArchiveException {
240         File downloadTo = new File(workingDir, "dwca-download");

242         DownloadUtil.download(archiveUrl, downloadTo);
243         return openArchive(downloadTo, dwca);

246     /**
247      * Opens an archive from a local file and decompresses or copies it into the given archive directory.
248      * Make sure the archive directory does not contain files already!
249      *
250      * @param archiveFile the location of a compressed archive or single data file
251      * @param archiveDir  empty, writable directory used to keep decompress archive in
252      */
253     public static Archive openArchive(File archiveFile, File archiveDir) throws IOException, UnsupportedArchiveException {
254         // try to decompress archive
255         try {
256             List<File> files = CompressionUtil.decompressFile(archiveDir, archiveFile);
257             // continue to read archive from the tmp dir
258             return openArchive(archiveDir);
259 
260         } catch (CompressionUtil.UnsupportedCompressionType e) {
261             // If its a text file only we will get this exception - but also for corrupt compressions
262             // try to open as text file only
263             return openArchive(archiveFile);
264         }
265     }

267     /**
268      * @param unzippedFolderLocation the location of an expanded archive directory or just a single dwc text file
269      */
270     public static Archive openArchive(File unzippedFolderLocation) throws IOException, UnsupportedArchiveException {
271         Archive archive = new Archive();
272         if (!unzippedFolderLocation.exists()) {
273           throw new FileNotFoundException("Archive folder not existing: " + unzippedFolderLocation.getAbsolutePath());
274         }
275         archive.setLocation(unzippedFolderLocation);
276 
277         File mf = null;
278         // see if we can find a meta.xml descriptor file
279         if (unzippedFolderLocation.isFile()) {
280             String suffix = unzippedFolderLocation.getName().substring(unzippedFolderLocation.getName().lastIndexOf("."));
281             if (suffix.equalsIgnoreCase(".xml")) {
282                 // could be a metafile on its own pointing to remote data files...
283                 mf = unzippedFolderLocation;
284             }
285         } else {
286           
287           // HACK: http://dev.gbif.org/issues/browse/POR-2396
288           // Accommodate archives coming from legacy IPTs which put a "\" before each filename
289           Iterator<File> iter = FileUtils.iterateFiles(unzippedFolderLocation, new String[]{"xml", "txt"}, false);
290           while (iter.hasNext()) {
291             File f = iter.next();
292             if (f.getName().startsWith("\\")) {
293               String orig = f.getName();
294               String replacement = f.getName().replaceFirst("\\\\", "");
295               LOG.info("Renaming file from {} to {}", orig, replacement);
296               f.renameTo(new File(unzippedFolderLocation, replacement));
297             }
298           }
299           
300           mf = new File(unzippedFolderLocation, "meta.xml");
301         }
302         // read metadata
303         if (mf != null && mf.exists()) {
304             // read metafile
305             readMetaDescriptor(archive, new FileInputStream(mf), true);
306             if (archive.getMetadataLocation() == null) {
307                 // search for known metadata filenames
308                 File emlFile = new File(mf.getParentFile(), "eml.xml");
309                 if (emlFile.exists()) {
310                     archive.setMetadataLocation("eml.xml");
311                 }
312             }
313         } else {
314             // try to detect data files ourselves as best as we can...
315             // currently support a single data file or a folder which contains a single data file
316             if (unzippedFolderLocation.isFile()) {
317                 ArchiveFile coreFile = readFileHeaders(unzippedFolderLocation);
318                 archive.setCore(coreFile);
319             } else {
320                 // folder. see if we got only 1 file in there...
321                 List<File> dataFiles = new ArrayList<File>();
322                 FilenameFilter ff = new SuffixFileFilter(".csv", IOCase.INSENSITIVE);
323                 dataFiles.addAll(Arrays.asList(unzippedFolderLocation.listFiles(ff)));
324                 ff = new SuffixFileFilter(".txt", IOCase.INSENSITIVE);
325                 dataFiles.addAll(Arrays.asList(unzippedFolderLocation.listFiles(ff)));
326 
327                 if (dataFiles.size() == 1) {
328                     // set pointer to data file
329                     File dataFile = new File(unzippedFolderLocation, dataFiles.get(0).getName());
330                     archive.setLocation(unzippedFolderLocation);
331                     if (archive.getMetadataLocation() == null && unzippedFolderLocation.isDirectory()) {
332                         // search for known metadata filenames
333                         File emlFile = new File(unzippedFolderLocation, "eml.xml");
334                         if (emlFile.exists()) {
335                             archive.setMetadataLocation("eml.xml");
336                         }
337                     }
338                     ArchiveFile coreFile = readFileHeaders(dataFile);
339                     coreFile.getLocations().clear();
340                     coreFile.addLocation(dataFile.getName());
341                     archive.setCore(coreFile);
342                 } else {
343                     throw new UnsupportedArchiveException(
344                         "The archive given is a folder with more or less than 1 data files having a txt or csv suffix");
345                 }
346             }
347         }
348         // final validation
349         validateArchive(archive);
350         // report basic stats
351         LOG.debug("Archive contains " + archive.getExtensions().size() + " described extension files");
352         LOG.debug("Archive contains " + archive.getCore().getFields().size() + " core properties");
353         return archive;
354     }

366     private static ArchiveFile readFileHeaders(File dataFile) throws UnsupportedArchiveException, IOException {
367         ArchiveFile dwcFile = new ArchiveFile();
368         dwcFile.addLocation(null);
369         dwcFile.setIgnoreHeaderLines(1);
370 
371         CSVReader reader = CSVReaderFactory.build(dataFile);
372 
373         // copy found delimiters & encoding
374         dwcFile.setEncoding(reader.encoding);
375         dwcFile.setFieldsTerminatedBy(reader.delimiter);
376         dwcFile.setFieldsEnclosedBy(reader.quoteChar);
377 
378         // detect dwc terms as good as we can based on header row
379         String[] headers = reader.header;
380         int index = 0;
381         for (String head : headers) {
382             // there are never any quotes in term names - remove them just in case the csvreader didnt recognize them
383             if (head != null && head.length() > 1) {
384                 try {
385                     Term dt = TERM_FACTORY.findTerm(head);
386                     ArchiveField field = new ArchiveField(index, dt, null, DataType.string);
387                     if (dwcFile.getId() == null &&
388                         (dt.equals(DwcTerm.occurrenceID) || dt.equals(DwcTerm.taxonID) || dt.equals(DcTerm.identifier))) {
389                         dwcFile.setId(field);
390                     }
391                     dwcFile.addField(field);
392                 } catch (IllegalArgumentException e) {
393                     LOG.warn("Illegal term name >>{}<< found in header, ignore column {}", head, index);
394                 }
395             }
396             index++;
397         }
398 
399         return dwcFile;
400     }

402     private static void readMetaDescriptor(Archive archive, InputStream metaDescriptor, boolean normaliseTerms)
403         throws UnsupportedArchiveException {
404 
405         try {
406             SAXParser p = SAX_FACTORY.newSAXParser();
407             MetaHandler mh = new MetaHandler(archive);
408             LOG.debug("Reading archive metadata file");
409 //    p.parse(metaDescriptor, mh);
410             p.parse(new BomSafeInputStreamWrapper(metaDescriptor), mh);
411         } catch (Exception e1) {
412             LOG.warn("Exception caught", e1);
413             throw new UnsupportedArchiveException(e1);
414         }
415     }

417     private static void validateArchive(Archive archive) throws UnsupportedArchiveException {
418         validateCoreFile(archive.getCore(), !archive.getExtensions().isEmpty());
419         for (ArchiveFile af : archive.getExtensions()) {
420             validateExtensionFile(af);
421         }
422     }
423 
424     private static void validateCoreFile(ArchiveFile f, boolean hasExtensions) throws UnsupportedArchiveException {
425         if (hasExtensions) {
426             if (f.getId() == null) {
427                 LOG.warn(
428                     "DwC-A core data file " + f.getTitle() + " is lacking an id column. No extensions allowed in this case");
429             }
430         }
431         validateFile(f);
432     }
433 
434     private static void validateExtensionFile(ArchiveFile f) throws UnsupportedArchiveException {
435         if (f.getId() == null) {
436             throw new UnsupportedArchiveException(
437                 "DwC-A data file " + f.getTitle() + " requires an id or foreign key to the core id");
438         }
439         validateFile(f);
440     }
441 
442     private static void validateFile(ArchiveFile f) throws UnsupportedArchiveException {
443         if (f == null) {
444             throw new UnsupportedArchiveException("DwC-A data file is NULL");
445         }
446         if (f.getLocationFile() == null) {
447             throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a location");
448         }
449         if (f.getEncoding() == null) {
450             throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a character encoding");
451         }
