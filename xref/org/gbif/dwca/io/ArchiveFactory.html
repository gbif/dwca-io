13  package org.gbif.dwca.io;
14  
15  import org.gbif.dwc.DwcFiles;
16  import org.gbif.dwc.meta.DwcMetaFiles;
17  import org.gbif.util.DownloadUtil;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.net.URL;
22  
23  import com.google.common.io.Files;
24  import org.apache.commons.lang3.StringUtils;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  /**
29   * Factory used to build {@link Archive} object from a DarwinCore Archive file.
30   * 
31   * @author mdoering
32   *
33   */
34  public class ArchiveFactory {
35  
36    private static final Logger LOG = LoggerFactory.getLogger(ArchiveFactory.class);
37  
38    /**
39     * Opens an archive from a URL, downloading and decompressing it.
40     *
41     * @param archiveUrl the location of a compressed archive or single data file
42     * @param workingDir writable directory to download to and decompress archive
43     */
44    public static Archive openArchive(URL archiveUrl, File workingDir) throws IOException, UnsupportedArchiveException {
45      File downloadTo = new File(workingDir, "dwca-download");
46      File dwca = new File(workingDir, "dwca");
47      DownloadUtil.download(archiveUrl, downloadTo);
48      return openArchive(downloadTo, dwca);
49    }
50  
51    /**
52     * Opens an archive from a local file and decompresses or copies it into the given archive directory.
53     * Make sure the archive directory does not contain files already, any existing files will be removed!
54     *
55     * If the source archive is an uncompressed, single data file and a valid archive, it will be copied as is
56     * to the archiveDir.
57     *
58     * @param archiveFile the location of a compressed archive or single data file
59     * @param archiveDir  empty, writable directory used to keep decompress archive in
60     */
61    public static Archive openArchive(File archiveFile, File archiveDir) throws IOException, UnsupportedArchiveException {
62  
63      try {
64        Archive archive = DwcFiles.fromCompressed(archiveFile.toPath(), archiveDir.toPath());
65        // keep validation for backward compatibility
66        return validateArchive(archive);
67      }
68      catch (UnsupportedArchiveException uaEx){
69        LOG.debug("Could not uncompress archive [{}], try to read as single text file", archiveFile, uaEx);
70        // If its a text file only we will get this exception - but also for corrupt compressions
71        // try to open as text file only and if successful copy file to archive dir
72        Archive arch = openArchiveDataFile(archiveFile);
73        Files.copy(archiveFile, new File(archiveDir, archiveFile.getName()));
74        return arch;
75      }
76    }
77  
78    /**
79     * Opens a dwca archive which is just a single decompressed data file with headers, e.g. a csv or tab delimited file
80     */
81    public static Archive openArchiveDataFile(File dataFile) throws IOException, UnsupportedArchiveException {
82      Archive archive = DwcFiles.fromLocation(dataFile.toPath());
83  
84      // for backward compatibility reason we keep this behavior to check if we have a metadata file next to
85      // a single file.
86      if(StringUtils.isBlank(archive.getMetadataLocation())) {
87        // check if we also have a metadata file next to this data file
88        DwcMetaFiles.discoverMetadataFile(dataFile.getParentFile().toPath())
89                .ifPresent(archive::setMetadataLocation);
90      }
91  
92      // final validation
93      return validateArchive(archive);
94    }
95  
96    /**
97     * @param dwcaFolder the location of an expanded dwc archive directory or just a single dwc text file
98     */
99    public static Archive openArchive(File dwcaFolder) throws IOException, UnsupportedArchiveException {
100 
101     Archive archive = DwcFiles.fromLocation(dwcaFolder.toPath());
102 
103     // final validation
104     return validateArchive(archive);
105   }
106 
107   private static Archive validateArchive(Archive archive) throws UnsupportedArchiveException {
108     validateCoreFile(archive.getCore(), !archive.getExtensions().isEmpty());
109     for (ArchiveFile af : archive.getExtensions()) {
110       validateExtensionFile(af);
111     }
112     // report basic stats
113     LOG.debug("Archive contains " + archive.getExtensions().size() + " described extension files");
114     LOG.debug("Archive contains " + archive.getCore().getFields().size() + " core properties");
115     return archive;
116   }
117 
118   private static void validateCoreFile(ArchiveFile f, boolean hasExtensions) throws UnsupportedArchiveException {
119     if (hasExtensions) {
120       if (f.getId() == null) {
121         LOG.warn(
122           "DwC-A core data file " + f.getTitle() + " is lacking an id column. No extensions allowed in this case");
123       }
124     }
125     validateFile(f);
126   }
127 
128   private static void validateExtensionFile(ArchiveFile f) throws UnsupportedArchiveException {
129     if (f.getId() == null) {
130       throw new UnsupportedArchiveException(
131         "DwC-A data file " + f.getTitle() + " requires an id or foreign key to the core id");
132     }
133     validateFile(f);
134   }
135 
136   private static void validateFile(ArchiveFile f) throws UnsupportedArchiveException {
137     if (f == null) {
138       throw new UnsupportedArchiveException("DwC-A data file is NULL");
139     }
140     if (f.getLocationFile() == null) {
141       throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a location");
142     }
143     if (f.getEncoding() == null) {
144       throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a character encoding");
145     }
146   }
