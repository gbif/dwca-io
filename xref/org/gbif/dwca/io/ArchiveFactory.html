61      private static Character getFirstChar(String x) throws UnsupportedArchiveException {
62        if (x == null || x.length() == 0) {
63          return null;
64        }
65        if (x.length() == 1) {
66          return x.charAt(0);
67        }
68        if (x.equalsIgnoreCase("\\t")) {
69          return '\t';
70        }
71        if (x.equalsIgnoreCase("\\n")) {
72          return '\n';
73        }
74        if (x.equalsIgnoreCase("\\r")) {
75          return '\r';
76        }
77        if (x.length() > 1) {
78          throw new UnsupportedArchiveException(
79            "Only darwin core archives with a single quotation character are supported, but found >>>" + x + "<<<");
80        }
81        return ' ';
82      }

84      private static void makeLocationPathsAbsolute(ArchiveFile af, File root) {
85        // I know this is verbose and stupid, but its easy coded now without the hassle of deep copying lists, etc...
86        List<String> newLocs = new ArrayList<String>();
87        for (String loc : af.getLocations()) {
88          newLocs.add(new File(root, af.getLocation()).getAbsolutePath());
89        }
90        af.getLocations().clear();
91        for (String loc : newLocs) {
92          af.getLocations().add(loc);
93        }
94      }

96      private static String unescapeBackslash(String x) {
97        if (x == null || x.length() == 0) {
98          return null;
99        }
100       return x.replaceAll("\\\\t", String.valueOf('\t')).replaceAll("\\\\n", String.valueOf('\n'))
101         .replaceAll("\\\\r", String.valueOf('\r')).replaceAll("\\\\f", String.valueOf('\f'));
102     }

104     private ArchiveFile buildArchiveFile(Attributes attr) throws UnsupportedArchiveException {
105       ArchiveFile dwcFile = new ArchiveFile();
106 
107       // extract the File attributes
108       if (getAttr(attr, "encoding") != null) {
109         dwcFile.setEncoding(getAttr(attr, "encoding"));
110       }
111       if (getAttr(attr, "fieldsTerminatedBy") != null) {
112         dwcFile.setFieldsTerminatedBy(unescapeBackslash(getAttr(attr, "fieldsTerminatedBy")));
113       }
114       if (getAttr(attr, "fieldsEnclosedBy") != null) {
115         dwcFile.setFieldsEnclosedBy(getFirstChar(getAttr(attr, "fieldsEnclosedBy")));
116       }
117       if (getAttr(attr, "linesTerminatedBy") != null) {
118         dwcFile.setLinesTerminatedBy(unescapeBackslash(getAttr(attr, "linesTerminatedBy")));
119       }
120       if (getAttr(attr, "rowType") != null) {
121         dwcFile.setRowType(TERM_FACTORY.findTerm(getAttr(attr, "rowType")));
122       }
123       String ignoreHeaderLines = getAttr(attr, "ignoreHeaderLines");
124       try {
125         dwcFile.setIgnoreHeaderLines(Integer.parseInt(ignoreHeaderLines));
126       } catch (NumberFormatException ignored) { // swallow null or bad value
127       }
128       return dwcFile;
129     }

131     /**
132      * Build an ArchiveField object based on xml attributes.
133      */
134     private ArchiveField buildField(Attributes attributes) {
135       // build field
136       Term term = TERM_FACTORY.findTerm(getAttr(attributes, "term"));
137       String defaultValue = getAttr(attributes, "default");
138       DataType type = DataType.findByXmlSchemaType(getAttr(attributes, "type"));
139       if (type == null) {
140         type = DataType.string;
141       }
142       String indexAsString = getAttr(attributes, "index");
143       Integer index = null;
144       if (indexAsString != null) {
145         // let bad errors be thrown up
146         try {
147           index = Integer.parseInt(indexAsString);
148         } catch (NumberFormatException e) {
149           throw new UnsupportedArchiveException(e);
150         }
151       }
152       String delimiter = getAttr(attributes, "delimitedBy");
153       return new ArchiveField(index, term, defaultValue, type, delimiter);
154     }

158       // calling the super method to stringify the character buffer

165         //      makeLocationPathsAbsolute(af, archive.getLocation());
166         archive.setCore(af);
167       } else if (localName.equalsIgnoreCase("extension")) {

171         if (af.getId() != null && af.getId().getIndex() != null) {
172           archive.addExtension(af);
173         } else {
174           log.warn("Skipping extension [" + af.getRowType() + "] with no index attribute");
175         }
176       } else if (localName.equalsIgnoreCase("location")) {
177         // a file location
178         af.addLocation(content);
179       }

183     private String getAttr(Attributes attributes, String key) {
184       String val = null;
185       if (attributes != null) {
186         // try without NS
187         val = attributes.getValue("", key);
188         if (val == null) {
189           // try with dwca NS if nothing found
190           val = attributes.getValue(NS_DWCA, key);
191         }
192       }
193       return Strings.isNullOrEmpty(val) ? null : val;
194     }
195 
196     @Override
197     public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
198       super.startElement(uri, localName, qName, attributes);
199       if (localName.equalsIgnoreCase("archive") || localName.equalsIgnoreCase("stararchive")) {
200         // metadata location
201         archive.setMetadataLocation(getAttr(attributes, "metadata"));
202       } else if (localName.equalsIgnoreCase("core") || localName.equalsIgnoreCase("extension")) {
203         // archive/extension
204         af = new ArchiveFile();
205         if (localName.equalsIgnoreCase("core") || localName.equalsIgnoreCase("extension")) {
206           // archive/core or archive/extension
207           af = buildArchiveFile(attributes);
208         }
209       } else if (localName.equalsIgnoreCase("coreid") || localName.equalsIgnoreCase("id")) {
210         ArchiveField field = buildField(attributes);
211         if (af != null) {
212           af.setId(field);
213         } else {
214           log.warn(localName + " field found outside of an archive file");
215         }
216       } else if (localName.equalsIgnoreCase("field")) {
217         ArchiveField field = buildField(attributes);
218         if (af != null) {
219           af.addField(field);
220         } else {
221           log.warn("field found outside of an archive file");
222         }
223       }
224     }

248   /**
249    * Opens an archive from a local file and decompresses or copies it into the given archive directory.
250    * Make sure the archive directory does not contain files already, any existing files will be removed!
251    *
252    * If the source archive is an uncompressed, single data file and a valid archive, it will be copied as is
253    * to the archiveDir.
254    *
255    * @param archiveFile the location of a compressed archive or single data file
256    * @param archiveDir  empty, writable directory used to keep decompress archive in
257    */
258   public static Archive openArchive(File archiveFile, File archiveDir) throws IOException, UnsupportedArchiveException {
259     if (archiveDir.exists()) {
260       // clean up any existing folder
261       LOG.debug("Deleting existing archive folder [{}]", archiveDir.getAbsolutePath());
262       org.gbif.utils.file.FileUtils.deleteDirectoryRecursively(archiveDir);
263     }
264     FileUtils.forceMkdir(archiveDir);
265     // try to decompress archive
266     try {
267       CompressionUtil.decompressFile(archiveDir, archiveFile, true);
268       // we keep subfolder, but often the entire archive is within one subfolder. Remove that root folder if present
269       File[] rootFiles = archiveDir.listFiles((FileFilter) HiddenFileFilter.VISIBLE);
270       if (rootFiles.length == 1) {
271         File root = rootFiles[0];
272         if (root.isDirectory()) {
273           // single root dir, flatten structure
274           LOG.debug("Removing single root folder {} found in decompressed archive", root.getAbsoluteFile());
275           for (File f : FileUtils.listFiles(root, TrueFileFilter.TRUE, null)) {
276             File f2 = new File(archiveDir, f.getName());
277             f.renameTo(f2);
278           }
279         }
280       }
281       // continue to read archive from the tmp dir
282       return openArchive(archiveDir);
283 
284     } catch (CompressionUtil.UnsupportedCompressionType e) {
285       LOG.debug("Could not uncompress archive [{}], try to read as single text file", archiveFile, e);
286       // If its a text file only we will get this exception - but also for corrupt compressions
287       // try to open as text file only and if successful copy file to archive dir
288       Archive arch = openArchiveDataFile(archiveFile);
289       Files.copy(archiveFile, new File(archiveDir, archiveFile.getName()));
290       return arch;
291     }
292   }

294   /**
295    * Opens a dwca archive which is just a single decompressed data file with headers, e.g. a csv or tab delimited file
296    */
297   public static Archive openArchiveDataFile(File dataFile) throws IOException, UnsupportedArchiveException {
298     Archive archive = new Archive();
299     archive.setLocation(dataFile);
300 
301     ArchiveFile coreFile = readFileHeaders(dataFile);
302     archive.setCore(coreFile);
303 
304     // check if we also have a metadata file next to this data file
305     discoverMetadataFile(archive, dataFile.getParentFile());
306 
307     // final validation
308     return validateArchive(archive);
309   }

311   /**
312    * @param dwcaFolder the location of an expanded dwc archive directory or just a single dwc text file
313    */
314   public static Archive openArchive(File dwcaFolder) throws IOException, UnsupportedArchiveException {
315     if (!dwcaFolder.exists()) {
316       throw new FileNotFoundException("Archive folder not existing: " + dwcaFolder.getAbsolutePath());
317     }
318     // delegate to open data file method if its a single file, not a folder
319     if (dwcaFolder.isFile()) {
320       return openArchiveDataFile(dwcaFolder);
321     }
322 
323     Archive archive = new Archive();
324     archive.setLocation(dwcaFolder);
325 
326     // Accommodate archives coming from legacy IPTs which put a "\" before each filename
327     // http://dev.gbif.org/issues/browse/POR-2396
328     // https://code.google.com/p/gbif-providertoolkit/issues/detail?id=1015
329     Iterator<File> iter = FileUtils.iterateFiles(dwcaFolder, new String[] {"xml", "txt"}, false);
330     while (iter.hasNext()) {
331       File f = iter.next();
332       if (f.getName().startsWith("\\")) {
333         String orig = f.getName();
334         String replacement = f.getName().replaceFirst("\\\\", "");
335         LOG.info("Renaming file from {} to {}", orig, replacement);
336         f.renameTo(new File(dwcaFolder, replacement));
337       }
338     }
339 
340     // read metadata
341     File mf = new File(dwcaFolder, "meta.xml");
342     if (mf.exists()) {
343       // read metafile
344       readMetaDescriptor(archive, new FileInputStream(mf), true);
345 
346     } else {
347       // meta.xml lacking.
348       // Try to detect data files ourselves as best as we can.
349       // look for a single, visible text data file
350       List<File> dataFiles = new ArrayList<File>();
351       for (String suffix : Lists.newArrayList(".csv", ".txt", ".tab", ".text", ".data")) {
352         FileFilter ff =
353           FileFilterUtils.and(FileFilterUtils.suffixFileFilter(suffix, IOCase.INSENSITIVE), HiddenFileFilter.VISIBLE);
354         dataFiles.addAll(Arrays.asList(dwcaFolder.listFiles(ff)));
355       }
356 
357       if (dataFiles.size() == 1) {
358         File dataFile = new File(dwcaFolder, dataFiles.get(0).getName());
359         ArchiveFile coreFile = readFileHeaders(dataFile);
360         coreFile.getLocations().clear();
361         coreFile.addLocation(dataFile.getName());
362         archive.setCore(coreFile);
363 
364       } else {
365         throw new UnsupportedArchiveException(
366           "The archive given is a folder with more or less than 1 data files having a csv, txt or tab suffix");
367       }
368     }
369 
370     // check if we also have a metadata file next to this data file
371     discoverMetadataFile(archive, mf.getParentFile());
372 
373     // final validation
374     return validateArchive(archive);
375   }

378   private static void discoverMetadataFile(Archive archive, File folder) {
379     if (archive.getMetadataLocation() == null) {
380       // search for popular metadata filenames
381       for (String metadataFN : Lists.newArrayList("eml.xml", "metadata.xml")) {
382         File emlFile = new File(folder, metadataFN);
383         if (emlFile.exists()) {
384           archive.setMetadataLocation(metadataFN);
385           break;
386         }
387       }
388     }
389   }

391   private static ArchiveFile readFileHeaders(File dataFile) throws UnsupportedArchiveException, IOException {
392     ArchiveFile dwcFile = new ArchiveFile();
393     dwcFile.addLocation(null);
394     dwcFile.setIgnoreHeaderLines(1);
395 
396     CSVReader reader = CSVReaderFactory.build(dataFile);
397 
398     // copy found delimiters & encoding
399     dwcFile.setEncoding(reader.encoding);
400     dwcFile.setFieldsTerminatedBy(reader.delimiter);
401     dwcFile.setFieldsEnclosedBy(reader.quoteChar);
402 
403     // detect dwc terms as good as we can based on header row
404     String[] headers = reader.header;
405     int index = 0;
406     for (String head : headers) {
407       // there are never any quotes in term names - remove them just in case the csvreader didnt recognize them
408       if (head != null && head.length() > 1) {
409         try {
410           Term dt = TERM_FACTORY.findTerm(head);
411           ArchiveField field = new ArchiveField(index, dt, null, DataType.string);
412           if (dwcFile.getId() == null && (dt.equals(DwcTerm.occurrenceID) || dt.equals(DwcTerm.taxonID) || dt
413             .equals(DwcTerm.eventID) || dt.equals(DcTerm.identifier))) {
414             dwcFile.setId(field);
415 
416             // Set the rowType corresponding to the type of id encountered (e.g. dwc:Taxon for taxonID).
417             // Please note the ordering of ids is important, and the first id encountered will be used
418             // unless the generic id (dc:identifier) was encountered first.
419             // Ideally only one id will be used per rowType anyways.
420             if (dwcFile.getRowType() == null && dwcFile.getId().getTerm() != DcTerm.identifier) {
421               if (dt.equals(DwcTerm.occurrenceID)) {
422                 dwcFile.setRowType(DwcTerm.Occurrence);
423               } else if (dt.equals(DwcTerm.taxonID)) {
424                 dwcFile.setRowType(DwcTerm.Taxon);
425               } else if (dt.equals(DwcTerm.eventID)) {
426                 dwcFile.setRowType(DwcTerm.Event);
427               }
428             }
429           }
430           dwcFile.addField(field);
431         } catch (IllegalArgumentException e) {
432           LOG.warn("Illegal term name >>{}<< found in header, ignore column {}", head, index);
433         }
434       }
435       index++;
436     }
437 
438     return dwcFile;
439   }

441   private static void readMetaDescriptor(Archive archive, InputStream metaDescriptor, boolean normaliseTerms)
442     throws UnsupportedArchiveException {
443 
444     try {
445       SAXParser p = SAX_FACTORY.newSAXParser();
446       MetaHandler mh = new MetaHandler(archive);
447       LOG.debug("Reading archive metadata file");
448       //    p.parse(metaDescriptor, mh);
449       p.parse(new BomSafeInputStreamWrapper(metaDescriptor), mh);
450     } catch (Exception e1) {
451       LOG.warn("Exception caught", e1);
452       throw new UnsupportedArchiveException(e1);
453     }
454   }

456   private static Archive validateArchive(Archive archive) throws UnsupportedArchiveException {
457     validateCoreFile(archive.getCore(), !archive.getExtensions().isEmpty());
458     for (ArchiveFile af : archive.getExtensions()) {
459       validateExtensionFile(af);
460     }
461     // report basic stats
462     LOG.debug("Archive contains " + archive.getExtensions().size() + " described extension files");
463     LOG.debug("Archive contains " + archive.getCore().getFields().size() + " core properties");
464     return archive;
465   }
466 
467   private static void validateCoreFile(ArchiveFile f, boolean hasExtensions) throws UnsupportedArchiveException {
468     if (hasExtensions) {
469       if (f.getId() == null) {
470         LOG.warn(
471           "DwC-A core data file " + f.getTitle() + " is lacking an id column. No extensions allowed in this case");
472       }
473     }
474     validateFile(f);
475   }
476 
477   private static void validateExtensionFile(ArchiveFile f) throws UnsupportedArchiveException {
478     if (f.getId() == null) {
479       throw new UnsupportedArchiveException(
480         "DwC-A data file " + f.getTitle() + " requires an id or foreign key to the core id");
481     }
482     validateFile(f);
483   }
484 
485   private static void validateFile(ArchiveFile f) throws UnsupportedArchiveException {
486     if (f == null) {
487       throw new UnsupportedArchiveException("DwC-A data file is NULL");
488     }
489     if (f.getLocationFile() == null) {
490       throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a location");
491     }
492     if (f.getEncoding() == null) {
493       throw new UnsupportedArchiveException("DwC-A data file " + f.getTitle() + " requires a character encoding");
494     }
