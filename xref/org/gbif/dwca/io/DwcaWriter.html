45  /**
46   * Simple writer class to create valid dwc archives using tab data files.
47   * The meta.xml descriptor is generated automatically and an optional EML metadata document can be added.
48   * The archive is NOT compressed but the final product is a directory with all the necessary files.
49   * For usage of this class please @see DwcaWriterTest.
50   */
51  public class DwcaWriter {
52    private Logger log = LoggerFactory.getLogger(DwcaWriter.class);
53    private final File dir;
54    private final boolean useHeaders;
55    private long recordNum;
56    private String coreId;
57    private Map<Term, String> coreRow;
58    private final Term coreRowType;
59    private final Term coreIdTerm;
60    private final Map<Term, TabWriter> writers = Maps.newHashMap();
61    private final Set<Term> headersOut = Sets.newHashSet();
62    private final Map<Term, String> dataFileNames = Maps.newHashMap();
63    // key=rowType, value=columns
64    private final Map<Term, List<Term>> terms = Maps.newHashMap();
65    // key=rowType, value=default values per column
66    private final Map<Term, Map<Term, String>> defaultValues = Maps.newHashMap();
67    private Dataset eml;

90    /**
91     * If headers are used the first record must include all terms ever used for that file.
92     * If in subsequent rows additional terms are introduced an IllegalArgumentException is thrown.
93     * 
94     * @param coreRowType the core row type
95     * @param coreIdTerm the term of the id column
96     * @param dir the directory to create the archive in
97     * @param useHeaders if true the first row in every data file will include headers
98     */
99    public DwcaWriter(Term coreRowType, Term coreIdTerm, File dir, boolean useHeaders) throws IOException {
100     this.dir = dir;
101     this.coreRowType = coreRowType;
102     this.coreIdTerm = coreIdTerm;
103     this.useHeaders = useHeaders;
104     addRowType(coreRowType);
105   }

107   public static Map<Term, String> recordToMap(Record rec, ArchiveFile af) {
108     Map<Term, String> map = new HashMap<Term, String>();
109     for (Term t : af.getTerms()) {
110       map.put(t, rec.value(t));
111     }
112     return map;
113   }

115   public static String dataFileName(Term rowType) {
116     return rowType.simpleName().toLowerCase() + ".txt";
117   }

119   private void addRowType(Term rowType) throws IOException {
120     terms.put(rowType, new ArrayList<Term>());
121 
122     String dfn = dataFileName(rowType);
123     dataFileNames.put(rowType, dfn);
124     File df = new File(dir, dfn);
125     FileUtils.forceMkdir(df.getParentFile());
126     OutputStream out = new FileOutputStream(df);
127     TabWriter wr = new TabWriter(out);
128     writers.put(rowType, wr);
129   }

131   /**
132    * A new core record is started and the last core and all extension records are written.
133    * @param id the new records id
134    * @throws IOException
135    */
136   public void newRecord(String id) throws IOException {
137     // flush last record
138     flushLastCoreRecord();
139     // start new
140     recordNum++;
141     coreId = id;
142     coreRow = new HashMap<Term, String>();
143   }
144 
145   private void flushLastCoreRecord() throws IOException {
146     if (coreRow != null) {
147       writeRow(coreRow, coreRowType);
148     }
149   }
150 
151   public long getRecordsWritten() {
152     return recordNum;
153   }

155   private void writeRow(Map<Term, String> rowMap, Term rowType) throws IOException {
156     TabWriter writer = writers.get(rowType);
157     List<Term> columns = terms.get(rowType);
158     if (useHeaders && !headersOut.contains(rowType)){
159       // write header row
160       writeHeader(writer, rowType, columns);
161     }
162 
163     // make sure coreId is not null for extensions
164     if (coreRowType != rowType && coreId == null){
165       log.warn("Adding an {} extension record to a core without an Id! Skip this record", rowType);
166 
167     } else {
168       String[] row = new String[columns.size() + 1];
169       row[0] = coreId;
170       for (Map.Entry<Term, String> conceptTermStringEntry : rowMap.entrySet()) {
171         int column = 1 + columns.indexOf(conceptTermStringEntry.getKey());
172         row[column] = conceptTermStringEntry.getValue();
173       }
174       writer.write(row);
175     }
176   }
177 
178   private void writeHeader(TabWriter writer, Term rowType, List<Term> columns) throws IOException {
179     int idx = 0;
180     String[] row = new String[columns.size() + 1];
181     Term idTerm;
182     if (DwcTerm.Taxon == coreRowType){
183       idTerm = DwcTerm.taxonID;
184     } else if (DwcTerm.Occurrence == coreRowType){
185       idTerm = DwcTerm.occurrenceID;
186     } else if (DwcTerm.Identification == coreRowType){
187       idTerm = DwcTerm.identificationID;
188     } else if (DwcTerm.Event == coreRowType){
189       idTerm = DwcTerm.eventID;
190     } else {
191       // default to generic dc identifier for id column
192       idTerm = DcTerm.identifier;
193     }
194     row[idx] = idTerm.simpleName();
195 
196     for (Term term : columns) {
197       idx ++;
198       row[idx] = term.simpleName();
199     }
200     writer.write(row);
201 
202     headersOut.add(rowType);
203   }

206   /**
207    * Add a single value for the current core record.
208    * Calling this method requires that #newRecord() has been called at least once,
209    * otherwise an IllegalStateException is thrown.
210    * @param term
211    * @param value
212    */
213   public void addCoreColumn(Term term, String value) {
214     // ensure we do not overwrite the coreIdTerm if one is defined
215     if (coreIdTerm != null && coreIdTerm.equals(term)) {
216       throw new IllegalStateException("You cannot add a term that was specified as coreId term");
217     }
218     
219     List<Term> coreTerms = terms.get(coreRowType);
220     if (!coreTerms.contains(term)) {
221       if (useHeaders && recordNum>1){
222         throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
223       }
224       coreTerms.add(term);
225     }
226     try {
227       coreRow.put(term, value);
228     } catch (NullPointerException e) {
229       // no core record has been started yet
230       throw new IllegalStateException("No core record has been created yet. Call newRecord() at least once");
231     }
232   }

236    * 
237    * @param term

242   }

244   /**
245    * Add a default value to a term of the provided rowType.
246    * 
247    * @param rowType
248    * @param term
249    * @param defaultValue
250    */
251   public void addDefaultValue(Term rowType, Term term, String defaultValue){
252     
253     if(!defaultValues.containsKey(rowType)){
254       defaultValues.put(rowType, new HashMap<Term, String>());
255     }
256     Map<Term,String> currentDefaultValues= defaultValues.get(rowType);
257     if(currentDefaultValues.containsKey(term)){
258       throw new IllegalStateException("The default value of term "+ term + " is already defined");
259     }
260     currentDefaultValues.put(term, defaultValue);
261   }

270   /**
271    * Add an extension record associated with the current core record.
272    * 
273    * @param rowType
274    * @param row
275    * @throws IOException
276    */
277   public void addExtensionRecord(Term rowType, Map<Term, String> row) throws IOException {
278     // make sure we know the extension rowtype
279     if (!terms.containsKey(rowType)) {
280       addRowType(rowType);
281     }
282     
283     // make sure we know all terms
284     List<Term> knownTerms = terms.get(rowType);
285     final boolean isFirst = knownTerms.isEmpty();
286     for (Term term : row.keySet()) {
287       if (!knownTerms.contains(term)) {
288         if (useHeaders && !isFirst){
289           throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
290         }
291         knownTerms.add(term);
292       }
293     }
294 
295     // write extension record
296     writeRow(row, rowType);
297   }

303   /**
304    * Writes meta.xml and eml.xml to the archive and closes tab writers.
305    *
306    */
307   public void close() throws IOException {
308     addEml();
309     addMeta();
310     // flush last record
311     flushLastCoreRecord();
312     // TODO: add missing columns in second iteration of data files
313 
314     // close writers
315     for (TabWriter w : writers.values()) {
316       w.close();
317     }
318   }

320   private void addEml() throws IOException {
321     if (eml != null) {
322       Writer writer = new FileWriter(new File(dir, "eml.xml"));
323       EMLWriter.write(eml, writer);
324     }
325   }

327   private void addMeta() throws IOException {
328     File metaFile = new File(dir, "meta.xml");
329 
330     Archive arch = new Archive();
331     if (eml != null) {
332       arch.setMetadataLocation("eml.xml");
333     }
334     arch.setCore(buildArchiveFile(arch, coreRowType, coreIdTerm));
335     for (Term rowType : this.terms.keySet()) {
336       if (!coreRowType.equals(rowType)) {
337         arch.addExtension(buildArchiveFile(arch, rowType, null));
338       }
339     }
340     MetaDescriptorWriter.writeMetaFile(metaFile, arch);
341   }

343   /**
344    * Build an ArchiveFile for core or extension(s).
345    *
346    * @param archive
347    * @param rowType
348    * @param idTerm the term of the id column, may be null
349    * @return
350    */
351   private ArchiveFile buildArchiveFile(Archive archive, Term rowType, Term idTerm) {
352     ArchiveFile af = ArchiveFile.buildTabFile();
353     af.setArchive(archive);
354     af.addLocation(dataFileNames.get(rowType));
355 
356     af.setEncoding("utf-8");
357     af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
358     af.setRowType(rowType);
359 
360     ArchiveField id = new ArchiveField();
361     id.setIndex(0);
362     af.setId(id);
363     // always use the index 0 for idTerm
364     if (idTerm != null) {
365       af.addField(buildArchiveField(0, idTerm));
366     }
367     
368     Map<Term,String> termDefaultValueMap = defaultValues.get(rowType);
369     List<Term> rowTypeTerms = terms.get(rowType);
370     int idx = 0;
371     String defaultValue;
372     for (Term c : rowTypeTerms) {
373       idx++;
374       defaultValue = (termDefaultValueMap !=null ? termDefaultValueMap.get(c) : null);
375       af.addField(buildArchiveField(idx, c, defaultValue));
376     }
377     
378     // check if default values are provided for this rowType
379     if(termDefaultValueMap != null){
380       for (Term t : termDefaultValueMap.keySet()) {
381         if(!rowTypeTerms.contains(t)){
382           af.addField(buildArchiveFieldDefaultValue(t, termDefaultValueMap.get(t)));
383         }
384       }
385     }
386 
387     return af;
388   }

390   /**
391    * Build an ArchiveField with a defaultValue and no index.
392    *
393    * @param term
394    * @param defaultValue
395    * @return
396    */
397   private ArchiveField buildArchiveFieldDefaultValue(Term term, String defaultValue){
398     Preconditions.checkNotNull(term, "Can't use a null term");
399     Preconditions.checkNotNull(defaultValue, "Can't use a null defaultValue");
400 
401     return new ArchiveField(term, defaultValue);
402   }

404   /**
405    * Build an ArchiveField with no defaultValue.
406    *
407    * @param idx
408    * @param term
409    * @return
410    */
411   private ArchiveField buildArchiveField(Integer idx, Term term){
412     return buildArchiveField(idx, term, null);
413   }

415   /**
416    *
417    * Build an ArchiveField from optional parameters.
418    *
419    * @param idx index or null
420    * @param term term or null
421    * @param defaultValue default value or null
422    * @return
423    */
424   private ArchiveField buildArchiveField(Integer idx, Term term, String defaultValue){
425     Preconditions.checkNotNull(idx, "Can't use a null index");
426     Preconditions.checkNotNull(term, "Can't use a null term");
427 
428     ArchiveField field = new ArchiveField(idx, term);
429 
430     if (StringUtils.isNotBlank(defaultValue)){
431       field.setDefaultValue(defaultValue);
432     }
433     return field;
434   }
