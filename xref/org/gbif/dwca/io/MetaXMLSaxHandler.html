11  /**
12   * SAX handler to parse a meta.xml descriptor for dwc archives. It populates a given archive instance and ignores
13   * namespaces. The parser needs to be namespace aware!
14   */
15  class MetaXMLSaxHandler extends SimpleSaxHandler {
16  
17    private static final TermFactory TERM_FACTORY = TermFactory.instance();
18    private static final String NS_DWCA = "http://rs.tdwg.org/dwc/text/";
19  
20    private final Archive archive;
21    private ArchiveFile af;
22  
23    MetaXMLSaxHandler(Archive archive) {
24      this.archive = archive;
25    }

27    private static Character getFirstChar(String x) throws UnsupportedArchiveException {
28      if (x == null || x.length() == 0) {
29        return null;
30      }
31      if (x.length() == 1) {
32        return x.charAt(0);
33      }
34      if (x.equalsIgnoreCase("\\t")) {
35        return '\t';
36      }
37      if (x.equalsIgnoreCase("\\n")) {
38        return '\n';
39      }
40      if (x.equalsIgnoreCase("\\r")) {
41        return '\r';
42      }
43      if (x.length() > 1) {
44        throw new UnsupportedArchiveException(
45                "Only darwin core archives with a single quotation character are supported, but found >>>" + x + "<<<");
46      }
47      return ' ';
48    }
49  
50    private static String unescapeBackslash(String x) {
51      if (x == null || x.length() == 0) {
52        return null;
53      }
54      return x.replaceAll("\\\\t", String.valueOf('\t')).replaceAll("\\\\n", String.valueOf('\n'))
55              .replaceAll("\\\\r", String.valueOf('\r')).replaceAll("\\\\f", String.valueOf('\f'));
56    }
57  
58    private ArchiveFile buildArchiveFile(Attributes attr) throws UnsupportedArchiveException {
59      ArchiveFile dwcFile = new ArchiveFile();
60  
61      // extract the File attributes
62      if (getAttr(attr, "encoding") != null) {
63        dwcFile.setEncoding(getAttr(attr, "encoding"));
64      }
65      if (getAttr(attr, "fieldsTerminatedBy") != null) {
66        dwcFile.setFieldsTerminatedBy(unescapeBackslash(getAttr(attr, "fieldsTerminatedBy")));
67      }
68      if (getAttr(attr, "fieldsEnclosedBy") != null) {
69        dwcFile.setFieldsEnclosedBy(getFirstChar(getAttr(attr, "fieldsEnclosedBy")));
70      }
71      if (getAttr(attr, "linesTerminatedBy") != null) {
72        dwcFile.setLinesTerminatedBy(unescapeBackslash(getAttr(attr, "linesTerminatedBy")));
73      }
74      if (getAttr(attr, "rowType") != null) {
75        dwcFile.setRowType(TERM_FACTORY.findTerm(getAttr(attr, "rowType")));
76      }
77      String ignoreHeaderLines = getAttr(attr, "ignoreHeaderLines");
78      try {
79        dwcFile.setIgnoreHeaderLines(Integer.parseInt(ignoreHeaderLines));
80      } catch (NumberFormatException ignored) { // swallow null or bad value
81      }
82      return dwcFile;
83    }
84  
85    /**
86     * Build an ArchiveField object based on xml attributes.
87     */
88    private ArchiveField buildField(Attributes attributes) {
89      // build field
90      Term term = TERM_FACTORY.findTerm(getAttr(attributes, "term"));
91      String defaultValue = getAttr(attributes, "default");
92      String vocabulary = getAttr(attributes, "vocabulary");
93      ArchiveField.DataType type = ArchiveField.DataType.findByXmlSchemaType(getAttr(attributes, "type"));
94      if (type == null) {
95        type = ArchiveField.DataType.string;
96      }
97      String indexAsString = getAttr(attributes, "index");
98      Integer index = null;
99      if (indexAsString != null) {
100       // let bad errors be thrown up
101       try {
102         index = Integer.parseInt(indexAsString);
103       } catch (NumberFormatException e) {
104         throw new UnsupportedArchiveException(e);
105       }
106     }
107     String delimiter = getAttr(attributes, "delimitedBy");
108     return new ArchiveField(index, term, defaultValue, type, delimiter, vocabulary);
109   }

113     // calling the super method to stringify the character buffer
114     super.endElement(uri, localName, qName);
115 
116     if (localName.equalsIgnoreCase("archive")) {
117       // archive
118     } else if (localName.equalsIgnoreCase("core")) {
119       // update location to absolute path incl archive path
120       //      makeLocationPathsAbsolute(af, archive.getLocation());
121       archive.setCore(af);
122     } else if (localName.equalsIgnoreCase("extension")) {
123       // update location to absolute path incl archive path
124       //      makeLocationPathsAbsolute(af, archive.getLocation());
125 
126       if (af.getId() != null && af.getId().getIndex() != null) {
127         archive.addExtension(af);
128       } else {
129         log.warn("Skipping extension [" + af.getRowType() + "] with no index attribute");
130       }
131     } else if (localName.equalsIgnoreCase("location")) {
132       // a file location
133       af.addLocation(content);
134     }

138   /**
139    * Get attribute from a key
140    *
141    * @param attributes
142    * @param key
143    * @return attributes value or null
144    */
145   private String getAttr(Attributes attributes, String key) {
146     String val = null;
147     if (attributes != null) {
148       // try without NS
149       val = attributes.getValue("", key);
150       if (val == null) {
151         // try with dwca NS if nothing found
152         val = attributes.getValue(NS_DWCA, key);
153       }
154     }
155     return Strings.isNullOrEmpty(val) ? null : val;
156   }

158   @Override
159   public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
160     super.startElement(uri, localName, qName, attributes);
161     if (localName.equalsIgnoreCase("archive") || localName.equalsIgnoreCase("stararchive")) {
162       // metadata location
163       archive.setMetadataLocation(getAttr(attributes, "metadata"));
164     } else if (localName.equalsIgnoreCase("core") || localName.equalsIgnoreCase("extension")) {
165       // archive/extension
166       af = new ArchiveFile();
167       if (localName.equalsIgnoreCase("core") || localName.equalsIgnoreCase("extension")) {
168         // archive/core or archive/extension
169         af = buildArchiveFile(attributes);
170       }
171     } else if (localName.equalsIgnoreCase("coreid") || localName.equalsIgnoreCase("id")) {
172       ArchiveField field = buildField(attributes);
173       if (af != null) {
174         af.setId(field);
175       } else {
176         log.warn(localName + " field found outside of an archive file");
177       }
178     } else if (localName.equalsIgnoreCase("field")) {
179       ArchiveField field = buildField(attributes);
180       if (af != null) {
181         af.addField(field);
182       } else {
183         log.warn("field found outside of an archive file");
184       }
185     }
186   }
