14  /**
15   * Static utility methods related to {@link TermTabularDataFileReader} instances.
16   */
17  public class TermTabularFiles {
18  
19    /**
20     * Get a new TermTabularDataFileReader for {@link Term}.
21     *
22     * @param in
23     * @param delimiterChar
24     * @param headerLine
25     * @param columnMapping
26     * @return
27     */
28    public static TermTabularDataFileReader newTermMappedTabularFileReader(InputStream in, char delimiterChar,
29                                                                                   boolean headerLine,
30                                                                                   Term[] columnMapping){
31      Preconditions.checkNotNull(columnMapping, "columnMapping must be provided");
32  
33      TabularDataFileReader<List<String>> tabularFileReader = TabularFiles
34              .newTabularFileReader(in, delimiterChar, headerLine);
35      return new TermTabularDataFileReader(tabularFileReader, columnMapping);
36    }
37  
38  
39    /**
40     * Get a new TermTabularDataFileReader from a {@link ArchiveFile} definition for content as {@link InputStream}.
41     *
42     * @param in          the content to read
43     * @param archiveFile the "definition" of the archive
44     * @param headerLine  is the header line included in the content to read ?
45     *
46     * @return
47     */
48    public static TermTabularDataFileReader newDwcTabularFileReader(InputStream in, ArchiveFile archiveFile,
49                                                                    boolean headerLine) {
50      Preconditions.checkNotNull(archiveFile, "DarwinCore archive fields must be provided");
51      Preconditions.checkArgument(archiveFile.getFieldsTerminatedBy().length() == 1,
52              "DarwinCore archive getFieldsTerminatedBy must be 1 char");
53  
54      TabularDataFileReader<List<String>> tabularFileReader = TabularFiles
55              .newTabularFileReader(in, archiveFile.getFieldsTerminatedBy().charAt(0), headerLine);
56  
57      return new DwcTabularDataFileReader(tabularFileReader, archiveFile.getId(),
58              Lists.newArrayList(archiveFile.getFields().values()));
59    }
60  
61  }
