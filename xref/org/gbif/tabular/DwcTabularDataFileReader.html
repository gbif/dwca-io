3   import org.gbif.dwc.terms.Term;
4   import org.gbif.dwc.terms.TermFactory;
5   import org.gbif.dwca.io.ArchiveField;
6   import org.gbif.utils.file.tabular.TabularDataFileReader;
7   
8   import java.io.IOException;
9   import java.util.List;
10  import java.util.Map;
11  
12  import com.google.common.base.Strings;
13  import com.google.common.collect.Maps;
14  
15  /**
16   * Specialized TermTabularDataFileReader for DarwinCore archive file.
17   * In addition of reading the underlying csv file this class will handle id field and default values that can
18   * be provided in DarwinCore.
19   */
20  public class DwcTabularDataFileReader extends TermTabularDataFileReader {
21  
22    //refers to a column in the archive representing the id of the line that is not mapped to a real term
23    public static Term DEFAULT_ID_TERM = TermFactory.instance().findTerm("dwcaId");
24  
25    private Map<Term, String> defaultValues = Maps.newHashMap();
26  
27    public DwcTabularDataFileReader(TabularDataFileReader<List<String>> tabularDataFileReader, ArchiveField idField,
28                                    List<ArchiveField> fields) {
29      super(tabularDataFileReader, buildColumnMapping(idField, fields));
30  
31      for(ArchiveField af : fields) {
32        if(!Strings.isNullOrEmpty(af.getDefaultValue())){
33          defaultValues.put(af.getTerm(), af.getDefaultValue());
34        }
35      }
36    }
37  
38    private static Term[] buildColumnMapping(ArchiveField idField, List<ArchiveField> dwcTerms){
39      Map<Integer, Term> columnMapping = Maps.newTreeMap();
40      columnMapping.put(idField.getIndex(), idField.getTerm() != null ? idField.getTerm() : DEFAULT_ID_TERM);
41      for(ArchiveField af : dwcTerms) {
42        if(af.getIndex() != null && af.getIndex() >= 0){
43          columnMapping.put(af.getIndex(), af.getTerm());
44        }
45      }
46      return columnMapping.values().toArray(new Term[0]);
47    }
48  
49    @Override
50    public TermTabularDataLine read() throws IOException {
51      TermTabularDataLine line = super.read();
52  
53      if(line == null){
54        return null;
55      }
56  
57      if(defaultValues.isEmpty()){
58        return line;
59      }
60  
61      Map<Term, String> mappedData = Maps.newHashMap(line.getMappedData());
62      int numberOfColumn = line.getNumberOfColumn();
63  
64      for(Term term : defaultValues.keySet()) {
65        if(mappedData.containsKey(term)){
66          if(Strings.isNullOrEmpty(mappedData.get(term))){
67            mappedData.put(term, defaultValues.get(term));
68          }
69        }
70        else {
71          mappedData.put(term, defaultValues.get(term));
72          numberOfColumn++;
73        }
74      }
75  
76      return new TermTabularDataLine(line.getLineNumber(), mappedData, numberOfColumn);
77    }
78  
79  }
