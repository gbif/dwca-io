3   import java.io.File;
4   import java.io.FileOutputStream;
5   import java.io.IOException;
6   import java.net.URL;
7   import java.nio.channels.Channels;
8   import java.nio.channels.ReadableByteChannel;
9   
10  /**
11   * A simple download utils that uses the native http capabilities of the URL class and NIO.
12   * It does not handle redirects and other advanced http options - please use the gbif-httputil HttpUtils class for
13   * those cases which makes use of the heavier but more robust HttpClient.
14   */
15  public class DownloadUtil {
16  
17    /**
18     * A private Utils class.
19     */
20    private DownloadUtil() {
21    }
22  
23    /**
24     * Copies the content of a URL to a local file.
25     *
26     * @param url source url
27     * @param out file to write content to
28     */
29    public static void download(URL url, File out) throws IOException {
30      ReadableByteChannel rbc = Channels.newChannel(url.openStream());
31      FileOutputStream fos = new FileOutputStream(out);
32      fos.getChannel().transferFrom(rbc, 0, 1 << 24);
33      fos.close();
34      rbc.close();
35    }
36  
37    /**
38     * Copies the content of a URL to a local file.
39     *
40     * @param url source url
41     * @param out file to write content to
42     */
43    public static void download(String url, File out) throws IOException {
44      URL u = new URL(url);
45      download(u, out);
46    }
47  
48  }
