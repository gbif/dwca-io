65    private static final TermFactory TERM_FACTORY = TermFactory.instance();
66    private static final String DEFAULT_ENDLINE_CHAR = "\n";
67    private static final char DEFAULT_DELIMITER_CHAR = ',';

69    /**
70     * Predefined mapping between {@link Term} and its rowType.
71     * Ordering is important since the first found will be used.
72     */
73    private static final Map<Term, Term> TERM_TO_ROW_TYPE;
74    static {
75      Map<Term, Term> idToRowType = new LinkedHashMap<>();
76      idToRowType.put(DwcTerm.occurrenceID, DwcTerm.Occurrence);
77      idToRowType.put(DwcTerm.taxonID, DwcTerm.Taxon);
78      idToRowType.put(DwcTerm.eventID, DwcTerm.Event);
79      TERM_TO_ROW_TYPE = Collections.unmodifiableMap(idToRowType);
80    }

82    /**
83     * Terms that can represent an identifier within a file
84     */
85    private static final List<Term> ID_TERMS = Collections.unmodifiableList(
86            Arrays.asList(DwcTerm.occurrenceID, DwcTerm.taxonID, DwcTerm.eventID, DcTerm.identifier));

88    private static final List<String> DATA_FILE_SUFFICES = ImmutableList.of(".csv", ".txt", ".tsv", ".tab", ".text", ".data", ".dwca");

93    private static void applyIpt205Patch(final Path dwcFolder) {
94      // Accommodate archives coming from legacy IPTs which put a "\" before each filename
95      // http://dev.gbif.org/issues/browse/POR-2396
96      // https://code.google.com/p/gbif-providertoolkit/issues/detail?id=1015
97      Iterator<File> iter = FileUtils.iterateFiles(dwcFolder.toFile(), new String[]{"xml", "txt"}, false);
98      while (iter.hasNext()) {
99        File f = iter.next();
100       if (f.getName().startsWith("\\")) {
101         String orig = f.getName();
102         String replacement = f.getName().replaceFirst("\\\\", "");
103         LOG.info("Renaming file from {} to {}", orig, replacement);
104         f.renameTo(new File(dwcFolder.toFile(), replacement));
105       }
106     }
107   }

109   private static List<File> extractPossibleDataFile(File dwcFolder) {
110     List<File> dataFiles = new ArrayList<>();
111     for (String suffix : DATA_FILE_SUFFICES) {
112       FileFilter ff = FileFilterUtils.and(
113               FileFilterUtils.suffixFileFilter(suffix, IOCase.INSENSITIVE), HiddenFileFilter.VISIBLE
114       );
115       dataFiles.addAll(Arrays.asList(dwcFolder.listFiles(ff)));
116     }
117     return dataFiles;
118   }

120   /**
121    * Creates an {@link Archive} from a single file.
122    *
123    * @param dwcFile represents a single data file or a metadata file.
124    * @return a new {@link Archive}, never null.
125    * @throws IOException
126    */
127   private static Archive archiveFromSingleFile(Path dwcFile) throws IOException {
128     Archive archive = new Archive();
129     archive.setLocation(dwcFile.toFile());
130     archive.setDwcLayout(DwcLayout.fromFile(dwcFile.toFile()));
131 
132     Optional<String> possibleMetadataFile = DwcMetaFiles.discoverMetadataFile(dwcFile);
133     if(possibleMetadataFile.isPresent()) {
134       archive.setMetadataLocation(possibleMetadataFile.get());
135     }
136     else{
137       ArchiveFile coreFile = fromSingleFile(dwcFile);
138       coreFile.addLocation(dwcFile.getFileName().toString());
139       archive.setCore(coreFile);
140     }
141 
142     return archive;
143   }

145   /**
146    * @param dwcLocation the location of an expanded dwc archive directory or just a single dwc text file
147    * @return new {@link Archive}, never null. But, the {@link Archive} can be empty (e.g. no core)
148    */
149   static Archive fromLocation(Path dwcLocation) throws IOException, UnsupportedArchiveException {
150     if (!Files.exists(dwcLocation)) {
151       throw new FileNotFoundException("dwcLocation does not exist: " + dwcLocation.toAbsolutePath());
152     }
153     // delegate to {@link #archiveFromSingleFile) if its a single file, not a folder
154     if (Files.isRegularFile(dwcLocation)) {
155       return archiveFromSingleFile(dwcLocation);
156     }
157 
158     Archive archive = new Archive();
159     applyIpt205Patch(dwcLocation);
160 
161     // Check for meta descriptor
162     Path metaDescriptorFile = dwcLocation.resolve(Archive.META_FN);
163     if (Files.exists(metaDescriptorFile)) {
164       // read metaDescriptor file
165       try {
166         archive = DwcMetaFiles.fromMetaDescriptor(new FileInputStream(metaDescriptorFile.toFile()));
167       } catch (SAXException | IOException e) {
168         // using UnsupportedArchiveException for backward compatibility but IOException would be fine here
169         throw new UnsupportedArchiveException(e);
170       }
171     } else {
172       // meta.xml lacking.
173       // Try to detect data files ourselves as best as we can.
174       // look for a single, visible text data file
175       List<File> dataFiles = extractPossibleDataFile(dwcLocation.toFile());
176       if (dataFiles.size() == 1) {
177         archive = archiveFromSingleFile(dataFiles.get(0).toPath());
178       }
179     }
180 
181     // check if we also have a metadata file next to this data file
182     DwcMetaFiles.discoverMetadataFile(dwcLocation)
183             .ifPresent(archive::setMetadataLocation);
184 
185     archive.setLocation(dwcLocation.toFile());
186     archive.setDwcLayout(DwcLayout.DIRECTORY_ROOT);
187 
188     return archive;
189   }

191   static Archive fromCompressed(Path dwcaLocation, Path destination) throws IOException, UnsupportedArchiveException {
192     if (!Files.exists(dwcaLocation)) {
193       throw new FileNotFoundException("dwcaLocation does not exist: " + dwcaLocation.toAbsolutePath());
194     }
195 
196     if (Files.exists(destination)) {
197       // clean up any existing folder
198       LOG.debug("Deleting existing archive folder [{}]", destination.toAbsolutePath());
199       org.gbif.utils.file.FileUtils.deleteDirectoryRecursively(destination.toFile());
200     }
201     FileUtils.forceMkdir(destination.toFile());
202     // try to decompress archive
203     try {
204       CompressionUtil.decompressFile(destination.toFile(), dwcaLocation.toFile(), true);
205       // we keep subfolder, but often the entire archive is within one subfolder. Remove that root folder if present
206       File[] rootFiles = destination.toFile().listFiles((FileFilter) HiddenFileFilter.VISIBLE);
207       if (rootFiles.length == 1) {
208         File root = rootFiles[0];
209         if (root.isDirectory()) {
210           // single root dir, flatten structure
211           LOG.debug("Removing single root folder {} found in decompressed archive", root.getAbsoluteFile());
212           for (File f : FileUtils.listFiles(root, TrueFileFilter.TRUE, null)) {
213             File f2 = new File(destination.toFile(), f.getName());
214             f.renameTo(f2);
215           }
216         }
217       }
218       // continue to read archive from the tmp dir
219       return fromLocation(destination);
220     } catch (CompressionUtil.UnsupportedCompressionType e) {
221       throw new UnsupportedArchiveException(e);
222     }
223   }

225   /**
226    * Return a {@link ArchiveFile} based on a single data file.
227    * Delimiter, quote char and encoding will be extracted from the file using {@link TabularFileMetadataExtractor}.
228    * Columns will be determined by extracting the first line of the file.
229    * The end of line character can only be {@link #DEFAULT_ENDLINE_CHAR}.
230    * @param dataFile
231    * @return
232    * @throws UnsupportedArchiveException
233    * @throws IOException
234    */
235   static ArchiveFile fromSingleFile(Path dataFile) throws UnsupportedArchiveException, IOException {
236     Preconditions.checkArgument(Files.isRegularFile(dataFile), "dataFile shall be a file");
237     ArchiveFile dwcFile = new ArchiveFile();
238    // dwcFile.addLocation(null);
239     dwcFile.setIgnoreHeaderLines(1);
240 
241     TabularFileMetadata tabularFileMetadata = TabularFileMetadataExtractor.extractTabularFileMetadata(dataFile);
242     dwcFile.setFieldsTerminatedBy(Optional.ofNullable(tabularFileMetadata.getDelimiter()).orElse(DEFAULT_DELIMITER_CHAR).toString());
243     dwcFile.setFieldsEnclosedBy(tabularFileMetadata.getQuotedBy());
244     dwcFile.setEncoding(tabularFileMetadata.getEncoding().name());
245 
246     List<String> headers;
247     try (TabularDataFileReader<List<String>> reader =
248                  TabularFiles.newTabularFileReader(Files.newBufferedReader(dataFile, tabularFileMetadata.getEncoding()),
249                          dwcFile.getFieldsTerminatedBy().charAt(0), DEFAULT_ENDLINE_CHAR,
250                          dwcFile.getFieldsEnclosedBy(), true)) {
251       headers = reader.getHeaderLine() == null ? Collections.emptyList() : reader.getHeaderLine();
252     }
253 
254     // detect dwc terms as good as we can based on header row
255     int index = 0;
256     for (String head : headers) {
257       // there are never any quotes in term names - remove them just in case the we didn't recognize them
258       if (head != null && head.length() > 1) {
259         try {
260           Term dt = TERM_FACTORY.findPropertyTerm(head);
261           dwcFile.addField(new ArchiveField(index, dt, null, ArchiveField.DataType.string));
262         } catch (IllegalArgumentException e) {
263           LOG.warn("Illegal term name >>{}<< found in header, ignore column {}", head, index);
264         }
265       }
266       index++;
267     }
268 
269     List<Term> headerAsTerm = dwcFile.getFields().keySet()
270             .stream()
271             .collect(Collectors.toList());
272 
273     determineRecordIdentifier(headerAsTerm).ifPresent(
274             t -> dwcFile.setId(dwcFile.getField(t))
275     );
276 
277     determineRowType(headerAsTerm).ifPresent(dwcFile::setRowType);
278     return dwcFile;
279   }

281   /**
282    * Tries to determine the rowType based on a list of {@link Term}.
283    *
284    * @param terms the list can contain null values
285    *
286    * @return {@link Term} as {@code Optional} or {@code Optional.empty()} if can not be determined
287    */
288   static Optional<Term> determineRowType(List<Term> terms) {
289     return TERM_TO_ROW_TYPE.entrySet().stream()
290             .filter(ke -> terms.contains(ke.getKey()))
291             .map(Map.Entry::getValue).findFirst();
292   }
293 
294   /**
295    * Tries to determine the record identifier based on a list of {@link Term}.
296    *
297    * @param terms the list can contain null values
298    *
299    * @return {@link Term} as {@code Optional} or {@code Optional.empty()} if can not be determined
300    */
301   static Optional<Term> determineRecordIdentifier(List<Term> terms) {
302     //try to find the first matching term respecting the order defined by ID_TERMS
303     return ID_TERMS.stream()
304             .filter(terms::contains)
305             .findFirst();
306   }
