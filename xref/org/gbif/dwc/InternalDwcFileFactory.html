41    /**
42     * Predefined mapping between {@link Term} and its rowType.
43     * Ordering is important since the first found will be used.
44     */
45    private static final Map<Term, Term> TERM_TO_ROW_TYPE;
46    static {
47      Map<Term, Term> idToRowType = new LinkedHashMap<>();
48      idToRowType.put(DwcTerm.occurrenceID, DwcTerm.Occurrence);
49      idToRowType.put(DwcTerm.taxonID, DwcTerm.Taxon);
50      idToRowType.put(DwcTerm.eventID, DwcTerm.Event);
51      TERM_TO_ROW_TYPE = Collections.unmodifiableMap(idToRowType);
52    }

65    private static void applyIpt205Patch(final Path dwcFolder) {
66      // Accommodate archives coming from legacy IPTs which put a "\" before each filename
67      // http://dev.gbif.org/issues/browse/POR-2396
68      // https://code.google.com/p/gbif-providertoolkit/issues/detail?id=1015
69      Iterator<File> iter = FileUtils.iterateFiles(dwcFolder.toFile(), new String[]{"xml", "txt"}, false);
70      while (iter.hasNext()) {
71        File f = iter.next();
72        if (f.getName().startsWith("\\")) {
73          String orig = f.getName();
74          String replacement = f.getName().replaceFirst("\\\\", "");
75          LOG.info("Renaming file from {} to {}", orig, replacement);
76          f.renameTo(new File(dwcFolder.toFile(), replacement));
77        }
78      }
79    }

81    private static List<File> extractPossibleDataFile(File dwcFolder) {
82      List<File> dataFiles = new ArrayList<>();
83      for (String suffix : DATA_FILE_SUFFICES) {
84        FileFilter ff = FileFilterUtils.and(
85                FileFilterUtils.suffixFileFilter(suffix, IOCase.INSENSITIVE), HiddenFileFilter.VISIBLE
86        );
87        dataFiles.addAll(Arrays.asList(dwcFolder.listFiles(ff)));
88      }
89      return dataFiles;
90    }

92    /**
93     * Creates an {@link Archive} from a single file.
94     *
95     * @param dwcFile represents a single data file or a metadata file.
96     * @return a new {@link Archive}, never null.
97     * @throws IOException
98     */
99    private static Archive archiveFromSingleFile(Path dwcFile) throws IOException {
100     Archive archive = new Archive();
101     archive.setLocation(dwcFile.toFile());
102     archive.setDwcLayout(DwcLayout.fromFile(dwcFile.toFile()));
103 
104     Optional<String> possibleMetadataFile = DwcMetaFiles.discoverMetadataFile(dwcFile);
105     if(possibleMetadataFile.isPresent()) {
106       archive.setMetadataLocation(possibleMetadataFile.get());
107     }
108     else{
109       ArchiveFile coreFile = fromSingleFile(dwcFile);
110       coreFile.addLocation(dwcFile.getFileName().toString());
111       archive.setCore(coreFile);
112     }
113 
114     return archive;
115   }

117   /**
118    * @param dwcLocation the location of an expanded dwc archive directory or just a single dwc text file
119    * @return new {@link Archive}, never null. But, the {@link Archive} can be empty (e.g. no core)
120    */
121   static Archive fromLocation(Path dwcLocation) throws IOException, UnsupportedArchiveException {
122     if (!Files.exists(dwcLocation)) {
123       throw new FileNotFoundException("dwcLocation does not exist: " + dwcLocation.toAbsolutePath());
124     }
125     // delegate to {@link #archiveFromSingleFile) if its a single file, not a folder
126     if (Files.isRegularFile(dwcLocation)) {
127       return archiveFromSingleFile(dwcLocation);
128     }
129 
130     Archive archive = new Archive();
131     applyIpt205Patch(dwcLocation);
132 
133     // Check for meta descriptor
134     Path metaDescriptorFile = dwcLocation.resolve(Archive.META_FN);
135     if (Files.exists(metaDescriptorFile)) {
136       // read metaDescriptor file
137       try {
138         archive = DwcMetaFiles.fromMetaDescriptor(new FileInputStream(metaDescriptorFile.toFile()));
139       } catch (SAXException | IOException e) {
140         // using UnsupportedArchiveException for backward compatibility but IOException would be fine here
141         throw new UnsupportedArchiveException(e);
142       }
143     } else {
144       // meta.xml lacking.
145       // Try to detect data files ourselves as best as we can.
146       // look for a single, visible text data file
147       List<File> dataFiles = extractPossibleDataFile(dwcLocation.toFile());
148       if (dataFiles.size() == 1) {
149         archive = archiveFromSingleFile(dataFiles.get(0).toPath());
150       }
151     }
152 
153     // check if we also have a metadata file next to this data file
154     DwcMetaFiles.discoverMetadataFile(dwcLocation)
155             .ifPresent(archive::setMetadataLocation);
156 
157     archive.setLocation(dwcLocation.toFile());
158     archive.setDwcLayout(DwcLayout.DIRECTORY_ROOT);
159 
160     return archive;
161   }

163   static Archive fromCompressed(Path dwcaLocation, Path destination) throws IOException, UnsupportedArchiveException {
164     if (!Files.exists(dwcaLocation)) {
165       throw new FileNotFoundException("dwcaLocation does not exist: " + dwcaLocation.toAbsolutePath());
166     }
167 
168     if (Files.exists(destination)) {
169       // clean up any existing folder
170       LOG.debug("Deleting existing archive folder [{}]", destination.toAbsolutePath());
171       org.gbif.utils.file.FileUtils.deleteDirectoryRecursively(destination.toFile());
172     }
173     FileUtils.forceMkdir(destination.toFile());
174     // try to decompress archive
175     try {
176       CompressionUtil.decompressFile(destination.toFile(), dwcaLocation.toFile(), true);
177       // we keep subfolder, but often the entire archive is within one subfolder. Remove that root folder if present
178       File[] rootFiles = destination.toFile().listFiles((FileFilter) HiddenFileFilter.VISIBLE);
179       if (rootFiles.length == 1) {
180         File root = rootFiles[0];
181         if (root.isDirectory()) {
182           // single root dir, flatten structure
183           LOG.debug("Removing single root folder {} found in decompressed archive", root.getAbsoluteFile());
184           for (File f : FileUtils.listFiles(root, TrueFileFilter.TRUE, null)) {
185             File f2 = new File(destination.toFile(), f.getName());
186             f.renameTo(f2);
187           }
188         }
189       }
190       // continue to read archive from the tmp dir
191       return fromLocation(destination);
192     } catch (CompressionUtil.UnsupportedCompressionType e) {
193       throw new UnsupportedArchiveException(e);
194     }
195   }

197   /**
198    * Return a {@link ArchiveFile} based on a single data file.
199    * Delimiter, quote char and encoding will be extracted from the file using {@link TabularFileMetadataExtractor}.
200    * Columns will be determined by extracting the first line of the file.
201    * The end of line character can only be {@link #DEFAULT_ENDLINE_CHAR}.
202    * @param dataFile
203    * @return
204    * @throws UnsupportedArchiveException
205    * @throws IOException
206    */
207   static ArchiveFile fromSingleFile(Path dataFile) throws UnsupportedArchiveException, IOException {
208     Preconditions.checkArgument(Files.isRegularFile(dataFile), "dataFile shall be a file");
209     ArchiveFile dwcFile = new ArchiveFile();
210    // dwcFile.addLocation(null);
211     dwcFile.setIgnoreHeaderLines(1);
212 
213     TabularFileMetadata tabularFileMetadata = TabularFileMetadataExtractor.extractTabularFileMetadata(dataFile);
214     dwcFile.setFieldsTerminatedBy(Optional.ofNullable(tabularFileMetadata.getDelimiter()).orElse(DEFAULT_DELIMITER_CHAR).toString());
215     dwcFile.setFieldsEnclosedBy(tabularFileMetadata.getQuotedBy());
216     dwcFile.setEncoding(tabularFileMetadata.getEncoding().name());
217 
218     List<String> headers;
219     try (TabularDataFileReader<List<String>> reader =
220                  TabularFiles.newTabularFileReader(Files.newBufferedReader(dataFile, tabularFileMetadata.getEncoding()),
221                          dwcFile.getFieldsTerminatedBy().charAt(0), DEFAULT_ENDLINE_CHAR,
222                          dwcFile.getFieldsEnclosedBy(), true)) {
223       headers = reader.getHeaderLine() == null ? Collections.emptyList() : reader.getHeaderLine();
224     }
225 
226     // detect dwc terms as good as we can based on header row
227     int index = 0;
228     for (String head : headers) {
229       // there are never any quotes in term names - remove them just in case the we didn't recognize them
230       if (head != null && head.length() > 1) {
231         try {
232           Term dt = TERM_FACTORY.findPropertyTerm(head);
233           dwcFile.addField(new ArchiveField(index, dt, null, ArchiveField.DataType.string));
234         } catch (IllegalArgumentException e) {
235           LOG.warn("Illegal term name >>{}<< found in header, ignore column {}", head, index);
236         }
237       }
238       index++;
239     }
240 
241     List<Term> headerAsTerm = dwcFile.getFields().keySet()
242             .stream()
243             .collect(Collectors.toList());
244 
245     determineRecordIdentifier(headerAsTerm).ifPresent(
246             t -> dwcFile.setId(dwcFile.getField(t))
247     );
248 
249     determineRowType(headerAsTerm).ifPresent(dwcFile::setRowType);
250     return dwcFile;
251   }

253   /**
254    * Tries to determine the rowType based on a list of {@link Term}.
255    *
256    * @param terms the list can contain null values
257    *
258    * @return {@link Term} as {@code Optional} or {@code Optional.empty()} if can not be determined
259    */
260   static Optional<Term> determineRowType(List<Term> terms) {
261     return TERM_TO_ROW_TYPE.entrySet().stream()
262             .filter(ke -> terms.contains(ke.getKey()))
263             .map(Map.Entry::getValue).findFirst();
264   }
265 
266   /**
267    * Tries to determine the record identifier based on a list of {@link Term}.
268    *
269    * @param terms the list can contain null values
270    *
271    * @return {@link Term} as {@code Optional} or {@code Optional.empty()} if can not be determined
272    */
273   static Optional<Term> determineRecordIdentifier(List<Term> terms) {
274     //try to find the first matching term respecting the order defined by ID_TERMS
275     return ID_TERMS.stream()
276             .filter(terms::contains)
277             .findFirst();
278   }
