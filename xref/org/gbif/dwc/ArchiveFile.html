70    /**
71     * Comparator to sort by column index, then term name.
72     */
73    private static final Comparator<ArchiveField> AF_IDX_COMPARATOR = (o1, o2) -> {
74      if (o1.getIndex() == null && o2.getIndex() == null) {
75        return 0;
76      } else if (o1.getIndex() == null) {
77        return -1;
78      } else if (o2.getIndex() == null) {
79        return 1;
80      } else if (o1.getIndex().equals(o2.getIndex())) {
81        return o1.getTerm().qualifiedName().compareTo(o2.getTerm().qualifiedName());
82      } else {
83        return o1.getIndex().compareTo(o2.getIndex());
84      }
85    };

117   protected void validateAsCore(boolean hasExtensions) throws UnsupportedArchiveException {
118     if (hasExtensions) {
119       if (id == null) {
120         LOG.warn("DwC-A core data file »" + title + "« is lacking an id column. No extensions allowed in this case");
121       }
122     }
123     validate();
124   }

126   protected void validateAsExtension() throws UnsupportedArchiveException {
127     if (id == null) {
128       throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires an id or foreign key to the core id");
129     }
130     validate();
131   }

133   protected void validate() throws UnsupportedArchiveException {
134     if (getLocation() == null) {
135       throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a location");
136     }
137     if (!getLocationFile().exists()) {
138       throw new UnsupportedArchiveException("DwC-A data file »" + title + "« does not exist");
139     }
140     if (encoding == null) {
141       throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a character encoding");
142     }
143   }

145   /**
146    * Normalizes and sorts the data file, so it is suitable for in-order iteration as a core or extension file.
147    *
148    * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
149    */
150   /*
151    * Sorting implies a normalization phase to ensure we sort the file properly.
152    * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
153    * we want to sort.
154    *
155    * This method is synchronized to prevent multiple threads trying to normalize/sort a file at the same time,
156    * and uses an advisory lock to avoid conflict between multiple processes.
157    */
158   protected synchronized boolean normalizeAndSort() throws IOException {
159     File fileToSort = getLocationFile();
160     File sortedFile = getLocationFileSorted(getLocationFile());
161 
162     File lockFile = getLocationLockFile(getLocationFile());
163     RandomAccessFile lockFileRA = new RandomAccessFile(lockFile, "rw");
164 
165     FileLock lock = lockFileRA.getChannel().tryLock();
166     if (lock == null) {
167       LOG.warn("Another process has locked this DWCA for initialization; waiting until the lock is released.");
168       lock = lockFileRA.getChannel().lock();
169       LOG.warn("Other process has released lock; lock taken, proceeding.");
170     }
171 
172     try {
173       // If we already sorted the file and its source didn't change we can avoid doing it again
174       if (sortedFile.exists() && Files.getLastModifiedTime(sortedFile.toPath()).toInstant().isAfter(
175           Files.getLastModifiedTime(fileToSort.toPath()).toInstant())) {
176         LOG.debug("File {} is already sorted ({}B)", sortedFile, sortedFile.length());
177         return false;
178       }
179 
180       File normalizedFile = normalizeIfRequired();
181       if (normalizedFile != null) {
182         fileToSort = normalizedFile;
183       }
184 
185       FILE_UTILS.sort(fileToSort, getLocationFileSorted(getLocationFile()), getEncoding(),
186           getId().getIndex(), getFieldsTerminatedBy(), getFieldsEnclosedBy(),
187           TabularFileNormalizer.NORMALIZED_END_OF_LINE, getIgnoreHeaderLines());
188 
189       if (normalizedFile != null) {
190         Files.deleteIfExists(normalizedFile.toPath());
191       }
192 
193       return true;
194     } finally {
195       lock.release();
196       lockFileRA.close();
197     }
198   }

200   /**
201    * Apply file normalization if required.
202    **
203    * @return normalizedFile or null if normalization was not applied
204    *
205    * @throws IOException
206    */
207   @VisibleForTesting
208   protected File normalizeIfRequired() throws IOException {
209     // If the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
210     // we can skip normalization
211     boolean normalizationRequired = ! TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(getLinesTerminatedBy())
212         || getFieldsEnclosedBy() != null;
213 
214     if (normalizationRequired) {
215       File normalizedFile = getLocationFileNormalized(getLocationFile());
216       TabularFileNormalizer.normalizeFile(getLocationFile().toPath(), normalizedFile.toPath(),
217           Charset.forName(getEncoding()), getFieldsTerminatedByChar(),
218           getLinesTerminatedBy(), getFieldsEnclosedBy());
219       return normalizedFile;
220     }
221     return null;
222   }

244         title = location.substring(location.lastIndexOf('/') + 1, location.length());
245       } else {
246         title = location;

305   /**
306    * Generates a list of {@link Term}s mapped to each column in the underlying file.
307    *
308    * A list may be empty when no Term is mapped at a specific position, or may contain more than one Term where
309    * the same column is mapped more than once.
310    *
311    * The length of the list is the maximum index used in the {@link ArchiveField}s + 1.
312    *
313    * @return Data structure representing the header of the file.
314    */
315   public List<List<Term>> getHeader() {
316     List<ArchiveField> archiveFieldsWithIndex = getFieldsSorted()
317             .stream().filter(af -> af.getIndex() != null)
318             .collect(Collectors.toList());
319 
320     Optional<Integer> idIndex = id != null ? Optional.of(id.getIndex()) : Optional.empty();
321 
322     if (archiveFieldsWithIndex.isEmpty() && !idIndex.isPresent()){
323       return new ArrayList<>();
324     }
325 
326     int maxIndex = archiveFieldsWithIndex.stream().mapToInt(ArchiveField::getIndex).max().getAsInt();
327     maxIndex = Math.max(maxIndex, idIndex.orElse(-1));
328 
329     List<List<Term>> terms = new ArrayList<>();
330     for (int i = 0; i <= maxIndex; i++) {
331       terms.add(new ArrayList<Term>());
332     }
333     archiveFieldsWithIndex.stream().forEach(af -> terms.get(af.getIndex()).add(af.getTerm()));
334     // Assign default term for id column.
335     idIndex.ifPresent(idx -> terms.get(idx).add(DEFAULT_ID_TERM));
336     return terms;
337   }

339   /**
340    * Get default values mapped to their {@link Term} (if any).
341    *
342    * @return default values or {@code Optional.empty()} if none
343    */
344   public Optional<Map<Term, String>> getDefaultValues() {
345     //check if there is default value(s) defined (look for absence of index)
346     Map<Term, String> defaultValues = fields.values().stream()
347             .filter(af -> StringUtils.isNotBlank(af.getDefaultValue()))
348             .collect(Collectors.toMap(ArchiveField::getTerm, ArchiveField::getDefaultValue));
349     return defaultValues.isEmpty() ? Optional.empty() : Optional.of(defaultValues);
350   }

416   public File getLocationFile() {
417     File dataFile;
418     if (archive != null) {
419       if (getLocation() == null) {
420         // use only archive
421         dataFile = archive.getLocation();
422       } else if (getLocation().startsWith("/")) {
423         // absolute already
424         dataFile = new File(getLocation());
425       } else {
426         // use source file relative to archive dir
427         Path archiveLocation = archive.getLocation().toPath();
428         File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
429         dataFile = new File(directory, getLocation());
430       }
431     } else {
432       dataFile = new File(getLocation());
433     }
434 
435     return dataFile;
436   }

472   private Reader getReader(boolean sorted) throws IOException {
473     // ArchiveFile location, or Archive in case this is a fake single-file "archive".
474     File file = getLocationFile() != null ? getLocationFile() : getArchive().getLocation();
475     if (sorted) {
476       file = getLocationFileSorted(file);
477     }
478     return Files.newBufferedReader(file.toPath(), Charset.forName(getEncoding()));
479   }
480 
481   /**
482    * Get a {@link ClosableIterator} over the records in this file, optionally replacing nulls and entities.
483    *
484    * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
485    * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
486    */
487   public ClosableIterator<Record> iterator(boolean replaceNulls, boolean replaceEntities) {
488     try {
489       TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(false),
490           getFieldsTerminatedByChar(), getLinesTerminatedBy(), getFieldsEnclosedBy(),
491           areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
492       return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
493     } catch (IOException e) {
494       throw new UnsupportedArchiveException(e);
495     }
496   }

498   /**
499    * Build an iterator pointing to the sorted tabular file.
500    * The sorted tabular file is also assumed to have been normalized.
501    *
502    * @param replaceNulls
503    * @param replaceEntities
504    *
505    * @return
506    *
507    * @throws IOException
508    */
509   protected ClosableIterator<Record> sortedIterator(boolean replaceNulls, boolean replaceEntities) throws IOException {
510     TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(true),
511             getFieldsTerminatedByChar(), TabularFileNormalizer.NORMALIZED_END_OF_LINE, getFieldsEnclosedBy(),
512             areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
513     return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
514   }
