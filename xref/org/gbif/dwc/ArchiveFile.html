68    /**
69     * Comparator to sort by column index, then term name.
70     */
71    private static final Comparator<ArchiveField> AF_IDX_COMPARATOR = (o1, o2) -> {
72      if (o1.getIndex() == null && o2.getIndex() == null) {
73        return 0;
74      } else if (o1.getIndex() == null) {
75        return -1;
76      } else if (o2.getIndex() == null) {
77        return 1;
78      } else if (o1.getIndex().equals(o2.getIndex())) {
79        return o1.getTerm().qualifiedName().compareTo(o2.getTerm().qualifiedName());
80      } else {
81        return o1.getIndex().compareTo(o2.getIndex());
82      }
83    };

127     }
128     validate();

131   protected void validate() throws UnsupportedArchiveException {
132     if (getFirstLocation() == null) {
133       throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a location");
134     }
135     for (File f : getLocationFiles()) {
136       if (!f.exists()) {
137         throw new UnsupportedArchiveException("DwC-A data file »" + title + "« does not exist");
138       }
139     }
140     if (encoding == null) {
141       throw new UnsupportedArchiveException("DwC-A data file »" + title + "« requires a character encoding");
142     }
143   }

145   /**
146    * Normalizes and sorts the data file, so it is suitable for in-order iteration as a core or extension file.
147    *
148    * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
149    */
150   /*
151    * Sorting implies a normalization phase to ensure we sort the file properly.
152    * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
153    * we want to sort.
154    *
155    * This method is synchronized to prevent multiple threads trying to normalize/sort a file at the same time,
156    * and uses an advisory lock to avoid conflict between multiple processes.
157    */
158   protected synchronized boolean normalizeAndSort() throws IOException {
159     List<File> filesToSort = getLocationFiles();
160     File sortedFile = getLocationFileSorted(getFirstLocationFile());
161 
162     File lockFile = getLocationLockFile(getFirstLocationFile());
163     RandomAccessFile lockFileRA = new RandomAccessFile(lockFile, "rw");
164 
165     FileLock lock = lockFileRA.getChannel().tryLock();
166     if (lock == null) {
167       LOG.warn("Another process has locked this DWCA for initialization; waiting until the lock is released.");
168       lock = lockFileRA.getChannel().lock();
169       LOG.warn("Other process has released lock; lock taken, proceeding.");
170     }
171 
172     try {
173       // If we already sorted the file and its source didn't change we can avoid doing it again
174       long youngestFileTime = Long.MIN_VALUE;
175       for (File f : filesToSort) {
176         youngestFileTime = Math.max(youngestFileTime, f.lastModified());
177       }
178       if (sortedFile.exists()) {
179         if (sortedFile.lastModified() > youngestFileTime) {
180           LOG.debug("File {} is already sorted ({}B)", sortedFile, sortedFile.length());
181           return false;
182         } else {
183           LOG.info("Deleting existing (old) sorted file {} ({}B)", sortedFile, sortedFile.length());
184           sortedFile.delete();
185         }
186       }
187 
188       List<File> normalizedFiles = normalizeIfRequired();
189       if (normalizedFiles != null) {
190         filesToSort = normalizedFiles;
191       }
192 
193       // Sort to this temporary location, then move the file in place once the sort is completed.
194       // (Avoids leaving half-sorted files lying around.)
195       File temporarySortedFile = getLocationTempFileSorted(getFirstLocationFile());
196 
197       FILE_UTILS.sort(filesToSort, temporarySortedFile, getEncoding(),
198           getId().getIndex(), getFieldsTerminatedBy(), getFieldsEnclosedBy(),
199           TabularFileNormalizer.NORMALIZED_END_OF_LINE, getIgnoreHeaderLines());
200 
201       if (normalizedFiles != null) {
202         for (File f : normalizedFiles) {
203           Files.deleteIfExists(f.toPath());
204         }
205       }
206 
207       Files.move(temporarySortedFile.toPath(), sortedFile.toPath());
208       return true;
209     } finally {
210       lock.release();
211       lockFileRA.close();
212     }
213   }

215   /**
216    * Apply file normalization if required.
217    **
218    * @return normalizedFile or null if normalization was not applied
219    *
220    * @throws IOException
221    */
222   protected List<File> normalizeIfRequired() throws IOException {
223     // If the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
224     // we can skip normalization
225     boolean normalizationRequired = ! TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(getLinesTerminatedBy())
226         || getFieldsEnclosedBy() != null;
227 
228     if (normalizationRequired) {
229       List<File> normalizedFiles = new ArrayList<>();
230       for (File f : getLocationFiles()) {
231         File normalizedFile = getLocationFileNormalized(f);
232         TabularFileNormalizer.normalizeFile(f.toPath(), normalizedFile.toPath(),
233           Charset.forName(getEncoding()), getFieldsTerminatedByChar(),
234           getLinesTerminatedBy(), getFieldsEnclosedBy());
235         normalizedFiles.add(normalizedFile);
236       }
237       return normalizedFiles;
238     }
239     return null;
240   }

326   /**
327    * Generates a list of {@link Term}s mapped to each column in the underlying file.
328    *
329    * A list may be empty when no Term is mapped at a specific position, or may contain more than one Term where
330    * the same column is mapped more than once.
331    *
332    * The length of the list is the maximum index used in the {@link ArchiveField}s + 1.
333    *
334    * @return Data structure representing the header of the file.
335    */
336   public List<List<Term>> getHeader() {
337     List<ArchiveField> archiveFieldsWithIndex = getFieldsSorted()
338             .stream().filter(af -> af.getIndex() != null)
339             .collect(Collectors.toList());
340 
341     Optional<Integer> idIndex = id != null ? Optional.of(id.getIndex()) : Optional.empty();
342 
343     if (archiveFieldsWithIndex.isEmpty() && !idIndex.isPresent()){
344       return new ArrayList<>();
345     }
346 
347     int maxIndex = archiveFieldsWithIndex.stream().mapToInt(ArchiveField::getIndex).max().getAsInt();
348     maxIndex = Math.max(maxIndex, idIndex.orElse(-1));
349 
350     List<List<Term>> terms = new ArrayList<>();
351     for (int i = 0; i <= maxIndex; i++) {
352       terms.add(new ArrayList<Term>());
353     }
354     archiveFieldsWithIndex.stream().forEach(af -> terms.get(af.getIndex()).add(af.getTerm()));
355     // Assign default term for id column.
356     idIndex.ifPresent(idx -> terms.get(idx).add(DEFAULT_ID_TERM));
357     return terms;
358   }

361    * Get default values mapped to their {@link Term} (if any).
362    *
363    * @return default values or {@code Optional.empty()} if none
364    */
365   public Optional<Map<Term, String>> getDefaultValues() {
366     //check if there is default value(s) defined (look for absence of index)
367     Map<Term, String> defaultValues = fields.values().stream()
368             .filter(af -> StringUtils.isNotBlank(af.getDefaultValue()))
369             .collect(Collectors.toMap(ArchiveField::getTerm, ArchiveField::getDefaultValue));
370     return defaultValues.isEmpty() ? Optional.empty() : Optional.of(defaultValues);
371   }

440   public File getFirstLocationFile() {
441     File dataFile;
442     if (archive != null) {
443       if (getFirstLocation() == null) {
444         // use only archive
445         dataFile = archive.getLocation();
446       } else if (getFirstLocation().startsWith("/")) {
447         // absolute already
448         dataFile = new File(getFirstLocation());
449       } else {
450         // use source file relative to archive dir
451         Path archiveLocation = archive.getLocation().toPath();
452         File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
453         dataFile = new File(directory, getFirstLocation());
454       }
455     } else {
456       dataFile = new File(getFirstLocation());
457     }
458 
459     return dataFile;
460   }

476   /**
477    * Get Files for each location.  Note archive files may use more than one location.
478    */
479   public List<File> getLocationFiles() {
480     if (archive != null) {
481       if (getLocations() == null) {
482         // use only archive
483         return Collections.singletonList(archive.getLocation());
484       } else {
485         List<File> locations = new ArrayList<>();
486         for (String s : getLocations()) {
487           if (s.startsWith("/")) {
488             // absolute already
489             locations.add(new File(s));
490           } else {
491             // use source file relative to archive dir
492             Path archiveLocation = archive.getLocation().toPath();
493             File directory = Files.isDirectory(archiveLocation) ? archiveLocation.toFile() : archiveLocation.getParent().toFile();
494             locations.add(new File(directory, s));
495           }
496         }
497         return locations;
498       }
499     } else {
500       return Collections.singletonList(new File(getFirstLocation()));
501     }
502   }

534   private Reader getReader(boolean sorted) throws IOException {
535     // ArchiveFile location, or Archive in case this is a fake single-file "archive".
536     File file = getFirstLocationFile() != null ? getFirstLocationFile() : getArchive().getLocation();
537     if (sorted) {
538       file = getLocationFileSorted(file);
539     }
540     return Files.newBufferedReader(file.toPath(), Charset.forName(getEncoding()));
541   }

543   /**
544    * Get a {@link ClosableIterator} over the records in this file, optionally replacing nulls and entities.
545    *
546    * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
547    * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
548    */
549   public ClosableIterator<Record> iterator(boolean replaceNulls, boolean replaceEntities) {
550     try {
551       // Use the sortedIterator (sorted into a single file) if there are multiple files.
552       if (getLocations().size() > 1) {
553         return sortedIterator(replaceNulls, replaceEntities);
554       }
555 
556       TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(false),
557           getFieldsTerminatedByChar(), getLinesTerminatedBy(), getFieldsEnclosedBy(),
558           areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
559       return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
560     } catch (IOException e) {
561       throw new UnsupportedArchiveException(e);
562     }
563   }

565   /**
566    * Build an iterator pointing to the sorted tabular file.
567    * The sorted tabular file is also assumed to have been normalized.
568    *
569    * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
570    * @param replaceEntities if true HTML & XML entities in record values will be replaced with the interpreted value.
571    */
572   protected ClosableIterator<Record> sortedIterator(boolean replaceNulls, boolean replaceEntities) throws IOException {
573     TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(getReader(true),
574             getFieldsTerminatedByChar(), TabularFileNormalizer.NORMALIZED_END_OF_LINE, getFieldsEnclosedBy(),
575             areHeaderLinesIncluded(), getLinesToSkipBeforeHeader());
576     return new DwcRecordIterator(tabularFileReader, getId(), getFields(), getRowType(), replaceNulls, replaceEntities);
577   }
