36      this.extItSupplier = extItSupplier;
37    }
38  
39    /**
40     * Get a new {@link ClosableIterator} of {@link StarRecord}.

43     * @throws IOException
44     */
45    public ClosableIterator<StarRecord> iterator() throws IOException {
