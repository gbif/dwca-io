3   import java.util.regex.Pattern;

5   import org.apache.commons.lang3.StringEscapeUtils;

7   public class CleanUtils {
8     private static final Pattern NULL_REPL = Pattern.compile("^\\s*(null|\\\\N)?\\s*$", Pattern.CASE_INSENSITIVE);
9   
10    private CleanUtils() {
11    }
12  
13    /**
14     * Does basic entity replacements if requested to string values.
15     * @param value the original string
16     * @param nulls if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
17     * @param entities if true replaces html4, xml and numerical entities with their unicode character
18     */
19    public static String clean(String value, boolean nulls, boolean entities) {
20      if (value == null || (nulls && NULL_REPL.matcher(value).find()) ) {
21        return null;
22      }
23      return entities ? StringEscapeUtils.unescapeHtml4(value) : value;
24    }

26  }
