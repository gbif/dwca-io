1   package org.gbif.dwc.record;

3   import org.gbif.dwc.terms.Term;
4   
5   import java.util.ArrayList;
6   import java.util.Collection;
7   import java.util.HashMap;
8   import java.util.Iterator;
9   import java.util.List;
10  import java.util.Map;
11  import java.util.Set;
12  
13  public class StarRecordImpl implements StarRecord {
14  
15    private Record core;
16    private final Map<Term, List<Record>> extensions;
17  
18    public StarRecordImpl(Collection<Term> extensions) {
19      this.extensions = new HashMap<Term, List<Record>>();
20      for (Term rowType : extensions) {
21        this.extensions.put(rowType, new ArrayList<Record>());
22      }
23    }
24  
25    public void addRecord(Term rowType, Record record) {
26      if (!extensions.containsKey(rowType)) {
27        throw new IllegalArgumentException("RowType not supported");
28      }
29      extensions.get(rowType).add(record);
30    }
31  
32    /**
33     * @return the core record
34     */
35    public Record core() {
36      return core;
37    }
38  
39    public boolean hasExtension(Term rowType) {
40      return extensions.containsKey(rowType) && !extensions.get(rowType).isEmpty();
41    }
42  
43    /**
44     * Retrieves all extension records of a specific extension.
45     * If the requested extension is not mapped null will be returned.
46     *
47     * @param rowType the Term representing the rowType
48     *
49     * @return possibly empty list of extension record or null if extension is not mapped at all
50     */
51    public List<Record> extension(Term rowType) {
52      return extensions.get(rowType);
53    }
54  
55    /**
56     * Retrieves all extension records related to the core record across all extensions as a map.
57     */
58    public Map<Term, List<Record>> extensions() {
59      return extensions;
60    }
61  
62    /**
63     * Get a new iterator over all extension records, no matter to which extension they belong.
64     *
65     * @return the ExtensionRecord iterator, reusing the same instance for each call
66     */
67    public Iterator<Record> iterator() {
68      List<Record> records = new ArrayList<Record>();
69      for (List<Record> recs : extensions.values()) {
70        records.addAll(recs);
71      }
72      return records.iterator();
73    }
74  
75    public void newCoreRecord(Record core) {
76      this.core = core;
77      for (List<Record> lists : extensions.values()) {
78        lists.clear();
79      }
80    }
81  
82    /**
83     * @return set of extension rowTypes associated with this star record
84     */
85    public Set<Term> rowTypes() {
86      return extensions.keySet();
87    }
88  
89    /**
90     * @return the number of associated extension records across all rowTypes
91     */
92    public int size() {
93      int x = 0;
94      for (List<Record> recs : extensions.values()) {
95        x += recs.size();
96      }
97      return x;
98    }
99  
100   @Override
101   public String toString() {
102     return "StarRecord with core "+core.toString()+" and extensions "+extensions;
103   }
104 }
