1   package org.gbif.dwc.digester;

3   import org.gbif.dwc.extensions.ExtensionProperty;
4   import org.gbif.dwc.extensions.VocabulariesManager;
5   import org.gbif.dwc.extensions.Vocabulary;
6   
7   import java.net.URL;
8   
9   import org.apache.commons.digester3.Rule;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  import org.xml.sax.Attributes;
13  
14  /**
15   * This will call the root of the stack to find the url2thesaurus, and then set the appropriate thesaurus on the
16   * extension. Namespaces are completely ignored. The "thesaurus" attribute is searched for and if found, the thesaurus
17   * is set if found.
18   */
19  public class ThesaurusHandlingRule extends Rule {
20  
21    private static final Logger LOG = LoggerFactory.getLogger(ThesaurusHandlingRule.class);
22  
23    public static final String ATTRIBUTE_THESAURUS = "thesaurus";
24  
25    private VocabulariesManager vocabManager;
26  
27    public ThesaurusHandlingRule(VocabulariesManager vocabManager) {
28      super();
29      this.vocabManager = vocabManager;
30    }
31  
32    @Override
33    public void begin(String namespace, String name, Attributes attributes) throws Exception {
34  
35      for (int i = 0; i < attributes.getLength(); i++) {
36        if (ThesaurusHandlingRule.ATTRIBUTE_THESAURUS.equals(attributes.getQName(i))) {
37          Vocabulary tv = null;
38          try {
39            URL vocabURL = new URL(attributes.getValue(i));
40            tv = vocabManager.get(vocabURL);
41          } catch (Exception e) {
42            LOG.error("Could not load vocabulary with location {}: {}", new Object[]{attributes.getValue(i), e.getMessage(), e});
43          }
44  
45          if (tv != null) {
46            Object extensionPropertyAsObject = getDigester().peek();
47            if (extensionPropertyAsObject instanceof ExtensionProperty) {
48              ExtensionProperty eProperty = (ExtensionProperty) extensionPropertyAsObject;
49              eProperty.setVocabulary(tv);
50              LOG.debug("Vocabulary with URI[{}] added to extension property", tv.getUri());
51            }
52          } else {
53            LOG.warn("No vocabulary exists for the URL[{}]", attributes.getValue(i));
54          }
55  
56          break; // since we found the attribute
57        }
58      }
59    }
60  }
