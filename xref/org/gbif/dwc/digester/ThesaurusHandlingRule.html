27  /**
28   * This will call the root of the stack to find the url2thesaurus, and then set the appropriate thesaurus on the
29   * extension. Namespaces are completely ignored. The "thesaurus" attribute is searched for and if found, the thesaurus
30   * is set if found.
31   */
32  public class ThesaurusHandlingRule extends Rule {
33  
34    private static final Logger LOG = LoggerFactory.getLogger(ThesaurusHandlingRule.class);
35  
36    public static final String ATTRIBUTE_THESAURUS = "thesaurus";
37  
38    private VocabulariesManager vocabManager;
39  
40    public ThesaurusHandlingRule(VocabulariesManager vocabManager) {
41      super();
42      this.vocabManager = vocabManager;
43    }
44  
45    @Override
46    public void begin(String namespace, String name, Attributes attributes) throws Exception {
47  
48      for (int i = 0; i < attributes.getLength(); i++) {
49        if (ThesaurusHandlingRule.ATTRIBUTE_THESAURUS.equals(attributes.getQName(i))) {
50          Vocabulary tv = null;
51          try {
52            URL vocabURL = new URL(attributes.getValue(i));
53            tv = vocabManager.get(vocabURL);
54          } catch (Exception e) {
55            LOG.error("Could not load vocabulary with location {}: {}", new Object[]{attributes.getValue(i), e.getMessage(), e});
56          }
57  
58          if (tv != null) {
59            Object extensionPropertyAsObject = getDigester().peek();
60            if (extensionPropertyAsObject instanceof ExtensionProperty) {
61              ExtensionPropertyg/gbif/dwc/extensions/ExtensionProperty.html#ExtensionProperty">ExtensionProperty eProperty = (ExtensionProperty) extensionPropertyAsObject;
62              eProperty.setVocabulary(tv);
63              LOG.debug("Vocabulary with URI[{}] added to extension property", tv.getUri());
64            }
65          } else {
66            LOG.warn("No vocabulary exists for the URL[{}]", attributes.getValue(i));
67          }
68  
69          break; // since we found the attribute
70        }
71      }
72    }
73  }
