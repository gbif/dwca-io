86    /**
87     * Get a {@link ClosableIterator} on the provided {@link ArchiveFile}.
88     *
89     * @param source
90     * @param replaceNulls    if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
91     * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
92     */
93    public static ClosableIterator<Record> iterator(ArchiveFile source, boolean replaceNulls, boolean replaceEntities) throws IOException {
94      Objects.requireNonNull(source, "source ArchiveFile shall be provided");
95      TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(
96              Files.newBufferedReader(source.getLocationFile() != null ? source.getLocationFile().toPath() : source.getArchive().getLocation().toPath()),
97              getFieldsTerminatedBy(source.getFieldsTerminatedBy()), source.getLinesTerminatedBy(), source.getFieldsEnclosedBy(),
98              isHeaderLineIncluded(source.getIgnoreHeaderLines()), getLineToSkipBeforeHeader(source.getIgnoreHeaderLines()));
99      return new DwcRecordIterator(tabularFileReader, source.getId(), source.getFields(), source.getRowType(), replaceNulls, replaceEntities);
100   }

102   /**
103    * Get the fieldsTerminatedBy as char or throw exception.
104    *
105    * @param fieldsTerminatedBy
106    *
107    * @return
108    */
109   private static char getFieldsTerminatedBy(String fieldsTerminatedBy) {
110     Objects.requireNonNull(fieldsTerminatedBy, "fieldsTerminatedBy shall be provided");
111     if (fieldsTerminatedBy.length() != 1) {
112       throw new IllegalArgumentException();
113     }
114     return fieldsTerminatedBy.charAt(0);
115   }

135   private static Integer getLineToSkipBeforeHeader(Integer ignoreHeaderLines) {
136     if (ignoreHeaderLines != null && ignoreHeaderLines > 1) {

156   /**
157    * Prepare an {@link Archive} into a {@link NormalizedDwcArchive} which allows to get {@link StarRecord} {@link
158    * ClosableIterator}.
159    * This method will initiate the normalization process. This process can take some times depending on the size of
160    * files and number of extension.
161    *
162    * @param archive source archive
163    * @param replaceNulls if true replaces common, literal NULL values with real nulls, e.g. "\N" or "NULL"
164    * @param replaceEntities if true html & xml entities in record values will be replaced with the interpreted value.
165    *
166    * @return new {@link NormalizedDwcArchive} instance
167    *
168    * @throws IOException
169    */
170   public static NormalizedDwcArchive prepareArchive(final Archive archive, boolean replaceNulls, boolean replaceEntities) throws IOException {
171 
172     Objects.requireNonNull(archive, "archive shall be provided");
173     Objects.requireNonNull(archive.getCore(), "The archive shall have a core");
174 
175     //if no extensions are provided we don't need to sort the file
176     if (archive.getExtensions().isEmpty()) {
177       return new NormalizedDwcArchive(() -> iterator(archive.getCore(), replaceNulls, replaceEntities));
178     }
179 
180     //otherwise, we need to sort core + extensions
181     normalizeAndSortArchiveFiles(archive);
182 
183     return new NormalizedDwcArchive(() -> buildSortedIterator(archive.getCore(), replaceNulls, replaceEntities),
184             () -> buildSortedIteratorExt(archive, replaceNulls, replaceEntities));
185   }
186 
187   /**
188    * Sorts all files according in the {@link Archive} so that we can easily iterate over all files at once.
189    */
190   private static void normalizeAndSortArchiveFiles(Archive archive) throws IOException {
191     normalizeAndSort(archive.getCore());
192     for (ArchiveFile archiveFile : archive.getExtensions()) {
193       normalizeAndSort(archiveFile);
194     }
195   }

197   /**
198    * Sort a single {@link ArchiveFile}. Sorting implies a normalization phase to ensure we sort the file properly.
199    * Note that the file will not be sorted if the sorted file is already there and its date is later than the file
200    * we want to sort.
201    *
202    * @param archiveFile
203    *
204    * @return the file was sorted or not. If the file was not sorted it simply means it was not required.
205    *
206    * @throws IOException
207    */
208   @VisibleForTesting
209   protected static boolean normalizeAndSort(ArchiveFile archiveFile) throws IOException {
210 
211     File fileToSort = archiveFile.getLocationFile();
212     File sortedFile = ArchiveFile.getLocationFileSorted(archiveFile.getLocationFile());
213 
214     //if we already sorted the file and its source didn't change we can avoid doing it again
215     if (sortedFile.exists() && Files.getLastModifiedTime(sortedFile.toPath()).toInstant().isAfter(
216             Files.getLastModifiedTime(fileToSort.toPath()).toInstant())) {
217       return false;
218     }
219 
220     File normalizedFile = normalizeIfRequired(archiveFile);
221     if (normalizedFile != null) {
222       fileToSort = normalizedFile;
223     }
224 
225     F_UTILS.sort(fileToSort, ArchiveFile.getLocationFileSorted(archiveFile.getLocationFile()), archiveFile.getEncoding(),
226             archiveFile.getId().getIndex(), archiveFile.getFieldsTerminatedBy(), archiveFile.getFieldsEnclosedBy(),
227             TabularFileNormalizer.NORMALIZED_END_OF_LINE, archiveFile.getIgnoreHeaderLines());
228 
229     if (normalizedFile != null) {
230       Files.deleteIfExists(normalizedFile.toPath());
231     }
232 
233     return true;
234   }

236   /**
237    * Apply file normalization if required.
238    *
239    * @param archiveFile
240    *
241    * @return normalizedFile or null if normalization was not applied
242    *
243    * @throws IOException
244    */
245   @VisibleForTesting
246   protected static File normalizeIfRequired(ArchiveFile archiveFile) throws IOException {
247     //if the linesTerminatedBy used is the same as TabularFileNormalizer and no quoted cells are used
248     //we can skip normalization
249     boolean normalizationRequired =
250             !TabularFileNormalizer.NORMALIZED_END_OF_LINE.equals(archiveFile.getLinesTerminatedBy()) ||
251                     archiveFile.getFieldsEnclosedBy() != null;
252 
253     if (normalizationRequired) {
254       File normalizedFile = getLocationFileNormalized(archiveFile.getLocationFile());
255       TabularFileNormalizer.normalizeFile(archiveFile.getLocationFile().toPath(), normalizedFile.toPath(),
256               Charset.forName(archiveFile.getEncoding()), getFieldsTerminatedBy(archiveFile.getFieldsTerminatedBy()),
257               archiveFile.getLinesTerminatedBy(), archiveFile.getFieldsEnclosedBy());
258       return normalizedFile;
259     }
260     return null;
261   }
262 
263   private static File getLocationFileNormalized(File location) {
264     return new File(location.getParentFile(), location.getName() + "-normalized");
265   }

267   /**
268    * Build an iterator (pointing to the sorted tabular file) for each extensions of the {@link Archive}.
269    *
270    * @param archive
271    * @param replaceNulls
272    * @param replaceEntities
273    *
274    * @return
275    *
276    * @throws IOException
277    */
278   private static Map<Term, ClosableIterator<Record>> buildSortedIteratorExt(Archive archive,
279                                                                             boolean replaceNulls, boolean replaceEntities) throws IOException {
280     Map<Term, ClosableIterator<Record>> extensionIts = new HashMap<>();
281     for (ArchiveFile ext : archive.getExtensions()) {
282       extensionIts.put(ext.getRowType(), buildSortedIterator(ext, replaceNulls, replaceEntities));
283     }
284     return extensionIts;
285   }

287   /**
288    * Build an iterator on top of the provided {@link ArchiveFile} pointing to the sorted tabular file.
289    * The sorted tabular file is also assumed to have been normalized.
290    *
291    * @param af
292    * @param replaceNulls
293    * @param replaceEntities
294    *
295    * @return
296    *
297    * @throws IOException
298    */
299   private static ClosableIterator<Record> buildSortedIterator(ArchiveFile af,
300                                                               boolean replaceNulls, boolean replaceEntities) throws IOException {
301     TabularDataFileReader<List<String>> tabularFileReader = TabularFiles.newTabularFileReader(
302             Files.newBufferedReader(getLocationFileSorted(af.getLocationFile()).toPath()),
303             getFieldsTerminatedBy(af.getFieldsTerminatedBy()),
304             TabularFileNormalizer.NORMALIZED_END_OF_LINE, af.getFieldsEnclosedBy(),
305             isHeaderLineIncluded(af.getIgnoreHeaderLines()), getLineToSkipBeforeHeader(af.getIgnoreHeaderLines()));
306     return new DwcRecordIterator(tabularFileReader, af.getId(), af.getFields(), af.getRowType(), replaceNulls, replaceEntities);
307   }
