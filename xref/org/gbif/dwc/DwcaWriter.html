123   private void addRowType(Term rowType) throws IOException {
124     terms.put(rowType, new ArrayList<>());
125 
126     String dfn = dataFileName(rowType);
127     dataFileNames.put(rowType, dfn);
128     File df = new File(dir, dfn);
129     FileUtils.forceMkdir(df.getParentFile());
130     OutputStream out = new FileOutputStream(df);
131     TabWriterbWriter">TabWriter wr = new TabWriter(out);
132     writers.put(rowType, wr);

160   private void writeRow(Map<Term, String> rowMap, Term rowType) throws IOException {
161     TabWriter writer = writers.get(rowType);
162     List<Term> columns = terms.get(rowType);
163     if (useHeaders && !headersOut.contains(rowType)) {
164       // write header row
165       writeHeader(writer, rowType, columns);
166     }
167 
168     // make sure coreId is not null for extensions
169     if (coreRowType != rowType && coreId == null) {
170       LOG.warn("Adding an {} extension record to a core without an Id! Skip this record", rowType);
171 
172     } else {
173       String[] row = new String[columns.size() + 1];
174       row[0] = coreId;
175       for (Map.Entry<Term, String> conceptTermStringEntry : rowMap.entrySet()) {
176         int column = 1 + columns.indexOf(conceptTermStringEntry.getKey());
177         row[column] = conceptTermStringEntry.getValue();
178       }
179       writer.write(row);
180     }
181   }
182 
183   private void writeHeader(TabWriter writer, Term rowType, List<Term> columns) throws IOException {
184     int idx = 0;
185     String[] row = new String[columns.size() + 1];
186     Term idTerm;
187     if (DwcTerm.Taxon == coreRowType) {
188       idTerm = DwcTerm.taxonID;
189     } else if (DwcTerm.Occurrence == coreRowType) {
190       idTerm = DwcTerm.occurrenceID;
191     } else if (DwcTerm.Identification == coreRowType) {
192       idTerm = DwcTerm.identificationID;
193     } else if (DwcTerm.Event == coreRowType) {
194       idTerm = DwcTerm.eventID;
195     } else {
196       // default to generic dc identifier for id column
197       idTerm = DcTerm.identifier;
198     }
199     row[idx] = idTerm.simpleName();
200 
201     for (Term term : columns) {
202       idx++;
203       row[idx] = term.simpleName();
204     }
205     writer.write(row);
206 
207     headersOut.add(rowType);
208   }

211   /**
212    * Add a single value for the current core record.
213    * Calling this method requires that #newRecord() has been called at least once,
214    * otherwise an IllegalStateException is thrown.
215    *
216    * @param term
217    * @param value
218    */
219   public void addCoreColumn(Term term, String value) {
220     // ensure we do not overwrite the coreIdTerm if one is defined
221     if (coreIdTerm != null && coreIdTerm.equals(term)) {
222       throw new IllegalStateException("You cannot add a term that was specified as coreId term");
223     }
224 
225     List<Term> coreTerms = terms.get(coreRowType);
226     if (!coreTerms.contains(term)) {
227       if (useHeaders && recordNum > 1) {
228         throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
229       }
230       coreTerms.add(term);
231     }
232     try {
233       coreRow.put(term, value);
234     } catch (NullPointerException e) {
235       // no core record has been started yet
236       throw new IllegalStateException("No core record has been created yet. Call newRecord() at least once");
237     }
238   }

290    * Add a default value to a term of the provided rowType.
291    *
292    * @param rowType
293    * @param term
294    * @param defaultValue
295    */
296   public void addDefaultValue(Term rowType, Term term, String defaultValue) {
297 
298     if (!defaultValues.containsKey(rowType)) {
299       defaultValues.put(rowType, new HashMap<>());
300     }
301     Map<Term, String> currentDefaultValues = defaultValues.get(rowType);
302     if (currentDefaultValues.containsKey(term)) {
303       throw new IllegalStateException("The default value of term " + term + " is already defined");
304     }
305     currentDefaultValues.put(term, defaultValue);
306   }

340   /**
341    * Add an extension record associated with the current core record.
342    *
343    * @param rowType
344    * @param row
345    * @throws IOException
346    */
347   public void addExtensionRecord(Term rowType, Map<Term, String> row) throws IOException {
348     // make sure we know the extension rowtype
349     if (!terms.containsKey(rowType)) {
350       addRowType(rowType);
351     }
352 
353     // make sure we know all terms
354     List<Term> knownTerms = terms.get(rowType);
355     final boolean isFirst = knownTerms.isEmpty();
356     for (Term term : row.keySet()) {
357       if (!knownTerms.contains(term)) {
358         if (useHeaders && !isFirst) {
359           throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
360         }
361         knownTerms.add(term);
362       }
363     }
364 
365     // write extension record
366     writeRow(row, rowType);
367   }

431   private void addConstituents() throws IOException {
432     if (!constituents.isEmpty()) {
433       File ddir = new File(dir, Archive.CONSTITUENT_DIR);
434       ddir.mkdirs();
435       for (Map.Entry<String, String> de : constituents.entrySet()) {
436         writeMetadata(de.getValue(), new File(ddir, de.getKey() + ".xml"));
437       }
438     }
439   }

441   private void addMeta() throws IOException {
442     File metaFile = new File(dir, Archive.META_FN);
443 
444     Archiveive">Archive arch = new Archive();
445     if (metadataLocation != null) {
446       arch.setMetadataLocation(metadataLocation);
447     }
448     arch.setCore(buildArchiveFile(arch, coreRowType, coreIdTerm));
449     for (Term rowType : this.terms.keySet()) {
450       if (!coreRowType.equals(rowType)) {
451         arch.addExtension(buildArchiveFile(arch, rowType, null));
452       }
453     }
454     MetaDescriptorWriter.writeMetaFile(metaFile, arch);
455   }

457   /**
458    * Build an ArchiveFile for core or extension(s).
459    *
460    * @param archive
461    * @param rowType
462    * @param idTerm  the term of the id column, may be null
463    * @return
464    */
465   private ArchiveFile buildArchiveFile(Archive archive, Term rowType, Term idTerm) {
466     ArchiveFile af = ArchiveFile.buildTabFile();
467     af.setArchive(archive);
468     af.addLocation(dataFileNames.get(rowType));
469 
470     af.setEncoding("utf-8");
471     af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
472     af.setRowType(rowType);
473 
474     ArchiveFieldchiveField">ArchiveField id = new ArchiveField();
475     id.setIndex(0);
476     af.setId(id);
477     // always use the index 0 for idTerm
478     if (idTerm != null) {
479       af.addField(buildArchiveField(0, idTerm));
480     }
481 
482     Map<Term, String> termDefaultValueMap = defaultValues.get(rowType);
483     Map<Term, String> termMultiValueDelimiterMap = multiValueDelimiter.get(rowType);
484     List<Term> rowTypeTerms = terms.get(rowType);
485     int idx = 0;
486     String defaultValue;
487     String mvDelim;
488     for (Term c : rowTypeTerms) {
489       idx++;
490       defaultValue = (termDefaultValueMap != null ? termDefaultValueMap.get(c) : null);
491       mvDelim = (termMultiValueDelimiterMap != null ? termMultiValueDelimiterMap.get(c) : null);
492       af.addField(buildArchiveField(idx, c, defaultValue, mvDelim));
493     }
494 
495     // check if default values are provided for this rowType
496     if (termDefaultValueMap != null) {
497       for (Term t : termDefaultValueMap.keySet()) {
498         if (!rowTypeTerms.contains(t)) {
499           af.addField(buildArchiveFieldDefaultValue(t, termDefaultValueMap.get(t)));
500         }
501       }
502     }
503 
504     return af;
505   }
