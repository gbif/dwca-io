24    @Test
25    public void testIteratorDwc() throws Exception {
26      File csv = FileUtils.getClasspathFile("null_data.txt");
27      CSVReader reader = new CSVReader(csv, "utf8", "\t", null, 1);
28  
29      Archive arch = new Archive();
30      ArchiveField id = new ArchiveField(0, null, null, null);
31      ArchiveField taxStatus = new ArchiveField(1, DwcTerm.taxonomicStatus, "a", null);
32      ArchiveField dcmodified = new ArchiveField(6, DcTerm.modified, "mod", null);
33  
34      ArchiveFile core = mock(ArchiveFile.class);
35      when(core.getCSVReader()).thenReturn(reader);
36      when(core.getId()).thenReturn(id);
37      when(core.getArchive()).thenReturn(arch);
38      when(core.getField(Matchers.<Term>any())).thenReturn(taxStatus);
39      Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
40      fields.put(DwcTerm.taxonomicStatus, taxStatus);
41      fields.put(DcTerm.modified, dcmodified);
42      when(core.getFields()).thenReturn(fields);
43      when(core.hasTerm(eq(DwcTerm.taxonomicStatus))).thenReturn(true);
44      when(core.hasTerm(eq(DcTerm.modified))).thenReturn(true);
45      arch.setCore(core);
46  
47      Iterator<DarwinCoreRecord> iter = arch.iteratorDwc();
48      while (iter.hasNext()) {
49        DarwinCoreRecord rec = iter.next();
50        assertEquals("a", rec.getTaxonomicStatus());
51        assertEquals("mod", rec.getModified());
52      }
53    }

59      Archive arch = new Archive();

63      ArchiveField datasetId = new ArchiveField(1, DwcTerm.datasetID, null, null);
64      ArchiveField sciname = new ArchiveField(2, DwcTerm.scientificName, null, null);
65  
66      Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
67      fields.put(DwcTerm.taxonomicStatus, sciname);
68      fields.put(DwcTerm.datasetID, datasetId);
69  
70      Map<String, File> cons = arch.getConstituentMetadata();
