9   import java.io.File;
10  import java.io.IOException;
11  import java.util.Iterator;
12  
13  public class ScanArchiveForScientificName {
14  
15    public static void main(String[] args) throws IOException, UnsupportedArchiveException {
16      // opens csv files with headers or dwc-a direcotries with a meta.xml descriptor
17      Archive arch = ArchiveFactory
18        .openArchive(new File("/Volumes/Scratch/ecat-data-col/checklists/beac2c55-d889-4358-a414-b1db79ba3536/dwc-a"));
19  
20      // does scientific name exist?
21      if (!arch.getCore().hasTerm(DwcTerm.scientificName)) {
22        System.out.println("This application requires dwc-a with scientific names");
23        System.exit(1);
24      }
25  
26      // loop over core darwin core records
27      Iterator<DarwinCoreRecord> iter = arch.iteratorDwc();
28      DarwinCoreRecord dwc;
29      while (iter.hasNext()) {
30        dwc = iter.next();
31        if (dwc.getScientificName().startsWith("Ambispora callosa")) {
32          System.out.println(dwc.getScientificName());
33          System.out.println(dwc.getScientificNameAuthorship());
34          System.out.println(dwc);
35        }
36      }
37    }
38  
39  }
