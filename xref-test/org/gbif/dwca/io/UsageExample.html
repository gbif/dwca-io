11  import java.io.File;
12  import java.io.IOException;
13  
14  public class UsageExample {
15  
16    public static void main(String[] args) throws IOException, UnsupportedArchiveException {
17      // opens csv files with headers or dwc-a direcotries with a meta.xml descriptor
18      Archive arch = ArchiveFactory.openArchive(new File(args[0]));
19  
20      System.out.println("Reading archive from "+arch.getLocation().getAbsolutePath());
21      System.out.println("Archive of rowtype "+arch.getCore().getRowType()+" with "+arch.getExtensions().size()+" extensions");
22      // loop over star records. i.e. core with all linked extension records
23      for (StarRecord rec : arch) {
24        // print core ID + scientific name
25        System.out.println(rec.core().id()
26          + " sciname:" + rec.core().value(DwcTerm.scientificName)
27          + " bor:" + rec.core().value(DwcTerm.basisOfRecord)
28        );
29        // print out all rowTypes
30        for (Record erec : rec) {
31          // print out extension rowtype
32          System.out.println(erec.rowType() + ", id="+erec.value(DcTerm.identifier));
33        }
34      }
35    }
36  
37  }
