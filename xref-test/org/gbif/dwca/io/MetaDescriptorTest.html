48    public class SAXExtractTerms extends DefaultHandler2 {
49      private final List<String> terms;
50      public SAXExtractTerms(List<String> terms) {
51        this.terms = terms;
52      }
53  
54      @Override
55      public void startElement(String uri, String localName, String qName, Attributes atts) throws SAXException {
56        for (String attName : Lists.newArrayList("rowType", "term")) {
57          if (atts.getValue(attName) != null) {
58            terms.add(atts.getValue(attName));
59          }
60        }
61      }
62    }

64    @Test
65    public void testXml() throws Exception {
66      // read archive
67      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
68  
69      // write meta.xml
70      File tmpMeta = File.createTempFile("meta", ".xml");
71      System.out.println("Writing temporary test meta file to " + tmpMeta.getAbsolutePath());
72      MetaDescriptorWriter.writeMetaFile(tmpMeta, arch);
73  
74      // validate xml
75      System.out.println("Validate xml");
76      MetaValidator.validate(new FileInputStream(tmpMeta));
77  
78      // verify rowType & terms are URIs
79      List<String> terms = Lists.newArrayList();
80      SAXParserFactory spf = SAXParserFactory.newInstance();
81      spf.setNamespaceAware(true);
82      SAXParser saxParser = spf.newSAXParser();
83      XMLReader xmlReader = saxParser.getXMLReader();
84      xmlReader.setContentHandler(new SAXExtractTerms(terms));
85      xmlReader.parse(new InputSource(new FileInputStream(tmpMeta)));
86  
87      assertEquals(18, terms.size());
88      for (String term : terms) {
89        URI uri = URI.create(term);
90        assertNotNull(uri + " is no full URI term", uri.getScheme());
91        assertNotNull(uri + " is no full URI term", uri.getAuthority());
92        assertNotNull(uri + " is no full URI term", uri.getPath());
93      }
94    }

96    @Test
97    public void testRoundtrip() {
98      try {
99        // read archive
100       Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
101       assertNotNull(arch);
102       assertNotNull(arch.getCore());
103       assertTrue(arch.getCore().getId().getIndex() == 0);
104       assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
105       assertEquals(2, arch.getExtensions().size());
106       assertEquals("\t", arch.getCore().getFieldsTerminatedBy());
107       assertNull(arch.getCore().getField(DwcTerm.scientificName).getDelimitedBy());
108       assertEquals(";", arch.getCore().getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
109       assertEquals(NOMENCLATURAL_CODE_VOCABULARY, arch.getCore().getField(DwcTerm.nomenclaturalCode).getVocabulary());
110       assertEquals(LANGUAGE_VOCABULARY, arch.getExtension(GbifTerm.VernacularName).getField(DcTerm.language).getVocabulary());
111 
112       // write meta.xml
113       File tmpDwca = createTmpMeta(arch);
114 
115       Archive arch2 = ArchiveFactory.openArchive(tmpDwca);
116       // core props
117       ArchiveFile core = arch2.getCore();
118       assertNotNull(core);
119       assertNotNull(core.getId());
120       assertTrue(core.hasTerm(DwcTerm.scientificName));
121       assertEquals("DarwinCore.txt", core.getLocation());
122       assertEquals("\t", core.getFieldsTerminatedBy());
123       assertNull(core.getField(DwcTerm.scientificName).getDelimitedBy());
124       assertEquals(";", core.getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
125       assertEquals(NOMENCLATURAL_CODE_VOCABULARY, core.getField(DwcTerm.nomenclaturalCode).getVocabulary());
126 
127       for (ArchiveField f : arch.getCore().getFields().values()) {
128         assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
129         assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
130       }
131 
132       // extensions props
133       assertEquals(2, arch2.getExtensions().size());
134       Set<String> filenames = Sets.newHashSet("VernacularName.txt", "media.txt");
135       for (ArchiveFile ext : arch2.getExtensions()) {
136         assertTrue(filenames.contains(ext.getLocation()));
137         filenames.remove(ext.getLocation());
138       }
139       assertTrue(filenames.isEmpty());
140 
141     } catch (Exception e) {
142       e.printStackTrace();
143       fail();
144     }
145   }
