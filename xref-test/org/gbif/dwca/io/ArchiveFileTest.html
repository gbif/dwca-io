18  
19  import org.gbif.dwc.terms.DwcTerm;
20  import org.gbif.dwca.io.Archive;
21  import org.gbif.dwca.io.ArchiveFactory;
22  import org.gbif.dwca.io.ArchiveFile;
23  import org.gbif.dwca.io.UnsupportedArchiveException;
24  import org.gbif.dwca.record.Record;
25  import org.gbif.utils.file.FileUtils;
26  
27  import java.io.IOException;
28  
29  import org.junit.Test;
30  
31  import static org.junit.Assert.assertEquals;
32  import static org.junit.Assert.assertNotNull;
33  import static org.junit.Assert.assertTrue;
34  
35  public class ArchiveFileTest {
36  
37    @Test
38    public void testIterator() throws UnsupportedArchiveException, IOException {
39      // test proper archive
40      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc/DarwinCore.txt"));
41      ArchiveFile af = arch.getCore();
42  
43      assertNotNull(af);
44      assertNotNull(af.getId());
45      assertTrue(af.hasTerm(DwcTerm.scientificName));
46  
47      // test iterator
48      int counter = 0;
49      Record last = null;
50      for (Record rec : af) {
51        counter++;
52        last = rec;
53        if (counter == 1) {
54          assertEquals("1559060", rec.id());
55        }
56      }
57      assertEquals(3248, counter);
58      assertEquals("3082", last.id());
59    }
60  }
