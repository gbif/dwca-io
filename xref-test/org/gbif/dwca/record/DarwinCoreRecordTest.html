21    @Test
22    public void testProperties() {
23      final Set<String> IGNORE = ImmutableSet.of(DwcTerm.GROUP_MEASUREMENTORFACT, DwcTerm.GROUP_RESOURCERELATIONSHIP);
24  
25      DarwinCoreRecord dwc = new DarwinCoreRecord();
26      for (DwcTerm t : DwcTerm.values()) {
27        // only test non class terms
28        if (t.isClass()) {
29          continue;
30        }
31  
32        if (IGNORE.contains(t.getGroup())) {
33          try {
34            dwc.getProperty(t);
35            fail("term " + t.qualifiedName() + " should not exist on DarwinCoreRecord");
36          } catch (IllegalArgumentException e) {
37            // expected
38          }
39  
40        } else {
41          String val = new Date().toString();
42          dwc.setProperty(t, val);
43          assertEquals("missing term " + t.qualifiedName(), val, dwc.getProperty(t));
44        }
45      }
46    }
47  
48    @Test
49    public void testDefaultValue() {
50      final String DATASET = "ITIS";
51      ArchiveField id = new ArchiveField(0, DwcTerm.taxonID, null, null);
52      Set<ArchiveField> fields = new HashSet<ArchiveField>();
53      fields.add(new ArchiveField(1, DwcTerm.datasetName, DATASET, null));
54      RecordImpl rec = new RecordImpl(id, fields, DwcTerm.Taxon, true, true);
55  
56      String[] row = {"5432", "IPNI"};
57      rec.setRow(row);
58      assertEquals(row[1], rec.value(DwcTerm.datasetName));
59  
60      row = new String[] {"5432", null};
61      rec.setRow(row);
62      assertEquals(DATASET, rec.value(DwcTerm.datasetName));
63  
64      row = new String[] {"5432", ""};
65      rec.setRow(row);
66      assertEquals(DATASET, rec.value(DwcTerm.datasetName));
67    }
