19    @Test
20    public void testGetFullScientificName() {
21      DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca x brinkmannii A. Braun");
22      dwc.setScientificNameAuthorship("A. Braun");
23      assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
24      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
25  
26      dwc = new DarwinCoreTaxon("Festuca x brinkmannii A. Braun");
27      dwc.setScientificNameAuthorship("A.Braun");
28      assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
29      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
30  
31      dwc = new DarwinCoreTaxon("Festuca x brinkmannii");
32      dwc.setScientificNameAuthorship("A. Braun");
33      assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
34      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
35  
36      dwc = new DarwinCoreTaxon("Festuca A. Braun brinkmannii subsp. brinkmannii");
37      dwc.setScientificNameAuthorship("A. Braun");
38      assertEquals("Festuca A. Braun brinkmannii subsp. brinkmannii", dwc.getFullScientificName());
39      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
40  
41      dwc = new DarwinCoreTaxon("Festuca brinkmannii");
42      dwc.setScientificNameAuthorship("A. Braun");
43      assertEquals("Festuca brinkmannii A. Braun", dwc.getFullScientificName());
44      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
45  
46      dwc = new DarwinCoreTaxon(" ");
47      dwc.setScientificNameAuthorship("Rafinesque, 1795");
48      dwc.setKingdom("Plantae");
49      assertNull(dwc.getFullScientificName());
50      assertEquals("Plantae", dwc.getLowestScientificName());
51  
52      dwc = new DarwinCoreTaxon(" ");
53      dwc.setScientificNameAuthorship("L.");
54      dwc.setKingdom("Plantae");
55      dwc.setFamily("Asteraceae");
56      assertNull(dwc.getFullScientificName());
57      assertEquals("Asteraceae", dwc.getLowestScientificName());
58  
59      dwc = new DarwinCoreTaxon();
60      dwc.setScientificNameAuthorship("L.");
61      dwc.setKingdom("Plantae");
62      dwc.setGenus("Poa");
63      assertEquals("Poa", dwc.getFullScientificName());
64      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
65  
66      dwc.setSpecificEpithet("annua");
67      assertEquals("Poa annua L.", dwc.getFullScientificName());
68      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
69  
70      dwc.setScientificNameAuthorship("");
71      assertEquals("Poa annua", dwc.getFullScientificName());
72      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
73  
74      dwc.setScientificNameAuthorship("L.");
75      dwc.setInfraspecificEpithet("alpina");
76      assertEquals("Poa annua alpina L.", dwc.getFullScientificName());
77      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
78  
79      dwc.setVerbatimTaxonRank("var.");
80      assertEquals("Poa annua alpina L.", dwc.getFullScientificName());
81      assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
82  
83      // test not identical authorships
84      dwc = new DarwinCoreTaxon();
85      dwc.setScientificName("Aagaardia protensa Saether, 2000");
86      dwc.setGenus("Aagaardia");
87      dwc.setSpecificEpithet("protensa");
88      dwc.setScientificNameAuthorship("Saether 2000");
89      assertEquals(dwc.getScientificName(), dwc.getFullScientificName());
90  
91      // test genus
92      dwc = new DarwinCoreTaxon();
93      dwc.setScientificName("Aagaardia protensa Saether, 2000");
94      dwc.setGenus("Aagaardia");
95      dwc.setSpecificEpithet("protensa");
96      dwc.setScientificNameAuthorship("Saether 2000");
97      assertEquals(dwc.getScientificName(), dwc.getFullScientificName());
98  
99      // test authors with single letter that is part of the name
100     dwc = new DarwinCoreTaxon();
101     dwc.setScientificName("Trifolium repens");
102     dwc.setGenus("Trifolium");
103     dwc.setSpecificEpithet("repens");
104     dwc.setScientificNameAuthorship("L.");
105     assertEquals("Trifolium repens", dwc.getScientificName());
106     assertEquals("Trifolium repens L.", dwc.getFullScientificName());
107 
108     dwc = new DarwinCoreTaxon();
109     dwc.setScientificName("Trifolium thalii");
110     dwc.setScientificNameAuthorship("Vill.");
111     assertEquals("Trifolium thalii Vill.", dwc.getFullScientificName());
112 
113     dwc = new DarwinCoreTaxon();
114     dwc.setScientificName("Trifolium alpestre");
115     dwc.setScientificNameAuthorship("L.");
116     assertEquals("Trifolium alpestre L.", dwc.getFullScientificName());
117 
118     // autonyms ???

127   @Test
128   public void testInterpretedRank() {
129     DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca brinkmannii A. Braun");
130     dwc.setKingdom("PLANTAE");
131     dwc.setPhylum("Magnoliophyta");
132     dwc.setFamily("Poaceae");
133     dwc.setGenus("Festuca");
134     dwc.setTaxonRank("species");
135     assertEquals("species", dwc.getTaxonRankInterpreted());
136 
137     dwc.setSpecificEpithet("brinkmannii");
138     assertEquals("species", dwc.getTaxonRankInterpreted());
139 
140     dwc.setInfraspecificEpithet("nonexistii");
141     assertEquals("species", dwc.getTaxonRankInterpreted());
142 
143     dwc.setScientificName(null);
144     assertEquals("species", dwc.getTaxonRankInterpreted());
145 
146     dwc.setTaxonRank(null);
147     assertEquals("infraspecies", dwc.getTaxonRankInterpreted());
148 
149     dwc.setVerbatimTaxonRank("var.");
150     assertEquals("var.", dwc.getTaxonRankInterpreted());
151 
152     dwc.setSpecificEpithet(null);
153     assertEquals("genus", dwc.getTaxonRankInterpreted());
154 
155     dwc.setGenus(null);
156     assertEquals("family", dwc.getTaxonRankInterpreted());
157   }

159   @Test
160   public void testHigherClassification() {
161     DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca brinkmannii A. Braun");
162     dwc.setKingdom("PLANTAE");
163     dwc.setPhylum("Magnoliophyta");
164     dwc.setFamily("Poaceae");
165     dwc.setGenus("Festuca");
166     assertEquals("Plantae | Magnoliophyta | Poaceae | Festuca", dwc.buildHigherClassification(null));
167     assertEquals("Plantae | Magnoliophyta | Poaceae", dwc.buildHigherClassification('f'));
168     assertEquals("Plantae | Magnoliophyta", dwc.buildHigherClassification('o'));
169   }

171   @Test
172   public void testProperties() {
173     Set<String> groups = ImmutableSet.of(DwcTerm.GROUP_TAXON, DwcTerm.GROUP_RECORD);
174     DarwinCoreTaxon dwc = new DarwinCoreTaxon();
175     for (DwcTerm t : DwcTerm.values()) {
176       if (groups.contains(t.getGroup()) && !t.isClass()){
177         String val = new Date().toString();
178         dwc.setProperty(t, val);
179         try {
180           dwc.getProperty(t);
181           assertEquals(val, dwc.getProperty(t));
182         } catch (Exception e) {
183           System.err.println(e.getMessage());
184         }
185       }
186     }
187   }
