31  /**
32   * This test checks we don't get a race condition between two <strong>processes</strong> opening the same DWC-A at the
33   * same time.
34   *
35   * It needs to be run from the command line, in parallel.
36   *
37   * mvn exec:java -Dexec.mainClass=org.gbif.dwc.ParallelOpeningTest
38   *
39   * Monitor open files with something like:
40   *
41   * while :; do for i in `pgrep -f ParallelOpeningTest`; do echo $i && ls -l /proc/$i/fd | grep /tmp/par && echo; sleep 1; done; done
42   */
43  public class ParallelOpeningTest {
44    private static final Logger LOG = LoggerFactory.getLogger(ParallelOpeningTest.class);
45  
46    private Archive arch;
47  
48    /**
49     * Check we don't get a race condition when opening an archive which needs sorting.
50     *
51     * This downloads a large archive (we don't want it in Git) and checks it has at least one extension.
52     */
53    public void testParallelOpening() throws Exception {
54      Path archivePath = Paths.get("/tmp/parallel-opening-test-archive.zip");
55      Path extractToFolder = Paths.get("/tmp/parallel-opening-test-archive");
56      if (!archivePath.toFile().exists()) {
57        URL download = new URL("http://api.gbif.org/v1/occurrence/download/request/0012957-180131172636756.zip");
58        Files.copy(download.openStream(), archivePath);
59        extractToFolder.toFile().mkdir();
60        arch = DwcFiles.fromCompressed(archivePath, extractToFolder);
61      }
62  
63      arch = DwcFiles.fromLocation(extractToFolder);
64  
65      assertFalse(arch.getExtensions().isEmpty());
66  
67      // Initialize (i.e. normalize and sort) the archive in the background.
68      LOG.info("Initializing archive");
69      arch.initialize();
70      LOG.info("Initialization completed, locks should have been released.");
71  
72      int counter = 0;
73      for (StarRecord rec : arch) {
74        if (counter == 0) {
75          LOG.info("Read first record {}", rec);
76        }
77        counter++;
78      }
79  
80      LOG.info("Counted {} records", counter);
81  
82      LOG.info("All files should have been closed.");
83      Thread.sleep(30_000);
84  
85      assertTrue(counter > 500000, "Many records extracted");
86    }
87

90      pot.testParallelOpening();
91    }
92  }
