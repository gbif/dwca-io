195   public void testStarIteratorExtRecords() throws Exception {
196     File zip = FileUtils.getClasspathFile("checklist_980.zip");
197     File tmpDir = File.createTempFile("dwca-reader-test", ".tmp").getParentFile();
198     CompressionUtil.decompressFile(tmpDir, zip);
199     // read archive from this tmp dir
200     Archive arch = ArchiveFactory.openArchive(tmpDir);
201     int counter = 0;
202     int occCounter = 0;
203     Set<String> ids = Sets.newHashSet();
204     for (StarRecord rec : arch) {
205       counter++;
206       ids.add(rec.core().id());
207       List<Record> occs = rec.extension(DwcTerm.Occurrence);
208       occCounter += occs.size();
209     }
210     assertEquals("Core taxon file has 356 records", 356, counter);
211     assertEquals("Core taxon file has 356 unique ids", 356, ids.size());
212 
213     // read extension file on its own and extract core ids to be cross checked with core id set
214     CSVReader occReader = CSVReaderFactory.build(arch.getExtension(DwcTerm.Occurrence));
215     int occCounter2=0;
216     for (String[] rec : occReader) {
217       String id = rec[1];
218       occCounter2++;
219       assertTrue("Occurrence coreid " + id + " not existing", ids.contains(id));
220     }
221     assertEquals("Occurrence extension file has 740 records", 740, occCounter2);
222     assertEquals("Occurrence start extensions should be 740 records", 740, occCounter);
223   }
