105   @Test
106   public void testHeaderWriting() throws Exception {
107     File dwcaDir = FileUtils.createTempDir();
108     dwcaDir.deleteOnExit();
109     LOG.info("Test archive writer in {}", dwcaDir.getAbsolutePath());
110 
111     DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, dwcaDir, true);
112 
113     writer.newRecord("dummy1");
114     writer.addCoreColumn(DwcTerm.parentNameUsageID);
115     writer.addCoreColumn(DwcTerm.acceptedNameUsageID);
116     writer.addCoreColumn(DwcTerm.scientificName);
117     writer.addCoreColumn(GbifTerm.canonicalName);
118     writer.addCoreColumn(DwcTerm.taxonRank, "species");
119     writer.addCoreColumn(DwcTerm.taxonomicStatus);
120     writer.addCoreColumn(DwcTerm.kingdom);
121     writer.addCoreColumn(DwcTerm.phylum);
122     writer.addCoreColumn(DwcTerm.class_);
123     writer.addCoreColumn(DwcTerm.order);
124     writer.addCoreColumn(DwcTerm.family);
125     writer.addCoreColumn(GbifTerm.depth);
126     writer.addCoreColumn(GbifTerm.depthAccuracy);
127 
128     writer.newRecord("dummy2");
129     writer.addCoreColumn(DwcTerm.kingdom, "Plantae");
130     writer.addCoreColumn(DwcTerm.phylum);
131     writer.addCoreColumn(DwcTerm.class_);
132     writer.addCoreColumn(DwcTerm.order);
133     writer.addCoreColumn(DwcTerm.family, "Asteraceae");
134 
135     writer.newRecord("dummy3");
136     writer.addCoreColumn(GbifTerm.depth, "2");
137     writer.addCoreColumn(GbifTerm.depthAccuracy, "1");
138 
139     // define extension columns
140     Map<Term, String> eData = Maps.newHashMap();
141     // distributions
142     eData.put(DwcTerm.locality, "locality1");
143     eData.put(DwcTerm.occurrenceStatus, "present");
144     eData.put(DwcTerm.establishmentMeans, "alien");
145     writer.addExtensionRecord(GbifTerm.Distribution, eData);
146 
147     eData.put(DwcTerm.locality, "locality2");
148     writer.addExtensionRecord(GbifTerm.Distribution, eData);
149 
150     writer.close();
151 
152     File cf = new File(dwcaDir, writer.getDataFiles().get(DwcTerm.Taxon));
153     File df = new File(dwcaDir, writer.getDataFiles().get(GbifTerm.Distribution));
154 
155     // check if taxon file contains headers
156     String[] headers = getFirstRow(cf);
157     LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
158     assertEquals(14, headers.length);
159     assertEquals("taxonID", headers[0]);
160     assertEquals("parentNameUsageID", headers[1]);
161     assertEquals("kingdom", headers[7]);
162 
163     // check if extension file contains headers
164     headers = getFirstRow(df);
165     LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
166     assertEquals(4, headers.length);
167     assertEquals("taxonID", headers[0]);
168   }

170   private String[] getFirstRow(File f) throws IOException {
171     BufferedReader r = FileUtils.getUtf8Reader(f);
172     String firstRow = r.readLine();
173     return firstRow.split("\t");
174   }

176   @Test
177   public void testRoundtrip() {
178     try {
179       // read taxon archive
180       Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("archive-dwc").toPath());
181       assertEquals(2, arch.getExtensions().size());
182       int coreRecords = 0;
183       int allRecords = 0;
184 
185       // write taxon archive
186       File tempArch = FileUtils.createTempDir();
187       tempArch.deleteOnExit();
188       System.out.println("Writing temporary test archive to " + tempArch.getAbsolutePath());
189       DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, tempArch);
190       for (StarRecord rec : arch) {
191         // core
192         coreRecords++;
193         allRecords += rec.size();
194 
195         writer.newRecord(rec.core().id());
196         for (Term term : arch.getCore().getTerms()) {
197           writer.addCoreColumn(term, rec.core().value(term));
198         }
199         // extensions
200         for (Term rt : rec.extensions().keySet()) {
201           ArchiveFile af = arch.getExtension(rt);
202           // iterate over records for one extension
203           for (Record row : rec.extension(rt)) {
204             writer.addExtensionRecord(rt, DwcaWriter.recordToMap(row, af));
205           }
206         }
207       }
208       writer.close();
209 
210       // reread and compare
211       Archive arch2 = DwcFiles.fromLocation(tempArch.toPath());
212 
213       int coreRecords2 = 0;
214       int allRecords2 = 0;
215       for (StarRecord rec : arch2) {
216         // core
217         coreRecords2++;
218         allRecords2 += rec.size();
219       }
220 
221       // compare
222       assertEquals(coreRecords, coreRecords2);
223       assertEquals(allRecords, allRecords2);
224 
225 
226     } catch (Exception e) {
227       e.printStackTrace();
228       fail();
229     }
230   }

252   /**
253    * Test the writing of an archive that includes some default values in the core and in one extension.
254    */
255   @Test
256   public void testWriterUsingDefaultValues() throws Exception {
257     File dwcaDir = FileUtils.createTempDir();
258     dwcaDir.deleteOnExit();
259     LOG.info("Test archive writer in {}", dwcaDir.getAbsolutePath());
260 
261     DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, DwcTerm.taxonID, dwcaDir, true);
262 
263     writer.newRecord("dummy1");
264     writer.addCoreColumn(DwcTerm.parentNameUsageID, "1");
265     writer.addCoreColumn(DwcTerm.acceptedNameUsageID, "2");
266     writer.addCoreColumn(DwcTerm.countryCode);
267     
268     // add a VernacularName extension record
269     Map<Term,String> extensionRecord = new HashMap<>();
270     extensionRecord.put(DwcTerm.vernacularName, "Komodo Dragon");
271     extensionRecord.put(DcTerm.language, null);
272     writer.addExtensionRecord(GbifTerm.VernacularName, extensionRecord);
273     
274     writer.addCoreDefaultValue(DwcTerm.collectionCode, "A2Z");
275     writer.addCoreDefaultValue(DwcTerm.countryCode, "CA");
276     writer.addDefaultValue(GbifTerm.VernacularName, DcTerm.language, "en");
277     
278     // add a second records and overwrite the default value
279     writer.newRecord("dummy2");
280     writer.addCoreColumn(DwcTerm.parentNameUsageID, "1");
281     writer.addCoreColumn(DwcTerm.acceptedNameUsageID, "2");
282     writer.addCoreColumn(DwcTerm.countryCode, "ID");
283     
284     // add a VernacularName extension record
285     extensionRecord = new HashMap<>();
286     extensionRecord.put(DwcTerm.vernacularName, "Varano De Komodo");
287     extensionRecord.put(DcTerm.language, "es");
288     writer.addExtensionRecord(GbifTerm.VernacularName, extensionRecord);
289     
290     writer.close();
291 
292     // validate core content
293     Archive arch = DwcFiles.fromLocation(dwcaDir.toPath());
294     Iterator<Record> recIt = arch.getCore().iterator();
295     Record firstRecord = recIt.next();
296     assertEquals("dummy1", firstRecord.id());
297     assertEquals("dummy1", firstRecord.value(DwcTerm.taxonID));
298     assertEquals("A2Z", firstRecord.value(DwcTerm.collectionCode));
299     assertEquals("CA", firstRecord.value(DwcTerm.countryCode));
300     assertEquals("A2Z", arch.getCore().getField(DwcTerm.collectionCode).getDefaultValue());
301     assertEquals("CA", arch.getCore().getField(DwcTerm.countryCode).getDefaultValue());
302     
303     Record secondRecord = recIt.next();
304     assertEquals("dummy2", secondRecord.id());
305     assertEquals("dummy2", secondRecord.value(DwcTerm.taxonID));
306     assertEquals("A2Z", secondRecord.value(DwcTerm.collectionCode));
307     assertEquals("ID", secondRecord.value(DwcTerm.countryCode));
308     
309     // validate extension content
310     Iterator<Record> extRecIt = arch.getExtension(GbifTerm.VernacularName).iterator();
311     assertEquals("en", arch.getExtension(GbifTerm.VernacularName).getField(DcTerm.language).getDefaultValue());
312     firstRecord = extRecIt.next();
313     assertEquals("dummy1", firstRecord.id());
314     assertEquals("en", firstRecord.value(DcTerm.language));
315     
316     secondRecord = extRecIt.next();
317     assertEquals("dummy2", secondRecord.id());
318     assertEquals("es", secondRecord.value(DcTerm.language));
319   }
