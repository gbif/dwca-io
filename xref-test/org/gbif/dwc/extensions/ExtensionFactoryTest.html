36  public class ExtensionFactoryTest {
37  
38    @Test
39    public void testExtensionFactory() throws IOException, ParserConfigurationException, SAXException {
40      String vocabUrl = "http://rs.gbif.org/vocabulary/gbif/description_type.xml";
41      InputStream vocab = FileUtils.classpathStream("extension/vocabulary_gbif_description_type.xml");
42      InputStream extension = FileUtils.classpathStream("extension/extension_gbif_1.0_description.xml");
43  
44      VocabularyFactory vocabFactory = new VocabularyFactory(SAXUtils.getNsAwareSaxParserFactory());
45  
46      ThesaurusHandlingRule thr = new ThesaurusHandlingRule(new MockVocabulariesManager(vocabUrl, vocabFactory.build(vocab)));
47      ExtensionFactory factory = new ExtensionFactory(thr, SAXUtils.getNsAwareSaxParserFactory());
48  
49      Extension ext = factory.build(extension, new URL(vocabUrl), false);
50  
51      assertNotNull(ext);
52      assertNotNull(ext.getProperty(DcTerm.type).getVocabulary());
53    }
54  
55    /**
56     * Mock VocabulariesManager that support only 1 vocabulary
57     */
58    private static class MockVocabulariesManager implements VocabulariesManager{
59  
60      private final String vocabUrl;
61      private final Vocabulary vocab;
62  
63      public MockVocabulariesManager(String vocabUrl, Vocabulary vocab){
64        this.vocabUrl = vocabUrl;
65        this.vocab = vocab;
66      }
67  
68      @Override
69      public Vocabulary get(String s) {
70        return null;
71      }
72  
73      @Override
74      public Vocabulary get(URL url) {
75        if(vocabUrl.equalsIgnoreCase(url.toString())){
76          return vocab;
77        }
78        return null;
79      }
80  
81      @Override
82      public Map<String, String> getI18nVocab(String s, String s1) {
83        return null;
84      }
85  
86      @Override
87      public List<Vocabulary> list() {
88        return null;
89      }
90    }
91  }
