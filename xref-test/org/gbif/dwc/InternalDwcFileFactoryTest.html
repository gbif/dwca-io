63    /**
64     * Givin a compressed file, make sure we can uncompressed it, read the core and find the provided id.
65     * @param compressedFile
66     * @param id
67     * @throws IOException
68     */
69    private void assertIdInCompressed(Path compressedFile, String id) throws IOException {
70      File tmpDir = folder.newFolder();
71  
72      // open archive from zip
73      Archive arch = InternalDwcFileFactory.fromCompressed(compressedFile, tmpDir.toPath());
74      assertNotNull(arch.getCore().getId());
75      assertEquals(1, arch.getExtensions().size());
76  
77      boolean found = false;
78      try (ClosableIterator<Record> it = arch.getCore().iterator()){
79        while(it.hasNext()){
80          if(id.equals(it.next().id())){
81            found = true;
82            break;
83          }
84        }
85      } catch (Exception e) {
86        e.printStackTrace();
87      }
88      assertTrue("Can find the id " + id + " inside the archive " + compressedFile.getFileName(), found);
89    }
