13  public class UsageExample {
14  
15    @Test
16    public void testUsageExample() throws IOException {
17      Path myArchiveFile = FileUtils.getClasspathFile("checklist_980.zip").toPath();
18      Path extractToFolder = FileUtils.createTempDir().toPath();
19      extractToFolder.toFile().deleteOnExit();
20  
21      // Open the Darwin Core Archive
22      Archive dwcArchive = DwcFiles.fromCompressed(myArchiveFile, extractToFolder);
23  
24      System.out.println("Reading archive from " + dwcArchive.getLocation().getAbsolutePath());
25      System.out.println("Archive of rowtype " + dwcArchive.getCore().getRowType() + " with " + dwcArchive.getExtensions().size() + " extensions");
26  
27      // Loop over core records and display id, basis of record and scientific name
28      for (Record rec : dwcArchive.getCore()) {
29        System.out.println(String.format("%s: %s (%s)", rec.id(), rec.value(DwcTerm.basisOfRecord), rec.value(DwcTerm.scientificName)));
30      }
31  
32      // Loop over star records and display id, core record data, and extension data
33      for (StarRecord rec : dwcArchive) {
34        System.out.println(String.format("%s: %s", rec.core().id(), rec.core().value(DwcTerm.scientificName)));
35        if (rec.hasExtension(GbifTerm.VernacularName)) {
36          for (Record extRec : rec.extension(GbifTerm.VernacularName)) {
37            System.out.println(" - " + extRec.value(DwcTerm.vernacularName));
38          }
39        }
40      }
41    }
42  }
