31  public class StrTokenizerPerformance {
32  
33    private long test(StrTokenizer tokenizer, File source) throws IOException {
34      FileInputStream fis = new FileInputStream(source);
35      InputStreamReader reader = new InputStreamReader(fis, "utf8");
36      BufferedReader br = new BufferedReader(reader);
37  
38      // keep track of time while iterating
39      long start = System.currentTimeMillis();
40      String row = br.readLine();
41      while (row != null) {
42        tokenizer.reset(row);
43        String[] columns = tokenizer.getTokenArray();
44        row = br.readLine();
45      }
46      long dur = System.currentTimeMillis() - start;
47      br.close();
48      return dur;
49    }
50  
51    @Test
52    public void testCharVsStringPerformance() throws IOException {
53      File source = FileUtils.getClasspathFile("irmng.tail");
54  
55      // test CHAR
56      StrTokenizer tokenizer = new StrTokenizer();
57      tokenizer.setDelimiterChar('\t');
58      tokenizer.setEmptyTokenAsNull(true);
59      tokenizer.setIgnoreEmptyTokens(false);
60      long time = test(tokenizer, source);
61      System.out.println(time + " milliseconds for CHAR based tokenizer.");
62  
63      // test STRING
64      tokenizer = new StrTokenizer();
65      tokenizer.setDelimiterString("\t");
66      tokenizer.setEmptyTokenAsNull(true);
67      time = test(tokenizer, source);
68      System.out.println(time + " milliseconds for STRING based tokenizer.");
69    }
