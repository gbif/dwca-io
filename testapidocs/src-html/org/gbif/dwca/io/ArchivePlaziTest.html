001package org.gbif.dwca.io;
002
003import org.gbif.dwca.io.Archive;
004import org.gbif.dwca.io.ArchiveFactory;
005import org.gbif.dwca.record.StarRecord;
006import org.gbif.utils.file.FileUtils;
007
008import java.io.File;
009import java.io.IOException;
010
011import org.junit.Test;
012
013import static org.junit.Assert.assertEquals;
014import static org.junit.Assert.assertNotNull;
015
016public class ArchivePlaziTest {
017
018  private void assertNumberStarRecords(Archive arch, int expectedRecords) throws IOException {
019    int rows = 0;
020    for (StarRecord rec : arch) {
021      assertNotNull(rec);
022      rows++;
023    }
024    assertEquals(expectedRecords, rows);
025  }
026
027  @Test
028  public void testBuildReaderFile() throws IOException {
029    File zip = FileUtils.getClasspathFile("plazi/6632D8151686A3F8C71D4B5A5B1181A4.zip");
030    File tmpDir = FileUtils.createTempDir();
031    tmpDir.deleteOnExit();
032
033    Archive arch = ArchiveFactory.openArchive(zip, tmpDir);
034    assertNumberStarRecords(arch, 10);
035  }
036}
