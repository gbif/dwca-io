191  /**
192   * Test extension sorting verifying that all core records do have the right number of extension records attached
193   * when using the star record iterator.
194   */
195  @Test
196  public void testStarIteratorExtRecords() throws Exception {
197    File zip = FileUtils.getClasspathFile("checklist_980.zip");
198    File tmpDir = Files.createTempDirectory("dwca-io-test").toFile();
199    // read archive from this tmp dir
200    Archive arch = DwcFiles.fromCompressed(zip.toPath(), tmpDir.toPath());
201    int counter = 0;
202    int occCounter = 0;
203    Set<String> ids = new HashSet<>();
204    for (StarRecord rec : arch) {
205      counter++;
206      ids.add(rec.core().id());
207      List<Record> occs = rec.extension(DwcTerm.Occurrence);
208      occCounter += occs.size();
209    }
210    assertEquals("Core taxon file has 356 records", 356, counter);
211    assertEquals("Core taxon file has 356 unique ids", 356, ids.size());
212
213    // read extension file on its own and extract core ids to be cross checked with core id set
214    File file = arch.getExtension(DwcTerm.Occurrence).getLocationFile();
215
216    TabularDataFileReader<List<String>> occReader = TabularFiles.newTabularFileReader(
217        Files.newBufferedReader(file.toPath(), StandardCharsets.UTF_8),
218          ';', "\n", null,true, 0);
219
220    int occCounter2 = 0;
221    List<String> rec;
222    while ((rec = occReader.read()) != null) {
223      String id = rec.get(1);
224      occCounter2++;
225      assertTrue("Occurrence coreid " + id + " not existing", ids.contains(id));
226    }
227    assertEquals("Occurrence extension file has 740 records", 740, occCounter2);
228    assertEquals("Occurrence start extensions should be 740 records", 740, occCounter);
229  }
