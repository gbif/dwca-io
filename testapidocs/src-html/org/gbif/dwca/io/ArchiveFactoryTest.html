199  @Test
200  @Ignore("currently fails with only 661 records coming through instead of 740")
201  public void testStarIteratorExtRecords() throws Exception {
202    File zip = FileUtils.getClasspathFile("checklist_980.zip");
203    File tmpDir = File.createTempFile("dwca-io-test", ".tmp").getParentFile();
204    CompressionUtil.decompressFile(tmpDir, zip);
205    // read archive from this tmp dir
206    Archive arch = ArchiveFactory.openArchive(tmpDir);
207    int counter = 0;
208    int occCounter = 0;
209    Set<String> ids = Sets.newHashSet();
210    for (StarRecord rec : arch) {
211      counter++;
212      ids.add(rec.core().id());
213      List<Record> occs = rec.extension(DwcTerm.Occurrence);
214      occCounter += occs.size();
215    }
216    assertEquals("Core taxon file has 356 records", 356, counter);
217    assertEquals("Core taxon file has 356 unique ids", 356, ids.size());
218
219    // read extension file on its own and extract core ids to be cross checked with core id set
220    CSVReader occReader = CSVReaderFactory.build(arch.getExtension(DwcTerm.Occurrence));
221    int occCounter2=0;
222    for (String[] rec : occReader) {
223      String id = rec[1];
224      occCounter2++;
225      assertTrue("Occurrence coreid " + id + " not existing", ids.contains(id));
226    }
227    assertEquals("Occurrence extension file has 740 records", 740, occCounter2);
228    assertEquals("Occurrence start extensions should be 740 records", 740, occCounter);
229  }
