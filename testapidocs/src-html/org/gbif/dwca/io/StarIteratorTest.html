001package org.gbif.dwca.io;
002
003import org.gbif.dwc.terms.DcTerm;
004import org.gbif.dwc.terms.DwcTerm;
005import org.gbif.dwca.io.Archive;
006import org.gbif.dwca.io.ArchiveFactory;
007import org.gbif.dwca.io.UnsupportedArchiveException;
008import org.gbif.dwca.record.Record;
009import org.gbif.dwca.record.StarRecord;
010import org.gbif.utils.file.FileUtils;
011
012import java.io.IOException;
013
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017import static org.junit.Assert.assertNotNull;
018
019public class StarIteratorTest {
020
021  @Test
022  public void testIterator() throws IOException, UnsupportedArchiveException {
023    // test proper archive
024    Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
025    assertNotNull(arch.getCore());
026    assertEquals(2, arch.getExtensions().size());
027    int found = 0;
028    int extensionRecords = 0;
029    for (StarRecord rec : arch) {
030      //System.out.println(rec.core().id() + " --> " + rec.size());
031      // count all extension records
032      extensionRecords += rec.size();
033
034      if (rec.core().id().equals("544382")) {
035        found++;
036        assertEquals("Tursiops truncatus truncatus Montagu", rec.core().value(DwcTerm.scientificName));
037        // test extension iter
038        int i = 0;
039        for (Record er : rec) {
040          i++;
041        }
042        assertEquals(27, i);
043      } else if (rec.core().id().equals("105833")) {
044        found++;
045//        105833  Chinese river dolphin English
046//        105833  Chinese lake dolphin  English
047//        105833  Pei c’hi      Chinese
048//        105833  White flag dolphin  English
049        int i = 0;
050        for (Record er : rec) {
051          i++;
052          if ("Chinese river dolphin".equals(er.value(DwcTerm.vernacularName))) {
053            assertEquals("English", er.value(DcTerm.language));
054            assertEquals("English", er.value(DcTerm.language));
055          } else if ("Chinese lake dolphin".equals(er.value(DwcTerm.vernacularName))) {
056            assertEquals("English", er.value(DcTerm.language));
057          } else if ("Pei c’hi".equals(er.value(DwcTerm.vernacularName))) {
058            assertEquals("Chinese", er.value(DcTerm.language));
059          } else {
060            assertEquals("White flag dolphin", er.value(DwcTerm.vernacularName));
061            assertEquals("English", er.value(DcTerm.language));
062          }
063        }
064        assertEquals(4, i);
065      } else if (rec.core().id().equals("105838")) {
066        found++;
067//      105838  Delfin de La Plata  Spanish
068//      105838  Franciscana Spanish
069//      105838  Franciscano Portuguese
070//      105838  La Plata dolphin  English
071//      105838  Tonina  Spanish
072//      105838  Toninha Portuguese
073        int i = 0;
074        for (Record er : rec) {
075          i++;
076          if ("Delfin de La Plata".equals(er.value(DwcTerm.vernacularName))) {
077            assertEquals("Spanish", er.value(DcTerm.language));
078          } else if ("Franciscana".equals(er.value(DwcTerm.vernacularName))) {
079            assertEquals("Spanish", er.value(DcTerm.language));
080          } else if ("Franciscano".equals(er.value(DwcTerm.vernacularName))) {
081            assertEquals("Portuguese", er.value(DcTerm.language));
082          } else if ("La Plata dolphin".equals(er.value(DwcTerm.vernacularName))) {
083            assertEquals("English", er.value(DcTerm.language));
084          } else if ("Tonina".equals(er.value(DwcTerm.vernacularName))) {
085            assertEquals("Spanish", er.value(DcTerm.language));
086          } else {
087            assertEquals("Toninha", er.value(DwcTerm.vernacularName));
088            assertEquals("Portuguese", er.value(DcTerm.language));
089          }
090        }
091        assertEquals(6, i);
092      }
093
094    }
095    assertEquals(3, found);
096    assertEquals(1057, extensionRecords);
097  }
098}
