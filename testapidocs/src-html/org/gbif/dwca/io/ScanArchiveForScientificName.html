001package org.gbif.dwca.io;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwca.io.Archive;
005import org.gbif.dwca.io.ArchiveFactory;
006import org.gbif.dwca.io.UnsupportedArchiveException;
007import org.gbif.dwca.record.DarwinCoreRecord;
008
009import java.io.File;
010import java.io.IOException;
011import java.util.Iterator;
012
013public class ScanArchiveForScientificName {
014
015  public static void main(String[] args) throws IOException, UnsupportedArchiveException {
016    // opens csv files with headers or dwc-a direcotries with a meta.xml descriptor
017    Archive arch = ArchiveFactory
018      .openArchive(new File("/Volumes/Scratch/ecat-data-col/checklists/beac2c55-d889-4358-a414-b1db79ba3536/dwc-a"));
019
020    // does scientific name exist?
021    if (!arch.getCore().hasTerm(DwcTerm.scientificName)) {
022      System.out.println("This application requires dwc-a with scientific names");
023      System.exit(1);
024    }
025
026    // loop over core darwin core records
027    Iterator<DarwinCoreRecord> iter = arch.iteratorDwc();
028    DarwinCoreRecord dwc;
029    while (iter.hasNext()) {
030      dwc = iter.next();
031      if (dwc.getScientificName().startsWith("Ambispora callosa")) {
032        System.out.println(dwc.getScientificName());
033        System.out.println(dwc.getScientificNameAuthorship());
034        System.out.println(dwc);
035      }
036    }
037  }
038
039}
