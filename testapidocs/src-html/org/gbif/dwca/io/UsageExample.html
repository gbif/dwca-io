001package org.gbif.dwca.io;
002
003import org.gbif.dwc.terms.DcTerm;
004import org.gbif.dwc.terms.DwcTerm;
005import org.gbif.dwca.io.Archive;
006import org.gbif.dwca.io.ArchiveFactory;
007import org.gbif.dwca.io.UnsupportedArchiveException;
008import org.gbif.dwca.record.Record;
009import org.gbif.dwca.record.StarRecord;
010
011import java.io.File;
012import java.io.IOException;
013
014public class UsageExample {
015
016  public static void main(String[] args) throws IOException, UnsupportedArchiveException {
017    // opens csv files with headers or dwc-a direcotries with a meta.xml descriptor
018    Archive arch = ArchiveFactory.openArchive(new File(args[0]));
019
020    System.out.println("Reading archive from "+arch.getLocation().getAbsolutePath());
021    System.out.println("Archive of rowtype "+arch.getCore().getRowType()+" with "+arch.getExtensions().size()+" extensions");
022    // loop over star records. i.e. core with all linked extension records
023    for (StarRecord rec : arch) {
024      // print core ID + scientific name
025      System.out.println(rec.core().id()
026        + " sciname:" + rec.core().value(DwcTerm.scientificName)
027        + " bor:" + rec.core().value(DwcTerm.basisOfRecord)
028      );
029      // print out all rowTypes
030      for (Record erec : rec) {
031        // print out extension rowtype
032        System.out.println(erec.rowType() + ", id="+erec.value(DcTerm.identifier));
033      }
034    }
035  }
036
037}
