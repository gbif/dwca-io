026public class MetaDescriptorWriterTest {
027
028  @Test
029  public void testRoundtrip() {
030    try {
031      // read archive
032      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
033      assertNotNull(arch);
034      assertNotNull(arch.getCore());
035      assertTrue(arch.getCore().getId().getIndex() == 0);
036      assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
037      assertEquals(2, arch.getExtensions().size());
038      assertEquals("\t", arch.getCore().getFieldsTerminatedBy());
039      assertNull(arch.getCore().getField(DwcTerm.scientificName).getDelimitedBy());
040      assertEquals(";", arch.getCore().getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
041
042      // write meta.xml
043      File tmpDwca = createTmpMeta(arch);
044
045      Archive arch2 = ArchiveFactory.openArchive(tmpDwca);
046      // core props
047      ArchiveFile core = arch2.getCore();
048      assertNotNull(core);
049      assertNotNull(core.getId());
050      assertTrue(core.hasTerm(DwcTerm.scientificName));
051      assertEquals("DarwinCore.txt", core.getLocation());
052      assertEquals("\t", core.getFieldsTerminatedBy());
053      assertNull(arch.getCore().getField(DwcTerm.scientificName).getDelimitedBy());
054      assertEquals(";", arch.getCore().getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
055      for (ArchiveField f : arch.getCore().getFields().values()) {
056        assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
057        assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
058      }
059
060      // extensions props
061      assertEquals(2, arch2.getExtensions().size());
062      Set<String> filenames = Sets.newHashSet("VernacularName.txt", "media.txt");
063      for (ArchiveFile ext : arch2.getExtensions()) {
064        assertTrue(filenames.contains(ext.getLocation()));
065        filenames.remove(ext.getLocation());
066      }
067      assertTrue(filenames.isEmpty());
068
069    } catch (Exception e) {
070      e.printStackTrace();
071      fail();
072    }
073  }
074
075  private File createTmpMeta(Archive arch) throws IOException {
076    File tmpDir = Files.createTempDirectory("dwca-io-test").toFile();
077    tmpDir.deleteOnExit();
078    File tmpMeta = new File(tmpDir, "meta.xml");
079    System.out.println("Writing temporary test meta file to " + tmpMeta.getAbsolutePath());
080    MetaDescriptorWriter.writeMetaFile(tmpMeta, arch);
081    return tmpDir;
082  }
083
084  @Test
085  public void testRoundtripQuotes() {
086    try {
087      // read archive
088      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("xml-entity-meta"));
089      assertNotNull(arch);
090      assertNotNull(arch.getCore());
091      assertNotNull(arch.getCore().getId());
092      assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
093      assertEquals(1, arch.getExtensions().size());
094
095      // write meta.xml
096      File tmpDwca = createTmpMeta(arch);
097      Archive arch2 = ArchiveFactory.openArchive(tmpDwca);
098
099      // core props
100      ArchiveFile core = arch2.getCore();
101      assertNotNull(core);
102      assertNotNull(core.getId());
103      assertTrue(core.hasTerm(DwcTerm.scientificName));
104      assertEquals("test", core.getLocation());
105      for (ArchiveField f : arch.getCore().getFields().values()) {
106        assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
107        assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
108      }
109
110      // extensions props
111      assertEquals(1, arch2.getExtensions().size());
112      ArchiveFile ext = arch2.getExtensions().iterator().next();
113      assertEquals("test2", ext.getLocation());
114      assertEquals(2, ext.getFields().size());
115
116    } catch (Exception e) {
117      e.printStackTrace();
118      fail();
119    }
120  }
