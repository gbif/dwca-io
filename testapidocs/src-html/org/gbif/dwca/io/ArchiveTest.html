022public class ArchiveTest {
023
024  @Test
025  public void testIteratorDwc() throws Exception {
026    File csv = FileUtils.getClasspathFile("null_data.txt");
027    CSVReader reader = new CSVReader(csv, "utf8", "\t", null, 1);
028
029    Archive arch = new Archive();
030    ArchiveField id = new ArchiveField(0, null, null, null);
031    ArchiveField taxStatus = new ArchiveField(1, DwcTerm.taxonomicStatus, "a", null);
032    ArchiveField dcmodified = new ArchiveField(6, DcTerm.modified, "mod", null);
033
034    ArchiveFile core = mock(ArchiveFile.class);
035    when(core.getCSVReader()).thenReturn(reader);
036    when(core.getId()).thenReturn(id);
037    when(core.getArchive()).thenReturn(arch);
038    when(core.getField(Matchers.<Term>any())).thenReturn(taxStatus);
039    Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
040    fields.put(DwcTerm.taxonomicStatus, taxStatus);
041    fields.put(DcTerm.modified, dcmodified);
042    when(core.getFields()).thenReturn(fields);
043    when(core.hasTerm(eq(DwcTerm.taxonomicStatus))).thenReturn(true);
044    when(core.hasTerm(eq(DcTerm.modified))).thenReturn(true);
045    arch.setCore(core);
046
047    Iterator<DarwinCoreRecord> iter = arch.iteratorDwc();
048    while (iter.hasNext()) {
049      DarwinCoreRecord rec = iter.next();
050      assertEquals("a", rec.getTaxonomicStatus());
051      assertEquals("mod", rec.getModified());
052    }
053  }
054
055  @Test
056  public void testConstituents() throws Exception {
057    File dir = FileUtils.getClasspathFile("constituentsdwca");
058
059    Archive arch = new Archive();
060    arch.setLocation(dir);
061    arch.setMetadataLocation("eml.xml");
062    ArchiveField id = new ArchiveField(0, null, null, null);
063    ArchiveField datasetId = new ArchiveField(1, DwcTerm.datasetID, null, null);
064    ArchiveField sciname = new ArchiveField(2, DwcTerm.scientificName, null, null);
065
066    Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
067    fields.put(DwcTerm.taxonomicStatus, sciname);
068    fields.put(DwcTerm.datasetID, datasetId);
069
070    Map<String, File> cons = arch.getConstituentMetadata();
071    assertEquals(6, cons.size());
072    for (Map.Entry<String, File> c : cons.entrySet()) {
073      final String name = c.getKey();
074      final File file = c.getValue();
075      assertEquals(name, file.getName().split("\\.")[0]);
076    }
077  }
