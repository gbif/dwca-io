019import org.gbif.dwc.terms.DwcTerm;
020import org.gbif.dwc.terms.Term;
021import org.gbif.dwca.record.Record;
022import org.gbif.utils.file.FileUtils;
023
024import java.io.IOException;
025import java.util.Map;
026import java.util.Optional;
027
028import org.junit.Test;
029
030import static org.junit.Assert.assertEquals;
031import static org.junit.Assert.assertFalse;
032import static org.junit.Assert.assertNotNull;
033import static org.junit.Assert.assertNull;
034import static org.junit.Assert.assertTrue;
035
036/**
037 * Unit test related to {@link ArchiveFile}.
038 */
039public class ArchiveFileTest {
040
041  @Test
042  public void testIterator() throws UnsupportedArchiveException, IOException {
043    // test proper archive
044    Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc/DarwinCore.txt"));
045    ArchiveFile af = arch.getCore();
046
047    assertNotNull(af);
048    assertNotNull(af.getId());
049    assertTrue(af.hasTerm(DwcTerm.scientificName));
050
051    // test iterator
052    int counter = 0;
053    Record last = null;
054    for (Record rec : af) {
055      counter++;
056      last = rec;
057      if (counter == 1) {
058        assertEquals("1559060", rec.id());
059      }
060    }
061    assertEquals(3248, counter);
062    assertEquals("3082", last.id());
063  }
064
065  @Test
066  public void testRowTypeEquivalence() {
067    ArchiveFile af = new ArchiveFile();
068    af.setRowType(DwcTerm.Occurrence);
069    assertEquals(af.getRowType(), DwcTerm.Occurrence);
070    assertTrue(af.getRowType().equals(DwcTerm.Occurrence));
071    assertTrue(af.getRowType().qualifiedName().equals(DwcTerm.Occurrence.qualifiedName()));
072  }
073
074  @Test
075  public void testGetHeader() throws UnsupportedArchiveException, IOException {
076    ArchiveFile core = getCore("archive-dwc/DarwinCore.txt");
077    Term[] header = core.getHeader();
078    assertEquals(12, header.length);
079    assertEquals(DwcTerm.scientificName, header[2]);
080  }
081
082  @Test
083  public void testIdWithTermAssociated() throws UnsupportedArchiveException, IOException {
084    ArchiveFile core = getCore("meta-xml-variants/dwca-id-with-term");
085    Term[] header = core.getHeader();
086    assertEquals(5, header.length);
087    assertEquals(DwcTerm.occurrenceID, header[0]);
088    assertNull(header[3]);
089  }
090
091  @Test
092  public void testDefaultValues() throws UnsupportedArchiveException, IOException {
093    ArchiveFile core = getCore("meta-xml-variants/dwca-id-with-term");
094    Optional<Map<Term, String>> defaultValues = core.getDefaultValues();
095    assertTrue(defaultValues.isPresent());
096    assertEquals("Plantae", defaultValues.get().get(DwcTerm.kingdom));
097  }
098
099  @Test
100  public void testEmptyArchiveFile() {
101    ArchiveFile archiveFile = new ArchiveFile();
102    Term[] header = archiveFile.getHeader();
103    assertNotNull(header);
104    assertFalse(archiveFile.getDefaultValues().isPresent());
105  }
106
107  private ArchiveFile getCore(String testFilePath) throws IOException {
108    Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile(testFilePath));
109    return arch.getCore();
110  }
111
112}
