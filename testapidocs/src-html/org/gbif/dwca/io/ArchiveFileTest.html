017package org.gbif.dwca.io;
018
019import org.gbif.dwc.terms.DwcTerm;
020import org.gbif.dwca.io.Archive;
021import org.gbif.dwca.io.ArchiveFactory;
022import org.gbif.dwca.io.ArchiveFile;
023import org.gbif.dwca.io.UnsupportedArchiveException;
024import org.gbif.dwca.record.Record;
025import org.gbif.utils.file.FileUtils;
026
027import java.io.IOException;
028
029import org.junit.Test;
030
031import static org.junit.Assert.assertEquals;
032import static org.junit.Assert.assertNotNull;
033import static org.junit.Assert.assertTrue;
034
035public class ArchiveFileTest {
036
037  @Test
038  public void testIterator() throws UnsupportedArchiveException, IOException {
039    // test proper archive
040    Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc/DarwinCore.txt"));
041    ArchiveFile af = arch.getCore();
042
043    assertNotNull(af);
044    assertNotNull(af.getId());
045    assertTrue(af.hasTerm(DwcTerm.scientificName));
046
047    // test iterator
048    int counter = 0;
049    Record last = null;
050    for (Record rec : af) {
051      counter++;
052      last = rec;
053      if (counter == 1) {
054        assertEquals("1559060", rec.id());
055      }
056    }
057    assertEquals(3248, counter);
058    assertEquals("3082", last.id());
059  }
060
061  @Test
062  public void testRowTypeEquivalence() {
063    ArchiveFile af = new ArchiveFile();
064    af.setRowType(DwcTerm.Occurrence);
065    assertEquals(af.getRowType(), DwcTerm.Occurrence);
066    assertTrue(af.getRowType().equals(DwcTerm.Occurrence));
067    assertTrue(af.getRowType().qualifiedName().equals(DwcTerm.Occurrence.qualifiedName()));
068  }
069}
