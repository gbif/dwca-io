048  public class SAXExtractTerms extends DefaultHandler2 {
049    private final List<String> terms;
050    public SAXExtractTerms(List<String> terms) {
051      this.terms = terms;
052    }
053
054    @Override
055    public void startElement(String uri, String localName, String qName, Attributes atts) throws SAXException {
056      for (String attName : Lists.newArrayList("rowType", "term")) {
057        if (atts.getValue(attName) != null) {
058          terms.add(atts.getValue(attName));
059        }
060      }
061    }
062  }

064  @Test
065  public void testXml() throws Exception {
066    // read archive
067    Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
068
069    // write meta.xml
070    File tmpMeta = File.createTempFile("meta", ".xml");
071    System.out.println("Writing temporary test meta file to " + tmpMeta.getAbsolutePath());
072    MetaDescriptorWriter.writeMetaFile(tmpMeta, arch);
073
074    // validate xml
075    System.out.println("Validate xml");
076    MetaValidator.validate(new FileInputStream(tmpMeta));
077
078    // verify rowType & terms are URIs
079    List<String> terms = Lists.newArrayList();
080    SAXParserFactory spf = SAXParserFactory.newInstance();
081    spf.setNamespaceAware(true);
082    SAXParser saxParser = spf.newSAXParser();
083    XMLReader xmlReader = saxParser.getXMLReader();
084    xmlReader.setContentHandler(new SAXExtractTerms(terms));
085    xmlReader.parse(new InputSource(new FileInputStream(tmpMeta)));
086
087    assertEquals(18, terms.size());
088    for (String term : terms) {
089      URI uri = URI.create(term);
090      assertNotNull(uri + " is no full URI term", uri.getScheme());
091      assertNotNull(uri + " is no full URI term", uri.getAuthority());
092      assertNotNull(uri + " is no full URI term", uri.getPath());
093    }
094  }

096  @Test
097  public void testRoundtrip() {
098    try {
099      // read archive
100      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
101      assertNotNull(arch);
102      assertNotNull(arch.getCore());
103      assertTrue(arch.getCore().getId().getIndex() == 0);
104      assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
105      assertEquals(2, arch.getExtensions().size());
106      assertEquals("\t", arch.getCore().getFieldsTerminatedBy());
107      assertNull(arch.getCore().getField(DwcTerm.scientificName).getDelimitedBy());
108      assertEquals(";", arch.getCore().getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
109      assertEquals(NOMENCLATURAL_CODE_VOCABULARY, arch.getCore().getField(DwcTerm.nomenclaturalCode).getVocabulary());
110      assertEquals(LANGUAGE_VOCABULARY, arch.getExtension(GbifTerm.VernacularName).getField(DcTerm.language).getVocabulary());
111
112      // write meta.xml
113      File tmpDwca = createTmpMeta(arch);
114
115      Archive arch2 = ArchiveFactory.openArchive(tmpDwca);
116      // core props
117      ArchiveFile core = arch2.getCore();
118      assertNotNull(core);
119      assertNotNull(core.getId());
120      assertTrue(core.hasTerm(DwcTerm.scientificName));
121      assertEquals("DarwinCore.txt", core.getLocation());
122      assertEquals("\t", core.getFieldsTerminatedBy());
123      assertNull(core.getField(DwcTerm.scientificName).getDelimitedBy());
124      assertEquals(";", core.getField(DwcTerm.nomenclaturalStatus).getDelimitedBy());
125      assertEquals(NOMENCLATURAL_CODE_VOCABULARY, core.getField(DwcTerm.nomenclaturalCode).getVocabulary());
126
127      for (ArchiveField f : arch.getCore().getFields().values()) {
128        assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
129        assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
130      }
131
132      // extensions props
133      assertEquals(2, arch2.getExtensions().size());
134      Set<String> filenames = Sets.newHashSet("VernacularName.txt", "media.txt");
135      for (ArchiveFile ext : arch2.getExtensions()) {
136        assertTrue(filenames.contains(ext.getLocation()));
137        filenames.remove(ext.getLocation());
138      }
139      assertTrue(filenames.isEmpty());
140
141    } catch (Exception e) {
142      e.printStackTrace();
143      fail();
144    }
145  }

151    System.out.println("Writing temporary test meta file to " + tmpMeta.getAbsolutePath());

156  @Test
157  public void testRoundtripQuotes() {
158    try {
159      // read archive
160      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("xml-entity-meta"));
161      assertNotNull(arch);
162      assertNotNull(arch.getCore());
163      assertNotNull(arch.getCore().getId());
164      assertTrue(arch.getCore().hasTerm(DwcTerm.scientificName));
165      assertEquals(1, arch.getExtensions().size());
166
167      // write meta.xml
168      File tmpDwca = createTmpMeta(arch);
169      Archive arch2 = ArchiveFactory.openArchive(tmpDwca);
170
171      // core props
172      ArchiveFile core = arch2.getCore();
173      assertNotNull(core);
174      assertNotNull(core.getId());
175      assertTrue(core.hasTerm(DwcTerm.scientificName));
176      assertEquals("test", core.getLocation());
177      for (ArchiveField f : arch.getCore().getFields().values()) {
178        assertTrue(core.hasTerm(f.getTerm().qualifiedName()));
179        assertEquals(core.getField(f.getTerm().qualifiedName()).getIndex(), f.getIndex());
180      }
181
182      // extensions props
183      assertEquals(1, arch2.getExtensions().size());
184      ArchiveFile ext = arch2.getExtensions().iterator().next();
185      assertEquals("test2", ext.getLocation());
186      assertEquals(2, ext.getFields().size());
187
188    } catch (Exception e) {
189      e.printStackTrace();
190      fail();
191    }

201    // we can read only a meta.xml file as an Archive

212
213    // validate vocabulary
214    af = arch.getCore().getField(DwcTerm.nomenclaturalCode);
215    assertEquals(NOMENCLATURAL_CODE_VOCABULARY, af.getVocabulary());
216  }
