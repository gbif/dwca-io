001package org.gbif.dwca.record;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwca.io.ArchiveField;
005import org.gbif.dwca.record.DarwinCoreRecord;
006import org.gbif.dwca.record.RecordImpl;
007
008import java.util.Date;
009import java.util.HashSet;
010import java.util.Set;
011
012import com.google.common.collect.ImmutableSet;
013
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017import static org.junit.Assert.fail;
018
019public class DarwinCoreRecordTest {
020
021  @Test
022  public void testProperties() {
023    final Set<String> IGNORE = ImmutableSet.of(DwcTerm.GROUP_MEASUREMENTORFACT, DwcTerm.GROUP_RESOURCERELATIONSHIP);
024
025    DarwinCoreRecord dwc = new DarwinCoreRecord();
026    for (DwcTerm t : DwcTerm.values()) {
027      // only test non class terms
028      if (t.isClass()) {
029        continue;
030      }
031
032      if (IGNORE.contains(t.getGroup())) {
033        try {
034          dwc.getProperty(t);
035          fail("term " + t.qualifiedName() + " should not exist on DarwinCoreRecord");
036        } catch (IllegalArgumentException e) {
037          // expected
038        }
039
040      } else {
041        String val = new Date().toString();
042        dwc.setProperty(t, val);
043        assertEquals("missing term " + t.qualifiedName(), val, dwc.getProperty(t));
044      }
045    }
046  }
047
048  @Test
049  public void testDefaultValue() {
050    final String DATASET = "ITIS";
051    ArchiveField id = new ArchiveField(0, DwcTerm.taxonID, null, null);
052    Set<ArchiveField> fields = new HashSet<ArchiveField>();
053    fields.add(new ArchiveField(1, DwcTerm.datasetName, DATASET, null));
054    RecordImpl rec = new RecordImpl(id, fields, DwcTerm.Taxon, true, true);
055
056    String[] row = {"5432", "IPNI"};
057    rec.setRow(row);
058    assertEquals(row[1], rec.value(DwcTerm.datasetName));
059
060    row = new String[] {"5432", null};
061    rec.setRow(row);
062    assertEquals(DATASET, rec.value(DwcTerm.datasetName));
063
064    row = new String[] {"5432", ""};
065    rec.setRow(row);
066    assertEquals(DATASET, rec.value(DwcTerm.datasetName));
067  }
068}
