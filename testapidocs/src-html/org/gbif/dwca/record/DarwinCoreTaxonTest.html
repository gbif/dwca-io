019  @Test
020  public void testGetFullScientificName() {
021    DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca x brinkmannii A. Braun");
022    dwc.setScientificNameAuthorship("A. Braun");
023    assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
024    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
025
026    dwc = new DarwinCoreTaxon("Festuca x brinkmannii A. Braun");
027    dwc.setScientificNameAuthorship("A.Braun");
028    assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
029    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
030
031    dwc = new DarwinCoreTaxon("Festuca x brinkmannii");
032    dwc.setScientificNameAuthorship("A. Braun");
033    assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
034    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
035
036    dwc = new DarwinCoreTaxon("Festuca A. Braun brinkmannii subsp. brinkmannii");
037    dwc.setScientificNameAuthorship("A. Braun");
038    assertEquals("Festuca A. Braun brinkmannii subsp. brinkmannii", dwc.getFullScientificName());
039    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
040
041    dwc = new DarwinCoreTaxon("Festuca brinkmannii");
042    dwc.setScientificNameAuthorship("A. Braun");
043    assertEquals("Festuca brinkmannii A. Braun", dwc.getFullScientificName());
044    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
045
046    dwc = new DarwinCoreTaxon(" ");
047    dwc.setScientificNameAuthorship("Rafinesque, 1795");
048    dwc.setKingdom("Plantae");
049    assertNull(dwc.getFullScientificName());
050    assertEquals("Plantae", dwc.getLowestScientificName());
051
052    dwc = new DarwinCoreTaxon(" ");
053    dwc.setScientificNameAuthorship("L.");
054    dwc.setKingdom("Plantae");
055    dwc.setFamily("Asteraceae");
056    assertNull(dwc.getFullScientificName());
057    assertEquals("Asteraceae", dwc.getLowestScientificName());
058
059    dwc = new DarwinCoreTaxon();
060    dwc.setScientificNameAuthorship("L.");
061    dwc.setKingdom("Plantae");
062    dwc.setGenus("Poa");
063    assertEquals("Poa", dwc.getFullScientificName());
064    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
065
066    dwc.setSpecificEpithet("annua");
067    assertEquals("Poa annua L.", dwc.getFullScientificName());
068    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
069
070    dwc.setScientificNameAuthorship("");
071    assertEquals("Poa annua", dwc.getFullScientificName());
072    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
073
074    dwc.setScientificNameAuthorship("L.");
075    dwc.setInfraspecificEpithet("alpina");
076    assertEquals("Poa annua alpina L.", dwc.getFullScientificName());
077    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
078
079    dwc.setVerbatimTaxonRank("var.");
080    assertEquals("Poa annua alpina L.", dwc.getFullScientificName());
081    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
082
083    // test not identical authorships
084    dwc = new DarwinCoreTaxon();
085    dwc.setScientificName("Aagaardia protensa Saether, 2000");
086    dwc.setGenus("Aagaardia");
087    dwc.setSpecificEpithet("protensa");
088    dwc.setScientificNameAuthorship("Saether 2000");
089    assertEquals(dwc.getScientificName(), dwc.getFullScientificName());
090
091    // test genus
092    dwc = new DarwinCoreTaxon();
093    dwc.setScientificName("Aagaardia protensa Saether, 2000");
094    dwc.setGenus("Aagaardia");
095    dwc.setSpecificEpithet("protensa");
096    dwc.setScientificNameAuthorship("Saether 2000");
097    assertEquals(dwc.getScientificName(), dwc.getFullScientificName());
098
099    // test authors with single letter that is part of the name
100    dwc = new DarwinCoreTaxon();
101    dwc.setScientificName("Trifolium repens");
102    dwc.setGenus("Trifolium");
103    dwc.setSpecificEpithet("repens");
104    dwc.setScientificNameAuthorship("L.");
105    assertEquals("Trifolium repens", dwc.getScientificName());
106    assertEquals("Trifolium repens L.", dwc.getFullScientificName());
107
108    dwc = new DarwinCoreTaxon();
109    dwc.setScientificName("Trifolium thalii");
110    dwc.setScientificNameAuthorship("Vill.");
111    assertEquals("Trifolium thalii Vill.", dwc.getFullScientificName());
112
113    dwc = new DarwinCoreTaxon();
114    dwc.setScientificName("Trifolium alpestre");
115    dwc.setScientificNameAuthorship("L.");
116    assertEquals("Trifolium alpestre L.", dwc.getFullScientificName());
117
118    // autonyms ???
119    dwc = new DarwinCoreTaxon();
120    dwc.setScientificName("Trifolium alpestre alpestre");
121    dwc.setScientificNameAuthorship("L.");
122    // this would be really correct for autonyms...
123    //assertEquals("Trifolium alpestre L. alpestre", dwc.getFullScientificName());
124    assertEquals("Trifolium alpestre alpestre L.", dwc.getFullScientificName());
125  }

127  @Test
128  public void testInterpretedRank() {
129    DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca brinkmannii A. Braun");
130    dwc.setKingdom("PLANTAE");
131    dwc.setPhylum("Magnoliophyta");
132    dwc.setFamily("Poaceae");
133    dwc.setGenus("Festuca");
134    dwc.setTaxonRank("species");
135    assertEquals("species", dwc.getTaxonRankInterpreted());
136
137    dwc.setSpecificEpithet("brinkmannii");
138    assertEquals("species", dwc.getTaxonRankInterpreted());
139
140    dwc.setInfraspecificEpithet("nonexistii");
141    assertEquals("species", dwc.getTaxonRankInterpreted());
142
143    dwc.setScientificName(null);
144    assertEquals("species", dwc.getTaxonRankInterpreted());
145
146    dwc.setTaxonRank(null);
147    assertEquals("infraspecies", dwc.getTaxonRankInterpreted());
148
149    dwc.setVerbatimTaxonRank("var.");
150    assertEquals("var.", dwc.getTaxonRankInterpreted());
151
152    dwc.setSpecificEpithet(null);
153    assertEquals("genus", dwc.getTaxonRankInterpreted());
154
155    dwc.setGenus(null);
156    assertEquals("family", dwc.getTaxonRankInterpreted());
157  }

159  @Test
160  public void testHigherClassification() {
161    DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca brinkmannii A. Braun");
162    dwc.setKingdom("PLANTAE");
163    dwc.setPhylum("Magnoliophyta");
164    dwc.setFamily("Poaceae");
165    dwc.setGenus("Festuca");
166    assertEquals("Plantae | Magnoliophyta | Poaceae | Festuca", dwc.buildHigherClassification(null));
167    assertEquals("Plantae | Magnoliophyta | Poaceae", dwc.buildHigherClassification('f'));
168    assertEquals("Plantae | Magnoliophyta", dwc.buildHigherClassification('o'));
169  }

171  @Test
172  public void testProperties() {
173    Set<String> groups = ImmutableSet.of(DwcTerm.GROUP_TAXON, DwcTerm.GROUP_RECORD);
174    DarwinCoreTaxon dwc = new DarwinCoreTaxon();
175    for (DwcTerm t : DwcTerm.values()) {
176      if (groups.contains(t.getGroup()) && !t.isClass()){
177        String val = new Date().toString();
178        dwc.setProperty(t, val);
179        try {
180          dwc.getProperty(t);
181          assertEquals(val, dwc.getProperty(t));
182        } catch (Exception e) {
183          System.err.println(e.getMessage());
184        }
185      }
186    }
187  }
