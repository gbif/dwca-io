039  public void testCsvOptionalQuotes() throws IOException {
040    File csv = FileUtils.getClasspathFile("csv_optional_quotes_excel2008CSV.csv");
041    CSVReader reader = new CSVReader(csv, "utf8", ",", '"', 1);
042
043    String[] atom = reader.next();
044    assertEquals(3, atom.length);
045    assertEquals("1", atom[0]);
046    assertEquals("This has a, comma", atom[2]);
047
048    atom = reader.next();
049    assertEquals("I say this is only a \"quote\"", atom[2]);
050
051    atom = reader.next();
052    assertEquals("What though, \"if you have a quote\" and a comma", atom[2]);
053
054    atom = reader.next();
055    assertEquals("What, if we have a \"quote, which has a comma, or 2\"", atom[2]);
056
057    reader.close();
058  }

060  /**
061   * tests the csv reader with different number of header rows on the same file and compares the 4th line in the text
062   * file for each of them
063   */
064  @Test
065  public void testHeaderRows() throws IOException {
066    File source = FileUtils.getClasspathFile("iucn100.csv");
067    // assert the headers are the same, no matter how many rows we skip for the iterator
068    CSVReader reader = new CSVReader(source, "utf8", ",", '"', 1);
069    reader.next();
070    reader.next();
071    String[] row4h1 = reader.next();
072    reader.close();
073
074    reader = new CSVReader(source, "utf8", ",", '"', 0);
075    reader.next();
076    reader.next();
077    reader.next();
078    String[] row4h0 = reader.next();
079    reader.close();
080
081    reader = new CSVReader(source, "utf8", ",", '"', 3);
082    String[] row4h3 = reader.next();
083    reader.close();
084
085    assertTrue(row4h0.length == row4h1.length);
086    assertTrue(row4h0.length == row4h3.length);
087    int idx = row4h0.length;
088    while (idx > 0) {
089      idx--;
090      assertEquals(row4h0[idx], row4h1[idx]);
091      assertEquals(row4h0[idx], row4h3[idx]);
092    }
093  }

112  @Test
113  public void testIgnoreEmptyLines() throws IOException {
114    File csv = FileUtils.getClasspathFile("empty_line.tab");
115    CSVReader reader = new CSVReader(csv, "utf8", "\t", null, 1);
116    String[] ids = {"1", "5", "10", "12", "14", "20"};
117    int row = 0;
118    while (reader.hasNext()) {
119      String[] rec = reader.next();
120      assertEquals(ids[row], rec[0]);
121      row++;
122    }
123    assertTrue(reader.getEmptyLines().size() > 1);
124    assertTrue(reader.getEmptyLines().contains(6));
125    assertTrue(reader.getEmptyLines().contains(9));
126
127  }
