017package org.gbif.utils.html;
018
019import com.google.common.collect.ImmutableSet;
020import org.junit.Test;
021import org.owasp.html.PolicyFactory;
022import org.owasp.html.Sanitizers;
023
024import static junit.framework.TestCase.assertEquals;
025import static org.junit.Assert.assertTrue;
026
027/**
028 * Unit test for BeanHtmlSanitizer.
029 *
030 */
031public class BeanHtmlSanitizerTest {
032
033  @Test
034  public void testBeanHtmlSanitizer(){
035    //Pre-packaged HTML sanitizer policies that allows common block elements including <p>, <h1>, etc.
036    PolicyFactory policy = Sanitizers.BLOCKS;
037    TestBean testBean = new TestBean();
038    testBean.setCount(1);
039    testBean.setComment("Help to find your way on the Internet");
040    testBean.setUnsafeHtml("<p>      <a href=\"http://perdu.com/\" class=\"important\">" +
041            "        Help" +
042            "      </a></p>");
043
044    //demonstrate that inner beans are NOT sanitized
045    TestInnerBean innerBean = new TestInnerBean();
046    innerBean.setInnerString("<h1><script type=\"text/javascript\">echo(\"oops\");<script></h1");
047    testBean.setInnerBean(innerBean);
048
049    BeanHtmlSanitizer.sanitize(testBean, policy);
050
051    assertEquals("<p>Help</p>", testBean.getUnsafeHtml().replaceAll(" ", ""));
052    //ensure other fields were untouched
053    assertEquals("Help to find your way on the Internet", testBean.getComment());
054    assertEquals(1, testBean.getCount());
055    assertEquals("<h1><script type=\"text/javascript\">echo(\"oops\");<script></h1", testBean.getInnerBean().getInnerString());
056
057    // but we can sanitize the inner bean explicitly
058    BeanHtmlSanitizer.sanitize(innerBean, policy);
059    assertEquals("<h1></h1>", testBean.getInnerBean().getInnerString());
060  }
061
062  @Test
063  public void testBeanHtmlSanitizerExclusionList(){
064    PolicyFactory policy = Sanitizers.BLOCKS;
065    TestBean testBean = new TestBean();
066    testBean.setCount(1);
067    testBean.setComment("Help to find your way on the Internet");
068    testBean.setUnsafeHtml("<p>      <a href=\"http://perdu.com/\" class=\"important\">" +
069            "        Help" +
070            "      </a></p>");
071    // ask to exclude "unsafeHtml" property from sanitization
072    BeanHtmlSanitizer.sanitize(testBean, policy, ImmutableSet.of("unsafeHtml"));
073    assertTrue(testBean.getUnsafeHtml().contains("<a href=\"http://perdu.com/\" class=\"important\">"));
074  }
075
076
077  /**
078   * Requires to be public for testing BeanHtmlSanitizer.
079   *
080   */
081  public static class TestBean{
082    private String unsafeHtml;
083    private String comment;
084    private int count;
085
086    private TestInnerBean innerBean;
087
088    public String getUnsafeHtml() {
089      return unsafeHtml;
090    }
091
092    public void setUnsafeHtml(String unsafeHtml) {
093      this.unsafeHtml = unsafeHtml;
094    }
095
096    public String getComment() {
097      return comment;
098    }
099
100    public void setComment(String comment) {
101      this.comment = comment;
102    }
103
104    public int getCount() {
105      return count;
106    }
107
108    public void setCount(int count) {
109      this.count = count;
110    }
111
112    public TestInnerBean getInnerBean() {
113      return innerBean;
114    }
115
116    public void setInnerBean(TestInnerBean innerBean) {
117      this.innerBean = innerBean;
118    }
119
120  }
121
122  public static class TestInnerBean{
123    private String innerString;
124
125    public String getInnerString() {
126      return innerString;
127    }
128
129    public void setInnerString(String innerString) {
130      this.innerString = innerString;
131    }
132  }
133}
