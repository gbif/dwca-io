017package org.gbif.utils.collection;
018
019import java.util.HashMap;
020import java.util.LinkedList;
021import java.util.Map;
022import java.util.Queue;
023
024import org.junit.Test;
025
026import static org.junit.Assert.assertEquals;
027
028public class MapUtilsTest {
029
030
031  @Test
032  public void testMap() {
033    Map<String, Double> map = new HashMap<String, Double>(4);
034    map.put("A", 99.5);
035    map.put("B", 67.4);
036    map.put("C", 67.5);
037    map.put("D", 67.3);
038
039    Map<String, Double> mapSorted = MapUtils.sortByValue(map);
040
041    assertEquals(4, mapSorted.size());
042
043    Queue<String> expected = new LinkedList<String>();
044    expected.add("D");
045    expected.add("B");
046    expected.add("C");
047    expected.add("A");
048    for (String k : mapSorted.keySet()) {
049      assertEquals(expected.poll(), k);
050    }
051  }
052
053  @Test
054  public void testMap2() {
055    Map<Integer, String> map = new HashMap<Integer, String>(4);
056    map.put(5, "C");
057    map.put(4, "B");
058    map.put(99, "A");
059    map.put(3, "D");
060
061    Map<Integer, String> mapSorted = MapUtils.sortByValue(map);
062
063    assertEquals(4, mapSorted.size());
064
065    Queue<Integer> expected = new LinkedList<Integer>();
066    expected.add(99);
067    expected.add(4);
068    expected.add(5);
069    expected.add(3);
070    for (Integer k : mapSorted.keySet()) {
071      assertEquals(expected.poll(), k);
072    }
073  }
074}
