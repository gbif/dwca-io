016package org.gbif.utils.collection;
017
018import java.util.HashMap;
019import java.util.LinkedList;
020import java.util.Map;
021import java.util.Queue;
022
023import org.junit.jupiter.api.Test;
024
025import static org.junit.jupiter.api.Assertions.assertEquals;
026
027public class MapUtilsTest {
028
029
030  @Test
031  public void testMap() {
032    Map<String, Double> map = new HashMap<String, Double>(4);
033    map.put("A", 99.5);
034    map.put("B", 67.4);
035    map.put("C", 67.5);
036    map.put("D", 67.3);
037
038    Map<String, Double> mapSorted = MapUtils.sortByValue(map);
039
040    assertEquals(4, mapSorted.size());
041
042    Queue<String> expected = new LinkedList<String>();
043    expected.add("D");
044    expected.add("B");
045    expected.add("C");
046    expected.add("A");
047    for (String k : mapSorted.keySet()) {
048      assertEquals(expected.poll(), k);
049    }
050  }
051
052  @Test
053  public void testMap2() {
054    Map<Integer, String> map = new HashMap<Integer, String>(4);
055    map.put(5, "C");
056    map.put(4, "B");
057    map.put(99, "A");
058    map.put(3, "D");
059
060    Map<Integer, String> mapSorted = MapUtils.sortByValue(map);
061
062    assertEquals(4, mapSorted.size());
063
064    Queue<Integer> expected = new LinkedList<Integer>();
065    expected.add(99);
066    expected.add(4);
067    expected.add(5);
068    expected.add(3);
069    for (Integer k : mapSorted.keySet()) {
070      assertEquals(expected.poll(), k);
071    }
072  }
073}
