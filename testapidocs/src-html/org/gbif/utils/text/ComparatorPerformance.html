016package org.gbif.utils.text;
017
018import org.gbif.utils.file.FileUtils;
019
020import java.io.File;
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.Comparator;
025import java.util.LinkedList;
026import java.util.List;
027
028import org.junit.jupiter.api.Test;
029
030/**
031 * @author markus
032 */
033public class ComparatorPerformance {
034
035  private final String ENCODING = "UTF-8";
036
037  /**
038   * Comparing performance for various file soring methods.
039   * <p/>
040   * Executed on a MacPro with 9GB, 8-core 3GHz and 1TB disk
041   * sorting a 207MB large text file made from concatenation of irmng.tail:
042   * <p/>
043   * Sorting with unix sort took 5817 ms
044   * Sorting with org.gbif.utils.text.StringComparator and 10k lines in memory (200 parts) took 48968 ms
045   * Sorting with org.gbif.utils.text.CCollationComparator and 10k lines in memory (200 parts) took 49858 ms
046   * Sorting with org.gbif.utils.text.StringComparator and 100k lines in memory (20 parts) took 17962 ms
047   * Sorting with org.gbif.utils.text.CCollationComparator and 100k lines in memory (20 parts) took 14046 ms
048   * Sorting with org.gbif.utils.text.StringComparator and 1000k lines in memory (2 parts) took 15492 ms
049   * Sorting with org.gbif.utils.text.CCollationComparator and 1000k lines in memory (2 parts) took 14317 ms
050   */
051
052  @Test
053  public void testFileSorting() throws IOException {
054    // 10MB text file, big file used in results above was concatenated from this one
055    File source = FileUtils.getClasspathFile("sorting/irmng.tail");
056    File sorted = File.createTempFile("gbif-common-file-sort2", "sorted.txt");
057    sorted.deleteOnExit();
058    FileUtils futils = new FileUtils();
059
060    // test unix sort
061    long start = System.currentTimeMillis();
062    futils.sort(source, sorted, ENCODING, 0, "\t", null, "\n", 0);
063    long end = System.currentTimeMillis();
064    System.out.printf("Sorting with unix sort took %s ms%n", (end - start));
065
066    // sort with comparator to test
067    List<Comparator<String>> comparators = availableComparators();
068
069    for (Integer linesInMen : Collections.singletonList(100000)) {
070      FileUtils.setLinesPerMemorySort(linesInMen);
071      for (Comparator<String> comp : comparators) {
072
073        start = System.currentTimeMillis();
074        futils.sortInJava(source, sorted, ENCODING, comp, 0);
075        end = System.currentTimeMillis();
076
077        System.out.printf("Sorting with %s and %s lines in memory took %s ms%n", comp.getClass().getName(), linesInMen, (end - start));
078      }
079    }
080  }
081
082  private List<Comparator<String>> availableComparators() {
083    List<Comparator<String>> comparators = new ArrayList<>();
084    comparators.add(new StringComparator());
085    comparators.add(Comparator.nullsFirst(Comparator.naturalOrder()));
086    return comparators;
087  }
088
089  @Test
090  public void testVariousComparators() throws IOException {
091    // sort with comparator to test
092    List<Comparator<String>> comparators = availableComparators();
093    for (Comparator<String> comp : comparators) {
094      LinkedList<String> source = FileUtils.streamToList(FileUtils.classpathStream("sorting/irmng.tail"));
095      long start = System.currentTimeMillis();
096      source.sort(comp);
097      long end = System.currentTimeMillis();
098      System.out.println("Sorting with " + comp.getClass().getName() + " took " + (end - start) + " ms");
099    }
100  }
101}
