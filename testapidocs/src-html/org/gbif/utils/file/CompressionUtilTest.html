039  public static File createTempDirectory() throws IOException {
040
041    final File temp = File.createTempFile("temp", Long.toString(System.nanoTime()));
042
043    if (!temp.delete()) {
044      throw new IOException("Could not delete temp file: " + temp.getAbsolutePath());
045    }
046
047    if (!temp.mkdir()) {
048      throw new IOException("Could not create temp directory: " + temp.getAbsolutePath());
049    }
050    temp.deleteOnExit();
051
052    return temp;
053  }
054
055  private static void assureEqualContent(List<File> result, String metaContent, String dataContent) throws IOException {
056    for (File f : result) {
057      String x = FileUtils.readFileToString(f, "utf-8");
058      if ("meta.xml".equals(f.getName())) {
059        assertEquals(metaContent, x);
060      } else if ("quote_in_quote.csv".equals(f.getName())) {
061        assertEquals(dataContent, x);
062      } else {
063        fail("unexpected file");
064      }
065    }
066  }
067
068  public File classpathFile(String path) {
069    File f = null;
070    // relative path. Use classpath instead
071    URL url = getClass().getClassLoader().getResource(path);
072    if (url != null) {
073      f = new File(url.getFile());
074    }
075    return f;
076  }

078  @Test
079  public void testDecompress() throws IOException {
080    // meta.xml
081    File meta = classpathFile("compression/archive/meta.xml");
082    String metaContent = FileUtils.readFileToString(meta, "utf-8");
083    // quote_in_quote.csv
084    File data = classpathFile("compression/archive/quote_in_quote.csv");
085    String dataContent = FileUtils.readFileToString(data, "utf-8");
086
087    File tmpDir = createTempDirectory();
088    File testArchiveFile = classpathFile("compression/archive.zip");
089    List<File> result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
090    assertEquals(2, result.size());
091    assureEqualContent(result, metaContent, dataContent);
092
093    FileUtils.cleanDirectory(tmpDir);
094    testArchiveFile = classpathFile("compression/archive.tgz");
095    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
096    assertEquals(2, result.size());
097    assureEqualContent(result, metaContent, dataContent);
098
099    FileUtils.cleanDirectory(tmpDir);
100    testArchiveFile = classpathFile("compression/archive-zip.dat");
101    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
102    assertEquals(2, result.size());
103    assureEqualContent(result, metaContent, dataContent);
104
105    FileUtils.cleanDirectory(tmpDir);
106    testArchiveFile = classpathFile("compression/archive-tgz.dat");
107    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
108    assertEquals(2, result.size());
109    assureEqualContent(result, metaContent, dataContent);
110
111    FileUtils.cleanDirectory(tmpDir);
112    testArchiveFile = classpathFile("compression/archive.tar");
113    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
114    assertEquals(2, result.size());
115    assureEqualContent(result, metaContent, dataContent);
116
117    FileUtils.cleanDirectory(tmpDir);
118    testArchiveFile = classpathFile("compression/cate.zip");
119    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
120    assertEquals(3, result.size());

279  @Test
280  public void testZipFolder() throws IOException {
281    File zipWithDirs = File.createTempFile("aha", ".zip");
282    System.out.println(zipWithDirs.getAbsolutePath());
283
284    File zipWithoutDirs = File.createTempFile("aha", ".zip");
285    System.out.println(zipWithoutDirs.getAbsolutePath());
286    //tmp.deleteOnExit();
287    File testFolder = classpathFile("charsets");
288    // remember how many files we have in the root folder, exlcuding files in subdirectories
289    final int rootFileNum = testFolder.listFiles().length;
290
291    CompressionUtil.zipDir(testFolder, zipWithDirs, true);
292    CompressionUtil.zipDir(testFolder, zipWithoutDirs, false);
293
294    assertNotEquals(zipWithDirs.length(), zipWithoutDirs.length());
295
296    // now decompress the subdir zip and make sure we get the same amount of root files
297    File tmpDir = org.gbif.utils.file.FileUtils.createTempDir();
298    tmpDir.deleteOnExit();
299    CompressionUtil.unzipFile(tmpDir, zipWithDirs, true);
300    int decompressedRootFileNum = tmpDir.listFiles().length;
301    assertEquals(rootFileNum, decompressedRootFileNum);
302  }

321  /**
322   * Check that files are closed after use.
323   */
324  @Test
325  public void testFilesClosedCorrectly() throws Exception {
326    File tmpDir = createTempDirectory();
327    File testZippedFolder = classpathFile("compression/infozip64.zip");
328
329    // Unzip first, to use the code path.  (Various things like /dev/random are opened.)
330    CompressionUtil.unzipFile(tmpDir, testZippedFolder);
331
332    OperatingSystemMXBean os = ManagementFactory.getOperatingSystemMXBean();
333    long openFiles = 0;
334    if (os instanceof UnixOperatingSystemMXBean) {
335      openFiles = ((UnixOperatingSystemMXBean) os).getOpenFileDescriptorCount();
336    }
337
338    // From all the other tests.
339    List<String> files = Arrays.asList(
340      "compression/archive/meta.xml",
341      "compression/archive/quote_in_quote.csv",
342      "compression/archive.tgz",
343      "compression/archive-tgz.dat",
344      "compression/archive.zip",
345      "compression/archive-zip.dat",
346      "compression/cate.zip",
347      "compression/infozip64.zip",
348      "compression/test.txt.gz",
349      "compression/with_dot_svn.zip",
350      "compression/withSubdirsAndHiddenFiles.zip"
351    );
352
353    for (String file : files) {
354      tmpDir = createTempDirectory();
355      CompressionUtil.decompressFile(tmpDir, classpathFile(file));
356    }
357
358    if (os instanceof UnixOperatingSystemMXBean) {
359      assertEquals(openFiles, ((UnixOperatingSystemMXBean) os).getOpenFileDescriptorCount());
360    } else {
361      System.err.println("Cannot check files are closed except on Unix.");
362    }
363
364    // Try ls -l /proc/`pgrep -f java -n`/fd
365    // Thread.sleep(30000);
366  }
