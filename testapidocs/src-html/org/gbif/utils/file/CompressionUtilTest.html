042  public static File createTempDirectory() throws IOException {
043
044    final File temp = File.createTempFile("temp", Long.toString(System.nanoTime()));
045
046    if (!temp.delete()) {
047      throw new IOException("Could not delete temp file: " + temp.getAbsolutePath());
048    }
049
050    if (!temp.mkdir()) {
051      throw new IOException("Could not create temp directory: " + temp.getAbsolutePath());
052    }
053    temp.deleteOnExit();
054
055    return temp;
056  }

058  private static void assureEqualContent(List<File> result, String metaContent, String dataContent) throws IOException {
059    for (File f : result) {
060      String x = FileUtils.readFileToString(f, "utf-8");
061      if ("meta.xml".equals(f.getName())) {
062        assertEquals(metaContent, x);
063      } else if ("quote_in_quote.csv".equals(f.getName())) {
064        assertEquals(dataContent, x);
065      } else {
066        fail("unexpected file");
067      }
068    }
069  }
070
071  public File classpathFile(String path) {
072    File f = null;
073    // relative path. Use classpath instead
074    URL url = getClass().getClassLoader().getResource(path);
075    if (url != null) {
076      f = new File(url.getFile());
077    }
078    return f;
079  }

081  @Test
082  public void testDecompress() throws IOException {
083    // meta.xml
084    File meta = classpathFile("compression/archive/meta.xml");
085    String metaContent = FileUtils.readFileToString(meta, "utf-8");
086    // quote_in_quote.csv
087    File data = classpathFile("compression/archive/quote_in_quote.csv");
088    String dataContent = FileUtils.readFileToString(data, "utf-8");
089
090    File tmpDir = createTempDirectory();
091    File testArchiveFile = classpathFile("compression/archive.zip");
092    List<File> result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
093    assertEquals(2, result.size());
094    assureEqualContent(result, metaContent, dataContent);
095
096    FileUtils.cleanDirectory(tmpDir);
097    testArchiveFile = classpathFile("compression/archive.tgz");
098    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
099    assertEquals(2, result.size());
100    assureEqualContent(result, metaContent, dataContent);
101
102    FileUtils.cleanDirectory(tmpDir);
103    testArchiveFile = classpathFile("compression/archive-zip.dat");
104    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
105    assertEquals(2, result.size());
106    assureEqualContent(result, metaContent, dataContent);
107
108    FileUtils.cleanDirectory(tmpDir);
109    testArchiveFile = classpathFile("compression/archive-tgz.dat");
110    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
111    assertEquals(2, result.size());
112    assureEqualContent(result, metaContent, dataContent);
113
114    FileUtils.cleanDirectory(tmpDir);
115    testArchiveFile = classpathFile("compression/cate.zip");
116    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
117    assertEquals(3, result.size());
118  }

264  @Test
265  public void testZipFolder() throws IOException {
266    File zipWithDirs = File.createTempFile("aha", ".zip");
267    System.out.println(zipWithDirs.getAbsolutePath());
268
269    File zipWithoutDirs = File.createTempFile("aha", ".zip");
270    System.out.println(zipWithoutDirs.getAbsolutePath());
271    //tmp.deleteOnExit();
272    File testFolder = classpathFile("charsets");
273    // remember how many files we have in the root folder, exlcuding files in subdirectories
274    final int rootFileNum = testFolder.listFiles().length;
275
276    CompressionUtil.zipDir(testFolder, zipWithDirs, true);
277    CompressionUtil.zipDir(testFolder, zipWithoutDirs, false);
278
279    assertNotEquals(zipWithDirs.length(), zipWithoutDirs.length());
280
281    // now decompress the subdir zip and make sure we get the same amount of root files
282    File tmpDir = org.gbif.utils.file.FileUtils.createTempDir();
283    tmpDir.deleteOnExit();
284    CompressionUtil.unzipFile(tmpDir, zipWithDirs, true);
285    int decompressedRootFileNum = tmpDir.listFiles().length;
286    assertEquals(rootFileNum, decompressedRootFileNum);
287  }

306  /**
307   * Check that files are closed after use.
308   */
309  @Test
310  public void testFilesClosedCorrectly() throws Exception {
311    File tmpDir = createTempDirectory();
312    File testZippedFolder = classpathFile("compression/infozip64.zip");
313
314    // Unzip first, to use the code path.  (Various things like /dev/random are opened.)
315    CompressionUtil.unzipFile(tmpDir, testZippedFolder);
316
317    OperatingSystemMXBean os = ManagementFactory.getOperatingSystemMXBean();
318    long openFiles = 0;
319    if (os instanceof UnixOperatingSystemMXBean) {
320      openFiles = ((UnixOperatingSystemMXBean) os).getOpenFileDescriptorCount();
321    }
322
323    // From all the other tests.
324    List<String> files = ImmutableList.of(
325      "compression/archive/meta.xml",
326      "compression/archive/quote_in_quote.csv",
327      "compression/archive.tgz",
328      "compression/archive-tgz.dat",
329      "compression/archive.zip",
330      "compression/archive-zip.dat",
331      "compression/cate.zip",
332      "compression/infozip64.zip",
333      "compression/test.txt.gz",
334      "compression/with_dot_svn.zip",
335      "compression/withSubdirsAndHiddenFiles.zip"
336    );
337
338    for (String file : files) {
339      tmpDir = createTempDirectory();
340      CompressionUtil.decompressFile(tmpDir, classpathFile(file));
341    }
342
343    if (os instanceof UnixOperatingSystemMXBean) {
344      assertEquals(openFiles, ((UnixOperatingSystemMXBean) os).getOpenFileDescriptorCount());
345    } else {
346      System.err.println("Cannot check files are closed except on Unix.");
347    }
348
349    // Try ls -l /proc/`pgrep -f java -n`/fd
350    // Thread.sleep(30000);
351  }
