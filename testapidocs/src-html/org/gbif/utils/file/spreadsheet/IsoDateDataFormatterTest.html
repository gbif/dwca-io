001package org.gbif.utils.file.spreadsheet;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class IsoDateDataFormatterTest {
008
009  @Test
010  public void testIsoDateDateFormatter() {
011    IsoDateDataFormatter iddf = new IsoDateDataFormatter();

013    assertEquals("1990-01-02", iddf.formatRawCellContents(32875.0, 0, "mm/dd/yy"));

015  }
