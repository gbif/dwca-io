001package org.gbif.utils.file.spreadsheet;
002
003import org.gbif.utils.file.FileUtils;
004
005import java.io.File;
006import java.io.FileWriter;
007import java.io.IOException;
008
009import org.apache.poi.openxml4j.exceptions.InvalidFormatException;
010import org.junit.Rule;
011import org.junit.Test;
012import org.junit.rules.TemporaryFolder;
013
014import static junit.framework.TestCase.assertEquals;
015import static org.junit.Assert.assertTrue;
016
017/**
018 * {@link ExcelConverter} related tests
019 */
020public class ExcelConverterTest {
021
022  @Rule
023  public TemporaryFolder folder = new TemporaryFolder();
024
025  private static final String CSV_TEST_FILE_LOCATION = "spreadsheet/occurrence-workbook.csv";
026  private static final String XLSX_TEST_FILE_LOCATION = "spreadsheet/occurrence-workbook.xlsx";
027  private static final String XLS_TEST_FILE_LOCATION = "spreadsheet/occurrence-workbook.xls";
028
029  private final File testCsvFile = FileUtils.getClasspathFile(CSV_TEST_FILE_LOCATION);
030  private final File testXlsFile = FileUtils.getClasspathFile(XLS_TEST_FILE_LOCATION);
031  private final File testXlsxFile = FileUtils.getClasspathFile(XLSX_TEST_FILE_LOCATION);
032
033  @Test
034  public void testExcelConverter() throws Exception {
035    File testFile = folder.newFile();
036    //testFile = new File("/tmp/test.csv");
037    //System.out.println(testFile.toPath());
038
039    long lines = ExcelConverter.convert(testXlsFile.toPath(),
040      new CsvSpreadsheetConsumer(new FileWriter(testFile)));
041
042    assertTrue(org.apache.commons.io.FileUtils.contentEqualsIgnoreEOL(testFile, testCsvFile, "UTF-8"));
043    assertEquals(7L, lines);
044  }
045
046  @Test
047  public void testExcelXmlConverter() throws Exception {
048    File testFile = folder.newFile();
049    //testFile = new File("/tmp/test.csv");
050    //System.out.println(testFile.toPath());
051
052    long lines = ExcelXmlConverter.convert(testXlsxFile.toPath(),
053      new CsvSpreadsheetConsumer(new FileWriter(testFile)));
054
055    assertTrue(org.apache.commons.io.FileUtils.contentEqualsIgnoreEOL(testFile, testCsvFile, "UTF-8"));
056    assertEquals(7L, lines);
057  }
058}
