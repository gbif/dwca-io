048  @Test
049  public void testComputeDelimiterFrequencySums() {
050    List<String> sample = new ArrayList<>();
051    sample.add("ID\tName\tName2\tName3");
052    sample.add("1\ta\tb\tc,1");
053    sample.add("2\tc\td\te,2");
054    sample.add("3\tf\tg\th,3");
055
056    List<TabularFileMetadataExtractor.LineDelimiterStats> linesStats =
057            computeLineDelimiterStats(sample);
058    Map<Character, Integer> delimiterFrequencySums = TabularFileMetadataExtractor.computeDelimiterFrequencySums(linesStats);
059    // here, the delimiter that is used the most often is in fact the correct one
060    assertEquals(12, delimiterFrequencySums.get('\t').intValue());
061    assertEquals(3, delimiterFrequencySums.get(',').intValue());
062
063    //add a "noise" line to demonstrate the impact on this function
064    sample.add("4\ti\tj\tk,,,,,,,,,,,,,,,,,,,,,,,,,,,,,,,4");
065    linesStats =
066            computeLineDelimiterStats(sample);
067    delimiterFrequencySums = TabularFileMetadataExtractor.computeDelimiterFrequencySums(linesStats);
068    // here, the delimiter that is used the most often is the wrong one
069    assertEquals(15, delimiterFrequencySums.get('\t').intValue());
070    assertEquals(34, delimiterFrequencySums.get(',').intValue());
071  }

073  @Test
074  public void testComputeDelimiterDistinctFrequency() {
075    List<String> sample = new ArrayList<>();
076    sample.add("ID\tName\tName2\tName3");
077    sample.add("1\ta\tb\tc,1");
078    sample.add("2\tc\td\te,2");
079    sample.add("3\tf\tg\th,3");
080
081    List<TabularFileMetadataExtractor.LineDelimiterStats> linesStats =
082            computeLineDelimiterStats(sample);
083    Map<Character, Set<Integer>> delimiterDistinctFrequency = TabularFileMetadataExtractor.computeDelimiterDistinctFrequency(linesStats);
084
085    // here, the delimiter with the most stable frequency is the correct one
086    assertEquals(1, delimiterDistinctFrequency.get('\t').size());
087    assertEquals(2, delimiterDistinctFrequency.get(',').size());
088
089    sample.add("4\ti\t\"j\t\"\tk,4");
090    sample.add("5\tl\t\"m\t\t\"\tn,5");
091    linesStats =
092            computeLineDelimiterStats(sample);
093    delimiterDistinctFrequency = TabularFileMetadataExtractor.computeDelimiterDistinctFrequency(linesStats);
094    // here, the delimiter that is the most stable is now the wrong one (because of the delimiter inside the quoted text)
095    assertEquals(3, delimiterDistinctFrequency.get('\t').size());
096    assertEquals(2, delimiterDistinctFrequency.get(',').size());
097  }

099  @Test
100  public void testComputeDelimiterHighestFrequencyPerLine() {
101    List<String> sample = new ArrayList<>();
102    sample.add("ID\tName\tName2\tName3");
103    sample.add("1\ta\tb\tc,1");
104    sample.add("2\tc\td\te,2");
105    sample.add("3\tf\tg\th,3");
106
107    Map<Character, Long> delimiterDistinctFrequency = TabularFileMetadataExtractor.
108      computeDelimiterHighestFrequencyPerLine(sample);
109
110    assertEquals(4, delimiterDistinctFrequency.get('\t').intValue());
111    assertNull(delimiterDistinctFrequency.get(','));
112
113    //this line alone won't have an impact on computeDelimiterHighestFrequencyPerLine result
114    sample.add("4\ti\tj\tk,,,,,,,,,,,,,,,,,,,,,,,,,,,,,,,4");
115    delimiterDistinctFrequency = TabularFileMetadataExtractor.
116            computeDelimiterHighestFrequencyPerLine(sample);
117    assertEquals(4, delimiterDistinctFrequency.get('\t').intValue());
118    assertEquals(1, delimiterDistinctFrequency.get(',').intValue());
119  }

121  @Test
122  public void testExtractTabularMetadata() {
123    List<String> sample = new ArrayList<>();
124    sample.add("OccurrenceID,ScientificName,Locality");
125    sample.add("1,Gadus morhua,\"This has a, comma\"");
126    sample.add("2,Abies alba,\"I say this is only a \"\"quote\"\"\"");
127    sample.add("3,Pomatoma saltatrix,\"What though, \"\"if you have a quote\"\" and a comma\"");
128    sample.add("4,Yikes ofcourses,\"What, if we have a \"\"quote, which has a comma, or 2\"\"\"");
129
130    TabularFileMetadata metadata = TabularFileMetadataExtractor.extractTabularMetadata(sample);
131    assertEquals(Character.valueOf(',').charValue(), metadata.getDelimiter().charValue());
132    assertEquals(Character.valueOf('\"'), metadata.getQuotedBy());
133  }

158  @Test
159  public void testGetDelimiterWithHighestCount() {
160    //no delimiter
161    assertFalse(getDelimiterWithHighestCount("there is no delimiter here").isPresent());
162
163    assertEquals(Character.valueOf(',').charValue(), getDelimiterWithHighestCount("a,b,c,d,e").get().charValue());
164    assertEquals(Character.valueOf('|').charValue(), getDelimiterWithHighestCount("a|b,c|d|e").get().charValue());
165    assertEquals(Character.valueOf('\t').charValue(), getDelimiterWithHighestCount("a\tb\tc\td\te").get().charValue());
166    assertEquals(Character.valueOf(';').charValue(), getDelimiterWithHighestCount("a; b; c; d; e").get().charValue());
167  }

169  @Test
170  public void testGetQuoteCharWithHighestCount() {
171    //no quote character
172    assertFalse(getQuoteCharWithHighestCount("a,b,c,d", ',').isPresent());
173
174    // test double quote character and ensure the result is not affected by another quote character that is not used for quoting
175    assertEquals(Character.valueOf('\"').charValue(), getQuoteCharWithHighestCount("a,\"b,8\",c\'\'\'\'\'\'\'\'\'\'\',d", ',').get().charValue());
176
177    // test single quote character
178    assertEquals(Character.valueOf('\'').charValue(), getQuoteCharWithHighestCount("a,\'b,8\',c,d", ',').get().charValue());
179  }

217  private static void runExtractTabularFileMetadata(String classPathFile, Character expectedDelimiter, Character expectedQuoteChar,
218                                                    Charset expectedCharset) throws IOException {
219    Path source = FileUtils.getClasspathFile(classPathFile).toPath();
220    TabularFileMetadata tabFileMetadata = extractTabularFileMetadata(source);
221    Assert.assertEquals(expectedDelimiter.charValue(), tabFileMetadata.getDelimiter().charValue());
222
223    if(expectedQuoteChar == null) {
224      assertNull(tabFileMetadata.getQuotedBy());
225    }
226    else{
227      assertNotNull("Expect a quote character -> " + source, tabFileMetadata.getQuotedBy());
228      Assert.assertEquals("Source file -> " + source, expectedQuoteChar.charValue(), tabFileMetadata.getQuotedBy().charValue());
229    }
230
231    Assert.assertEquals(expectedCharset, tabFileMetadata.getEncoding());
232  }
