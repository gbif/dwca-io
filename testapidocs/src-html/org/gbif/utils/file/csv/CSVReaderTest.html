049  @Test
050  public void testCsvQuotedDelimiter() throws IOException {
051    String rows =
052            "12,\"not real\"\n"
053                    + "13,not \"real\"\n"
054                    + "\"14\",noting\n"
055                    + "15,\"not \"\"real\"\"\"\n"
056                    + "16,\"no, this is \"\"real\"\"\"\n";
057
058    InputStream stream = new ByteArrayInputStream(rows.getBytes(StandardCharsets.UTF_8));
059    CSVReader reader = new CSVReader(stream, UTF8, ",", '"', 0);
060
061    String[] rec = reader.next();
062    assertEquals(2, rec.length);
063    assertEquals("12", rec[0]);
064    assertEquals("not real", rec[1]);
065
066    rec = reader.next();
067    assertEquals(2, rec.length);
068    assertEquals("13", rec[0]);
069    assertEquals("not \"real\"", rec[1]);
070
071    rec = reader.next();
072    assertEquals(2, rec.length);
073    assertEquals("14", rec[0]);
074    assertEquals("noting", rec[1]);
075
076    rec = reader.next();
077    assertEquals(2, rec.length);
078    assertEquals("15", rec[0]);
079    assertEquals("not \"real\"", rec[1]);
080
081    rec = reader.next();
082    assertEquals(2, rec.length);
083    assertEquals("16", rec[0]);
084    assertEquals("no, this is \"real\"", rec[1]);
085
086    assertFalse(reader.hasNext());
087  }

090   * csv file with optional quotes generated by excel.
091   * single, double quotes and comma within a field are tested.

093  @Test
094  public void testCsvOptionalQuotes() throws IOException {
095    File csv = FileUtils.getClasspathFile("csv/csv_optional_quotes_excel2008.csv");
096    CSVReader reader = new CSVReader(csv, UTF8, ",", '"', 1);
097
098    String[] atom = reader.next();
099    assertEquals(3, atom.length);
100    assertEquals("1", atom[0]);
101    assertEquals("This has a, comma", atom[2]);
102
103    atom = reader.next();
104    assertEquals("I say this is only a \"quote\"", atom[2]);
105
106    atom = reader.next();
107    assertEquals("What though, \"if you have a quote\" and a comma", atom[2]);
108
109    atom = reader.next();
110    assertEquals("What, if we have a \"quote, which has a comma, or 2\"", atom[2]);
111
112    reader.close();
113  }

115  /**
116   * tests the csv reader with different number of header rows on the same file and compares the 4th line in the text
117   * file for each of them
118   */
119  @Test
120  public void testHeaderRows() throws IOException {
121    File source = FileUtils.getClasspathFile("csv/iucn100.csv");
122    // assert the headers are the same, no matter how many rows we skip for the iterator
123    CSVReader reader = new CSVReader(source, UTF8, ",", '"', 1);
124    reader.next();
125    reader.next();
126    String[] row4h1 = reader.next();
127    reader.close();
128
129    reader = new CSVReader(source, UTF8, ",", '"', 0);
130    reader.next();
131    reader.next();
132    reader.next();
133    String[] row4h0 = reader.next();
134    reader.close();
135
136    reader = new CSVReader(source, UTF8, ",", '"', 3);
137    String[] row4h3 = reader.next();
138    reader.close();
139
140    assertTrue(row4h0.length == row4h1.length);
141    assertTrue(row4h0.length == row4h3.length);
142    int idx = row4h0.length;
143    while (idx > 0) {
144      idx--;
145      assertEquals(row4h0[idx], row4h1[idx]);
146      assertEquals(row4h0[idx], row4h3[idx]);
147    }
148  }

165  @Test
166  public void testIgnoreEmptyLines() throws IOException {
167    File csv = FileUtils.getClasspathFile("csv/empty_line.tab");
168    CSVReader reader = new CSVReader(csv, UTF8, "\t", null, 1);
169    String[] ids = {"1", "5", "10", "12", "14", "20", "21", "", "30"};
170    int row = 0;
171    while (reader.hasNext()) {
172      String[] rec = reader.next();
173      assertEquals(ids[row], rec[0]);
174      row++;
175    }
176    assertTrue(reader.getEmptyLines().size() > 1);
177    assertTrue(reader.getEmptyLines().contains(6));
178    assertTrue(reader.getEmptyLines().contains(12));
179  }
