017package org.gbif.utils.file.csv;
018
019import org.gbif.utils.file.FileUtils;
020
021import java.io.File;
022import java.io.IOException;
023
024import org.junit.Test;
025
026import static org.junit.Assert.assertEquals;
027import static org.junit.Assert.assertNull;
028import static org.junit.Assert.assertTrue;
029
030/**
031 *
032 */
033public class CSVReaderFactoryTest {
034
035  @Test
036  public void detectCsvAlwaysQuoted() throws IOException {
037    File source = FileUtils.getClasspathFile("csv/csv_always_quoted.csv");
038    CSVReader reader = CSVReaderFactory.build(source);
039    assertEquals(",", reader.delimiter);
040    assertEquals(new Character('"'), reader.quoteChar);
041    assertEquals(1, reader.headerRows);
042    reader.close();
043  }
044
045  /**
046   * We dont want unquoted CSVs, See detectCsvOptionallyQuoted()
047   */
048  public void detectCsvUnquoted() throws IOException {
049    String[] files = {"csv/csv_unquoted.txt"};
050    for (String fn : files) {
051      File source = FileUtils.getClasspathFile(fn);
052      CSVReader reader = CSVReaderFactory.build(source);
053      assertEquals(",", reader.delimiter);
054      assertNull(reader.quoteChar);
055      assertEquals(1, reader.headerRows);
056      reader.close();
057    }
058  }
059
060  @Test
061  public void detectPipe() throws IOException {
062    String[] files = new String[] {"csv/pipe_separator.txt"};
063    for (String fn : files) {
064      File source = FileUtils.getClasspathFile(fn);
065      CSVReader reader = CSVReaderFactory.build(source);
066      assertEquals("|", reader.delimiter);
067      assertNull(reader.quoteChar);
068      assertEquals(1, reader.headerRows);
069      reader.close();
070    }
071  }
072
073  @Test
074  public void detectSemicolon() throws IOException {
075    String[] files = {"csv/semicolon_separator.csv"};
076    for (String fn : files) {
077      File source = FileUtils.getClasspathFile(fn);
078      CSVReader reader = CSVReaderFactory.build(source);
079      assertEquals(";", reader.delimiter);
080      assertNull(reader.quoteChar);
081      assertEquals(1, reader.headerRows);
082      reader.close();
083    }
084  }
085
086  /**
087   * As CSV files with rare optional quotes are hard to detect but cause problems
088   * we prefer to default to the " quotation in case comma seperated files are used.
089   * This is why test detectCsvUnquoted() is outcommented right now!
090   */
091  @Test
092  public void detectCsvOptionallyQuoted() throws IOException {
093    String[] files = {"csv/csv_optional_quotes_puma.csv", "csv/csv_optional_quotes_excel2008.csv",
094            "csv/csv_incl_single_quotes.csv", "csv/iucn100.csv", "csv/csv_unquoted.txt", "csv/csv_unquoted_coordinates.txt"};
095    for (String fn : files) {
096      File source = FileUtils.getClasspathFile(fn);
097      CSVReader reader = CSVReaderFactory.build(source);
098      assertEquals(",", reader.delimiter);
099      assertEquals(new Character('"'), reader.quoteChar);
100      assertEquals(1, reader.headerRows);
101      reader.close();
102    }
103  }
104
105  @Test
106  public void detectTab() throws IOException {
107    String[] files = {"csv/ipni.tab.txt", "csv/tab_separated_generic.txt", "csv/iucn100.tab.txt", "csv/ebird.tab.txt",
108            "csv/irmng.tail", "csv/MOBOT.tab.csv"};
109    for (String fn : files) {
110      File source = FileUtils.getClasspathFile(fn);
111      CSVReader reader = CSVReaderFactory.build(source);
112      assertEquals("Check " + fn, "\t", reader.delimiter);
113      assertNull(reader.quoteChar);
114      assertEquals(1, reader.headerRows);
115      reader.close();
116    }
117  }
118
119  @Test
120  public void detectTabQuoted() throws IOException {
121    String[] files = {"csv/eol/my_darwincore_tab_separated_quoted.txt",
122            "csv/eol/my_dataobject_tab_separated_quoted.txt", "csv/borza_tab_separated_quoted.txt"};
123    for (String fn : files) {
124      File source = FileUtils.getClasspathFile(fn);
125      CSVReader reader = CSVReaderFactory.build(source);
126      assertEquals("\t", reader.delimiter);
127      assertTrue(reader.quoteChar == '"');
128      assertEquals(1, reader.headerRows);
129      reader.close();
130    }
131  }
132
