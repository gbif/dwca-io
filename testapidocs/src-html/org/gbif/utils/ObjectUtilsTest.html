017package org.gbif.utils;
018
019import java.util.Collection;
020
021import com.google.common.collect.Lists;
022import org.junit.Test;
023
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertNull;
026
027/**
028 *
029 */
030public class ObjectUtilsTest {
031
032  @Test
033  public void testCoalesce() throws Exception {
034    assertNull(ObjectUtils.coalesce());
035    assertNull(ObjectUtils.coalesce((Integer)null));
036    assertNull(ObjectUtils.coalesce((Integer)null, (Integer)null));
037
038    assertEquals((Integer)13, ObjectUtils.coalesce(null, 13));
039    assertEquals((Integer)13, ObjectUtils.coalesce(null, 13, 14));
040    assertEquals((Integer)13, ObjectUtils.coalesce(13, 15));
041  }
042
043  @Test
044  public void testCoalesce1() throws Exception {
045    assertNull(ObjectUtils.coalesce((Collection) null));
046    assertNull(ObjectUtils.coalesce(Lists.newArrayList()));
047    assertNull(ObjectUtils.coalesce(Lists.newArrayList(null, null)));
048
049    assertEquals((Integer)13, ObjectUtils.coalesce(Lists.newArrayList(null, null, 13)));
050    assertEquals((Integer)13, ObjectUtils.coalesce(Lists.newArrayList(null, null, 13, 14)));
051    assertEquals((Integer)13, ObjectUtils.coalesce(Lists.newArrayList(13, null)));
052
053  }
054}
