017package org.gbif.utils.concurrent;
018
019import java.util.concurrent.ExecutorService;
020import java.util.concurrent.Executors;
021import java.util.concurrent.TimeUnit;
022
023import com.google.common.base.Throwables;
024import org.junit.Test;
025
026import static org.junit.Assert.assertTrue;
027
028/**
029 *
030 */
031public class ExecutorUtilsTest {
032
033  @Test
034  public void stopEmptyExec() throws Exception {
035    ExecutorService exec = Executors.newFixedThreadPool(2, new NamedThreadFactory("tata"));
036
037    ExecutorUtils.stop(exec, 1, TimeUnit.SECONDS);
038    assertTrue(exec.isTerminated());
039  }
040
041  @Test
042  public void stopExecWithJobs() throws Exception {
043    ExecutorService exec = Executors.newFixedThreadPool(2, new NamedThreadFactory("tata"));
044    exec.submit(new DontStop());
045    exec.submit(new DontStop());
046    exec.submit(new DontStop());
047
048    ExecutorUtils.stop(exec, 1, TimeUnit.SECONDS);
049    Thread.sleep(1000);
050    assertTrue(exec.isTerminated());
051  }
052
053  class DontStop implements Runnable {
054
055    @Override
056    public void run() {
057      long x = 0;
058      while (true) {
059        if (x == Long.MAX_VALUE) {
060          x = Long.MIN_VALUE;
061        }
062        x++;
063        try {
064          Thread.sleep(10);
065        } catch (InterruptedException e) {
066          Throwables.propagate(e);
067        }
068      }
069    }
070  }
071
072}
