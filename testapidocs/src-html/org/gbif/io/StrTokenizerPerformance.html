017package org.gbif.io;
018
019import org.gbif.utils.file.FileUtils;
020
021import java.io.BufferedReader;
022import java.io.File;
023import java.io.FileInputStream;
024import java.io.IOException;
025import java.io.InputStreamReader;
026
027import org.apache.commons.lang3.text.StrTokenizer;
028import org.junit.Test;
029
030
031public class StrTokenizerPerformance {
032
033  private long test(StrTokenizer tokenizer, File source) throws IOException {
034    FileInputStream fis = new FileInputStream(source);
035    InputStreamReader reader = new InputStreamReader(fis, "utf8");
036    BufferedReader br = new BufferedReader(reader);
037
038    // keep track of time while iterating
039    long start = System.currentTimeMillis();
040    String row = br.readLine();
041    while (row != null) {
042      tokenizer.reset(row);
043      String[] columns = tokenizer.getTokenArray();
044      row = br.readLine();
045    }
046    long dur = System.currentTimeMillis() - start;
047    br.close();
048    return dur;
049  }
050
051  @Test
052  public void testCharVsStringPerformance() throws IOException {
053    File source = FileUtils.getClasspathFile("irmng.tail");
054
055    // test CHAR
056    StrTokenizer tokenizer = new StrTokenizer();
057    tokenizer.setDelimiterChar('\t');
058    tokenizer.setEmptyTokenAsNull(true);
059    tokenizer.setIgnoreEmptyTokens(false);
060    long time = test(tokenizer, source);
061    System.out.println(time + " milliseconds for CHAR based tokenizer.");
062
063    // test STRING
064    tokenizer = new StrTokenizer();
065    tokenizer.setDelimiterString("\t");
066    tokenizer.setEmptyTokenAsNull(true);
067    time = test(tokenizer, source);
068    System.out.println(time + " milliseconds for STRING based tokenizer.");
069  }
070}
