019import java.io.IOException;

021import org.apache.commons.lang3.text.StrTokenizer;
022import org.junit.Test;
023

028public class StrTokenizerTest {
029
030  @Test
031  public void testCsvQuoted() throws IOException {
032    StrTokenizer tokenizer = new StrTokenizer();
033    tokenizer.setDelimiterString(",");
034    tokenizer.setQuoteChar('"');
035    tokenizer.setEmptyTokenAsNull(true);
036    tokenizer.setIgnoreEmptyTokens(false);
037
038    tokenizer.reset("121,432423, 9099053,\"Frieda karla L.,DC.\",Ahrens");
039    String[] columns = tokenizer.getTokenArray();
040    assertEquals("121", columns[0]);
041    assertEquals("432423", columns[1]);
042    assertEquals(" 9099053", columns[2]);
043    assertEquals("Frieda karla L.,DC.", columns[3]);
044    assertEquals("Ahrens", columns[4]);
045
046    tokenizer.reset("   ,4321");
047    columns = tokenizer.getTokenArray();
048    assertEquals("   ", columns[0]);
049    assertEquals("4321", columns[1]);
050
051    tokenizer.reset(" ,,,,zzz  ");
052    columns = tokenizer.getTokenArray();
053    assertEquals(" ", columns[0]);
054    assertNull(columns[1]);
055    assertNull(columns[2]);
056    assertNull(columns[3]);
057    assertEquals("zzz  ", columns[4]);
058
059    tokenizer.reset(",,,,zzz  ");
060    columns = tokenizer.getTokenArray();
061    assertNull(columns[0]);
062    assertNull(columns[1]);
063    assertNull(columns[2]);
064    assertNull(columns[3]);
065    assertEquals("zzz  ", columns[4]);
066  }

101    tokenizer.reset("121|432423| 9099053|\"Frieda karla L.|DC.\"|Ahrens");

103    assertEquals("121", columns[0]);

105    assertEquals(" 9099053", columns[2]);
