047  @TempDir
048  public File folder;

050  @Test
051  public void testFromCompressedZip() throws UnsupportedArchiveException, IOException {
052    // test zip with 1 extension file
053    File zip = FileUtils.getClasspathFile("archive-tax.zip");
054    assertIdInCompressed(zip.toPath(), "113775");
055  }
056
057  @Test
058  public void testFromCompressedTarGzip() throws UnsupportedArchiveException, IOException {
059    // test gziped tar file with 1 extension
060    File gzip = FileUtils.getClasspathFile("archive-tax.tar.gz");
061    assertIdInCompressed(gzip.toPath(), "113775");
062  }
063

066    // test zip with 1 extension file
067    Path none = Paths.get("/ping/pong/nuts");
068    // try to open archive
069    assertThrows(FileNotFoundException.class,
070        () -> InternalDwcFileFactory.fromCompressed(none, folder.toPath()));

072
073  @Test
074  public void testNonExistingFileFromLocation() {
075    File none = new File("/ping/pong/nuts");
076    assertThrows(FileNotFoundException.class, () -> InternalDwcFileFactory.fromLocation(none.toPath()));
077  }
078
079  /**
080   * Givin a compressed file, make sure we can uncompressed it, read the core and find the provided id.
081   */
082  private void assertIdInCompressed(Path compressedFile, String id) throws IOException {
083    File tmpDir = folder;
084
085    // open archive from zip
086    Archive arch = InternalDwcFileFactory.fromCompressed(compressedFile, tmpDir.toPath());
087    assertNotNull(arch.getCore().getId());
088    assertEquals(1, arch.getExtensions().size());
089
090    boolean found = false;
091    try (ClosableIterator<Record> it = arch.getCore().iterator()){
092      while(it.hasNext()){
093        if(id.equals(it.next().id())){
094          found = true;
095          break;
096        }
097      }
098    } catch (Exception e) {
099      e.printStackTrace();
100    }
101    assertTrue(found, "Can find the id " + id + " inside the archive " + compressedFile.getFileName());
102  }
103
104  @Test
105  public void testDetermineRowType() {
106    Optional<Term> rowType = InternalDwcFileFactory
107            .determineRowType(Arrays.asList(DwcTerm.decimalLatitude, DwcTerm.occurrenceID));
108    assertTrue(rowType.isPresent());
109    assertEquals(DwcTerm.Occurrence, rowType.get());
110  }
111
112  @Test
113  public void testDetermineRecordIdentifier() {
114    Optional<Term> id = InternalDwcFileFactory.determineRecordIdentifier(Arrays.asList(DwcTerm.decimalLatitude, DwcTerm.occurrenceID));
115    assertTrue(id.isPresent());
116    assertEquals(DwcTerm.occurrenceID, id.get());
117
118    id = InternalDwcFileFactory.determineRecordIdentifier(Arrays.asList(DwcTerm.taxonID, DwcTerm.scientificName));
119    assertTrue(id.isPresent());
120    assertEquals(DwcTerm.taxonID, id.get());
121
122    //eventId should be picked even if taxonID is there
123    id = InternalDwcFileFactory.determineRecordIdentifier(Arrays.asList(DwcTerm.eventID, DwcTerm.scientificName, DwcTerm.taxonID));
124    assertTrue(id.isPresent());
125    assertEquals(DwcTerm.taxonID, id.get());
126
127    id = InternalDwcFileFactory.determineRecordIdentifier(Arrays.asList(DwcTerm.decimalLongitude, DwcTerm.scientificName,
128            DcTerm.identifier));
129    assertTrue(id.isPresent());
130    assertEquals(DcTerm.identifier, id.get());
131
132    //eventId should be picked even if taxonID is there
133    id = InternalDwcFileFactory.determineRecordIdentifier(Arrays.asList(DwcTerm.decimalLongitude, DwcTerm.scientificName, DwcTerm.decimalLatitude));
134    assertFalse(id.isPresent());
135  }
