001package org.gbif.dwc;
002
003import org.gbif.dwc.record.StarRecord;
004import org.gbif.dwc.terms.DwcTerm;
005import org.gbif.dwc.terms.Term;
006import org.gbif.utils.file.FileUtils;
007import org.junit.Test;
008
009import java.io.File;
010import java.util.HashMap;
011import java.util.Map;
012
013import static org.junit.Assert.assertEquals;
014
015public class ArchiveTest {
016
017  /**
018   * Check we can handle the simplest of archives.
019   */
020  @Test
021  public void testIteratorDwc() throws Exception {
022    Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("simplest-archive").toPath());
023
024    int count = 0;
025    for (StarRecord rec : arch) {
026      assertEquals("Quercus alba", rec.core().value(DwcTerm.scientificName));
027      assertEquals("Species", rec.core().value(DwcTerm.taxonRank));
028      count++;
029    }
030    assertEquals(2, count);
031  }
032
033  /**
034   * Check we can handle an archive file with multiple header lines
035   */
036  @Test
037  public void testMultilineHeader() throws Exception {
038    Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("multiline-header").toPath());
039
040    int count = 0;
041    for (StarRecord rec : arch) {
042      assertEquals("Quercus alba", rec.core().value(DwcTerm.scientificName));
043      assertEquals("Species", rec.core().value(DwcTerm.taxonRank));
044      count++;
045    }
046    assertEquals(2, count);
047  }
048
049  @Test
050  public void testConstituents() throws Exception {
051    File dir = FileUtils.getClasspathFile("constituentsdwca");
052
053    Archive arch = new Archive();
054    arch.setLocation(dir);
055    arch.setMetadataLocation("eml.xml");
056    ArchiveField id = new ArchiveField(0, null, null, null);
057    ArchiveField datasetId = new ArchiveField(1, DwcTerm.datasetID, null, null);
058    ArchiveField sciname = new ArchiveField(2, DwcTerm.scientificName, null, null);
059
060    Map<Term, ArchiveField> fields = new HashMap<Term, ArchiveField>();
061    fields.put(DwcTerm.taxonomicStatus, sciname);
062    fields.put(DwcTerm.datasetID, datasetId);
063
064    Map<String, File> cons = arch.getConstituentMetadata();
065    assertEquals(6, cons.size());
066    for (Map.Entry<String, File> c : cons.entrySet()) {
067      final String name = c.getKey();
068      final File file = c.getValue();
069      assertEquals(name, file.getName().split("\\.")[0]);
070    }
071  }
072}
