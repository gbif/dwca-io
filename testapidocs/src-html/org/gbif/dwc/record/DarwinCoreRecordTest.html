001package org.gbif.dwc.record;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwc.text.ArchiveField;
005
006import java.util.Date;
007import java.util.HashSet;
008import java.util.Set;
009
010import com.google.common.collect.ImmutableSet;
011import org.junit.Test;
012
013import static org.junit.Assert.assertEquals;
014import static org.junit.Assert.fail;
015
016public class DarwinCoreRecordTest {
017
018  @Test
019  public void testProperties() {
020    final Set<String> IGNORE = ImmutableSet.of(DwcTerm.GROUP_MEASUREMENTORFACT, DwcTerm.GROUP_RESOURCERELATIONSHIP);
021
022    DarwinCoreRecord dwc = new DarwinCoreRecord();
023    for (DwcTerm t : DwcTerm.values()) {
024      // only test non class terms
025      if (t.isClass()) {
026        continue;
027      }
028
029      if (IGNORE.contains(t.getGroup())) {
030        try {
031          dwc.getProperty(t);
032          fail("term " + t.qualifiedName() + " should not exist on DarwinCoreRecord");
033        } catch (IllegalArgumentException e) {
034          // expected
035        }
036
037      } else {
038        String val = new Date().toString();
039        dwc.setProperty(t, val);
040        assertEquals("missing term " + t.qualifiedName(), val, dwc.getProperty(t));
041      }
042    }
043  }
044
045  @Test
046  public void testDefaultValue() {
047    final String DATASET = "ITIS";
048    ArchiveField id = new ArchiveField(0, DwcTerm.taxonID, null, null);
049    Set<ArchiveField> fields = new HashSet<ArchiveField>();
050    fields.add(new ArchiveField(1, DwcTerm.datasetName, DATASET, null));
051    RecordImpl rec = new RecordImpl(id, fields, DwcTerm.Taxon, true, true);
052
053    String[] row = {"5432", "IPNI"};
054    rec.setRow(row);
055    assertEquals(row[1], rec.value(DwcTerm.datasetName));
056
057    row = new String[] {"5432", null};
058    rec.setRow(row);
059    assertEquals(DATASET, rec.value(DwcTerm.datasetName));
060
061    row = new String[] {"5432", ""};
062    rec.setRow(row);
063    assertEquals(DATASET, rec.value(DwcTerm.datasetName));
064  }
065}
