087  @Test
088  public void testHeaderWriting() throws Exception {
089    File dwcaDir = FileUtils.createTempDir();
090    dwcaDir.deleteOnExit();
091    LOG.info("Test archive writer in {}", dwcaDir.getAbsolutePath());
092
093    DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, dwcaDir, true);
094
095    writer.newRecord("dummy1");
096    writer.addCoreColumn(DwcTerm.parentNameUsageID);
097    writer.addCoreColumn(DwcTerm.acceptedNameUsageID);
098    writer.addCoreColumn(DwcTerm.scientificName);
099    writer.addCoreColumn(GbifTerm.canonicalName);
100    writer.addCoreColumn(DwcTerm.taxonRank, "species");
101    writer.addCoreColumn(DwcTerm.taxonomicStatus);
102    writer.addCoreColumn(DwcTerm.kingdom);
103    writer.addCoreColumn(DwcTerm.phylum);
104    writer.addCoreColumn(DwcTerm.class_);
105    writer.addCoreColumn(DwcTerm.order);
106    writer.addCoreColumn(DwcTerm.family);
107    writer.addCoreColumn(GbifTerm.depth);
108    writer.addCoreColumn(GbifTerm.depthAccuracy);
109
110    writer.newRecord("dummy2");
111    writer.addCoreColumn(DwcTerm.kingdom, "Plantae");
112    writer.addCoreColumn(DwcTerm.phylum);
113    writer.addCoreColumn(DwcTerm.class_);
114    writer.addCoreColumn(DwcTerm.order);
115    writer.addCoreColumn(DwcTerm.family, "Asteraceae");
116
117    writer.newRecord("dummy3");
118    writer.addCoreColumn(GbifTerm.depth, "2");
119    writer.addCoreColumn(GbifTerm.depthAccuracy, "1");
120
121    // define extension columns
122    Map<Term, String> eData = Maps.newHashMap();
123    // distributions
124    eData.put(DwcTerm.locality, "locality1");
125    eData.put(DwcTerm.occurrenceStatus, "present");
126    eData.put(DwcTerm.establishmentMeans, "alien");
127    writer.addExtensionRecord(GbifTerm.Distribution, eData);
128
129    eData.put(DwcTerm.locality, "locality2");
130    writer.addExtensionRecord(GbifTerm.Distribution, eData);
131
132    writer.close();
133
134    File cf = new File(dwcaDir, writer.getDataFiles().get(DwcTerm.Taxon));
135    File df = new File(dwcaDir, writer.getDataFiles().get(GbifTerm.Distribution));
136
137    // check if taxon file contains headers
138    String[] headers = getFirstRow(cf);
139    LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
140    assertEquals(14, headers.length);
141    assertEquals("taxonID", headers[0]);
142    assertEquals("parentNameUsageID", headers[1]);
143    assertEquals("kingdom", headers[7]);
144
145    // check if extension file contains headers
146    headers = getFirstRow(df);
147    LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
148    assertEquals(4, headers.length);
149    assertEquals("taxonID", headers[0]);
150  }

152  private String[] getFirstRow(File f) throws IOException {
153    BufferedReader r = FileUtils.getUtf8Reader(f);
154    String firstRow = r.readLine();
155    return firstRow.split("\t");
156  }

158  @Test
159  public void testRoundtrip() throws Exception {
160    try {
161      // read taxon archive
162      Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("archive-dwc").toPath());
163      assertEquals(2, arch.getExtensions().size());
164      int coreRecords = 0;
165      int allRecords = 0;
166
167      // write taxon archive
168      File tempArch = FileUtils.createTempDir();
169      tempArch.deleteOnExit();
170      System.out.println("Writing temporary test archive to " + tempArch.getAbsolutePath());
171      DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, tempArch);
172      for (StarRecord rec : arch) {
173        // core
174        coreRecords++;
175        allRecords += rec.size();
176
177        writer.newRecord(rec.core().id());
178        for (Term term : arch.getCore().getTerms()) {
179          writer.addCoreColumn(term, rec.core().value(term));
180        }
181        // extensions
182        for (Term rt : rec.extensions().keySet()) {
183          ArchiveFile af = arch.getExtension(rt);
184          // iterate over records for one extension
185          for (Record row : rec.extension(rt)) {
186            writer.addExtensionRecord(rt, DwcaWriter.recordToMap(row, af));
187          }
188        }
189      }
190      writer.close();
191
192      // reread and compare
193      Archive arch2 = DwcFiles.fromLocation(tempArch.toPath());
194
195      int coreRecords2 = 0;
196      int allRecords2 = 0;
197      for (StarRecord rec : arch2) {
198        // core
199        coreRecords2++;
200        allRecords2 += rec.size();
201      }
202
203      // compare
204      assertEquals(coreRecords, coreRecords2);
205      assertEquals(allRecords, allRecords2);
206
207
208    } catch (Exception e) {
209      e.printStackTrace();
210      fail();
211    }
212  }

234  /**
235   * Test the writing of an archive that includes some default values in the core and in one extension.
236   * 
237   * @throws Exception
238   */
239  @Test
240  public void testWriterUsingDefaultValues() throws Exception {
241    File dwcaDir = FileUtils.createTempDir();
242    dwcaDir.deleteOnExit();
243    LOG.info("Test archive writer in {}", dwcaDir.getAbsolutePath());
244
245    DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, DwcTerm.taxonID, dwcaDir, true);
246
247    writer.newRecord("dummy1");
248    writer.addCoreColumn(DwcTerm.parentNameUsageID, "1");
249    writer.addCoreColumn(DwcTerm.acceptedNameUsageID, "2");
250    writer.addCoreColumn(DwcTerm.countryCode);
251    
252    // add a VernacularName extension record
253    Map<Term,String> extensionRecord = new HashMap<Term, String>();
254    extensionRecord.put(DwcTerm.vernacularName, "Komodo Dragon");
255    extensionRecord.put(DcTerm.language, null);
256    writer.addExtensionRecord(GbifTerm.VernacularName, extensionRecord);
257    
258    writer.addCoreDefaultValue(DwcTerm.collectionCode, "A2Z");
259    writer.addCoreDefaultValue(DwcTerm.countryCode, "CA");
260    writer.addDefaultValue(GbifTerm.VernacularName, DcTerm.language, "en");
261    
262    // add a second records and overwrite the default value
263    writer.newRecord("dummy2");
264    writer.addCoreColumn(DwcTerm.parentNameUsageID, "1");
265    writer.addCoreColumn(DwcTerm.acceptedNameUsageID, "2");
266    writer.addCoreColumn(DwcTerm.countryCode, "ID");
267    
268    // add a VernacularName extension record
269    extensionRecord = new HashMap<Term, String>();
270    extensionRecord.put(DwcTerm.vernacularName, "Varano De Komodo");
271    extensionRecord.put(DcTerm.language, "es");
272    writer.addExtensionRecord(GbifTerm.VernacularName, extensionRecord);
273    
274    writer.close();
275
276    // validate core content
277    Archive arch = DwcFiles.fromLocation(dwcaDir.toPath());
278    Iterator<Record> recIt = arch.getCore().iterator();
279    Record firstRecord = recIt.next();
280    assertEquals("dummy1", firstRecord.id());
281    assertEquals("dummy1", firstRecord.value(DwcTerm.taxonID));
282    assertEquals("A2Z", firstRecord.value(DwcTerm.collectionCode));
283    assertEquals("CA", firstRecord.value(DwcTerm.countryCode));
284    assertEquals("A2Z", arch.getCore().getField(DwcTerm.collectionCode).getDefaultValue());
285    assertEquals("CA", arch.getCore().getField(DwcTerm.countryCode).getDefaultValue());
286    
287    Record secondRecord = recIt.next();
288    assertEquals("dummy2", secondRecord.id());
289    assertEquals("dummy2", secondRecord.value(DwcTerm.taxonID));
290    assertEquals("A2Z", secondRecord.value(DwcTerm.collectionCode));
291    assertEquals("ID", secondRecord.value(DwcTerm.countryCode));
292    
293    // validate extension content
294    Iterator<Record> extRecIt = arch.getExtension(GbifTerm.VernacularName).iterator();
295    assertEquals("en", arch.getExtension(GbifTerm.VernacularName).getField(DcTerm.language).getDefaultValue());
296    firstRecord = extRecIt.next();
297    assertEquals("dummy1", firstRecord.id());
298    assertEquals("en", firstRecord.value(DcTerm.language));
299    
300    secondRecord = extRecIt.next();
301    assertEquals("dummy2", secondRecord.id());
302    assertEquals("es", secondRecord.value(DcTerm.language));
303  }
