017 * Created by markus on 17/02/2017.
018 */
019public class DwcaStreamWriterTest {
020
021  @Test(expected = IllegalStateException.class)
022  public void writeEmpty() throws Exception {
023    File dwca = FileUtils.createTempDir();
024    try (DwcaStreamWriter dwcaWriter = new DwcaStreamWriter(dwca, DwcTerm.Taxon, DwcTerm.taxonID, true)){
025
026    } finally {
027      org.apache.commons.io.FileUtils.deleteQuietly(dwca);
028    }
029  }
030
031  @Test
032  public void write() throws Exception {
033    File dwca = FileUtils.createTempDir();
034    Map<Term, Integer> mapping = ImmutableMap.of(
035        DwcTerm.taxonID, 0,
036        DwcTerm.scientificName, 1,
037        DwcTerm.taxonRank, 2);
038    try (DwcaStreamWriter dwcaWriter = new DwcaStreamWriter(dwca, DwcTerm.Taxon, DwcTerm.taxonID, true)){
039      String m = "<eml/>";
040      dwcaWriter.addMetadata(m, "eml.xml");
041      dwcaWriter.write(DwcTerm.Taxon, 0, mapping, ImmutableList.<String[]>builder()
042          .add(new String[] { "tax-1", "Abies Mill.", "genus" })
043          .add(new String[] { "tax-2", "Abies alba Mill.", "species" })
044          .add(new String[] { "tax-3", "Piceae abies L.", "species" })
045          .add(new String[] { "tax-4", "Piceae abies subsp. helvetica L.", "subspecies" })
046          .build()
047      );
048
049      Archive arch = DwcFiles.fromLocation(dwca.toPath());
050      assertEquals("eml.xml", arch.getMetadataLocation());
051      assertEquals("<eml/>", arch.getMetadata());
052
053    } finally {
054      org.apache.commons.io.FileUtils.deleteQuietly(dwca);
055    }
056  }
057
058}
