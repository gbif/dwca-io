001package org.gbif.dwc.extensions;
002
003import org.gbif.digester.ThesaurusHandlingRule;
004import org.gbif.dwc.terms.DcTerm;
005import org.gbif.utils.file.FileUtils;
006import org.gbif.xml.SAXUtils;
007
008import java.io.IOException;
009import java.io.InputStream;
010import java.net.URL;
011import java.util.List;
012import java.util.Map;
013import javax.xml.parsers.ParserConfigurationException;
014
015import org.junit.Test;
016import org.xml.sax.SAXException;
017
018import static junit.framework.TestCase.assertNotNull;
019
020/**
021 *
022 */
023public class ExtensionFactoryTest {
024
025  private String VOCAB_URL = "http://rs.gbif.org/vocabulary/gbif/description_type.xml";
026  private InputStream VOCAB = FileUtils.classpathStream("extension/vocabulary_gbif_description_type.xml");
027  private InputStream EXTENSION = FileUtils.classpathStream("extension/extension_gbif_1.0_description.xml");
028
029  private VocabularyFactory VOCAB_FACTORY = new VocabularyFactory(SAXUtils.getNsAwareSaxParserFactory());
030
031  public ExtensionFactoryTest() throws IOException {
032  }
033
034  @Test
035  public void testExtensionFactory() throws IOException, ParserConfigurationException, SAXException {
036
037    ThesaurusHandlingRule thr = new ThesaurusHandlingRule(new MockVocabulariesManager(VOCAB_URL, VOCAB_FACTORY.build(VOCAB)));
038    ExtensionFactory factory = new ExtensionFactory(thr, SAXUtils.getNsAwareSaxParserFactory());
039
040    Extension ext = factory.build(EXTENSION, new URL(VOCAB_URL), false);
041
042    assertNotNull(ext);
043    assertNotNull(ext.getProperty(DcTerm.type).getVocabulary());
044  }
045
046  /**
047   * Mock VocabulariesManager that support only 1 vocabulary
048   */
049  private static class MockVocabulariesManager implements VocabulariesManager{
050
051    private final String vocabUrl;
052    private final Vocabulary vocab;
053
054    public MockVocabulariesManager(String vocabUrl, Vocabulary vocab){
055      this.vocabUrl = vocabUrl;
056      this.vocab = vocab;
057    }
058
059    @Override
060    public Vocabulary get(String s) {
061      return null;
062    }
063
064    @Override
065    public Vocabulary get(URL url) {
066      if(vocabUrl.equalsIgnoreCase(url.toString())){
067        return vocab;
068      }
069      return null;
070    }
071
072    @Override
073    public Map<String, String> getI18nVocab(String s, String s1) {
074      return null;
075    }
076
077    @Override
078    public List<Vocabulary> list() {
079      return null;
080    }
081  }
082}
