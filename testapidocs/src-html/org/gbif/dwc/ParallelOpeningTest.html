016package org.gbif.dwc;
017
018import org.gbif.dwc.record.StarRecord;
019
020import java.net.URL;
021import java.nio.file.Files;
022import java.nio.file.Path;
023import java.nio.file.Paths;
024
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import static org.junit.jupiter.api.Assertions.assertFalse;
029import static org.junit.jupiter.api.Assertions.assertTrue;
030
031/**
032 * This test checks we don't get a race condition between two <strong>processes</strong> opening the same DWC-A at the
033 * same time.
034 *
035 * It needs to be run from the command line, in parallel.
036 *
037 * mvn exec:java -Dexec.mainClass=org.gbif.dwc.ParallelOpeningTest
038 *
039 * Monitor open files with something like:
040 *
041 * while :; do for i in `pgrep -f ParallelOpeningTest`; do echo $i && ls -l /proc/$i/fd | grep /tmp/par && echo; sleep 1; done; done
042 */
043public class ParallelOpeningTest {
044  private static final Logger LOG = LoggerFactory.getLogger(ParallelOpeningTest.class);
045
046  private Archive arch;
047
048  /**
049   * Check we don't get a race condition when opening an archive which needs sorting.
050   *
051   * This downloads a large archive (we don't want it in Git) and checks it has at least one extension.
052   */
053  public void testParallelOpening() throws Exception {
054    Path archivePath = Paths.get("/tmp/parallel-opening-test-archive.zip");
055    Path extractToFolder = Paths.get("/tmp/parallel-opening-test-archive");
056    if (!archivePath.toFile().exists()) {
057      URL download = new URL("http://api.gbif.org/v1/occurrence/download/request/0012957-180131172636756.zip");
058      Files.copy(download.openStream(), archivePath);
059      extractToFolder.toFile().mkdir();
060      arch = DwcFiles.fromCompressed(archivePath, extractToFolder);
061    }
062
063    arch = DwcFiles.fromLocation(extractToFolder);
064
065    assertFalse(arch.getExtensions().isEmpty());
066
067    // Initialize (i.e. normalize and sort) the archive in the background.
068    LOG.info("Initializing archive");
069    arch.initialize();
070    LOG.info("Initialization completed, locks should have been released.");
071
072    int counter = 0;
073    for (StarRecord rec : arch) {
074      if (counter == 0) {
075        LOG.info("Read first record {}", rec);
076      }
077      counter++;
078    }
079
080    LOG.info("Counted {} records", counter);
081
082    LOG.info("All files should have been closed.");
083    Thread.sleep(30_000);
084
085    assertTrue(counter > 500000, "Many records extracted");
086  }
087
088  public static void main(String... args) throws Exception {
089    ParallelOpeningTest pot = new ParallelOpeningTest();
090    pot.testParallelOpening();
091  }
092}
