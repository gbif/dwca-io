001package org.gbif.dwc;
002
003import org.gbif.dwc.record.StarRecord;
004import org.gbif.utils.file.FileUtils;
005import org.junit.Assert;
006import org.junit.Test;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.net.URL;
011import java.nio.file.Files;
012import java.nio.file.Path;
013
014public class ParallelOpeningTest implements Runnable {
015  private static final Logger LOG = LoggerFactory.getLogger(ParallelOpeningTest.class);
016
017  private Archive arch;
018
019  /**
020   * Check we don't get a race condition when opening an archive which needs sorting.
021   *
022   * This downloads a large archive (we don't want it in Git) and checks it has at least one extension.
023   */
024  @Test
025  public void testParallelOpening() throws Exception {
026    URL ipni = new URL("http://rs.gbif.org/datasets/itis.zip");
027    Path archivePath = FileUtils.createTempDir().toPath().resolve("itis.zip");
028    Files.copy(ipni.openStream(), archivePath);
029
030    Path extractToFolder = FileUtils.createTempDir().toPath();
031    extractToFolder.toFile().deleteOnExit();
032
033    arch = DwcFiles.fromCompressed(archivePath, extractToFolder);
034
035    Assert.assertFalse(arch.getExtensions().isEmpty());
036
037    // Initialize (i.e. normalize and sort) the archive in the background.
038    new Thread(this).start();
039    // Wait Â½s, which should be enough for the CSV header to be written, but not the sorted data rows.
040    Thread.sleep(500);
041
042    int counter = 0;
043    for (StarRecord rec : arch) {
044      if (counter == 0) {
045        LOG.info("Read first record {}", rec);
046      }
047      counter++;
048    }
049
050    LOG.info("Counted {} records", counter);
051
052    Assert.assertTrue("Many records extracted", counter > 500000);
053
054    Files.delete(archivePath);
055    FileUtils.deleteDirectoryRecursively(extractToFolder.toFile());
056  }
057
058  @Override
059  public void run() {
060    LOG.info("Initializing archive in a background thread");
061    try {
062      arch.initialize();
063    } catch (Exception e) {
064      throw new RuntimeException(e);
065    }
066  }
067}
