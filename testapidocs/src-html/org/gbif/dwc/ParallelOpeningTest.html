001package org.gbif.dwc;
002
003import org.gbif.dwc.record.StarRecord;
004import org.junit.Assert;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.net.URL;
009import java.nio.file.Files;
010import java.nio.file.Path;
011import java.nio.file.Paths;
012
013/**
014 * This test checks we don't get a race condition between two <strong>processes</strong> opening the same DWC-A at the
015 * same time.
016 *
017 * It needs to be run from the command line, in parallel.
018 *
019 * mvn exec:java -Dexec.mainClass=org.gbif.dwc.ParallelOpeningTest
020 *
021 * Monitor open files with something like:
022 *
023 * while :; do for i in `pgrep -f ParallelOpeningTest`; do echo $i && ls -l /proc/$i/fd | grep /tmp/par && echo; sleep 1; done; done
024 */
025public class ParallelOpeningTest {
026  private static final Logger LOG = LoggerFactory.getLogger(ParallelOpeningTest.class);
027
028  private Archive arch;
029
030  /**
031   * Check we don't get a race condition when opening an archive which needs sorting.
032   *
033   * This downloads a large archive (we don't want it in Git) and checks it has at least one extension.
034   */
035  public void testParallelOpening() throws Exception {
036    Path archivePath = Paths.get("/tmp/parallel-opening-test-archive.zip");
037    Path extractToFolder = Paths.get("/tmp/parallel-opening-test-archive");
038    if (!archivePath.toFile().exists()) {
039      URL download = new URL("http://api.gbif.org/v1/occurrence/download/request/0012957-180131172636756.zip");
040      Files.copy(download.openStream(), archivePath);
041      extractToFolder.toFile().mkdir();
042      arch = DwcFiles.fromCompressed(archivePath, extractToFolder);
043    }
044
045    arch = DwcFiles.fromLocation(extractToFolder);
046
047    Assert.assertFalse(arch.getExtensions().isEmpty());
048
049    // Initialize (i.e. normalize and sort) the archive in the background.
050    LOG.info("Initializing archive");
051    arch.initialize();
052    LOG.info("Initialization completed, locks should have been released.");
053
054    int counter = 0;
055    for (StarRecord rec : arch) {
056      if (counter == 0) {
057        LOG.info("Read first record {}", rec);
058      }
059      counter++;
060    }
061
062    LOG.info("Counted {} records", counter);
063
064    LOG.info("All files should have been closed.");
065    Thread.sleep(30_000);
066
067    Assert.assertTrue("Many records extracted", counter > 500000);
068  }
069
070  public static void main(String... args) throws Exception {
071    ParallelOpeningTest pot = new ParallelOpeningTest();
072    pot.testParallelOpening();
073  }
074}
