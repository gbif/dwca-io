014package org.gbif.dwc;
015
016import org.gbif.dwc.record.Record;
017import org.gbif.dwc.record.StarRecord;
018import org.gbif.dwc.terms.DwcTerm;
019import org.gbif.dwc.terms.GbifTerm;
020import org.gbif.utils.file.FileUtils;
021
022import java.io.IOException;
023import java.nio.file.Path;
024
025import org.junit.jupiter.api.Test;
026
027public class UsageExample {
028
029  @Test
030  public void testUsageExample() throws IOException {
031    Path myArchiveFile = FileUtils.getClasspathFile("checklist_980.zip").toPath();
032    Path extractToFolder = FileUtils.createTempDir().toPath();
033    extractToFolder.toFile().deleteOnExit();
034
035    // Open the Darwin Core Archive
036    Archive dwcArchive = DwcFiles.fromCompressed(myArchiveFile, extractToFolder);
037
038    System.out.println("Reading archive from " + dwcArchive.getLocation().getAbsolutePath());
039    System.out.println("Archive of rowtype " + dwcArchive.getCore().getRowType() + " with " + dwcArchive.getExtensions().size() + " extensions");
040
041    // Loop over core records and display id, basis of record and scientific name
042    for (Record rec : dwcArchive.getCore()) {
043      System.out.printf("%s: %s (%s)%n", rec.id(), rec.value(DwcTerm.basisOfRecord), rec.value(DwcTerm.scientificName));
044    }
045
046    // Loop over star records and display id, core record data, and extension data
047    for (StarRecord rec : dwcArchive) {
048      System.out.printf("%s: %s%n", rec.core().id(), rec.core().value(DwcTerm.scientificName));
049      if (rec.hasExtension(GbifTerm.VernacularName)) {
050        for (Record extRec : rec.extension(GbifTerm.VernacularName)) {
051          System.out.println(" - " + extRec.value(DwcTerm.vernacularName));
052        }
053      }
054    }
055  }
056}
