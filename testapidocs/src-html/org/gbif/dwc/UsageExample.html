001package org.gbif.dwc;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwc.record.Record;
005import org.gbif.dwc.record.StarRecord;
006import org.gbif.dwc.terms.GbifTerm;
007import org.gbif.utils.file.FileUtils;
008import org.junit.Test;
009
010import java.io.IOException;
011import java.nio.file.Path;
012
013public class UsageExample {
014
015  @Test
016  public void testUsageExample() throws IOException {
017    Path myArchiveFile = FileUtils.getClasspathFile("checklist_980.zip").toPath();
018    Path extractToFolder = FileUtils.createTempDir().toPath();
019    extractToFolder.toFile().deleteOnExit();
020
021    // Open the Darwin Core Archive
022    Archive dwcArchive = DwcFiles.fromCompressed(myArchiveFile, extractToFolder);
023
024    System.out.println("Reading archive from " + dwcArchive.getLocation().getAbsolutePath());
025    System.out.println("Archive of rowtype " + dwcArchive.getCore().getRowType() + " with " + dwcArchive.getExtensions().size() + " extensions");
026
027    // Loop over core records and display id, basis of record and scientific name
028    for (Record rec : dwcArchive.getCore()) {
029      System.out.println(String.format("%s: %s (%s)", rec.id(), rec.value(DwcTerm.basisOfRecord), rec.value(DwcTerm.scientificName)));
030    }
031
032    // Loop over star records and display id, core record data, and extension data
033    for (StarRecord rec : dwcArchive) {
034      System.out.println(String.format("%s: %s", rec.core().id(), rec.core().value(DwcTerm.scientificName)));
035      if (rec.hasExtension(GbifTerm.VernacularName)) {
036        for (Record extRec : rec.extension(GbifTerm.VernacularName)) {
037          System.out.println(" - " + extRec.value(DwcTerm.vernacularName));
038        }
039      }
040    }
041  }
042}
