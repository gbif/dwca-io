055  @Test
056  public void testStartRecordIterator() throws IOException {
057
058    Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("archive-dwc").toPath());
059    NormalizedDwcArchive nda = DwcFiles.prepareArchive(arch, false, false);
060    try(ClosableIterator<StarRecord> it = nda.iterator()){
061      assertNotNull(arch.getCore());
062      assertEquals(2, arch.getExtensions().size());
063      int found = 0;
064      int extensionRecords = 0;
065      while (it.hasNext()) {
066        StarRecord rec = it.next();
067        //System.out.println(rec.core().id() + " --> " + rec.size());
068        // count all extension records
069        extensionRecords += rec.size();
070
071        if (rec.core().id().equals("544382")) {
072          found++;
073          assertEquals("Tursiops truncatus truncatus Montagu", rec.core().value(DwcTerm.scientificName));
074          // test extension iter
075          int i = 0;
076          for (Record er : rec) {
077            i++;
078          }
079          assertEquals(27, i);
080        } else if (rec.core().id().equals("105833")) {
081          found++;
082//        105833  Chinese river dolphin English
083//        105833  Chinese lake dolphin  English
084//        105833  Pei c’hi      Chinese
085//        105833  White flag dolphin  English
086          int i = 0;
087          for (Record er : rec) {
088            i++;
089            if ("Chinese river dolphin".equals(er.value(DwcTerm.vernacularName))) {
090              assertEquals("English", er.value(DcTerm.language));
091              assertEquals("English", er.value(DcTerm.language));
092            } else if ("Chinese lake dolphin".equals(er.value(DwcTerm.vernacularName))) {
093              assertEquals("English", er.value(DcTerm.language));
094            } else if ("Pei c’hi".equals(er.value(DwcTerm.vernacularName))) {
095              assertEquals("Chinese", er.value(DcTerm.language));
096            } else {
097              assertEquals("White flag dolphin", er.value(DwcTerm.vernacularName));
098              assertEquals("English", er.value(DcTerm.language));
099            }
100          }
101          assertEquals(4, i);
102        } else if (rec.core().id().equals("105838")) {
103          found++;
104//      105838  Delfin de La Plata  Spanish
105//      105838  Franciscana Spanish
106//      105838  Franciscano Portuguese
107//      105838  La Plata dolphin  English
108//      105838  Tonina  Spanish
109//      105838  Toninha Portuguese
110          int i = 0;
111          for (Record er : rec) {
112            i++;
113            if ("Delfin de La Plata".equals(er.value(DwcTerm.vernacularName))) {
114              assertEquals("Spanish", er.value(DcTerm.language));
115            } else if ("Franciscana".equals(er.value(DwcTerm.vernacularName))) {
116              assertEquals("Spanish", er.value(DcTerm.language));
117            } else if ("Franciscano".equals(er.value(DwcTerm.vernacularName))) {
118              assertEquals("Portuguese", er.value(DcTerm.language));
119            } else if ("La Plata dolphin".equals(er.value(DwcTerm.vernacularName))) {
120              assertEquals("English", er.value(DcTerm.language));
121            } else if ("Tonina".equals(er.value(DwcTerm.vernacularName))) {
122              assertEquals("Spanish", er.value(DcTerm.language));
123            } else {
124              assertEquals("Toninha", er.value(DwcTerm.vernacularName));
125              assertEquals("Portuguese", er.value(DcTerm.language));
126            }
127          }
128          assertEquals(6, i);
129        }
130      }
131      assertEquals(3, found);
132      assertEquals(1057, extensionRecords);
133    } catch (Exception e) {
134      e.printStackTrace();
135      fail();
136    }
137
138  }

140  @Test
141  public void testNormalizeAndSort() throws IOException, InterruptedException {
142
143    Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("archive-dwc").toPath());
144    ArchiveFile core = arch.getCore();
145    File sortedFile = ArchiveFile.getLocationFileSorted(core.getLocationFile());
146
147    //ensure the sorted file for the core doesn't exist
148    if(sortedFile.exists()) {
149      sortedFile.delete();
150    }
151    assertTrue(DwcFiles.normalizeAndSort(arch.getCore()));
152    assertTrue(sortedFile.exists());
153
154    //call the method again. Should return false since we already have the sorted file available.
155    assertFalse(DwcFiles.normalizeAndSort(arch.getCore()));
156  }
