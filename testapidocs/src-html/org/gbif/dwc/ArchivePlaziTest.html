001package org.gbif.dwc;
002
003import org.gbif.dwc.record.StarRecord;
004import org.gbif.utils.file.FileUtils;
005
006import java.io.File;
007import java.io.IOException;
008
009import org.junit.Test;
010
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNotNull;
013
014public class ArchivePlaziTest {
015
016  private void assertNumberStarRecords(Archive arch, int expectedRecords) throws IOException {
017    int rows = 0;
018    for (StarRecord rec : arch) {
019      assertNotNull(rec);
020      rows++;
021    }
022    assertEquals(expectedRecords, rows);
023  }
024
025  @Test
026  public void testBuildReaderFile() throws IOException {
027    File zip = FileUtils.getClasspathFile("plazi/6632D8151686A3F8C71D4B5A5B1181A4.zip");
028    File tmpDir = FileUtils.createTempDir();
029    tmpDir.deleteOnExit();
030
031    Archive arch = DwcFiles.fromCompressed(zip.toPath(), tmpDir.toPath());
032    assertNumberStarRecords(arch, 10);
033  }
034}
