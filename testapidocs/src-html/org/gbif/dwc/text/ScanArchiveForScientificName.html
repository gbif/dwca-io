001package org.gbif.dwc.text;
002
003import org.gbif.dwc.record.DarwinCoreRecord;
004import org.gbif.dwc.terms.DwcTerm;
005
006import java.io.File;
007import java.io.IOException;
008import java.util.Iterator;
009
010public class ScanArchiveForScientificName {
011
012  public static void main(String[] args) throws IOException, UnsupportedArchiveException {
013    // opens csv files with headers or dwc-a direcotries with a meta.xml descriptor
014    Archive arch = ArchiveFactory
015      .openArchive(new File("/Volumes/Scratch/ecat-data-col/checklists/beac2c55-d889-4358-a414-b1db79ba3536/dwc-a"));
016
017    // does scientific name exist?
018    if (!arch.getCore().hasTerm(DwcTerm.scientificName)) {
019      System.out.println("This application requires dwc-a with scientific names");
020      System.exit(1);
021    }
022
023    // loop over core darwin core records
024    Iterator<DarwinCoreRecord> iter = arch.iteratorDwc();
025    DarwinCoreRecord dwc;
026    while (iter.hasNext()) {
027      dwc = iter.next();
028      if (dwc.getScientificName().startsWith("Ambispora callosa")) {
029        System.out.println(dwc.getScientificName());
030        System.out.println(dwc.getScientificNameAuthorship());
031        System.out.println(dwc);
032      }
033    }
034  }
035
036}
