085  @Test
086  public void testHeaderWriting() throws Exception {
087    File dwcaDir = FileUtils.createTempDir();
088    dwcaDir.deleteOnExit();
089    LOG.info("Test archive writer in {}", dwcaDir.getAbsolutePath());
090
091    DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, dwcaDir, true);
092
093    writer.newRecord("dummy1");
094    writer.addCoreColumn(DwcTerm.parentNameUsageID, null);
095    writer.addCoreColumn(DwcTerm.acceptedNameUsageID, null);
096    writer.addCoreColumn(DwcTerm.scientificName, null);
097    writer.addCoreColumn(GbifTerm.canonicalName, null);
098    writer.addCoreColumn(DwcTerm.taxonRank, "species");
099    writer.addCoreColumn(DwcTerm.taxonomicStatus, null);
100    writer.addCoreColumn(DwcTerm.kingdom, null);
101    writer.addCoreColumn(DwcTerm.phylum, null);
102    writer.addCoreColumn(DwcTerm.class_, null);
103    writer.addCoreColumn(DwcTerm.order, null);
104    writer.addCoreColumn(DwcTerm.family, null);
105    writer.addCoreColumn(GbifTerm.depth, null);
106    writer.addCoreColumn(GbifTerm.depthAccuracy, null);
107
108    writer.newRecord("dummy2");
109    writer.addCoreColumn(DwcTerm.kingdom, "Plantae");
110    writer.addCoreColumn(DwcTerm.phylum, null);
111    writer.addCoreColumn(DwcTerm.class_, null);
112    writer.addCoreColumn(DwcTerm.order, null);
113    writer.addCoreColumn(DwcTerm.family, "Asteraceae");
114
115    writer.newRecord("dummy3");
116    writer.addCoreColumn(GbifTerm.depth, "2");
117    writer.addCoreColumn(GbifTerm.depthAccuracy, "1");
118
119    // define extension columns
120    Map<Term, String> eData = Maps.newHashMap();
121    // distributions
122    eData.put(DwcTerm.locality, "locality1");
123    eData.put(DwcTerm.occurrenceStatus, "present");
124    eData.put(DwcTerm.establishmentMeans, "alien");
125    writer.addExtensionRecord(GbifTerm.Distribution, eData);
126
127    eData.put(DwcTerm.locality, "locality2");
128    writer.addExtensionRecord(GbifTerm.Distribution, eData);
129
130    writer.close();
131
132    File cf = new File(dwcaDir, writer.getDataFiles().get(DwcTerm.Taxon));
133    File df = new File(dwcaDir, writer.getDataFiles().get(GbifTerm.Distribution));
134
135    // check if taxon file contains headers
136    String[] headers = getFirstRow(cf);
137    LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
138    assertEquals(14, headers.length);
139    assertEquals("taxonID", headers[0]);
140    assertEquals("parentNameUsageID", headers[1]);
141    assertEquals("kingdom", headers[7]);
142
143    // check if extension file contains headers
144    headers = getFirstRow(df);
145    LOG.debug(Joiner.on("; ").useForNull("NULL").join(headers));
146    assertEquals(4, headers.length);
147    assertEquals("taxonID", headers[0]);
148  }

150  private String[] getFirstRow(File f) throws IOException {
151    BufferedReader r = FileUtils.getUtf8Reader(f);
152    String firstRow = r.readLine();
153    return firstRow.split("\t");
154  }

156  @Test
157  public void testRoundtrip() throws Exception {
158    try {
159      TermFactory termFactory = TermFactory.instance();
160
161      // read taxon archive
162      Archive arch = ArchiveFactory.openArchive(FileUtils.getClasspathFile("archive-dwc"));
163      assertEquals(2, arch.getExtensions().size());
164      int coreRecords = 0;
165      int allRecords = 0;
166
167      // write taxon archive
168      File tempArch = FileUtils.createTempDir();
169      tempArch.deleteOnExit();
170      System.out.println("Writing temporary test archive to " + tempArch.getAbsolutePath());
171      DwcaWriter writer = new DwcaWriter(DwcTerm.Taxon, tempArch);
172      for (StarRecord rec : arch) {
173        // core
174        coreRecords++;
175        allRecords += rec.size();
176
177        writer.newRecord(rec.core().id());
178        for (Term term : arch.getCore().getTerms()) {
179          writer.addCoreColumn(term, rec.core().value(term));
180        }
181        // extensions
182        for (String rt : rec.extensions().keySet()) {
183          Term rowType = termFactory.findTerm(rt);
184          ArchiveFile af = arch.getExtension(rowType);
185          // iterate over records for one extension
186          for (Record row : rec.extension(rt)) {
187            writer.addExtensionRecord(rowType, DwcaWriter.recordToMap(row, af));
188          }
189        }
190      }
191      writer.close();
192
193      // reread and compare
194      Archive arch2 = ArchiveFactory.openArchive(tempArch);
195      int coreRecords2 = 0;
196      int allRecords2 = 0;
197      for (StarRecord rec : arch2) {
198        // core
199        coreRecords2++;
200        allRecords2 += rec.size();
201      }
202
203      // compare
204      assertEquals(coreRecords, coreRecords2);
205      assertEquals(allRecords, allRecords2);
206
207
208    } catch (Exception e) {
209      e.printStackTrace();
210      fail();
211    }
212  }
