001package org.gbif.dwc.text;
002
003import org.gbif.dwc.record.Record;
004import org.gbif.dwc.record.StarRecordImpl;
005import org.gbif.dwc.terms.DcTerm;
006import org.gbif.dwc.terms.DwcTerm;
007
008import java.io.File;
009import java.io.IOException;
010
011public class UsageExample {
012
013  public static void main(String[] args) throws IOException, UnsupportedArchiveException {
014    // opens csv files with headers or dwc-a direcotries with a meta.xml descriptor
015    Archive arch = ArchiveFactory.openArchive(new File(args[0]));
016
017    System.out.println("Reading archive from "+arch.getLocation().getAbsolutePath());
018    System.out.println("Archive of rowtype "+arch.getCore().getRowType()+" with "+arch.getExtensions().size()+" extensions");
019    // loop over star records. i.e. core with all linked extension records
020    for (StarRecordImpl rec : arch) {
021      // print core ID + scientific name
022      System.out.println(rec.core().id()
023        + " sciname:" + rec.core().value(DwcTerm.scientificName)
024        + " bor:" + rec.core().value(DwcTerm.basisOfRecord)
025      );
026      // print out all rowTypes
027      for (Record erec : rec) {
028        // print out extension rowtype
029        System.out.println(erec.rowType() + ", id="+erec.value(DcTerm.identifier));
030      }
031    }
032  }
033
034}
