001package org.gbif.dwc;
002
003import org.gbif.dwc.terms.DcTerm;
004import org.gbif.dwc.terms.DwcTerm;
005import org.gbif.dwc.record.Record;
006import org.gbif.dwc.record.StarRecord;
007import org.gbif.utils.file.FileUtils;
008
009import java.io.IOException;
010
011import org.junit.Test;
012
013import static org.junit.Assert.assertEquals;
014import static org.junit.Assert.assertNotNull;
015
016public class StarIteratorTest {
017
018  @Test
019  public void testIterator() throws IOException, UnsupportedArchiveException {
020    // test proper archive
021    Archive arch = DwcFiles.fromLocation(FileUtils.getClasspathFile("archive-dwc").toPath());
022    assertNotNull(arch.getCore());
023    assertEquals(2, arch.getExtensions().size());
024    int found = 0;
025    int extensionRecords = 0;
026    for (StarRecord rec : arch) {
027      //System.out.println(rec.core().id() + " --> " + rec.size());
028      // count all extension records
029      extensionRecords += rec.size();
030
031      if (rec.core().id().equals("544382")) {
032        found++;
033        assertEquals("Tursiops truncatus truncatus Montagu", rec.core().value(DwcTerm.scientificName));
034        // test extension iter
035        int i = 0;
036        for (Record er : rec) {
037          i++;
038        }
039        assertEquals(27, i);
040      } else if (rec.core().id().equals("105833")) {
041        found++;
042//        105833  Chinese river dolphin English
043//        105833  Chinese lake dolphin  English
044//        105833  Pei c’hi      Chinese
045//        105833  White flag dolphin  English
046        int i = 0;
047        for (Record er : rec) {
048          i++;
049          if ("Chinese river dolphin".equals(er.value(DwcTerm.vernacularName))) {
050            assertEquals("English", er.value(DcTerm.language));
051            assertEquals("English", er.value(DcTerm.language));
052          } else if ("Chinese lake dolphin".equals(er.value(DwcTerm.vernacularName))) {
053            assertEquals("English", er.value(DcTerm.language));
054          } else if ("Pei c’hi".equals(er.value(DwcTerm.vernacularName))) {
055            assertEquals("Chinese", er.value(DcTerm.language));
056          } else {
057            assertEquals("White flag dolphin", er.value(DwcTerm.vernacularName));
058            assertEquals("English", er.value(DcTerm.language));
059          }
060        }
061        assertEquals(4, i);
062      } else if (rec.core().id().equals("105838")) {
063        found++;
064//      105838  Delfin de La Plata  Spanish
065//      105838  Franciscana Spanish
066//      105838  Franciscano Portuguese
067//      105838  La Plata dolphin  English
068//      105838  Tonina  Spanish
069//      105838  Toninha Portuguese
070        int i = 0;
071        for (Record er : rec) {
072          i++;
073          if ("Delfin de La Plata".equals(er.value(DwcTerm.vernacularName))) {
074            assertEquals("Spanish", er.value(DcTerm.language));
075          } else if ("Franciscana".equals(er.value(DwcTerm.vernacularName))) {
076            assertEquals("Spanish", er.value(DcTerm.language));
077          } else if ("Franciscano".equals(er.value(DwcTerm.vernacularName))) {
078            assertEquals("Portuguese", er.value(DcTerm.language));
079          } else if ("La Plata dolphin".equals(er.value(DwcTerm.vernacularName))) {
080            assertEquals("English", er.value(DcTerm.language));
081          } else if ("Tonina".equals(er.value(DwcTerm.vernacularName))) {
082            assertEquals("Spanish", er.value(DcTerm.language));
083          } else {
084            assertEquals("Toninha", er.value(DwcTerm.vernacularName));
085            assertEquals("Portuguese", er.value(DcTerm.language));
086          }
087        }
088        assertEquals(6, i);
089      }
090
091    }
092    assertEquals(3, found);
093    assertEquals(1057, extensionRecords);
094  }
095}
